\chapter{算子代换法}
\section{算子的定义与运算性质}
\begin{definition}[算子 $E$]
    针对椭圆方程，定义作用于任意两点 $M(x_M, y_M)$ 与 $N(x_N, y_N)$ 的二元函数：
$$ E(M, N) = \frac{x_M x_N}{a^2} + \frac{y_M y_N}{b^2} $$
\end{definition}
因此 $E(M,N)=E(N,M)$。设任意三点 $A(x_A, y_A), B(x_B, y_B), C(x_C, y_C)$，并设
\[
P=\alpha A+\beta B,
\]
也就是 $x_P = \alpha x_A + \beta x_B,\ y_P = \alpha y_A + \beta y_B$。
\begin{align*}
E(P, C) &= E( \alpha A + \beta B, C) \\
&=\frac{(\alpha x_A + \beta x_B)x_C}{a^2} + \frac{(\alpha y_A + \beta y_B)y_C}{b^2}  = \frac{\alpha x_A x_C}{a^2} + \frac{\beta x_B x_C}{a^2} + \frac{\alpha y_A y_C}{b^2} + \frac{\beta y_B y_C}{b^2} \\
&= \alpha \left( \frac{x_A x_C}{a^2} + \frac{y_A y_C}{b^2} \right) + \beta \left( \frac{x_B x_C}{a^2} + \frac{y_B y_C}{b^2} \right) =\alpha E(A,C) + \beta E(B,C) 
\end{align*}
其余情况类似。
\begin{definition}[算子 $T$]
再定义
$$ T(M, N) = E(M, N) - 1 = \frac{x_M x_N}{a^2} + \frac{y_M y_N}{b^2} - 1. $$
当 $M=N$ 且 $M$ 在椭圆上时，有 $T(M,M)=0$。
\end{definition}
同样有 $T(M,N)=T(N,M)$。若点 $P$ 在直线 $AB$ 上，且
\[
P=\alpha A+\beta B,\qquad \alpha+\beta=1,
\]
则
\begin{align*}
  T(P,C) &= E(P,C) - 1  =E( \alpha  A + \beta B,C) - 1  \\
  &=\alpha E(A,C) + \beta E(B,C) - 1 =\alpha E(A,C) + \beta E(B,C)-\alpha - \beta\\
  &= \alpha E(A,C) - \alpha + \beta E(B,C) - \beta= \alpha T(A,C) + \beta T(B,C)
\end{align*}
\begin{definition}[外积 $W$]
对于以原点为起点的两个位置向量 $\overrightarrow{OM} = (x_M, y_M)$ 与 $\overrightarrow{ON} = (x_N, y_N)$，定义：
$$ W(M, N) = x_M y_N - x_N y_M $$
\end{definition}
\begin{itemize}
\item $W(M,N) = x_M y_N - x_N y_M = -W(N,M)$。
\item 由 $W(M,M) = -W(M,M)$ 可得 $2W(M,M) = 0$，所以 $W(M,M) = 0$。
\item 分配律：设存在任意三点（或位置向量） $A(x_A, y_A), B(x_B, y_B), C(x_C, y_C)$。设 $P = \alpha A + \beta B$，则
\begin{align*}W(P, C) &= W(\alpha A + \beta B, C)= (\alpha x_A + \beta x_B)y_C - x_C(\alpha y_A + \beta y_B) \\&= \alpha x_A y_C + \beta x_B y_C - \alpha x_C y_A - \beta x_C y_B \\&= \alpha (x_A y_C - x_C y_A) + \beta (x_B y_C - x_C y_B) \\&= \alpha W(A, C) + \beta W(B, C)\end{align*}
\item 再由反对称性 $W(M,N) = -W(N,M)$，可得第二个变元也满足分配律：
\begin{align*}W(C, P) &= W(C, \alpha A + \beta B) = -W(\alpha A + \beta B, C)= -\big(\alpha W(A, C) + \beta W(B, C)\big) \\&= \alpha \big(-W(A, C)\big) + \beta \big(-W(B, C)\big) = \alpha W(C, A) + \beta W(C, B)\end{align*}
\end{itemize}

\begin{definition}[内积 $I$]
$$ I(M, N) = x_M x_N + y_M y_N $$
\end{definition}
\begin{itemize}
    \item 对称性：$I(M,N) = I(N,M)$，即内积满足交换律。
    \item 对第一变元的线性性：对于任意实数 $\alpha,\beta$ 及点 $A,B,C$，有
    $$ I(\alpha A + \beta B, C) = \alpha I(A,C) + \beta I(B,C). $$
    \item 对第二变元的线性性：结合对称性与第一变元的线性性，可得
    $$ I(C, \alpha A + \beta B) = \alpha I(C,A) + \beta I(C,B). $$
    \item 自内积与模长的关系：$I(M,M) = x_M^2 + y_M^2 = |OM|^2$，即点 $M$ 到原点距离的平方。
    \item 与标准内积的联系：$I(M,N) = \overrightarrow{OM} \cdot \overrightarrow{ON}$，即欧几里得空间中的点积。
\end{itemize}

\begin{definition}[直线式 $F$]
    对于三点共线的$A(x_A,y_A),B(x_B,y_B),X(x,y)$，记
    \[F_{AB}(X)=y(x_A-x_B)+x(y_B-y_A)-(x_Ay_B-x_By_A)\]
\end{definition}
\begin{itemize}
    \item 对称性：$F_{AB}(X)$ 关于前两个变元是反对称的，即
    $$ F_{AB}(X) = -F(B,A,X). $$
    特别地，$F(A,A,X)=0$ 恒成立。
    \item 退化性：$F_{AB}(X)=0$ 当且仅当点 $X$ 位于直线 $AB$ 上。且 $F(A,B,A)=F(A,B,B)=0$。
    \item 三线性：将点视为向量（坐标），$F$ 对每个变元都是线性的。即对任意实数 $\alpha,\beta$ 及任意点（或向量）$A,A',B,B',X,X'$，有
    \begin{align*}
        F(\alpha A+\beta A',\,B,\,X) &= \alpha F_{AB}(X) + \beta F(A',B,X), \\
        F(A,\,\alpha B+\beta B',\,X) &= \alpha F_{AB}(X) + \beta F(A,B',X), \\
        F(A,\,B,\,\alpha X+\beta X') &= \alpha F_{AB}(X) + \beta F(A,B,X').
    \end{align*}
    \item 与外积 $W$ 的关系：
    $$ F_{AB}(X) = y(x_A-x_B) + x(y_B-y_A) - W(A,B), $$
    其中 $W(A,B)=x_Ay_B-x_By_A$。更紧凑地，利用三阶行列式可写为
    $$ F_{AB}(X) = -\begin{vmatrix}
        x & y & 1 \\
        x_A & y_A & 1 \\
        x_B & y_B & 1
    \end{vmatrix}. $$
    \item 轮换性质：由行列式的完全反对称性，有
    $$ F_{AB}(X) = F(B,X,A) = F(X,A,B) = -F(B,A,X) = -F(A,X,B) = -F(X,B,A). $$
\end{itemize}

\section{几个基本结论}

\begin{conclusion}[定比公式及其推论]
已知二次曲线 $C$ 上有两点 $A, B$。直线 $AB$ 上有一动点 $P$ 满足 $\overrightarrow{AP} = \lambda \overrightarrow{PB}$。则必然成立：
\begin{enumerate}
    \item 比例关系：
    \[\lambda = \frac{T(P,A)}{T(P,B)},~~~\alpha=\frac{T(P,B)}{T(A,B)}=\frac12\frac{T(P,P)}{T(P,A)},~~~\beta=\frac{T(P,A)}{T(A,B)}=\frac12\frac{T(P,P)}{T(P,B)}\]
    \item 由双线性展开可得：$T(A,B) = T(A,P) + T(P,B)$
    \item 整理得：$T(P,P) \cdot T(A,B) = 2 T(P,A) \cdot T(P,B)\Longleftrightarrow T(P,P) = \frac{2 T(P,A) T(P,B)}{T(P,A) + T(P,B)}$
\end{enumerate}
\end{conclusion}
由定比分点公式，
\[
x_P = \frac{x_A+\lambda x_B}{1+\lambda},\qquad y_P=\frac{y_A+\lambda y_B}{1+\lambda}.
\]
由于点 $A, B$ 在曲线 $\mathcal{C}$ 上，所以$T(A,A) = T(B,B) = 0$。\\
\textbf{插入律：}用定比点差法可得：
\begin{align*}
&T(P,A) = T(\frac{A+\lambda B}{1+\lambda},A)=\frac{1}{1+\lambda} T(A,A) +\frac{\lambda}{1+\lambda} T(B,A)=\frac{\lambda}{1+\lambda} T(B,A)=\beta T(A,B)\\
&T(P,B) =T(\frac{A+\lambda B}{1+\lambda},B)=\frac{1}{1+\lambda} T(A,B) +\frac{\lambda}{1+\lambda} T(B,B)=\frac{1}{1+\lambda} T(A,B)=\alpha T(A,B)\\
&\implies T(P,A)+T(P,B)=\frac{\lambda}{1+\lambda} T(B,A)+\frac{1}{1+\lambda} T(A,B)= T(A,B)
\end{align*}

\textbf{调和律（相乘法）：}
将 $T(P,A)$ 与 $T(P,B)$ 两式相乘，并用$T(A,A)=T(B,B)=0$配凑:
\begin{align*}
T(P,A) \cdot T(P,B) &= \left( \frac{\lambda}{1+\lambda} T(A,B) \right) \cdot \left( \frac{1}{1+\lambda} T(A,B) \right) = \frac{\lambda\big[T(A,B)\big]^2}{(1+\lambda)^2} =\dfrac12\frac{2\lambda T(A,B)}{(1+\lambda)^2} T(A,B)\\
&=\frac{T(A,B)}2\left(\frac{1}{(1+\lambda)^2} T(A,A) + \frac{2\lambda}{(1+\lambda)^2} T(A,B) + \frac{\lambda^2}{(1+\lambda)^2} T(B,B)\right)\\
\! =\frac12T(A,B)\cdot T(P,P)
\end{align*}
再结合插入律 $T(A,B) = T(P,A) + T(P,B)$，可化为：
$$ T(P,P) = \frac{2 T(P,A) T(P,B)}{T(P,A) + T(P,B)} $$
\begin{align*}
&T(P,A) = T(\frac{A+\lambda B}{1+\lambda},A)=\frac{1}{1+\lambda} T(A,A) +\frac{\lambda}{1+\lambda} T(B,A)=\frac{\lambda}{1+\lambda} T(B,A)=\beta T(A,B)\\
&T(P,B) =T(\frac{A+\lambda B}{1+\lambda},B)=\frac{1}{1+\lambda} T(A,B) +\frac{\lambda}{1+\lambda} T(B,B)=\frac{1}{1+\lambda} T(A,B)=\alpha T(A,B)\\
&\implies\lambda = \frac{T(P,A)}{T(P,B)}=\frac{T(P,A)\cdot T(P,B)}{T(P,B)^2}=\frac{T(P,P)\cdot T(A,B)}{2T(P,B)^2}=\frac{2T(P,A)^2}{T(P,P)\cdot T(A,B)}\\
&\implies\alpha=\frac{T(P,B)}{T(A,B)}=\frac{T(P,B)\cdot T(P,A)}{T(A,B)\cdot T(P,A)}=\frac{\frac12T(A,B)\cdot T(P,P)}{T(A,B)\cdot T(P,A)}=\frac12\frac{T(P,P)}{T(P,A)}\\
&\implies\beta=\frac{T(P,A)}{T(A,B)}=\frac{T(P,A)\cdot T(P,B)}{T(A,B)\cdot T(P,B)}=\frac{\frac12T(A,B)\cdot T(P,P)}{T(A,B)\cdot T(P,B)}=\frac12\frac{T(P,P)}{T(P,B)}
\end{align*}

\begin{conclusion}[差分律（中点共轭律）]
已知二次曲线 $C$ 上有两点 $A, B$。直线 $AB$ 上一动点 $P$ 满足 $\overrightarrow{AP} = \lambda \overrightarrow{PB}$。则：
$$ E(P, \overrightarrow{BA}) = T(P,A) - T(P,B) = \frac{\lambda - 1}{\lambda + 1} T(A,B) $$
\textbf{点差法：}当 $P$ 为弦 $AB$ 的中点时，必然成立 $E(P, \overrightarrow{BA}) = 0$。将其展开为坐标形式即：
$$ \frac{x_P(x_A-x_B)}{a^2} + \frac{y_P(y_A-y_B)}{b^2} = 0 \implies k_{OP} \cdot k_{AB} = -\frac{b^2}{a^2} $$
\end{conclusion}

\textbf{差分律：}
将 $T(P,A)$ 与 $T(P,B)$ 两式相减，正好得到：
\begin{align*}
T(P,A) - T(P,B) &= \frac{\lambda}{1+\lambda} T(A,B) - \frac{1}{1+\lambda} T(A,B) = \frac{\lambda - 1}{\lambda + 1} T(A,B)\\
T(P,A) - T(P,B) &= \big(E(P,A) - 1\big) - \big(E(P,B) - 1\big) = E(P,A) - E(P,B)=E(P, \overrightarrow{BA})
\end{align*}

\begin{conclusion}[一个内积关系式]
椭圆上两点 $A, B$ 坐标为$(x_1,y_1),(x_2,y_2)$，中点为$M$，弦 $AB$ 对应的极点（即过 $A,B$ 的切线交点）为 $P$
$$ \frac{x_1y_1+x_2y_2}{x_1y_2+x_2y_1} = E(A,B) = 2E(M,M) - 1 = \frac{1 - T(P,P)}{1 + T(P,P)}  $$
\end{conclusion}
用平方差公式得到
\[
\frac{x_1y_1+x_2y_2}{x_1y_2+x_2y_1}=E(A,B).
\]
\begin{align*}
\frac{x_1y_1+x_2y_2}{x_1y_2+x_2y_1} & = \frac{(x_1y_1+x_2y_2) \cdot W(A,B)}{(x_1y_2+x_2y_1) \cdot W(A,B)} = \frac{(x_1y_1+x_2y_2)(x_1y_2-x_2y_1)}{x_1^2 y_2^2 - x_2^2 y_1^2} \\
&=\frac{x_1^2 y_1 y_2 - x_1 x_2 y_1^2 + x_1 x_2 y_2^2 - x_2^2 y_1 y_2}{ x_1^2 \left[ b^2 \left(1 - \frac{x_2^2}{a^2}\right) \right] - x_2^2 \left[ b^2 \left(1 - \frac{x_1^2}{a^2}\right) \right] }=\frac{y_1y_2(x_1^2 - x_2^2) + x_1x_2(y_2^2 - y_1^2)}{b^2 (x_1^2 - x_2^2) }\\
&=\frac{ b^2 (x_1^2 - x_2^2) E(A,B) }{ b^2 (x_1^2 - x_2^2) } = E(A,B)
\end{align*}
设中点 $M = \frac{A+B}{2}$。对它取内积并利用双线性分配律展开：
\begin{align*} E(M,M) &= E\left(\frac{A+B}{2}, \frac{A+B}{2}\right) = \frac{1}{4} \big[ E(A,A) + 2E(A,B) + E(B,B) \big] =\frac{1 + E(A,B)}{2} \\
    \implies E(A,B) &= 2E(M,M) - 1\end{align*}

假设$\overrightarrow{OA}$ 与 $\overrightarrow{OB}$ 不共线，它们可作为平面的一组基。因此可设：
$ P = \alpha A + \beta B $，由于点 $A, B$ 在曲线上，$P$在$A,B$的极线上，故
\begin{align*}
1=E(P, A) &= E(\alpha A + \beta B, A) = \alpha E(A,A) + \beta E(B,A) \implies \alpha + \beta E(A,B) = 1 \\
1=E(P, B) &= E(\alpha A + \beta B, B) = \alpha E(A,B) + \beta E(B,B) \implies \alpha E(A,B) + \beta = 1\\
\text{相减：}&(\alpha - \beta) - (\alpha - \beta) E(A,B) = 0 \implies (\alpha - \beta)\big[1 - E(A,B)\big] = 0
\end{align*}
对于椭圆上两个不同的点 $A \neq B$，必有 $E(A-B, A-B)= 2 - 2E(A,B) > 0$。故$ \alpha = \beta $，从而解得伸缩系数：
$$ \alpha = \beta = \frac{1}{1 + E(A,B)} \implies P = \frac{A + B}{1 + E(A,B)} $$
极点 $P$ 的自内积 $E(P,P)$ 为：
\begin{align*}
E(P,P) &= E\left( \frac{A+B}{1+E(A,B)}, \frac{A+B}{1+E(A,B)} \right) = \frac{E(A,A) + E(B,B) + 2E(A,B)}{\big[1+E(A,B)\big]^2} \\
&= \frac{2 + 2E(A,B)}{\big[1+E(A,B)\big]^2} = \frac{2\big[1 + E(A,B)\big]}{\big[1+E(A,B)\big]^2} = \frac{2}{1 + E(A,B)}
\end{align*}
反解出 $E(A,B)$ 的表达式：
$$ 1 + E(A,B) = \frac{2}{E(P,P)} \implies E(A,B) = \frac{2}{E(P,P)} - 1= \frac{1 - T(P,P)}{1 + T(P,P)} $$

\emph{用参数方程核对也可得到同样结果。} 设 $A(a\cos\alpha, b\sin\alpha), B(a\cos\beta, b\sin\beta)$。\\
分子为：
$$ x_1y_1 + x_2y_2 = ab\cos\alpha\sin\alpha + ab\cos\beta\sin\beta = \frac{1}{2}ab(\sin 2\alpha + \sin 2\beta) = ab\sin(\alpha+\beta)\cos(\alpha-\beta) $$
分母为：
$$ x_1y_2 + x_2y_1 = ab\cos\alpha\sin\beta + ab\cos\beta\sin\alpha = ab\sin(\alpha+\beta) $$
两式相除可得
\[
\frac{x_1y_1+x_2y_2}{x_1y_2+x_2y_1}=\cos(\alpha-\beta).
\]
另一方面，
$$ E(A,B) = \frac{a\cos\alpha \cdot a\cos\beta}{a^2} + \frac{b\sin\alpha \cdot b\sin\beta}{b^2} = \cos\alpha\cos\beta + \sin\alpha\sin\beta = \cos(\alpha-\beta) $$
所以两者相等。再由半角公式，也可把 $T(P,P)$ 写成 $\tan^2\frac{\alpha-\beta}{2}$。

\begin{conclusion}[一个化简公式]
椭圆上两点$(x_1,y_1),(x_2,y_2)$，非椭圆上一点$P(m,n)$，则
\[(x_1y_2+x_2y_1)\left(\dfrac{m^2}{a^2}+\dfrac{n^2}{b^2}\right)=m(y_1+y_2) + n(x_1+x_2) - 2mn \]
\end{conclusion}
直线 $AB$ 上的动点 $P(m,n)$ 可写成 $P = \alpha A + \beta B$，其中
\[
m = \alpha x_1 + \beta x_2,\qquad n = \alpha y_1 + \beta y_2,\qquad \alpha+\beta=1.
\]
\begin{align*}
&(x_1y_2+x_2y_1)\left(\dfrac{m^2}{a^2}+\dfrac{n^2}{b^2}\right) \\
&= (x_1y_2+x_2y_1) E(P, P) \\
&= (x_1y_2+x_2y_1) E(\alpha A + \beta B, \alpha A + \beta B)= (x_1y_2+x_2y_1) \big[ \alpha^2 E(A,A) + \beta^2 E(B,B) + 2\alpha\beta E(A,B) \big] \\
&= (x_1y_2+x_2y_1) \big[ \alpha^2 + \beta^2 + 2\alpha\beta E(A,B) \big] = (\alpha^2 + \beta^2)(x_1y_2+x_2y_1) + 2\alpha\beta (x_1y_1 + x_2y_2) \\
&= 1 \cdot (x_1y_2+x_2y_1) - 2\alpha\beta (x_1y_2+x_2y_1) + 2\alpha(1-\alpha) x_1y_1 + 2\beta(1-\beta) x_2y_2\\
&= (\alpha+\beta)(x_1y_2+x_2y_1) + 2\alpha x_1y_1 - 2\alpha^2 x_1y_1 + 2\beta x_2y_2 - 2\beta^2 x_2y_2 - 2\alpha\beta(x_1y_2+x_2y_1)  \\
&= \big[ \alpha x_1y_2 + \beta x_1y_2 + \alpha x_2y_1 + \beta x_2y_1 + 2\alpha x_1y_1 + 2\beta x_2y_2 \big] - 2 \big( \alpha^2 x_1y_1 + \beta^2 x_2y_2 + \alpha\beta x_1y_2 + \alpha\beta x_2y_1 \big)  \\
&= \big[ \alpha x_1(y_1+y_2) + \alpha y_1(x_1+x_2) + \beta x_2(y_1+y_2) + \beta y_2(x_1+x_2) \big] - 2 \big( \alpha x_1 + \beta x_2 \big)\big( \alpha y_1 + \beta y_2 \big) \\
&= \big[ (\alpha x_1 + \beta x_2)(y_1+y_2) + (\alpha y_1 + \beta y_2)(x_1+x_2) \big] - 2 x_P y_P  \\
&= m(y_1+y_2) + n(x_1+x_2) - 2mn
\end{align*}


\begin{conclusion}[直线参数方程]
设有一条以定点 $P(x_0, y_0)$ 为基点、以向量 $\vec{v} = (v_x, v_y)$ 为方向向量的射线（或直线），其参数方程可写成 $X = P + t\vec{v}$。把 $X$ 代入二次曲线方程后，得到关于 $t$ 的二次方程：
$$ T(X,X) = E(\vec{v},\vec{v}) t^2 + 2T(P, \vec{v}) t + T(P,P) $$
这里记
\[
T(P, \vec{v})=E(P, \vec{v}) = \frac{x_0 v_x}{a^2} + \frac{y_0 v_y}{b^2}.
\]
由韦达定理，
\[ t_1 + t_2 = \frac{-2T(P,\vec{v})}{E(\vec{v},\vec{v})}, \quad t_1 t_2 = \frac{T(P,P)}{E(\vec{v},\vec{v})} \]
\end{conclusion}
\vspace{-20pt}
\begin{align*}
T(X,X)& = E(P + t\vec{v}, P + t\vec{v}) - 1 = E(P, P+t\vec{v}) + E(t\vec{v}, P+t\vec{v})-1\\
&= E(P, P+t\vec{v}) +t E(\vec{v}, P+t\vec{v})-1= (E(P,P) + tE(P,\vec{v})) + t(E(\vec{v},P) + tE(\vec{v},\vec{v}))-1 \\
&=E(P,P) + 2tE(P,\vec{v}) + t^2E(\vec{v},\vec{v})-1\\
&=T(P,P)+2tE(P,\vec{v}) + t^2E(\vec{v},\vec{v})
\end{align*}
因此，点写成参数式以后，曲线方程就化为关于 $t$ 的二次方程。

\begin{conclusion}[斜率与外积的关系]
定义水平方向单位向量 $\vec{i} = (1, 0)$ 和竖直方向单位向量 $\vec{j} = (0, 1)$。
对于平面上任意一条以向量 $\vec{v}$ 为方向的直线，其斜率 $k$ 满足
$$ k = \frac{W(\vec{i}, \vec{v})}{W(\vec{v}, \vec{j})} $$
\end{conclusion}
\textbf{证明：}
设 $\vec{v} = (v_x, v_y)$。由定义，
\[
W(\vec{i}, \vec{v}) = v_y,\qquad W(\vec{v}, \vec{j}) = v_x.
\]
两式相除便得 $k = \frac{v_y}{v_x}$。
\begin{remark}
因此，用 $W$ 处理斜率差时，通常比直接通分更简洁。
\end{remark}

\begin{conclusion}[调和线束中的一个关系式]
对于任意定点 $P$ 与共线的三点 $A, B, C$，有
$$ \frac{k_{PA}-k_{PC}}{k_{PB}-k_{PC}} \cdot \frac{x_A-x_P}{x_B-x_P} = -\frac{T(C,A)}{T(C,B)} = \frac{\overrightarrow{AC}}{\overrightarrow{BC}} $$
\end{conclusion}

\textbf{证明：}
计算斜率差：
$$ k_{PA} - k_{PC} = \frac{y_A-y_P}{x_A-x_P} - \frac{y_C-y_P}{x_C-x_P} = \frac{W(\overrightarrow{PC}, \overrightarrow{PA})}{(x_A-x_P)(x_C-x_P)} $$
同理可得 $k_{PB}-k_{PC}$。两式相除后，
$$ \frac{k_{PA}-k_{PC}}{k_{PB}-k_{PC}} = \frac{W(\overrightarrow{PC}, \overrightarrow{PA})}{W(\overrightarrow{PC}, \overrightarrow{PB})} \cdot \frac{x_B-x_P}{x_A-x_P} $$
因为 $A, B, C$ 共线，可按定比写成
\[
C = \frac{A + \lambda B}{1+\lambda},\qquad \lambda = \frac{\overrightarrow{AC}}{\overrightarrow{CB}} = \frac{T(C,A)}{T(C,B)}.
\]
代入外积式，
\begin{align*}
W(\overrightarrow{PC}, \overrightarrow{PA}) &= W\left(\frac{\overrightarrow{PA} + \lambda \overrightarrow{PB}}{1+\lambda}, \overrightarrow{PA}\right) = -\frac{\lambda}{1+\lambda} W(\overrightarrow{PA}, \overrightarrow{PB}) \\
W(\overrightarrow{PC}, \overrightarrow{PB}) &= W\left(\frac{\overrightarrow{PA} + \lambda \overrightarrow{PB}}{1+\lambda}, \overrightarrow{PB}\right) = \frac{1}{1+\lambda} W(\overrightarrow{PA}, \overrightarrow{PB})
\end{align*}
所以
$$ \frac{W(\overrightarrow{PC}, \overrightarrow{PA})}{W(\overrightarrow{PC}, \overrightarrow{PB})} = -\lambda = -\frac{T(C,A)}{T(C,B)} \implies\frac{k_{PA}-k_{PC}}{k_{PB}-k_{PC}} \frac{x_A-x_P}{x_B-x_P} = -\lambda = \frac{\overrightarrow{AC}}{\overrightarrow{BC}} $$

\begin{conclusion}[一个斜率关系式]
椭圆$\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$上有$P(x_0,y_0),A(x_1,y_1),B(x_2,y_2)$三个点，令$k_1=k_{PA},k_2=k_{PB}$，那么
    $$\frac{a^2}{b^2} k_1 k_2 - \frac{x_1 + x_2}{y_1 - y_2}(k_1 - k_2) + 1 = 0 $$
\end{conclusion}
已知椭圆上有三个定点 $P(x,y)$、$A(x_1,y_1)$、$B(x_2,y_2)$。设
\[
\vec{u} = A - P,\qquad \vec{v} = B - P,
\]
则
\[
k_1 = \frac{u_y}{u_x},\qquad k_2 = \frac{v_y}{v_x}.
\]
由斜率定义可写出斜率差：
$$k_1 - k_2 = \frac{u_y}{u_x} - \frac{v_y}{v_x} = \frac{u_y v_x - v_y u_x}{u_x v_x}=\frac{-W(\vec{u}, \vec{v})}{u_x v_x}$$
内积可写为：
$$E(\vec{u}, \vec{v}) = u_x v_x \left( \frac{1}{a^2} + \frac{1}{b^2} \frac{u_y v_y}{u_x v_x} \right) = u_x v_x \left( \frac{1}{a^2} + \frac{k_1 k_2}{b^2} \right)\implies a^2 E(\vec{u}, \vec{v}) = u_x v_x \left( 1 + \frac{a^2}{b^2} k_1 k_2 \right)$$
两式相除可得
\[
\frac{\frac{a^2}{b^2} k_1 k_2 + 1}{k_1 - k_2} = \frac{a^2 E(\vec{u}, \vec{v})}{-W(\vec{u}, \vec{v})}.
\]
\begin{itemize}
\item 引入 $A, B$ 的中点 $M = \frac{A+B}{2}$，这意味 $A+B = 2M$。
\item 分子部分为：
\begin{align*}
E(A-P, B-P) &= E(A,B)+ E(P,P) - (E(A,P) + E(B,P)) \\
&= E(A,B)+ E(P,P) -E(A+B, P)=E(A,B)+ E(P,P)-2E(M, P)\\
&= (2E(M,M) - 1) - 2E(M,P) + 1 =  {2E(M, M-P)}
\end{align*}
\item 分母部分由分配律得：
\begin{align*}
W(A-P, B-P) &= W(A,B) - W(A,P) - W(P,B) + W(P,P)\\
&= W(A,B) - W(A,P) + W(B,P)=W(A,B)-W(A-B, P)\\
&=W\left(M + \frac{A-B}{2},M - \frac{A-B}{2}\right)- W(A-B, P) \\
&=  W(A-B, M) - W(A-B, P) =W(A-B, M-P)
\end{align*}
\end{itemize}
把化简后的分子分母代回：
$$\frac{a^2 E(\vec{u}, \vec{v})}{-W(\vec{u}, \vec{v})} = \frac{2a^2 E(M, M-P)}{-W(A-B, M-P)}$$
把分子写成坐标形式：
$$2a^2 \left( \frac{x_M (x_M - x_P)}{a^2} + \frac{y_M (y_M - y_P)}{b^2} \right) = 2 \left( x_M (x_M - x_P) + \frac{a^2}{b^2}y_M (y_M - y_P) \right)$$
由 $A,B$ 在椭圆上，可用点差关系
\[
E(M, A-B) = 0 \implies \frac{x_M (x_A - x_B)}{a^2} + \frac{y_M (y_A - y_B)}{b^2} = 0.
\]
于是
\[
x_M = - \frac{a^2}{b^2} y_M \frac{y_A - y_B}{x_A - x_B}.
\]
把它代入上式中的 $x_M$：
\begin{align*}
\text{分子} &= 2 \left( {- \frac{a^2}{b^2} y_M \frac{y_A - y_B}{x_A - x_B}} (x_M - x_P) + \frac{a^2}{b^2} y_M (y_M - y_P) \right)\\
&=2 \frac{a^2}{b^2} \frac{y_M}{x_A - x_B} \Big[ -(y_A - y_B)(x_M - x_P) + (x_A - x_B)(y_M - y_P) \Big]\\
&= \frac{2a^2 y_M}{b^2 (x_A - x_B)} W(A-B, M-P)
\end{align*}
所以
$$\frac{ \frac{2a^2 y_M}{b^2 (x_A - x_B)} W(A-B, M-P) }{ -W(A-B, M-P) } = - \frac{2a^2 y_M}{b^2 (x_A - x_B)}$$
把 $2y_M = y_A + y_B$ 代回上面的结果：$$\text{右侧} = - \frac{a^2(y_A + y_B)}{b^2(x_A - x_B)}$$
再用一次点差法
\[
\frac{x_A + x_B}{a^2}(x_A - x_B) + \frac{y_A + y_B}{b^2}(y_A - y_B) = 0,
\]
移项得
\[
- \frac{a^2(y_A + y_B)}{b^2(x_A - x_B)} = \frac{x_A + x_B}{y_A - y_B} = \frac{x_1 + x_2}{y_1 - y_2}.
\]
因此
\[
\frac{\frac{a^2}{b^2} k_1 k_2 + 1}{k_1 - k_2} = \frac{x_1 + x_2}{y_1 - y_2},
\]
即
\[
\frac{a^2}{b^2} k_1 k_2 - \frac{x_1 + x_2}{y_1 - y_2}(k_1 - k_2) + 1 = 0.
\]

\begin{conclusion}[拉格朗日恒等式]
对于平面上任意两个向量 $\vec{u}, \vec{v}$，其外积 $W$ 与椭圆内积 $E$ 满足
$$ \big[W(\vec{u},\vec{v})\big]^2 = a^2 b^2 \Big[ E(\vec{u},\vec{u})E(\vec{v},\vec{v}) - \big[E(\vec{u},\vec{v})\big]^2 \Big] $$
\end{conclusion}
设 $\vec{u}=(x_1, y_1), \vec{v}=(x_2, y_2)$。
左侧外积的平方：$W(\vec{u},\vec{v})^2 = (x_1 y_2 - x_2 y_1)^2 = x_1^2 y_2^2 + x_2^2 y_1^2 - 2x_1 x_2 y_1 y_2$。
计算内积的平方并同乘 $a^2b^2$：$a^2b^2 E(\vec{u},\vec{v})^2 = a^2b^2 \left(\frac{x_1 x_2}{a^2} + \frac{y_1 y_2}{b^2}\right)^2 = \frac{b^2}{a^2}x_1^2 x_2^2 + \frac{a^2}{b^2}y_1^2 y_2^2 + 2x_1 x_2 y_1 y_2$。
将这两式相加：
\begin{align*}
W(\vec{u},\vec{v})^2 + a^2b^2 E(\vec{u},\vec{v})^2 &= x_1^2 y_2^2 + x_2^2 y_1^2 + \frac{b^2}{a^2}x_1^2 x_2^2 + \frac{a^2}{b^2}y_1^2 y_2^2 \\
&= x_1^2 \left( y_2^2 + \frac{b^2}{a^2}x_2^2 \right) + y_1^2 \left( x_2^2 + \frac{a^2}{b^2}y_2^2 \right)
\end{align*}
括号中的式子可写为
$y_2^2 + \frac{b^2}{a^2}x_2^2 = b^2 \left( \frac{x_2^2}{a^2} + \frac{y_2^2}{b^2} \right) = b^2 E(\vec{v},\vec{v})$。同理，后半部分的括号等于 $a^2 E(\vec{v},\vec{v})$。
于是
$$ x_1^2 \big[b^2 E(\vec{v},\vec{v})\big] + y_1^2 \big[a^2 E(\vec{v},\vec{v})\big] = a^2 b^2 E(\vec{v},\vec{v}) \left(\frac{x_1^2}{a^2} + \frac{y_1^2}{b^2}\right) = a^2 b^2 E(\vec{v},\vec{v}) E(\vec{u},\vec{u}) $$
移项即得。

\begin{conclusion}[弦长公式]
已知直线过定点 $P$，且方向向量为 $\vec{v}$，交二次曲线于 $A, B$ 两点。则弦长 $|AB|$ 可写成
$$ |AB| = \frac{2\sqrt{\big[E(P, \vec{v})\big]^2 - E(\vec{v}, \vec{v})T(P, P)}}{E(\vec{v}, \vec{v})} \sqrt{I(\vec{v}, \vec{v})} = 2 \frac{\sqrt{E(\vec{v},\vec{v}) - \frac{\big[W(P,\vec{v})\big]^2}{a^2 b^2}}}{E(\vec{v},\vec{v})} \cdot \sqrt{I(\vec{v},\vec{v})} $$
特别地，当 $P$ 是弦 $AB$ 的中点时，公式化为
\[
|AB| = 2\sqrt{\frac{-T(P,P)}{E(\vec{v}, \vec{v})}} \sqrt{I(\vec{v}, \vec{v})}.
\]
\end{conclusion}
把直线上动点 $X = P + t\vec{v}$ 代入曲线方程，得
$$ E(\vec{v},\vec{v})t^2 + 2E(P, \vec{v})t + T(P,P) = 0 $$
设交点对应的参数为 $t_1, t_2$。弦长就是参数之差乘以方向向量的欧氏长度：$|AB| = |t_1 - t_2| \cdot |\vec{v}|$。
由判别式可得
$$ |t_1 - t_2| = \frac{\sqrt{4\big[E(P, \vec{v})\big]^2 - 4E(\vec{v}, \vec{v})T(P, P)}}{E(\vec{v}, \vec{v})} $$
由 $T(P,P) = E(P,P) - 1$ 以及前面的拉格朗日恒等式，
\begin{align*}
\frac{\Delta}{4} &= E(P,\vec{v})^2 - \big(E(P,P) - 1\big)E(\vec{v},\vec{v}) = E(\vec{v},\vec{v}) - \Big[ E(P,P)E(\vec{v},\vec{v}) - E(P,\vec{v})^2 \Big] \\
&= E(\vec{v},\vec{v}) - \frac{\big[W(P,\vec{v})\big]^2}{a^2 b^2}
\end{align*}
又因为 $|\vec{v}| = \sqrt{I(\vec{v}, \vec{v})}$，相乘即得所求公式。

\begin{conclusion}[原点三角形 $OAB$ 面积公式]
已知椭圆上有两点 $A, B$。则由原点 $O$ 与 $A,B$ 构成的三角形面积可写成
$$ S_{\triangle OAB} = \frac{1}{2} ab \sqrt{-T(A,B) \big(T(A,B)+2\big)} $$
\end{conclusion}
三角形面积等于外积绝对值的一半：$S_{\triangle OAB} = \frac{1}{2} |W(A,B)|$。
既然 $A, B$ 都在椭圆上，那么把它们代入极化式必定等于 $0$：$T(A,A)=0 \implies E(A,A) = 1$。同理 $E(B,B) = 1$。
代入刚证的拉格朗日恒等式：
$$ \big[W(A,B)\big]^2 = a^2 b^2 \big(1 \times 1 - E(A,B)^2\big) = a^2 b^2 \big(1 - E(A,B)^2\big) $$
再代入 $E(A,B) = T(A,B) + 1$：
$$ 1 - \big(T(A,B)+1\big)^2 = 1 - \big(T(A,B)^2 + 2T(A,B) + 1\big) = -T(A,B)\big(T(A,B)+2\big) $$
开根号并乘以 $1/2$，即得到所需面积公式。

\begin{conclusion}[内接三角形的海伦型公式]
已知椭圆内接三角形 $\triangle ABC$。
记
$$ t_1 = -T(B,C), \quad t_2 = -T(A,C), \quad t_3 = -T(A,B) $$
则三角形面积满足
$$ S_{\triangle ABC} = \frac{1}{2} a b \sqrt{ 2t_1 t_2 + 2t_2 t_3 + 2t_3 t_1 - t_1^2 - t_2^2 - t_3^2 } $$
又因为 $A,B,C$ 同在一个椭圆上，根号内还可以化为 $2t_1 t_2 t_3$，于是
\[
S_{\triangle ABC} = \frac{\sqrt{2}}{2} ab \sqrt{t_1 t_2 t_3}.
\]
\end{conclusion}
内接三角形的面积为两条边向量外积的一半：$S_{\triangle ABC} = \frac{1}{2} \big| W(B-A, C-A) \big|$。
设向量 $\vec{u} = B-A, \vec{v} = C-A$。由拉格朗日恒等式，
$$ \big[W(\vec{u}, \vec{v})\big]^2 = a^2 b^2 \Big[ E(\vec{u}, \vec{u}) E(\vec{v}, \vec{v}) - \big[E(\vec{u}, \vec{v})\big]^2 \Big] $$
又因为 $E(A,A)=E(B,B)=E(C,C)=1$，所以
\begin{itemize}
\item $E(\vec{u}, \vec{u}) = E(B-A, B-A) = E(B,B) - 2E(A,B) + E(A,A) = 2 - 2E(A,B) = -2T(A,B) = 2t_3$。
\item 同理，$E(\vec{v}, \vec{v}) = E(C-A, C-A) = -2T(A,C) = 2t_2$。
\item
\[
E(\vec{u}, \vec{v}) = E(B-A, C-A) = E(B,C) - E(A,B) - E(A,C) + E(A,A).
\]
写成 $T$ 的形式，得
\[
E(\vec{u}, \vec{v}) = -t_1 + t_3 + t_2.
\]
\end{itemize}
代回即可得到
\begin{align*}
\frac{W^2}{a^2 b^2} &= (2t_3)(2t_2) - (t_3 + t_2 - t_1)^2 \\
&= 4t_2 t_3 - (t_3^2 + t_2^2 + t_1^2 + 2t_2 t_3 - 2t_1 t_3 - 2t_1 t_2) \\
&= 2t_1 t_2 + 2t_2 t_3 + 2t_3 t_1 - t_1^2 - t_2^2 - t_3^2
\end{align*}
开根号并乘上 $\frac{1}{2}$，便得到结论。

\section{圆锥曲线经典性质的证明}

\begin{definition}[极点与极线]
对于二次曲线内（或外）的任意定点 $P$，满足方程 $T(P, X) = 0$ 的所有动点 $X$ 构成一条直线，称为点 $P$ 对应的极线（Polar Line）。
\end{definition}

\begin{conclusion}[完全四边形极点极线定理]
已知二次曲线 $\mathcal{C}$ 内（或外）有一点 $P$。过 $P$ 作两条弦 $AB$ 与 $CD$。连接 $AD, BC$ 交于点 $Q$；连接 $AC, BD$ 交于点 $R$。
则交点 $Q$ 和 $R$ 都在点 $P$ 的极线上，即
$$ T(P, Q) = 0 \quad \text{且} \quad T(P, R) = 0 $$
\end{conclusion}
设 $P = \alpha_1 A + \beta_1 B = \alpha_2 C + \beta_2 D$，其中$\alpha_1+\beta_1 = \alpha_2+\beta_2 = 1$。为求直线 $AD$ 与 $BC$ 的交点 $Q$，把式子整理为
$$ \alpha_1 A - \beta_2 D = \alpha_2 C - \beta_1 B $$
设两侧系数和为 $k$，则
$$ Q = \frac{\alpha_1 A - \beta_2 D}{k} = \frac{\alpha_2 C - \beta_1 B}{k} $$
左式说明 $Q$ 在 $AD$ 上，右式说明 $Q$ 在 $BC$ 上。对 $Q$ 代入 $T(P,\cdot)$，得
\begin{align*}
T(P,Q) &= T\left(P, \frac{\alpha_1 A - \beta_2 D}{k}\right) = \frac{1}{k} \big[ \alpha_1 T(P,A) - \beta_2 T(P,D) \big]\\
&= \frac{1}{k} \big[\frac12T(P,P)-\frac12 T(P,P) \big]=0,~~~\text{定比律}
\end{align*}
所以 $Q$ 在点 $P$ 的极线上。对点 $R$ 也同理。若 $R=AC\cap BD$，则
$$ \alpha_1 A - \alpha_2 C = \beta_2 D - \beta_1 B $$
设两侧系数和为 $m = \alpha_1 - \alpha_2 = \beta_2 - \beta_1$，于是
\begin{align*}
T(P,R) &= T\left(P, \frac{\alpha_1 A - \alpha_2 C}{m}\right) = \frac{1}{m} \big[ \alpha_1 T(P,A) - \alpha_2 T(P,C) \big] \\
&= \frac{1}{m} \left[ \frac{1}{2} T(P,P) - \frac{1}{2} T(P,P) \right] = 0
\end{align*}
所以 $R$ 也在点 $P$ 的极线上。

\begin{conclusion}[坎迪定理（Candy's Theorem）与 蝴蝶定理]
    已知二次曲线 $\mathcal{C}$ 上有弦 $PQ$，直线 $PQ$ 上有一动点 $M$。过 $M$ 作两弦 $AB$ 和 $CD$。连结 $AD, BC$ 分别交直线 $PQ$ 于点 $X, Y$。则关于有向线段有
    $$ \frac{1}{MX} + \frac{1}{MY} = \frac{1}{MP} + \frac{1}{MQ}. $$
    \textbf{推论（蝴蝶定理）：}当 $M$ 为弦 $PQ$ 中点时，$M$ 也是 $XY$ 的中点。
\end{conclusion}
以 $M$ 为基点，设直线 $PQ$ 的方向向量为 $\vec{v}$，弦 $AB, CD$ 的方向向量分别为 $\vec{u}, \vec{w}$。
由直线参数方程，若 $X = M + t\vec{v}$，则交点 $P, Q$ 对应的参数 $t_P, t_Q$ 是方程 $T(X,X)=0$ 的两根，因此
\[
\frac{1}{t_P} + \frac{1}{t_Q} = \frac{t_P+t_Q}{t_Pt_Q} = \frac{-2E(M, \vec{v})}{T(M,M)}.
\]
同理，
\[
\frac{1}{t_A} + \frac{1}{t_B} = \frac{-2E(M, \vec{u})}{T(M,M)}, \qquad \frac{1}{t_C} + \frac{1}{t_D} = \frac{-2E(M, \vec{w})}{T(M,M)}.
\]

点 $X$ 在 $PQ$ 上，设其参数为 $x$，即 $X = M + x\vec{v}$。同时 $A = M+t_A\vec{u}$，$D = M+t_D\vec{w}$。由于 $X, A, D$ 三点共线，有
\[
W(\overrightarrow{AX}, \overrightarrow{AD}) = 0.
\]
展开得
\[
W(x\vec{v} - t_A\vec{u}, \ t_D\vec{w} - t_A\vec{u}) = x t_D W(\vec{v}, \vec{w}) - x t_A W(\vec{v}, \vec{u}) - t_A t_D W(\vec{u}, \vec{w}) = 0.
\]
两边同除以 $x \cdot t_A \cdot t_D \cdot W(\vec{u}, \vec{w})$，并利用反对称性，可得
\[
\frac{1}{x} = \frac{W(\vec{v}, \vec{w})}{W(\vec{u}, \vec{w})} \frac{1}{t_A} + \frac{W(\vec{u}, \vec{v})}{W(\vec{u}, \vec{w})} \frac{1}{t_D}.
\]
同理，若 $Y = BC \cap PQ$ 对应参数为 $y$，则
\[
\frac{1}{y} = \frac{W(\vec{v}, \vec{w})}{W(\vec{u}, \vec{w})} \frac{1}{t_B} + \frac{W(\vec{u}, \vec{v})}{W(\vec{u}, \vec{w})} \frac{1}{t_C}.
\]

将两式相加，并代入前面的倒数和公式：
\begin{align*}
\frac{1}{x} + \frac{1}{y} &= \frac{W(\vec{v}, \vec{w})}{W(\vec{u}, \vec{w})}\left(\frac{1}{t_A} + \frac{1}{t_B}\right) + \frac{W(\vec{u}, \vec{v})}{W(\vec{u}, \vec{w})}\left(\frac{1}{t_C} + \frac{1}{t_D}\right) \\
&= \frac{W(\vec{v}, \vec{w})}{W(\vec{u}, \vec{w})}\left[ \frac{-2E(M, \vec{u})}{T(M,M)} \right] + \frac{W(\vec{u}, \vec{v})}{W(\vec{u}, \vec{w})}\left[ \frac{-2E(M, \vec{w})}{T(M,M)} \right] \\
&= \frac{-2}{T(M,M) \cdot W(\vec{u}, \vec{w})} \Big[ W(\vec{v}, \vec{w})E(M, \vec{u}) + W(\vec{u}, \vec{v})E(M, \vec{w}) \Big] \\
&= \frac{-2}{T(M,M) \cdot W(\vec{u}, \vec{w})} E\Big(M, \; W(\vec{v}, \vec{w})\vec{u} + W(\vec{u}, \vec{v})\vec{w} \Big) \\
&= \frac{-2}{T(M,M) \cdot W(\vec{u}, \vec{w})} E\Big(M, \; W(\vec{u}, \vec{w})\vec{v} \Big) \\
&= \frac{-2E(M, \vec{v})}{T(M,M)} \\
&= \frac{1}{t_P} + \frac{1}{t_Q}
\end{align*}
因为参数均对应于同一基准向量 $\vec{v}$ 所在的同一直线，参数倒数和等价于有向线段倒数和，得到：
$$ \frac{1}{MX} + \frac{1}{MY} = \frac{1}{MP} + \frac{1}{MQ} $$
\textbf{蝴蝶定理}：当 $M$ 为弦 $PQ$ 中点时，由前面的差分关系可得 $E(M, \vec{v}) = 0$。于是
\[
\frac{1}{x} + \frac{1}{y} = 0 \implies x = -y,
\]
从而 $M$ 也是 $XY$ 的中点。

\begin{conclusion}[帕斯卡定理（Pascal's Theorem）]
    圆锥曲线内接六边形 $A_1A_2A_3A_4A_5A_6$ 的三对对边交点 $P=A_1A_2 \cap A_4A_5$，$Q=A_2A_3 \cap A_5A_6$，$R=A_3A_4 \cap A_6A_1$ 必然共线。
\end{conclusion}
\textbf{证明：}
设直线 $\mathcal{L}$ 交二次曲线于 $A, B$ 两点，该直线上还有两个不同的定点 $K, M$。取同一方向向量 $\vec{v}$，把直线上的点分别写成
\[
X = K + t_K\vec{v},\qquad X = M + t_M\vec{v}.
\]
由直线参数方程可得
\[
t_{KA} t_{KB} = \frac{T(K, K)}{E(\vec{v}, \vec{v})},\qquad t_{MA} t_{MB} = \frac{T(M, M)}{E(\vec{v}, \vec{v})}.
\]
又因为
\[
\overrightarrow{KA} = t_{KA}\vec{v},\qquad \overrightarrow{KB} = t_{KB}\vec{v},\qquad
\overrightarrow{MA} = t_{MA}\vec{v},\qquad \overrightarrow{MB} = t_{MB}\vec{v},
\]
所以
\[
\frac{\overrightarrow{KA} \cdot \overrightarrow{KB}}{\overrightarrow{MA} \cdot \overrightarrow{MB}}
= \frac{t_{KA} \cdot t_{KB}}{t_{MA} \cdot t_{MB}}
= \frac{T(K,K)}{T(M,M)}.
\]

再取内接六边形中互不相邻的三条边所在直线，延长后交成参考三角形 $\triangle KLM$：
\begin{itemize}
    \item $K = A_1A_2 \cap A_3A_4$，所以边 $KM$ 是直线 $A_1A_2$，边 $KL$ 是直线 $A_3A_4$。
    \item $L = A_3A_4 \cap A_5A_6$，所以边 $LM$ 是直线 $A_5A_6$。
    \item $M = A_5A_6 \cap A_1A_2$。
\end{itemize}
于是 $A_4A_5,\ A_2A_3,\ A_6A_1$ 分别是截三角形 $\triangle KLM$ 的三条截线。对它们分别应用梅涅劳斯定理：
    \begin{itemize}
    \item 直线 $A_4A_5$ 与三边分别交于 $P,\ A_5,\ A_4$，因此
    \[
    \frac{\overrightarrow{KP}}{\overrightarrow{PM}} \cdot \frac{\overrightarrow{MA_5}}{\overrightarrow{A_5L}} \cdot \frac{\overrightarrow{LA_4}}{\overrightarrow{A_4K}} = -1.
    \]
    \item 直线 $A_2A_3$ 与三边分别交于 $A_2,\ Q,\ A_3$，因此
    \[
    \frac{\overrightarrow{KA_2}}{\overrightarrow{A_2M}} \cdot \frac{\overrightarrow{MQ}}{\overrightarrow{QL}} \cdot \frac{\overrightarrow{LA_3}}{\overrightarrow{A_3K}} = -1.
    \]
    \item 直线 $A_6A_1$ 与三边分别交于 $A_1,\ A_6,\ R$，因此
    \[
    \frac{\overrightarrow{KA_1}}{\overrightarrow{A_1M}} \cdot \frac{\overrightarrow{MA_6}}{\overrightarrow{A_6L}} \cdot \frac{\overrightarrow{LR}}{\overrightarrow{RK}} = -1.
    \]
\end{itemize}
将这三个等式连乘，并把与 $P,Q,R$ 有关的部分提出：
$$ \underbrace{\left( \frac{\overrightarrow{KP}}{\overrightarrow{PM}} \frac{\overrightarrow{MQ}}{\overrightarrow{QL}} \frac{\overrightarrow{LR}}{\overrightarrow{RK}} \right)}_{\text{目标待证组}} \cdot \underbrace{\left( \frac{\overrightarrow{KA_1}}{\overrightarrow{A_1M}} \frac{\overrightarrow{KA_2}}{\overrightarrow{A_2M}} \right)}_{\text{边 } KM \text{ 上的交点}} \cdot \underbrace{\left( \frac{\overrightarrow{LA_3}}{\overrightarrow{A_3K}} \frac{\overrightarrow{LA_4}}{\overrightarrow{A_4K}} \right)}_{\text{边 } KL \text{ 上的交点}} \cdot \underbrace{\left( \frac{\overrightarrow{MA_5}}{\overrightarrow{A_5L}} \frac{\overrightarrow{MA_6}}{\overrightarrow{A_6L}} \right)}_{\text{边 } LM \text{ 上的交点}} = -1 $$
将分母方向写成以基点为起点的线段后，等式化为
\[
\left( \frac{\overrightarrow{KP}}{\overrightarrow{PM}} \frac{\overrightarrow{MQ}}{\overrightarrow{QL}} \frac{\overrightarrow{LR}}{\overrightarrow{RK}} \right)
\cdot \left( \frac{\overrightarrow{KA_1} \cdot \overrightarrow{KA_2}}{\overrightarrow{MA_1} \cdot \overrightarrow{MA_2}} \right)
\cdot \left( \frac{\overrightarrow{LA_3} \cdot \overrightarrow{LA_4}}{\overrightarrow{KA_3} \cdot \overrightarrow{KA_4}} \right)
\cdot \left( \frac{\overrightarrow{MA_5} \cdot \overrightarrow{MA_6}}{\overrightarrow{LA_5} \cdot \overrightarrow{LA_6}} \right)
= -1.
\]
再用上面的线段乘积关系，可分别把后三项改写为
\[
\frac{T(K,K)}{T(M,M)},\qquad \frac{T(L,L)}{T(K,K)},\qquad \frac{T(M,M)}{T(L,L)}.
\]
于是
\[
\left( \frac{\overrightarrow{KP}}{\overrightarrow{PM}} \frac{\overrightarrow{MQ}}{\overrightarrow{QL}} \frac{\overrightarrow{LR}}{\overrightarrow{RK}} \right)
\cdot \frac{T(K,K)}{T(M,M)} \cdot \frac{T(L,L)}{T(K,K)} \cdot \frac{T(M,M)}{T(L,L)} = -1.
\]
中间三项约去后得到
\[
\frac{\overrightarrow{KP}}{\overrightarrow{PM}} \cdot \frac{\overrightarrow{MQ}}{\overrightarrow{QL}} \cdot \frac{\overrightarrow{LR}}{\overrightarrow{RK}} = -1.
\]
而 $P,Q,R$ 分别在三角形 $\triangle KLM$ 的三边所在直线上，所以由梅涅劳斯定理的逆定理知，$P,Q,R$ 共线。

\begin{conclusion}[布列安桑定理（Brianchon's Theorem）]
圆锥曲线外切六边形 $V_1V_2V_3V_4V_5V_6$ 的三条主对角线 $V_1V_4, V_2V_5, V_3V_6$ 必然共点。
\end{conclusion}
\textbf{证明：}这一定理是帕斯卡定理的对偶形式。

\textbf{切线化为极线方程}
当直线与曲线相切于点 $A$ 时，设切线上动点 $X=A+t\vec{v}$。把它代入曲线方程 $T(X,X)=0$，相切对应于 $t=0$ 为二重根，因此一次项系数为 $2E(A,\vec{v})=0$。于是切线上任意动点均满足
\[
T(A, X) = T(A, A+t\vec{v}) = T(A,A) + tE(A,\vec{v}) = 0.
\]
因此方程 $T(P, X) = 0$ 就表示点 $P$ 的极线。
又因为 $T(P, Q) = T(Q, P)$，所以有对偶性质：
\[
\text{点 } P \text{ 在点 } Q \text{ 的极线上}\iff \text{点 } Q \text{ 在点 } P \text{ 的极线上}.
\]

\textbf{顶点的极线即切点弦}
设外切六边形各边与曲线的切点分别为 $A_1, A_2, A_3, A_4, A_5, A_6$（它们构成了内接六边形）。
外切顶点 $V_1$ 是切线 $A_1$ 与切线 $A_6$ 的交点。
因为 $V_1$ 在 $A_1$ 的极线上，由对偶律有 $T(A_1, V_1)=0 \implies T(V_1, A_1)=0$；同理 $T(V_1, A_6)=0$。
所以过 $A_1, A_6$ 的切点弦正是极点 $V_1$ 的极线。同理，切点弦 $A_3A_4$ 是极点 $V_4$ 的极线。

\textbf{对角线对偶为帕斯卡交点}
考察主对角线 $V_1 V_4$，它连接了极点 $V_1$ 与 $V_4$。
设弦 $A_1A_6$ 与 $A_3A_4$ 相交于点 $P'$。由于 $P'$ 同时在 $V_1$ 和 $V_4$ 的极线上，由对偶性质可得，$V_1$ 和 $V_4$ 也在 $P'$ 的极线上。
因此主对角线 $V_1V_4$ 正是点 $P'$ 的极线。
完全同理：主对角线 $V_2V_5$ 是交点 $Q' = A_1A_2 \cap A_4A_5$ 的极线；主对角线 $V_3V_6$ 是交点 $R' = A_2A_3 \cap A_5A_6$ 的极线。

由于 $A_1 \sim A_6$ 构成内接六边形，交点 $P', Q', R'$ 恰为三组对边的交点。由上述已证的\textbf{帕斯卡定理}，三点 $P', Q', R'$ 共线，设其所在直线为 $\mathcal{L}$。
共线三点的极线同过同一点，这个点即为直线 $\mathcal{L}$ 的极点。因此三条主对角线 $V_1V_4, V_2V_5, V_3V_6$ 共点。

\section{对称齐次化}
设椭圆 $\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$ 上一点 $P$ 引出弦 $PA,PB$，其斜率分别为 $k_1,k_2$。把
\[
PA:\ y-y_0=k_1(x-x_0),\qquad PB:\ y-y_0=k_2(x-x_0)
\]
相乘后得到双直线方程。再取关于原点对称的两点 $P,-P$，分别整理成齐次式并相加，得
\begin{align*}
    AB:&k_1k_2(x - x_0)^2 - (k_1+k_2)(x - x_0)(y - y_0) + (y - y_0)^2=0\\
    AB:&k_1k_2\frac{a^2}{b^2}(y + y_0)^2+ (k_1+k_2)(x + x_0)(y + y_0) + \frac{b^2}{a^2}(x + x_0)^2=0\\
\implies AB:&k_1k_2\left(x^2+\frac{a^2}{b^2}y^2+x_0^2+\frac{a^2}{b^2}y_0^2-2x_0x+2\frac{a^2}{b^2}y_0y\right)\\
&+(k_1+k_2)\left[(x - x_0+2x_0)(y - y_0+2y_0)-(x - x_0)(y - y_0)\right]\\
&+\left(\frac{b^2}{a^2}x^2+y^2+\frac{b^2}{a^2}x_0^2+y_0^2+2\frac{b^2}{a^2}x_0x-2y_0y\right)=0
\end{align*}
这里的 $x,y$ 代表弦与曲线交点的坐标，因此满足 $\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$。于是二次项可以消去，剩下一条由 $A,B$ 两点确定的直线，也就是 $AB$ 的方程：
\begin{align*}
\Longleftrightarrow AB:&k_1k_2\left(2a^2-2x_0x+2y_0y\right)+(k_1+k_2)(2y_0x+2x_0y)+2b^2+2\frac{b^2}{a^2}x_0x-2y_0y=0\\
\Longleftrightarrow AB:&k_1k_2\left(a^2-x_0x+y_0y\right)+(k_1+k_2)(y_0x+x_0y)+b^2+\frac{b^2}{a^2}x_0x-y_0y=0\\
\Longleftrightarrow AB:&\left( \frac{b^2}{a^2} x_0 - k_1k_2 x_0 + (k_1+k_2) y_0 \right) x + \left( k_1k_2 \frac{a^2}{b^2} y_0 - y_0 + (k_1+k_2) x_0 \right) y + k_1k_2 a^2 + b^2 = 0
\end{align*}
对双曲线 $\frac{x^2}{a^2}-\frac{y^2}{b^2}=1$ 也可作同样处理。取关于原点对称的两点 $P$ 和 $-P$，分别整理后相加，得到
\begin{align*}
AB:&\ k_1k_2(x - x_0)^2 - (k_1+k_2)(x - x_0)(y - y_0) + (y - y_0)^2=0\\
AB:&\ -k_1k_2\frac{a^2}{b^2}(y + y_0)^2+ (k_1+k_2)(x + x_0)(y + y_0) - \frac{b^2}{a^2}(x + x_0)^2=0\\
\implies AB:&\ k_1k_2\left[x^2-\frac{a^2}{b^2}y^2+x_0^2-\frac{a^2}{b^2}y_0^2-2x_0x-2\frac{a^2}{b^2}y_0y\right]\\
&\quad +(k_1+k_2)\big[(x - x_0+2x_0)(y - y_0+2y_0)-(x - x_0)(y - y_0)\big]\\
&\quad +\left[-\frac{b^2}{a^2}x^2+y^2-\frac{b^2}{a^2}x_0^2+y_0^2-2\frac{b^2}{a^2}x_0x-2y_0y\right]=0\\
\Longleftrightarrow AB:&\ k_1k_2\left[2a^2-2x_0x-2\frac{a^2}{b^2}y_0y\right]+(k_1+k_2)(2y_0x+2x_0y)-2b^2-2\frac{b^2}{a^2}x_0x-2y_0y=0\\
\Longleftrightarrow AB:&\ k_1k_2\left[a^2-x_0x-\frac{a^2}{b^2}y_0y\right]+(k_1+k_2)(y_0x+x_0y)-b^2-\frac{b^2}{a^2}x_0x-y_0y=0\\
\Leftrightarrow k_1k_2 a^2 - b^2 &\ + \left( -\frac{b^2}{a^2} x_0  - k_1k_2 x_0 + (k_1+k_2) y_0 \right) x + \left( -k_1k_2 \frac{a^2}{b^2} y_0 - y_0 + (k_1+k_2) x_0 \right) y  = 0
\end{align*}
对抛物线，可以取点 $P$ 与 $x$ 轴方向上的无穷远点来做同样的整理。对应到有限平面，就是过 $A,B$ 的两条水平线，所以
\begin{align*}
(y-y_A)(y-y_B)=0&\Leftrightarrow \left[  (y + y_0) - (y_A + y_0) \right] \left[ (y + y_0) - (y_B + y_0)\right] = 0\\
&\Leftrightarrow \left[  (y + y_0) - \frac{2p}{k_1} \right] \left[ (y + y_0) - \frac{2p}{k_2}\right] = 0\\
&\Leftrightarrow k_1k_2(y + y_0)^2 - 2p(k_1+k_2)(y + y_0) + 4p^2 = 0
\end{align*}
利用$y^2=2px$消去二次项得到
\begin{align*}
\Longleftrightarrow AB:&k_1k_2\left(2px+2px_0+2y_0y\right)-2p(k_1+k_2)(y+y_0)+4p^2=0\\
\Longleftrightarrow AB:&k_1k_2\left(px+px_0+y_0y\right)-p(k_1+k_2)(y+y_0)+2p^2=0\\
\Longleftrightarrow AB:&p k_1k_2 x + \left( k_1k_2 y_0 - p(k_1+k_2) \right) y + p k_1k_2 x_0 - p(k_1+k_2) y_0 + 2p^2 = 0
\end{align*}
因此，弦 $AB$ 的方程可以写成如下形式，其中 $P^*$ 表示 $P$ 关于 $x$ 轴的对称点：
\begin{itemize}
    \item 椭圆：$ (b^2 - k_1k_2 a^2) \cdot E(P^*, X) + (k_1+k_2) \cdot W(P^*, X) + (b^2 + K a^2) = 0$
    \item 双曲线：$(b^2 + k_1k_2 a^2) \cdot E(P^*, X) - S \cdot W(P^*, X) + (b^2 - k_1k_2 a^2) = 0$
    \item 抛物线：$ k_1k_2 \cdot T(P^*, X) + p (k_1+k_2) \cdot (y - y_{P^*}) - 2p^2 = 0 $
\end{itemize}
以下两个例子直接使用 $P^*$ 公式。

例如椭圆 $\frac{x^2}{4} + y^2 = 1$。基点为上顶点 $P_2(0,1)$，设两交点与基点连线的斜率和为 $S$。
\begin{itemize}
\item 椭圆参数为 $a^2=4, b^2=1$，基点 $(x_0,y_0)=(0,1)$，设斜率和为 $S=k_1+k_2$，斜率积为 $K=k_1k_2$。
    \item 代入前文得到的椭圆弦 $AB$ 方程公式：
$$ \left( \frac{1}{4} \cdot 0 - K \cdot 0 + S \cdot 1 \right) x + \left( K \cdot \frac{4}{1} \cdot 1 - 1 + S \cdot 0 \right) y + K \cdot 4 + 1 = 0 $$
\item 化简整理，得到原坐标系下直线 $AB$ 的方程为 $Sx + (4K-1)y + 4K + 1 = 0$。
\item 将直线方程按独立参数 $K$ 整理为 $K(4y+4) + (Sx-y+1) = 0$。令 $4y+4=0$ 且 $Sx-y+1=0$，解得 $y=-1, x=-2/S$。即直线恒过定点 $(-2/S, -1)$。若已知直线过定点 $(2,-1)$，则可反推得 $S=-1$。
\end{itemize}
椭圆 $x^2/4 + y^2/3 = 1$。基点为右顶点 $M(2,0)$，已知斜率积 $k_1k_2 = -1$。
\begin{itemize}
\item 椭圆参数为 $a^2=4, b^2=3$，基点 $(x_0,y_0)=(2,0)$，已知斜率积 $K=k_1k_2=-1$，设斜率和为 $S=k_1+k_2$。
    \item 代入前文得到的椭圆弦 $AB$ 方程公式：
$$ \left( \frac{3}{4} \cdot 2 - K \cdot 2 + S \cdot 0 \right) x + \left( K \cdot \frac{4}{3} \cdot 0 - 0 + S \cdot 2 \right) y + K \cdot 4 + 3 = 0 $$
\item 整理得到 $\left( \frac{3}{2} - 2K \right)x + 2Sy + 4K + 3 = 0$。将已知条件 $K=-1$ 代入，得 $\frac{7}{2}x + 2Sy - 1 = 0$，即 $7x + 4Sy - 2 = 0$。
\item 由于此直线方程中含有 $S$ 的项仅为 $4Sy$，故该直线独立于参数 $S$ 恒过定点。令 $y=0$，解得 $x = 2/7$，即直线恒过定点 $\left(\frac{2}{7}, 0\right)$。
\end{itemize}

\section{同解法}

\begin{conclusion}[直线的几个基本方程]
已知二次曲线的方程为 ${C}: T(X,X)=0$。
\begin{enumerate}
    \item \textbf{割线方程$AB$}：$ F_{AB}(X) = T(X,A) + T(X,B) - T(A,B) = 0 $
    \item \textbf{点 $A$ 处的切线方程}：$ T(X,A) = 0 $
    \item \textbf{切点弦方程（极线）}：从曲线外一点 $P$ 引两条切线，切点为 $A,B$：$AB: T(X,P) = 0 $
    \item \textbf{中点弦方程}：过曲线内一点 $P$ 作一条弦，若 $P$ 恰好是这条弦的中点，则该弦方程为 $T(X,P) = T(P,P)$
\end{enumerate}
\end{conclusion}
\begin{itemize}
\item 直线上任意点 $X$ 都可写成 $X = \alpha A + \beta B$，且 $\alpha + \beta = 1$。
分别计算 $T(X,A)$ 和 $T(X,B)$：
\begin{align*}
    T(X,A) &= T(\alpha A + \beta B, A) = \alpha T(A,A) + \beta T(B,A) = \beta T(A,B) \\
    T(X,B) &= T(\alpha A + \beta B, B) = \alpha T(A,B) + \beta T(B,B) = \alpha T(A,B)
\end{align*}
把两式相加，
$$ T(X,A) + T(X,B) = (\alpha + \beta) T(A,B) = 1 \cdot T(A,B) $$
移项即得 $T(X,A) + T(X,B) - T(A,B) = 0$。
\item 切线：令割线上的点 $B$ 趋于 $A$，上式变为 $T(X,A) + T(X,A) - T(A,A) = 0$。由于 $T(A,A)=0$，可得 $T(X,A) = 0$。
\item 切点弦：切线 $PA$ 的方程是 $T(X,A)=0$。把 $X=P$ 代入得 $T(P,A)=0$。由于 $T$ 是对称双线性式，$T(A,P)=T(P,A)=0$。同样，$T(B,P)=0$。因此 $A,B$ 都在直线 $T(X,P)=0$ 上，所以这条直线就是切点弦 $AB$。
\item 中点弦：设直线上任意一点 $X = P + \vec{v}$。因为 $P$ 是中点，交点参数方程的两根之和为 $0$，所以一次项系数为 $0$，即 $E(P, \vec{v}) = 0$。
代入展开：$E(P, X-P) = 0 \implies E(P,X) - E(P,P) = 0$。
再用恒等式 $E = T + 1$，便得 $T(X,P) = T(P,P)$。
\end{itemize}

\begin{conclusion}[几类常见交点曲线系]
设原二次曲线为 $\mathcal{C}: T(X,X)=0$。
\begin{enumerate}
    \item \textbf{四交点系}：过 $C$ 上 $A, B, C, D$ 四点的新二次曲线族：$$ T(X,X) + \lambda \cdot \underbrace{F_{AB}(X)}_{\text{割线}AB} \cdot \underbrace{F_{CD}(X)}_{\text{割线}CD} = 0 $$
    \item \textbf{三交一切系}：在 $A$ 点相切，且过 $C, D$ 的曲线系（让上面公式中的 $B \to A$）：$$ T(X,X) + \lambda \cdot \underbrace{T(X,A)}_{\text{A的切线}} \cdot \underbrace{F_{CD}(X)}_{\text{割线}CD}  = 0 $$
    \item \textbf{双切线系}：与 $C$ 在 $A, B$ 两点发生双重相切的曲线系（设切点弦 $AB$ 对应的极点为 $P$）：$$ T(X,X) + \lambda \cdot \big[T(X,P)\big]^2 = 0 $$
    \item \textbf{四阶接触曲线系}：在 $A$ 点发生四阶接触时（令双切系中 $P \to A$）：$$ T(X,X) + \lambda \cdot \big[T(X,A)\big]^2 = 0 $$
\end{enumerate}
\end{conclusion}

\begin{exercise}
椭圆外一点$P$向椭圆做两条切线，切线对的联合方程？
\end{exercise}
\begin{solution}
设点 $X$ 为从点 $P$ 引出的切线上异于 $P$ 的任意一点。此时直线 $PX$ 与椭圆有且仅有一个交点。
设直线 $PX$ 上的动点可表示为 $Y = P + t\vec{v}$，其中方向向量为 $\vec{v} = X - P$。
将点 $Y$ 的坐标代入椭圆极化方程 $T(Y,Y) = 0$ 中，利用双线性分配律展开可得
\[ T(P + t\vec{v}, P + t\vec{v}) = E(\vec{v},\vec{v})t^2 + 2E(P,\vec{v})t + T(P,P) = 0. \]
由于直线 $PX$ 为椭圆的切线，该关于 $t$ 的一元二次方程有两个相等的实数根。因此，其判别式 $\Delta$ 为零，即
\[ \Delta = 4\big[E(P,\vec{v})\big]^2 - 4E(\vec{v},\vec{v})T(P,P) = 0. \]
化简可得 $\big[E(P,\vec{v})\big]^2 = E(\vec{v},\vec{v})T(P,P)$。
将方向向量 $\vec{v} = X - P$ 代入内积中展开。对于一次项，有
\[ E(P, \vec{v}) = E(P, X - P) = E(P,X) - E(P,P) = T(P,X) - T(P,P). \]
对于二次项，有
\[ E(\vec{v}, \vec{v}) = E(X - P, X - P) = E(X,X) - 2E(P,X) + E(P,P) = T(X,X) - 2T(P,X) + T(P,P). \]
将上述展开式代回判别式为零的等式中，得到
\[ \big[T(P,X) - T(P,P)\big]^2 = \big[T(X,X) - 2T(P,X) + T(P,P)\big] T(P,P). \]
将等式两边展开，可得
\[ T(P,X)^2 - 2T(P,P)T(P,X) + T(P,P)^2 = T(P,P)T(X,X) - 2T(P,P)T(P,X) + T(P,P)^2. \]
消去等式两边相同的项 $-2T(P,P)T(P,X)$ 与 $T(P,P)^2$，得到切线对的联合方程
\[ T(P,X)^2 - T(P,P)T(X,X) = 0. \]
\end{solution}

\begin{exercise}
过椭圆上一点$P(x_0,y_0)$引弦$PA,PB$，试用$x_0,y_0,F(P,A,X),F(P,B,X)$表示弦$AB$方程。（注：$F(A,B,X):y(x_A-x_B)+x(y_B-y_A)-(x_Ay_B-x_By_A)$）
\end{exercise}
\begin{solution}
已知椭圆方程为 $T(X,X) = 0$。由于点 $P(x_0,y_0)$ 在椭圆上，所以 $T(P,P) = 0$。
过点 $P$ 的两条割线方程可分别表示为 $F(P,A,X)=0$ 与 $F(P,B,X)=0$。
椭圆与双直线 $F(P,A,X)F(P,B,X)=0$ 在点 $P$ 有共同的切向接触，并在 $A,B$ 两点相交。为了把剩余因子分离出来，设存在常数 $\mu$ 与一次式 $L_{AB}(X)$，使
\[ T(X,X) - \mu F(P,A,X)F(P,B,X) \equiv T(P,X) L_{AB}(X). \]
为求解常数 $\mu$，在切线 $T(P,X)=0$ 上取一点 $Q$，使得 $Q \neq P$。由椭圆在点 $P$ 处的切线方向向量为 $\vec{t} = (a^2 y_0, -b^2 x_0)$，可写出切线上一点
\[ Q = P + \vec{t} = (x_0 + a^2 y_0, y_0 - b^2 x_0). \]
将点 $X = Q$ 代入等式。由于点 $Q$ 位于切线 $T(P,X)=0$ 上，必有 $T(P,Q) = 0$，等式右侧归零：
\[ T(Q,Q) - \mu F(P,A,Q)F(P,B,Q) = 0. \]
对于 $T(Q,Q)$，展开可得
\[ T(Q,Q) = T(P+\vec{t}, P+\vec{t}) = T(P,P) + 2E(P,\vec{t}) + E(\vec{t},\vec{t}). \]
由于 $T(P,P)=0$ 且 $\vec{t}$ 为切线方向向量满足 $E(P,\vec{t})=0$，计算得 $T(Q,Q) = E(\vec{t},\vec{t}) = a^2b^2$。
将此结果代入，由于点 $Q$ 不在割线 $PA$ 与 $PB$ 上，有 $F(P,A,Q)F(P,B,Q) \neq 0$，解得
\[ \mu = \frac{a^2b^2}{F(P,A,Q)F(P,B,Q)}. \]
将求得的常数 $\mu$ 代回原恒等式，并解出 $L_{AB}(X)$，得到弦 $AB$ 的方程
\[ L_{AB}(X) = \frac{F(P,A,Q) F(P,B,Q) T(X,X) - a^2b^2 F(P,A,X) F(P,B,X)}{F(P,A,Q) F(P,B,Q) T(P,X)} = 0. \]
由于除式是一次多项式，被除式是二次多项式，所以商式必为一次多项式。设原坐标系下直线 $AB$ 的方程为
\[
L_{AB}(X) = Ax + By + C.
\]
把等式左右两端展开后，只比较 $x^2,\ y^2$ 和常数项：
等式左端（被除式）：
\begin{align*}
    \text{左端} &= \left(\frac{x^2}{a^2} + \frac{y^2}{b^2} - 1\right) - \mu \big[y - y_0 - k_1(x-x_0)\big] \big[y - y_0 - k_2(x-x_0)\big] \\
    &= \left(\frac{1}{a^2} - \mu k_1k_2\right) x^2 + \cdots xy + \left(\frac{1}{b^2} - \mu\right) y^2 + \cdots x + \cdots y \\
    &\quad - 1 - \mu(k_1x_0 - y_0)(k_2x_0 - y_0)
\end{align*}
等式右端（除式 $\times$ 商式）：
\begin{align*}
    \text{右端} &= \left(\frac{x_0}{a^2}x + \frac{y_0}{b^2}y - 1\right) (Ax + By + C) \\
    &= \frac{x_0}{a^2}A x^2 + \cdots xy + \frac{y_0}{b^2}B y^2 + \cdots x + \cdots y - C
\end{align*}
由于这是多项式恒等式，同类项系数必然相等，于是
\begin{align*}
    [x^2] \text{ 项}: \quad \frac{x_0}{a^2}A &= \frac{1}{a^2} - \mu k_1k_2 \implies A = \frac{1 - \mu a^2 k_1k_2}{x_0} = \frac{\mu}{x_0}\left(\frac{1}{\mu} - a^2 k_1k_2\right)\\
    [y^2] \text{ 项}: \quad \frac{y_0}{b^2}B &= \frac{1}{b^2} - \mu \implies B = \frac{1 - \mu b^2}{y_0} = \frac{\mu}{y_0}\left(\frac{1}{\mu} - b^2\right)\\
    \text{常数项}: \quad -C &= - 1 - \mu(k_1x_0 - y_0)(k_2x_0 - y_0) \implies C = \mu\left[ \frac{1}{\mu} + (k_1x_0 - y_0)(k_2x_0 - y_0) \right]
\end{align*}
把
\[
\frac{1}{\mu} = k_1k_2\frac{a^2}{b^2}y_0^2 + (k_1+k_2)x_0y_0 + \frac{b^2}{a^2}x_0^2
\]
代入中括号内，再利用点 $P$ 在椭圆上的条件 $\frac{x_0^2}{a^2} + \frac{y_0^2}{b^2} = 1$ 化简：
对于系数 $A$：
\begin{align*}
    A &= \frac{\mu}{x_0} \left[ k_1k_2\frac{a^2}{b^2}y_0^2 + (k_1+k_2)x_0y_0 + \frac{b^2}{a^2}x_0^2 - a^2 k_1k_2 \right] \\
    &= \frac{\mu}{x_0} \left[ k_1k_2 a^2 \left(\frac{y_0^2}{b^2} - 1\right) + (k_1+k_2)x_0y_0 + \frac{b^2}{a^2}x_0^2 \right] \\
    &= \frac{\mu}{x_0} \left[ k_1k_2 a^2 \left(-\frac{x_0^2}{a^2}\right) + (k_1+k_2)x_0y_0 + \frac{b^2}{a^2}x_0^2 \right] \\
    &= \mu \left( \frac{b^2}{a^2}x_0 - k_1k_2 x_0 + (k_1+k_2)y_0 \right)
\end{align*}
对于系数 $B$：
\begin{align*}
    B &= \frac{\mu}{y_0} \left[ k_1k_2\frac{a^2}{b^2}y_0^2 + (k_1+k_2)x_0y_0 + \frac{b^2}{a^2}x_0^2 - b^2 \right] \\
    &= \frac{\mu}{y_0} \left[ k_1k_2\frac{a^2}{b^2}y_0^2 + (k_1+k_2)x_0y_0 + b^2 \left(\frac{x_0^2}{a^2} - 1\right) \right] \\
    &= \frac{\mu}{y_0} \left[ k_1k_2\frac{a^2}{b^2}y_0^2 + (k_1+k_2)x_0y_0 + b^2 \left(-\frac{y_0^2}{b^2}\right) \right] \\
    &= \mu \left( k_1k_2\frac{a^2}{b^2}y_0 - y_0 + (k_1+k_2)x_0 \right)
\end{align*}
对于常数项 $C$：
\begin{align*}
    C &= \mu \left[ \left( k_1k_2\frac{a^2}{b^2}y_0^2 + (k_1+k_2)x_0y_0 + \frac{b^2}{a^2}x_0^2 \right) + (k_1x_0 - y_0)(k_2x_0 - y_0) \right] \\
    &= \mu \left[ k_1k_2\frac{a^2}{b^2}y_0^2 + (k_1+k_2)x_0y_0 + \frac{b^2}{a^2}x_0^2 + k_1k_2x_0^2 - (k_1+k_2)x_0y_0 + y_0^2 \right] \\
    &= \mu \left[ k_1k_2 a^2 \left( \frac{y_0^2}{b^2} + \frac{x_0^2}{a^2} \right) + b^2 \left( \frac{x_0^2}{a^2} + \frac{y_0^2}{b^2} \right) \right] \\
    &= \mu \left( k_1k_2 a^2 + b^2 \right)
\end{align*}
将求得的 $A, B, C$ 拼接在一起，得到商式 $L_{AB}(X)$：
\begin{align*}
    L_{AB}(X) = \mu\left[\left( \frac{b^2}{a^2} x_0 - k_1k_2 x_0 + (k_1+k_2) y_0 \right) x + \left( k_1k_2 \frac{a^2}{b^2} y_0 - y_0 + (k_1+k_2) x_0 \right) y + k_1k_2 a^2 + b^2\right]
\end{align*}
由于 $\mu \neq 0$，可将等式两边同时除以 $\mu$，从而得到原坐标系下的直线 $AB$ 方程：
\begin{align*}
    \Longleftrightarrow AB:&\left( \frac{b^2}{a^2} x_0 - k_1k_2 x_0 + (k_1+k_2) y_0 \right) x + \left( k_1k_2 \frac{a^2}{b^2} y_0 - y_0 + (k_1+k_2) x_0 \right) y + k_1k_2 a^2 + b^2 = 0
\end{align*}
\end{solution}


\begin{exercise}
过非椭圆一点$P(x_0,y_0)$引割线$PAB,PCD$，试用$x_0,y_0,F_{AB}(X),F_{CD}(X),T(P,X)$表示割线$AC,BD,AD,BC$的方程。
\end{exercise}
\begin{solution}
设过极点 $P$ 的割线 $PAB$ 和 $PCD$ 的方程分别为 $F_{AB}(X) = 0$ 与 $F_{CD}(X) = 0$。
因为过点 $P$ 的任一直线方程都可由
\[
T(P,X),\qquad F_{AB}(X),\qquad F_{CD}(X)
\]
线性表示，而对角线 $AC$ 不经过点 $P$，可设
\[ F_{AC}(X) = T(P,X) + \mu_1 F_{AB}(X) + \mu_2 F_{CD}(X) = 0. \]
代入点 $A$。因为 $A$ 在割线 $PAB$ 上，满足 $F(A,B,A) = 0$，所以
\[
T(P,A) + \mu_2 F(C,D,A) = 0,
\]
从而
\[
\mu_2 = -\frac{T(P,A)}{F(C,D,A)}.
\]
再代入点 $C$。因为 $C$ 在割线 $PCD$ 上，满足 $F(C,D,C) = 0$，所以
\[
T(P,C) + \mu_1 F(A,B,C) = 0,
\]
从而
\[
\mu_1 = -\frac{T(P,C)}{F(A,B,C)}.
\]
令 $k_1 = \frac{T(P,C)}{F(A,B,C)}$ 且 $k_2 = \frac{T(P,A)}{F(C,D,A)}$，则割线 $AC$ 的方程为
\[ F_{AC}(X) \equiv T(P,X) - k_1 F_{AB}(X) - k_2 F_{CD}(X) = 0. \]
点 $B$ 为直线 $PA$ 与椭圆的交点，所以可写成
\[
B = \frac{P + \lambda A}{1+\lambda}.
\]
又因为点 $B$ 在椭圆上，满足 $T(B,B) = 0$，展开得
\[ \frac{T(P,P) + 2\lambda T(P,A) + \lambda^2 T(A,A)}{(1+\lambda)^2} = 0. \]
已知点 $A$ 在椭圆上，有 $T(A,A) = 0$。由于 $B \neq A$，必有 $\lambda \neq 0$，等式化简为 $T(P,P) + 2\lambda T(P,A) = 0$，解得 $\lambda = -\frac{T(P,P)}{2T(P,A)}$。
将其代回上式并同乘分母，可写成
\[
B \propto 2T(P,A)P - T(P,P)A.
\]
设直线 $BD$ 的方程为 $F_{BD}(X) = 0$。把点 $B$ 代入，得
\[ F_{BD}\big( 2T(P,A)P - T(P,P)A \big) = 2T(P,A)F_{BD}(P) - T(P,P)F_{BD}(A) = 0. \]
由对称性 $T(P,A)=T(A,P)$，上式说明点 $A$ 满足
\[ 2F_{BD}(P)T(P,X) - T(P,P)F_{BD}(X) = 0. \]
同理，点 $D$ 为直线 $PC$ 与椭圆的交点，可写成
\[
D \propto 2T(P,C)P - T(P,P)C.
\]
把它代入 $F_{BD}(D) = 0$，同样可证点 $C$ 也满足上述一次方程。
由于点 $A$ 与点 $C$ 均满足该方程，且两点不重合，则该方程必定等价于过这两点的直线 $AC$ 的方程 $F_{AC}(X) = 0$。
反过来，直线 $BD$ 的方程可由直线 $AC$ 写成
\[ F_{BD}(X) \propto 2F_{AC}(P)T(P,X) - T(P,P)F_{AC}(X) = 0. \]
    计算常数 $F_{AC}(P)$ 的值。将点 $P$ 代入 $F_{AC}(X)$ 的表达式中。由于点 $P$ 是两条割线 $PAB$ 与 $PCD$ 的交点，满足 $F(A,B,P) = 0$ 且 $F(C,D,P) = 0$，代入后可得
\[ F_{AC}(P) = T(P,P) - 0 - 0 = T(P,P). \]
将 $F_{AC}(P) = T(P,P)$ 与 $F_{AC}(X)$ 的完整表达式代入 $F_{BD}(X)$ 的恒等式中，有
\[ F_{BD}(X) \propto 2T(P,P)T(P,X) - T(P,P) \left[ T(P,X) - k_1 F_{AB}(X) - k_2 F_{CD}(X) \right] = 0. \]
由于点 $P$ 位于椭圆外，$T(P,P) \neq 0$。将等式两边同除以 $T(P,P)$ 并合并同类项，得到
\[ 2T(P,X) - T(P,X) + k_1 F_{AB}(X) + k_2 F_{CD}(X) = 0. \]
最终得出对角割线 $BD$ 的方程为
\[ F_{BD}(X) \equiv T(P,X) + k_1 F_{AB}(X) + k_2 F_{CD}(X) = 0. \]
同理可推得对角割线 $AD$ 与 $BC$ 的方程。
由于割线 $AD$ 不经过点 $P$，也可设
\[ F_{AD}(X) = T(P,X) + \mu_1 F_{AB}(X) + \mu_2 F_{CD}(X) = 0. \]
将点 $X = A$ 代入该方程。因点 $A$ 位于割线 $PAB$ 上，满足 $F(A,B,A) = 0$。方程化简为 $T(P,A) + \mu_2 F(C,D,A) = 0$，解得
\[ \mu_2 = -\frac{T(P,A)}{F(C,D,A)} = -k_2. \]
将点 $X = D$ 代入。因点 $D$ 位于割线 $PCD$ 上，满足 $F(C,D,D) = 0$。方程化简为 $T(P,D) + \mu_1 F(A,B,D) = 0$，解得
\[ \mu_1 = -\frac{T(P,D)}{F(A,B,D)}. \]
为了计算 $\mu_1$，可仿照前面求点 $B$ 的方法。点 $D$ 是直线 $PC$ 与椭圆的另一交点，所以
\[
D \propto 2T(P,C)P - T(P,P)C.
\]
将其分别代入 $T(P,X)$ 与 $F_{AB}(X)$ 中，展开得
\begin{align*}
T(P,D) &\propto 2T(P,C)T(P,P) - T(P,P)T(P,C) = T(P,P)T(P,C). \\
F(A,B,D) &\propto 2T(P,C)F(A,B,P) - T(P,P)F(A,B,C).
\end{align*}
因为极点 $P$ 位于割线 $PAB$ 上，代入可得 $F(A,B,P) = 0$，故：
\[ F(A,B,D) \propto -T(P,P)F(A,B,C). \]
两式相除后，
\[ \mu_1 = -\frac{T(P,D)}{F(A,B,D)} = -\frac{T(P,P)T(P,C)}{-T(P,P)F(A,B,C)} = \frac{T(P,C)}{F(A,B,C)} = k_1. \]
代回得
\[ F_{AD}(X) \equiv T(P,X) + k_1 F_{AB}(X) - k_2 F_{CD}(X) = 0. \]
同理可求割线 $BC$。设
\[
F_{BC}(X) = T(P,X) + \nu_1 F_{AB}(X) + \nu_2 F_{CD}(X) = 0.
\]
将点 $X = C$ 代入，由 $F(C,D,C) = 0$ 解得 $\nu_1 = -\frac{T(P,C)}{F(A,B,C)} = -k_1$。\\
将点 $X = B$ 代入，由 $F(A,B,B) = 0$ 解得 $\nu_2 = -\frac{T(P,B)}{F(C,D,B)}$。\\
将前文求出的
\[
B \propto 2T(P,A)P - T(P,P)A
\]
代入 $T(P,B)$ 与 $F(C,D,B)$。其中 $P,C,D$ 共线，故 $F(C,D,P)=0$，于是：
\begin{align*}
T(P,B) &\propto 2T(P,A)T(P,P) - T(P,P)T(P,A) = T(P,P)T(P,A) \\
F(C,D,B) &\propto 2T(P,A)\underbrace{F(C,D,P)}_{=0} - T(P,P)F(C,D,A) = -T(P,P)F(C,D,A)
\end{align*}
同样相除可得
\[ \nu_2 = -\frac{T(P,B)}{F(C,D,B)} = -\frac{T(P,P)T(P,A)}{-T(P,P)F(C,D,A)} = \frac{T(P,A)}{F(C,D,A)} = k_2. \]
因此，割线 $BC$ 的方程为
\[ F_{BC}(X) \equiv T(P,X) - k_1 F_{AB}(X) + k_2 F_{CD}(X) = 0. \]
于是四条直线都可写成
\begin{align*}
    AC:&\quad T(P,X) - k_1 F_{AB}(X) - k_2 F_{CD}(X) = 0 \\
    BD:&\quad T(P,X) + k_1 F_{AB}(X) + k_2 F_{CD}(X) = 0 \\
    AD:&\quad T(P,X) + k_1 F_{AB}(X) - k_2 F_{CD}(X) = 0 \\
    BC:&\quad T(P,X) - k_1 F_{AB}(X) + k_2 F_{CD}(X) = 0 
\end{align*}
把这些方程两两相加即可得到完全四边形的极线结论：
\begin{enumerate}
    \item 将对角线 $AC$ 与 $BD$ 的方程相加，
    \[ F_{AC}(X) + F_{BD}(X) = 2T(P,X) = 0 \implies T(P,X) = 0. \]
    因而对角线交点 $Q = AC \cap BD$ 满足 $T(P,Q) = 0$，即 $Q$ 在点 $P$ 的极线上。
    
    \item 同理，将侧边弦 $AD$ 与 $BC$ 的方程相加：
    \[ F_{AD}(X) + F_{BC}(X) = 2T(P,X) = 0 \implies T(P,X) = 0. \]
    所以对边交点 $R = AD \cap BC$ 也满足 $T(P,R) = 0$，即 $R$ 也在点 $P$ 的极线上。
\end{enumerate}
\end{solution}

\begin{exercise}[蝴蝶定理]
已知椭圆$\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$，点$P$是非椭圆上一点，$A(x_1,y_1),B(x_2,y_2)$在椭圆上，连接$AP,BP$交椭圆于$C,D$，求$CD$的方程。
\end{exercise}
\begin{solution}
设点 $C$ 为直线 $AP$ 与椭圆的交点。由于 $P, A, C$ 三点共线，可写成
\[
C = \frac{P + \lambda A}{1+\lambda}.
\]
又因为点 $C$ 位于椭圆上，所以 $T(C,C) = 0$。展开得
\[ \frac{T(P,P) + 2\lambda T(P,A) + \lambda^2 T(A,A)}{(1+\lambda)^2} = 0. \]
已知点 $A$ 在椭圆上，有 $T(A,A) = 0$。由于 $C \neq A$，必有 $\lambda \neq 0$。等式化简为 $T(P,P) + 2\lambda T(P,A) = 0$，解得常数 $\lambda = -\frac{T(P,P)}{2T(P,A)}$。
将 $\lambda$ 的值代回并同乘分母，可得
\[ C \propto 2T(P,A)P - T(P,P)A. \]
同理，可得
\[ D \propto 2T(P,B)P - T(P,P)B. \]
设待求弦 $CD$ 的方程为一次线性方程 $L_{CD}(X) = 0$。
将点 $C$ 代入该方程，可得
\[ L_{CD}(C) = L_{CD}\big(2T(P,A)P - T(P,P)A\big) = 2T(P,A)L_{CD}(P) - T(P,P)L_{CD}(A) = 0. \]
由对称性 $T(P,A) = T(A,P)$，整理得点 $A$ 满足
\[ 2L_{CD}(P)T(P,X) - T(P,P)L_{CD}(X) = 0. \]
同理，将点 $D$ 代入 $L_{CD}(D) = 0$ 中，展开可得
\[ 2T(P,B)L_{CD}(P) - T(P,P)L_{CD}(B) = 0. \]
故点 $B$ 也满足上述一次线性方程。
由于直线 $AB$ 上的两个不同点 $A$ 和 $B$ 都满足该方程，所以它就是直线 $AB$ 的方程，即
\[ L_{AB}(X) \propto 2L_{CD}(P)T(P,X) - T(P,P)L_{CD}(X). \]
反过来，直线 $CD$ 的方程可由直线 $AB$ 写成
\[ L_{CD}(X) \propto 2L_{AB}(P)T(P,X) - T(P,P)L_{AB}(X) = 0. \]
设已知弦 $AB$ 的方程为 $L_{AB}(X) = F_{AB}(X) = 0$。将它及其在点 $P$ 处的值 $L_{AB}(P) = F(A,B,P)$ 代入上式，得
\[ 2 F(A,B,P) T(P,X) - T(P,P) F_{AB}(X) = 0. \]
移项整理，最终得出弦 $CD$ 的方程为
\[ T(P,P) F_{AB}(X) = 2 T(P,X) F(A,B,P). \]
\end{solution}

\begin{exercise}
已知 \(A,B,C,D\) 为椭圆 \(\frac{x^2}{4} + \frac{y^2}{3} = 1\) 上四点，\(AC \cap BD = Q, Q(1,1)\)，且 \(\overrightarrow{AQ} = 2\overrightarrow{QC}, \overrightarrow{BQ} = 2\overrightarrow{QD}\)，求直线 \(CD,BC\) 方程；
\end{exercise}
\begin{solution}
记
\[
E(M, N) = \frac{x_M x_N}{4} + \frac{y_M y_N}{3},\qquad W(M, N) = x_M y_N - x_N y_M.
\]
椭圆方程就是 $E(X, X) = 1$，并且
\[
E(Q, Q) = \frac{1}{4} + \frac{1}{3} = \frac{7}{12}.
\]
（1）直线 $CD$ 可由 $\overrightarrow{AQ} = 2\overrightarrow{QC}$ 求出：
\[
A = 3Q - 2C.
\]
代入 $E(A,A) = 1$，
\begin{align*}
E(3Q - 2C, 3Q - 2C) &= 9E(Q,Q) - 12E(Q,C) + 4E(C,C)\\
 &= 9E(Q,Q) - 12E(Q,C) + 4 = 1 \\
\implies& 8E(Q,C) - 3E(Q,Q) - 1 = 0
\end{align*}
同理，
\[
8E(Q,D) - 3E(Q,Q) - 1 = 0.
\]
所以直线 $CD$ 的方程为
\[ 8E(Q,X) - 3E(Q,Q) - 1 = 0 \]
代入 $E(Q,Q) = \frac{7}{12}$，得
\[
16E(Q,X) - 11 = 0.
\]
将 $E(Q,X) = \frac{x}{4} + \frac{y}{3}$ 展开，可得
\[
12x + 16y - 33 = 0.
\]

（2）求直线 $BC$。由
\[
C = \frac{3}{2}Q - \frac{1}{2}A
\]
及 $E(C,C)=E(A,A)=1$，得
\[
9E(Q,Q) - 6E(Q,A) + 1 = 4,
\]
即
\[
2E(Q,A) - 3E(Q,Q) + 1 = 0.
\]
同理，
\[
2E(Q,B) - 3E(Q,Q) + 1 = 0.
\]
所以直线 $AB$ 的方程为
\[
2E(Q,X) - 3E(Q,Q) + 1 = 0.
\]
代入 $E(Q,Q) = \frac{7}{12}$，得
\[
8E(Q,X) - 3 = 0.
\]

（3）过 $A,B,C,D$ 四点的二次曲线系为：
\begin{align*}
& E(X,X) - 1 + \lambda \big( 8E(Q,X) - 3 \big)\big( 16E(Q,X) - 11 \big) = 0\\
\Longleftrightarrow & E(X,X) - 1 + \lambda \left( 128E(Q,X)^2 - 136E(Q,X) + 33 \right) = 0
\end{align*}
由拉格朗日恒等式，
\[ W(Q,X)^2 = a^2b^2 \Big( E(Q,Q)E(X,X) - E(Q,X)^2 \Big)=12 \Big( E(Q,Q)E(X,X) - E(Q,X)^2 \Big)\]
代入 $E(Q,Q) = \frac{7}{12}$，得
\[ W(Q,X)^2 = 12 \left( \frac{7}{12}E(X,X) - E(Q,X)^2 \right) = 7E(X,X) - 12E(Q,X)^2. \]
所以
\[ E(X,X) = \frac{12}{7}E(Q,X)^2 + \frac{1}{7}W(Q,X)^2. \]
代入上式并合并同类项，
\[ \left( \frac{12}{7} + 128\lambda \right) E(Q,X)^2 - 136\lambda E(Q,X) + (33\lambda - 1) + \frac{1}{7}W(Q,X)^2 = 0\]
若曲线系退化为一对过同一交点的直线，则左端可分解为两个一次因子的乘积。此处式子只含 $E(Q,X)$ 与 $W(Q,X)$ 两个一次量，因此可把两个一次因子写成
\[
\alpha E(Q,X)+\beta W(Q,X)+\gamma,\qquad
\alpha E(Q,X)-\beta W(Q,X)+\gamma.
\]
这样乘开后没有 $E(Q,X)W(Q,X)$ 交叉项，与上式的结构相符。于是关于 $E(Q,X)$ 的二次式必须成为完全平方，故判别式为 $0$：
\[ \Delta = (-136\lambda)^2 - 4 \left( \frac{12}{7} + 128\lambda \right) (33\lambda - 1) = 0. \]
化简得
\[
700\lambda^2 + 125\lambda + 3 = 0,
\]
因此
\[
\lambda_1 = -\frac{1}{35},\qquad \lambda_2 = -\frac{3}{20}.
\]
其中 $AC \cup BD$ 也在这个曲线系里。因为 $Q=AC\cap BD$，取 $X=Q$ 时应使退化式为 $0$。此时 $W(Q,Q)=0$，且 $E(Q,Q)=\frac{7}{12}$。代入上式可得 $\lambda=-\frac{3}{20}$，故这一根对应 $AC \cup BD$，另一根
\[
\lambda = -\frac{1}{35}
\]
对应 $AD \cup BC$。

把 $\lambda = -\frac{1}{35}$ 代回，
\[
\left( \frac{12}{7} + 128\lambda \right) E(Q,X)^2 - 136\lambda E(Q,X) + (33\lambda - 1)
= -\frac{68}{35}(E(Q,X)-1)^2.
\]
于是
\[
-\frac{68}{35}(E(Q,X) - 1)^2 + \frac{1}{7}W(Q,X)^2 = 0.
\]
两边同乘 $35$，
\[
5W(Q,X)^2 - 68(E(Q,X) - 1)^2 = 0.
\]
分解得
\[
\sqrt{5}W(Q,X) \pm 2\sqrt{17}(E(Q,X) - 1) = 0.
\]
再代入
\[
W(Q,X)=y-x,\qquad E(Q,X)-1=\frac{x}{4}+\frac{y}{3}-1=\frac{3x+4y-12}{12},
\]
便得
\[
\sqrt{5}(y - x) \pm 2\sqrt{17}\left(\frac{3x+4y-12}{12}\right) = 0.
\]
整理后
\[
(3\sqrt{17} \mp 6\sqrt{5})x + (4\sqrt{17} \pm 6\sqrt{5})y - 12\sqrt{17} = 0.
\]
这两条直线就是 $AD$ 与 $BC$。
\end{solution}

\begin{exercise}
    已知 \(P\) 为椭圆 \(C: \frac{x^2}{4} + \frac{y^2}{3} = 1\) 上一动点，过点 \(P\) 作圆 \(O: x^2 + y^2 = 1\) 的两条切线分别交椭圆 \(C\) 于 \(A,B\) 两点，求证：直线 \(OP\) 与直线 \(AB\) 的斜率之积为定值；
\end{exercise}
\begin{solution}
设点 $P(x_0, y_0)$ 满足方程 $\frac{x_0^2}{4} + \frac{y_0^2}{3} = 1$。
对于圆 $O: x^2 + y^2 = 1$，记其极化式为 $T_O(M, N) = x_M x_N + y_M y_N - 1$。
根据切线方程，过点 $P$ 向圆 $O$ 作出的两条切线 $PA, PB$ 的联合方程为
\[ T_O(P, X)^2 - T_O(P, P)T_O(X, X)=(x_0 x + y_0 y - 1)^2 - (x_0^2 + y_0^2 - 1)(x^2 + y^2 - 1) = 0\]
展开左边，
\begin{align*}
& (x_0^2 x^2 + y_0^2 y^2 + 1 + 2x_0 y_0 xy - 2x_0 x - 2y_0 y) \\
& - (x_0^2 x^2 + x_0^2 y^2 - x_0^2 + y_0^2 x^2 + y_0^2 y^2 - y_0^2 - x^2 - y^2 + 1) = 0.
\end{align*}
合并同类项，得
\[ (1 - y_0^2)x^2 + (1 - x_0^2)y^2 + 2x_0 y_0 xy - 2x_0 x - 2y_0 y + x_0^2 + y_0^2 = 0. \]
公共弦 $AB$ 可由切线对与椭圆方程的线性组合分离出来。切线对与椭圆方程相减后仍经过 $A,B$，因此取它与椭圆方程的线性组合，使其中分解出椭圆在 $P$ 处的切线和公共弦 $AB$。设存在实数 $\lambda$，使对任意点 $(x, y)$ 都有
\begin{align*}
& (1 - y_0^2)x^2 + (1 - x_0^2)y^2 + 2x_0 y_0 xy - 2x_0 x - 2y_0 y + x_0^2 + y_0^2 + \lambda\left(\frac{x^2}{4} + \frac{y^2}{3} - 1\right) \\
& \equiv \left(\frac{x_0}{4} x + \frac{y_0}{3} y - 1\right)(Ax + By + C_0).
\end{align*}
展开右边，
\[ \frac{A x_0}{4}x^2 + \frac{B y_0}{3}y^2 + \left(\frac{B x_0}{4} + \frac{A y_0}{3}\right)xy + \left(\frac{C_0 x_0}{4} - A\right)x + \left(\frac{C_0 y_0}{3} - B\right)y - C_0. \]
比较两边系数，求 $A,B,C_0$。
\begin{itemize}
\item 对于常数项，左侧为 $x_0^2 + y_0^2 - \lambda$，右侧为 $-C_0$，得到 $\lambda = x_0^2 + y_0^2 + C_0$。
\item 对于 $x$ 的一次项，左侧为 $-2x_0$，右侧为 $\frac{C_0 x_0}{4} - A$，解得 $A = x_0\left(\frac{C_0}{4} + 2\right)$。
\item 对于 $y$ 的一次项，左侧为 $-2y_0$，右侧为 $\frac{C_0 y_0}{3} - B$，解得 $B = y_0\left(\frac{C_0}{3} + 2\right)$。
\item 对于 $xy$ 的交叉项，左侧为 $2x_0 y_0$，右侧为 $\frac{B x_0}{4} + \frac{A y_0}{3}$。将所求得的 $A$ 与 $B$ 的表达式代入，可得
\begin{align*}
2x_0 y_0 &= \frac{x_0}{4} y_0 \left(\frac{C_0}{3} + 2\right) + \frac{y_0}{3} x_0 \left(\frac{C_0}{4} + 2\right) \\
&= x_0 y_0 \left( \frac{C_0}{12} + \frac{1}{2} + \frac{C_0}{12} + \frac{2}{3} \right) \\
&= x_0 y_0 \left( \frac{C_0}{6} + \frac{7}{6} \right).
\end{align*}
当 $x_0 y_0 \neq 0$ 时，等式两端约去非零因子 $x_0 y_0$，得到
\[ 2 = \frac{C_0}{6} + \frac{7}{6}. \]
解得 $C_0 = 5$。
\item 对于 $x_0 y_0 = 0$ 的情形，由 $\frac{x_0^2}{4} + \frac{y_0^2}{3} = 1$ 可知，当 $x_0 = 0$ 时 $y_0^2 = 3$。比较 $x^2$ 项系数，左侧为 $1 - y_0^2 + \frac{\lambda}{4} = -2 + \frac{3 + C_0}{4}$，右侧为 $0$，解得 $C_0 = 5$。当 $y_0 = 0$ 时 $x_0^2 = 4$。比较 $y^2$ 项系数，左侧为 $1 - x_0^2 + \frac{\lambda}{3} = -3 + \frac{4 + C_0}{3}$，右侧为 $0$，同样解得 $C_0 = 5$。因此常数 $C_0 = 5$ 恒成立。
将 $C_0 = 5$ 代回 $A, B$ 的表达式，分别求得 $A = \frac{13}{4} x_0$ 与 $B = \frac{11}{3} y_0$。
\end{itemize}
因此，割线 $AB$ 的直线方程为 $\frac{13}{4} x_0 x + \frac{11}{3} y_0 y + 5 = 0$。将等号两侧同乘 $12$，可化为一般式
\[ 39x_0 x + 44y_0 y + 60 = 0. \]
由该方程中 $x,y$ 的系数可得直线 $AB$ 的斜率为 $k_{AB} = -\frac{39x_0}{44y_0}$。
已知直线 $OP$ 的斜率为 $k_{OP} = \frac{y_0}{x_0}$。将两斜率相乘，计算可得
\[ k_{OP} \cdot k_{AB} = \frac{y_0}{x_0} \cdot \left(-\frac{39x_0}{44y_0}\right) = -\frac{39}{44}. \]
该乘积为不依赖于点 $P$ 的常数，故直线 $OP$ 与直线 $AB$ 的斜率之积为定值。
\end{solution}


\begin{exercise}
已知 \(A,B,C,D\) 为椭圆 \(E: \frac{x^2}{2} + y^2 = 1\) 上四点，直线 \(AC, BD\) 均过点 \(F_2(1,0)\)，
\begin{enumerate}
\item 若直线 \(AB\) 斜率为 1，求证：直线 \(CD\) 过定点；
\item 若直线 \(AB\) 过点 \(F_1(-1,0)\)，求证：\(\frac{k_{AB}}{k_{CD}}\) 为定值；
\end{enumerate}
\end{exercise}
\begin{solution}
对于椭圆 $E: \frac{x^2}{2} + y^2 = 1$，记其极化式为 $T(M, N) = \frac{x_M x_N}{2} + y_M y_N - 1$。点 $A, B, C, D$ 均在椭圆上，满足$T(A,A) = T(B,B) = T(C,C) = T(D,D) = 0$。记定点 $F_2(1,0)$ 为 $P$，已知直线 $AC$ 与 $BD$ 均过点 $P$。
由于点 $A, P, C$ 共线，可写成
\[
C = \frac{P + \lambda A}{1+\lambda}.
\]
将其代入 $T(C,C) = 0$，得
\begin{align*}
\frac{1}{(1+\lambda)^2} \left[ T(P,P) + 2\lambda T(P,A) + \lambda^2 T(A,A) \right] = 0\implies \lambda = -\frac{T(P,P)}{2T(P,A)}
\end{align*}
设直线 $CD$ 的方程为 $L_{CD}(X) = 0$。把点 $C$ 代入，得
\begin{align*}
L_{CD}(C) = L_{CD}\left(\frac{P + \lambda A}{1+\lambda}\right) = \frac{L_{CD}(P) + \lambda L_{CD}(A)}{1+\lambda} = 0
\end{align*}
即 $L_{CD}(P) + \lambda L_{CD}(A) = 0$。将 $\lambda$ 的表达式代入，可得
\begin{align*}
L_{CD}(P) - \frac{T(P,P)}{2T(P,A)} L_{CD}(A) = 0.
\end{align*}
等式两边同乘 $2T(P,A)$，并利用极化式的交换律 $T(P,A) = T(A,P)$，可将其重写为
\begin{align*}
2L_{CD}(P)T(P,A) - T(P,P)L_{CD}(A) = 0.
\end{align*}
    因此，左端这一线性表达式对点 $A$ 取零值，即
    \begin{align*}
    2L_{CD}(P)T(P,X) - T(P,P)L_{CD}(X) = 0.
    \end{align*}
    同理，点 $B$ 也满足这个方程。
因为 $A,B$ 是直线 $AB$ 上的两个不同点，所以
\[
L_{AB}(X) \propto 2L_{CD}(P)T(P,X) - T(P,P)L_{CD}(X).
\]
令 $X=P$，得
\[
L_{AB}(P) \propto L_{CD}(P)T(P,P).
\]
消去比例常数后，可把 $CD$ 的方程写成
\[
2L_{AB}(P) T(P,X) - T(P,P) L_{AB}(X) = 0.
\]
对于基点 $P(1,0)$，代入极化式计算可得
\begin{align*}
T(P,P) &= \frac{1 \times 1}{2} + 0 - 1 = -\frac{1}{2}, \\
T(P,X) &= \frac{1 \cdot x}{2} + 0 \cdot y - 1 = \frac{x-2}{2}.
\end{align*}
将其代入所推导的直线 $CD$ 等效方程中，可得
\begin{align*}
2L_{AB}(1,0)\left(\frac{x-2}{2}\right) - \left(-\frac{1}{2}\right) L_{AB}(X) = 0.
\end{align*}
化简得
\begin{align*}
L_{AB}(1,0)(x - 2) + \frac{1}{2} L_{AB}(X) = 0.
\end{align*}
方程两边同乘 $2$，得
\begin{align*}
2L_{AB}(1,0)(x - 2) + L_{AB}(X) = 0.
\end{align*}
当直线 $AB$ 的斜率为 $1$ 时，设直线 $AB$ 的方程为 $L_{AB}(X) = x - y + m = 0$。
将定点 $P(1,0)$ 代入，得 $L_{AB}(1,0) = 1 - 0 + m = m + 1$。
代入上式，得到
\begin{align*}
2(m+1)(x - 2) + (x - y + m) = 0\implies(3x - y - 4) + m(2x - 3) = 0.
\end{align*}
若 $CD$ 恒过一点，则该点代入后所得式子应对任意 $m$ 成立，因此
\begin{align*}
\begin{cases}
2x - 3 = 0 \\
3x - y - 4 = 0
\end{cases}
\end{align*}
由第一式解得 $x = \frac{3}{2}$，代入第二式可得 $3\left(\frac{3}{2}\right) - y - 4 = 0$，解得 $y = \frac{1}{2}$。
由此可得，直线 $CD$ 恒过定点 $\left(\frac{3}{2}, \frac{1}{2}\right)$。
当直线 $AB$ 过点 $F_1(-1,0)$ 时，设其方程为 $L_{AB}(X) = k(x + 1) - y = 0$，即 $kx - y + k = 0$。
将定点 $P(1,0)$ 代入，可得常数项 $L_{AB}(1,0) = k(1) - 0 + k = 2k$。
代入上式，得到
\begin{align*}
2(2k)(x - 2) + (kx - y + k) = 0.
\end{align*}
展开并化简得
\begin{align*}
4kx - 8k + kx - y + k = 0,
\end{align*}
即
\begin{align*}
5kx - y - 7k = 0.
\end{align*}
因此，直线 $CD$ 的方程是 $y = 5kx - 7k$，其斜率为 $k_{CD} = 5k$。
结合直线 $AB$ 的斜率 $k_{AB} = k$，当 $k \neq 0$ 时，可得两者斜率的比值为
\begin{align*}
\frac{k_{AB}}{k_{CD}} = \frac{k}{5k} = \frac{1}{5}.
\end{align*}
因此比值 $\frac{k_{AB}}{k_{CD}}$ 为定值 $\frac{1}{5}$。
\end{solution}

\begin{exercise}
已知 \(R,S\) 是 \(x\) 轴上两定点，过 \(S\) 的直线交椭圆 \(\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1\) 于 \(A,B\) 两点，分别延长 \(AR,BR\) 再次交椭圆于 \(D,C\) 两点。求证：直线 \(CD\) 过定点；
\end{exercise}
\begin{solution}
记椭圆 $\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1$ 的极化式为 $T(M, N) = \frac{x_M x_N}{a^2} + \frac{y_M y_N}{b^2} - 1$。
对于该椭圆上的任意一点 $X$，恒有 $T(X, X) = 0$。
已知直线 $AB$ 经过定点 $S(s, 0)$，设其方程为 $L_{AB}(X) = 0$。由于点 $S$ 位于直线 $AB$ 上，所以 $L_{AB}(S) = 0$。\\
根据题意，点 $R, A, D$ 三点共线，因此可写成：
\[ A = \frac{R + \lambda D}{1+\lambda} \]
由于点 $A$ 位于椭圆上，将上式代入椭圆方程，得
\[ T\left(\frac{R + \lambda D}{1+\lambda}, \frac{R + \lambda D}{1+\lambda}\right) = \frac{T(R, R) + 2\lambda T(R, D) + \lambda^2 T(D, D)}{(1+\lambda)^2} = 0. \]
已知点 $D$ 亦位于椭圆上，满足 $T(D, D) = 0$。代入上式，分子部分化简为：
\[ T(R, R) + 2\lambda T(R, D) = 0. \]
由于直线 $AR$ 与椭圆交于相异两点 $A, D$，故 $\lambda \neq 0$ 且 $T(R, D) \neq 0$。解得
\[ \lambda = -\frac{T(R, R)}{2T(R, D)}. \]
将其代回点 $A$ 的表达式中，分子与分母同乘 $2T(R, D)$，得到
\[ A = \frac{2T(R, D)R - T(R, R)D}{2T(R, D) - T(R, R)}. \]
由于点 $A$ 位于直线 $AB$ 上，将上式代入 $L_{AB}(A) = 0$，展开得
\[ \frac{2T(R, D)L_{AB}(R) - T(R, R)L_{AB}(D)}{2T(R, D) - T(R, R)} = 0. \]
等式两端同乘非零分母，得到点 $D$ 所满足的一个一阶线性关系式：
\[ 2L_{AB}(R)T(R, D) - T(R, R)L_{AB}(D) = 0. \]
另一方面，点 $R,B,C$ 三点也共线，按同样方法可得
\[ 2L_{AB}(R)T(R, C) - T(R, R)L_{AB}(C) = 0. \]
因此，点 $C,D$ 都满足
\[ 2L_{AB}(R)T(R, X) - T(R, R)L_{AB}(X) = 0. \]
因此直线 $CD$ 的方程为：
\[ L_{CD}(X) = 2L_{AB}(R)T(R, X) - T(R, R)L_{AB}(X) = 0. \]
记直线 $CD$ 的定点为 $T$。由于直线 $AB$ 满足 $L_{AB}(S)=0$，要使 $CD$ 恒过 $T$，应使 $L_{CD}(T)$ 在这个约束下恒为零。
已知定点 $R(r,0)$ 与 $S(s,0)$ 都在 $x$ 轴上，且整个条件关于 $x$ 轴对称。若直线 $CD$ 的定点唯一，它也应位于这条对称轴上。于是将候选点写成
\[ T = \alpha S + \beta R,\qquad \alpha + \beta = 1. \]
把点 $T$ 代入直线 $AB$ 的方程，得
\[ L_{AB}(T) = \alpha L_{AB}(S) + \beta L_{AB}(R). \]
代入条件 $L_{AB}(S) = 0$，上式化简为 $L_{AB}(T) = \beta L_{AB}(R)$。
同时，将点 $T$ 代入极化式中展开：
\[ T(R, T) = \alpha T(R, S) + \beta T(R, R). \]
将 $L_{AB}(T)$ 与 $T(R, T)$ 共同代入直线 $CD$ 的方程 $L_{CD}(T) = 0$ 中：
\[ 2L_{AB}(R)\big[\alpha T(R, S) + \beta T(R, R)\big] - T(R, R)\big[\beta L_{AB}(R)\big] = 0. \]
对于一般情形下不经过点 $R$ 的割线 $AB$，有 $L_{AB}(R) \neq 0$。将等式两端同除以 $L_{AB}(R)$，得到
\[ 2\alpha T(R, S) + 2\beta T(R, R) - \beta T(R, R) = 0 \implies 2\alpha T(R, S) + \beta T(R, R) = 0. \]
结合 $\alpha + \beta = 1$，将 $\alpha = 1 - \beta$ 代入得到：
\[ 2(1 - \beta) T(R, S) + \beta T(R, R) = 0 \implies \beta \big[2T(R, S) - T(R, R)\big] = 2T(R, S). \]
解得参数：
\[ \beta = \frac{2T(R, S)}{2T(R, S) - T(R, R)}, \quad \alpha = 1 - \beta = \frac{-T(R, R)}{2T(R, S) - T(R, R)}. \]
此时 $L_{CD}(T)=0$ 对任意过 $S$ 的割线 $AB$ 成立，而参数 $\alpha,\beta$ 只由定点 $R,S$ 决定，不随直线 $AB$ 变化，所以点 $T$ 是定点。\\
将 $S(s, 0)$ 与 $R(r, 0)$ 代入：
\[ T(R, S) = \frac{rs}{a^2} - 1, \quad T(R, R) = \frac{r^2}{a^2} - 1. \]
计算参数的分母部分：
\[ 2T(R, S) - T(R, R) = 2\left(\frac{rs}{a^2} - 1\right) - \left(\frac{r^2}{a^2} - 1\right) = \frac{2rs - r^2 - a^2}{a^2}. \]
由于 $R, S$ 的纵坐标均为 $0$，点 $T$ 的纵坐标 $y_T = \alpha \cdot 0 + \beta \cdot 0 = 0$。点 $T$ 的横坐标 $x_T$ 计算如下：
\begin{align*}
x_T &= \alpha s + \beta r = \frac{-s T(R, R) + 2r T(R, S)}{2T(R, S) - T(R, R)} \\
&= \frac{-s\left(\frac{r^2 - a^2}{a^2}\right) + 2r\left(\frac{rs - a^2}{a^2}\right)}{\frac{2rs - r^2 - a^2}{a^2}} \\
&= \frac{-sr^2 + sa^2 + 2r^2s - 2ra^2}{2rs - r^2 - a^2} \\
&= \frac{r^2s + a^2s - 2a^2r}{2rs - r^2 - a^2}.
\end{align*}
将分子与分母同时乘 $-1$，整理得：
\[ x_T = \frac{2a^2r - a^2s - r^2s}{a^2 + r^2 - 2rs}. \]
由此可得，在 $a^2 + r^2 - 2rs \neq 0$ 时，直线 $CD$ 恒经过 $x$ 轴上的定点 $\left( \frac{2a^2r - a^2s - r^2s}{a^2 + r^2 - 2rs}, 0 \right)$。\\
直线 $CD$ 与 $AB$ 的斜率比值如下。
设直线 $AB$ 的方程为 $x = my + s$，其斜率为 $k_{AB} = \frac{1}{m}$。它也可写成 $L_{AB}(X) = x_X - m y_X - s$。
代入定点 $R(r, 0)$ 得到 $L_{AB}(R) = r - s$。对于动点 $X(x, y)$，有 $L_{AB}(X) = x - my - s$。
代入直线 $CD$ 的方程：
\[ 2(r - s)\left(\frac{rx}{a^2} - 1\right) - \left(\frac{r^2 - a^2}{a^2}\right)(x - my - s) = 0. \]
等号两端同乘 $a^2$ 展开：
\[ 2(r - s)(rx - a^2) - (r^2 - a^2)(x - my - s) = 0. \]
整理 $x$ 项与 $y$ 项的系数。对 $x$ 的系数，
\[ 2r(r - s) - (r^2 - a^2) = 2r^2 - 2rs - r^2 + a^2 = r^2 - 2rs + a^2. \]
对于 $y$ 的系数：
\[ -(r^2 - a^2)(-m) = m(r^2 - a^2). \]
所以直线 $CD$ 可写成 $(r^2 - 2rs + a^2)x + m(r^2 - a^2)y + C = 0$，其斜率为
\[ k_{CD} = -\frac{r^2 - 2rs + a^2}{m(r^2 - a^2)} = \frac{a^2 + r^2 - 2rs}{m(a^2 - r^2)}. \]
两直线斜率相除，参数 $m$ 消去，得
\[ \frac{k_{CD}}{k_{AB}} = \frac{\frac{a^2 + r^2 - 2rs}{m(a^2 - r^2)}}{\frac{1}{m}} = \frac{a^2 + r^2 - 2rs}{a^2 - r^2}. \]

\end{solution}


\begin{conclusion}[弦对原点张角的联合方程]
    已知不经过原点的直线交椭圆于 $A, B$ 两点。设这条直线对应的极点为 $M$，即直线方程写成 $E(M,X) = 1$。那么连接原点与交点的两条射线 $OA, OB$ 构成的一对直线满足
    $$ E(X,X) - \big[E(M,X)\big]^2 = 0 $$
\end{conclusion}
把点 $A,B$ 代入可知，这个方程经过 $A,B$；又因为它是二次齐次式，所以表示过原点的一对直线。设二次齐次式为 $Ax^2+Bxy+Cy^2=0$，对应两条直线的斜率乘积为 $\frac{A}{C}$。在 $OA \perp OB$ 的情形下有 $\frac{A}{C}=-1$，即 $A+C=0$，因此比较 $x^2$ 与 $y^2$ 的系数得：
\begin{itemize}
\item $E(X,X)$ 提供：$x^2$ 的系数是 $\frac{1}{a^2}$，$y^2$ 的系数是 $\frac{1}{b^2}$。
\item $\big[E(M,X)\big]^2$ 提供：$x^2$ 的系数是 $\frac{x_M^2}{a^4}$，$y^2$ 的系数是 $\frac{y_M^2}{b^4}$。因此
\[
\left(\frac{1}{a^2} - \frac{x_M^2}{a^4}\right) + \left(\frac{1}{b^2} - \frac{y_M^2}{b^4}\right) = 0
\implies \frac{1}{a^2} + \frac{1}{b^2} = \frac{x_M^2}{a^4} + \frac{y_M^2}{b^4}.
\]
\end{itemize}
曲线外一点引两条切线的联合方程如下。

\begin{conclusion}[外切线对联合方程与蒙日圆]
    从二次曲线外一点 $P$ 作原曲线的两条切线，则其联合方程为
    $$ \big[T(P,X)\big]^2 = T(P,P) \cdot T(X,X) $$
    \textbf{推论（蒙日圆）：} 若这两条切线互相垂直，点 $P$ 的轨迹满足 $x_P^2 + y_P^2 = a^2 + b^2$。
\end{conclusion}
设 $X$ 是这两条切线上的任意一点。连结 $P$ 和 $X$，取线段 $PX$ 上任意一点
\[
Y = \frac{P + \lambda X}{1+\lambda}.
\]
因为直线 $PX$ 是切线，所以它与椭圆只有一个交点。把 $Y$ 代入 $E(Y,Y)=1$，得
\[
E\left(\frac{P+\lambda X}{1+\lambda}, \frac{P+\lambda X}{1+\lambda}\right) = \frac{E(P,P) + 2\lambda E(P,X) + \lambda^2 E(X,X)}{(1+\lambda)^2} = 1.
\]
整理后可写成
\[
\big(E(P,P) - 1\big) + 2\lambda \big(E(P,X) - 1\big) + \lambda^2 \big(E(X,X) - 1\big) = 0.
\]
再用 $T = E - 1$，便得到
$$ T(P,P) + 2\lambda T(P,X) + \lambda^2 T(X,X) = 0 $$
由于直线 $PX$ 是切线，这个关于 $\lambda$ 的一元二次方程有重根，因此判别式为零：
\[
\Delta = \big[2T(P,X)\big]^2 - 4 \cdot T(P,P) \cdot T(X,X) = 0.
\]
于是
\[
\big[T(P,X)\big]^2 = T(P,P) \cdot T(X,X).
\]
\textbf{由此可求蒙日圆轨迹}\\
若两条切线互相垂直，则它们的联合方程中 $x^2$ 与 $y^2$ 的系数和为 $0$。对
\[
\big[T(P,X)\big]^2 = T(P,P) \cdot T(X,X)
\]
比较系数：
\begin{itemize}
\item 左边展开来自于 $\big(\frac{x_P x}{a^2} + \frac{y_P y}{b^2} - 1\big)^2$，$x^2$ 系数为 $\frac{x_P^2}{a^4}$，$y^2$ 系数为 $\frac{y_P^2}{b^4}$。
\item 右边展开来自于 $T(P,P) \cdot \big(\frac{x^2}{a^2} + \frac{y^2}{b^2} - 1\big)$，$x^2$ 系数为 $T(P,P)\cdot\frac{1}{a^2}$，$y^2$ 系数为 $T(P,P)\cdot\frac{1}{b^2}$。
\end{itemize}
于是
$$ \left( \frac{x_P^2}{a^4} - \frac{T(P,P)}{a^2} \right) + \left( \frac{y_P^2}{b^4} - \frac{T(P,P)}{b^2} \right) = 0 \implies \frac{x_P^2}{a^4} + \frac{y_P^2}{b^4} = T(P,P)\left(\frac{1}{a^2} + \frac{1}{b^2}\right) $$
把 $T(P,P) = \frac{x_P^2}{a^2} + \frac{y_P^2}{b^2} - 1$ 代入右侧：
$$ \frac{x_P^2}{a^4} + \frac{y_P^2}{b^4} = {\left(\frac{x_P^2}{a^4}\right)} + \frac{x_P^2}{a^2 b^2} + \frac{y_P^2}{a^2 b^2} + {\left(\frac{y_P^2}{b^4}\right)} - \left(\frac{1}{a^2} + \frac{1}{b^2}\right) $$
两边同含的 $\frac{x_P^2}{a^4}$ 和 $\frac{y_P^2}{b^4}$ 可以约去，于是
\[
0 = \frac{x_P^2 + y_P^2}{a^2 b^2} - \frac{a^2+b^2}{a^2 b^2}.
\]
两边同乘 $a^2b^2$，得到
\[
x_P^2 + y_P^2 = a^2 + b^2.
\]


\section{参数方程法说明}
从过左顶点 $S(-a,0)$ 的割线出发，取方向向量
\[
\vec{v} = (a, bt),
\]
作直线
\[
X=S+\tau \vec{v}.
\]
其中参数 $t$ 只用来表示方向。

代入 $T(X,X)=0$。由于 $S$ 在椭圆上，$T(S,S)=0$，所以
\[
\tau = \frac{-2E(S,\vec{v})}{E(\vec{v},\vec{v})} = \frac{-2(-1)}{1+t^2} = \frac{2}{1+t^2}.
\]
代回去，得到有理参数式
\[
X(t) = \left( -a + a\frac{2}{1+t^2}, \ 0 + bt\frac{2}{1+t^2} \right) = \left( a\frac{1-t^2}{1+t^2}, b\frac{2t}{1+t^2} \right).
\]
设极点坐标为 $P_{12}(x_0, y_0)$。若 $P_{12}$ 是弦 $M_1M_2$ 的极点，那么点 $M_1,M_2$ 都在同一条极线上，极线方程可以写成 $E(P_{12}, X) = 1$。
\[
E(P_{12}, X) = 1.
\]
把 $X(t)$ 代入，得
\[
\frac{x_0}{a}\left(\frac{1-t^2}{1+t^2}\right) + \frac{y_0}{b}\left(\frac{2t}{1+t^2}\right) = 1.
\]
两边同乘 $(1+t^2)$，整理为
\[
\left(1 + \frac{x_0}{a}\right)t^2 - \left(\frac{2y_0}{b}\right)t + \left(1 - \frac{x_0}{a}\right) = 0.
\]
若 $M_1,M_2$ 对应参数 $t_1,t_2$，那么 $t_1,t_2$ 正是这个方程的两根。由韦达定理，
\begin{align*}
    t_1 t_2 &= \frac{1 - x_0/a}{1 + x_0/a} \implies t_1 t_2 \left(1 + \frac{x_0}{a}\right) = 1 - \frac{x_0}{a} \implies {x_0 = a\frac{1-t_1t_2}{1+t_1t_2}} \\
    t_1 + t_2 &= \frac{2y_0/b}{1 + x_0/a} \implies \text{将} 1+\frac{x_0}{a} = \frac{2}{1+t_1t_2} \text{ 代入解纵坐标：} {y_0 = b\frac{t_1+t_2}{1+t_1t_2}}
\end{align*}
把上面刚求出的 $x_0, y_0$ 放进极线方程 $E(P_{12}, X) = 1$：
$$ \frac{1-t_1t_2}{1+t_1t_2} \frac{x}{a} + \frac{t_1+t_2}{1+t_1t_2} \frac{y}{b} = 1 $$
由对弦方程
\[
\frac{x_0}{a^2}x + \frac{y_0}{b^2}y = 1
\]
若写成 $Ax+By=1$，则 $x_0 = a^2A,\ y_0 = b^2B$。因此常见的判别常数可表示为
\[
\Delta_0 = a^2\left(\frac{x_0}{a^2}\right)^2 + b^2\left(\frac{y_0}{b^2}\right)^2 - 1 = \frac{x_0^2}{a^2} + \frac{y_0^2}{b^2} - 1 = T(P_{12},P_{12}).
\]
将 $x_0, y_0$ 代入，得
\begin{align*}
T(P_{12}, P_{12}) &= \left(\frac{1-t_1t_2}{1+t_1t_2}\right)^2 + \left(\frac{t_1+t_2}{1+t_1t_2}\right)^2 - 1 \\
&= \frac{(1 - 2t_1t_2 + t_1^2t_2^2) + (t_1^2 + 2t_1t_2 + t_2^2) - (1+2t_1t_2+t_1^2t_2^2)}{(1+t_1t_2)^2} \\
&= \frac{t_1^2 + t_2^2 - 2t_1t_2}{(1+t_1t_2)^2} = \left(\frac{t_1-t_2}{1+t_1t_2}\right)^2 \equiv C^2
\end{align*}
因此，
\[
\Delta_0 = \left(\frac{t_1-t_2}{1+t_1t_2}\right)^2,
\]
它也就是极点 $P_{12}$ 的极化值 $T(P_{12},P_{12})$。
