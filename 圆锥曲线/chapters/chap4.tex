\chapter{比值代换法}
\section{原始算子的定义与算术性质}
\begin{definition}{椭圆内积算子 $E$}{}
    针对椭圆方程，定义输入为任意两点 $M(x_M, y_M)$ 与 $N(x_N, y_N)$ 的二元函数（输出为实数）：
$$ E(M, N) = \frac{x_M x_N}{a^2} + \frac{y_M y_N}{b^2} $$
\end{definition}
显然，$E(M,N)=E(N,M)$满足交换律，下证$E$满足分配律：

设存在任意三点 $A(x_A, y_A), B(x_B, y_B), C(x_C, y_C)$。设点 $P$ 是 $A$ 和 $B$ 的仿射组合，即存在实数 $\alpha, \beta$，使得 $x_P = \alpha x_A + \beta x_B$ 且 $y_P = \alpha y_A + \beta y_B$。
\begin{align*}
E(P, C) &= E( \alpha A + \beta B, C) \\
&=\frac{(\alpha x_A + \beta x_B)x_C}{a^2} + \frac{(\alpha y_A + \beta y_B)y_C}{b^2}  = \frac{\alpha x_A x_C}{a^2} + \frac{\beta x_B x_C}{a^2} + \frac{\alpha y_A y_C}{b^2} + \frac{\beta y_B y_C}{b^2} \\
&= \alpha \left( \frac{x_A x_C}{a^2} + \frac{y_A y_C}{b^2} \right) + \beta \left( \frac{x_B x_C}{a^2} + \frac{y_B y_C}{b^2} \right) =\alpha E(A,C) + \beta E(B,C) 
\end{align*}

\begin{definition}{极化算子 $T$ }{}
定义包含曲线边界信息的极化算子，当 $M$ 与 $N$ 重合且位于椭圆上时，$T(M,M)=0$。
$$ T(M, N) = E(M, N) - 1 = \frac{x_M x_N}{a^2} + \frac{y_M y_N}{b^2} - 1 $$
\end{definition}
显然，$T(M,N)=T(N,M)$满足交换律，下证当$P,M,N$共线时，算子$T$满足分配律：设点 $P$ 是 $A$ 和 $B$ 的仿射组合，即存在实数 $\alpha, \beta$，使得 $x_P = \alpha x_A + \beta x_B$ 且 $y_P = \alpha y_A + \beta y_B$，且$\alpha+\beta=1$。
\begin{align*}
  T(P,C) &= E(P,C) - 1  =E( \alpha  A + \beta B,C) - 1  \\
  &=\alpha E(A,C) + \beta E(B,C) - 1 =\alpha E(A,C) + \beta E(B,C)-\alpha - \beta\\
  &= \alpha E(A,C) - \alpha + \beta E(B,C) - \beta= \alpha T(A,C) + \beta T(B,C)
\end{align*}

\begin{definition}{欧氏外积算子 $W$}{}
    为了处理共线、斜率以及面积，引入二维向量叉乘算子。对于以原点为起点的两个位置向量 $\overrightarrow{OM} = (x_M, y_M)$ 与 $\overrightarrow{ON} = (x_N, y_N)$，定义：
$$ W(M, N) = x_M y_N - x_N y_M $$
\end{definition}
证明 $W$ 的反对称性：$W(M,N) = x_M y_N - x_N y_M = -(x_N y_M - x_M y_N) = -W(N,M)$（根据提取公因式 $-1$）。进一步，由 $W(M,M) = -W(M,M)$，两边同时加上 $W(M,M)$ 得到 $2W(M,M) = 0 \Rightarrow W(M,M) = 0$。
\textit{【批注：该算子的绝对值代表由 $\overrightarrow{OM}$ 和 $\overrightarrow{ON}$ 构成的平行四边形的有向面积。】}

设存在任意三点（或位置向量） $A(x_A, y_A), B(x_B, y_B), C(x_C, y_C)$。设 $P$ 是 $A$ 和 $B$ 的线性组合，即存在实数 $\alpha, \beta$，使得 $x_P = \alpha x_A + \beta x_B$ 且 $y_P = \alpha y_A + \beta y_B$。对于第一变元，代入算子定义并展开：
\begin{align*}W(P, C) &= W(\alpha A + \beta B, C) \\&= (\alpha x_A + \beta x_B)y_C - x_C(\alpha y_A + \beta y_B) \\&= \alpha x_A y_C + \beta x_B y_C - \alpha x_C y_A - \beta x_C y_B \\&= \alpha (x_A y_C - x_C y_A) + \beta (x_B y_C - x_C y_B) \\&= \alpha W(A, C) + \beta W(B, C)\end{align*}
进一步，结合已证的反对称性 $W(M,N) = -W(N,M)$，可直接推导算子 $W$ 对第二变元也同样满足分配律（无需再次代入繁琐的坐标）：
\begin{align*}W(C, P) &= W(C, \alpha A + \beta B) \&= -W(\alpha A + \beta B, C) \\&= -\big(\alpha W(A, C) + \beta W(B, C)\big) \\&= \alpha \big(-W(A, C)\big) + \beta \big(-W(B, C)\big) \\&= \alpha W(C, A) + \beta W(C, B)\end{align*}

\begin{definition}{欧氏内积算子 $I$}{}
处理角度与长度时使用标准点积：
$$ I(M, N) = x_M x_N + y_M y_N $$
\end{definition}

\section{核心结论与推导证明}

\begin{conclusion}{定比律，插入律，调和律}{}
已知二次曲线 $C$ 上有两点 $A, B$。直线 $AB$ 上有一动点 $P$ 满足 $\overrightarrow{AP} = \lambda \overrightarrow{PB}$。则必然成立：
\begin{enumerate}
    \item 比例生成：
    \[\lambda = \frac{T(P,A)}{T(P,B)},~~~\alpha=\frac{T(P,B)}{T(A,B)}=\frac12\frac{T(P,P)}{T(P,A)},~~~\beta=\frac{T(P,A)}{T(A,B)}=\frac12\frac{T(P,P)}{T(P,B)}\]
    \item 插入拆分：$T(A,B) = T(A,P) + T(P,B)$
    \item 调和极化：$T(P,P) \cdot T(A,B) = 2 T(P,A) \cdot T(P,B)\Longleftrightarrow T(P,P) = \frac{2 T(P,A) T(P,B)}{T(P,A) + T(P,B)}$
\end{enumerate}
\end{conclusion}
定比分点有$ x_P = \frac{x_A+\lambda x_B}{1+\lambda},y_P=\frac{y_A+\lambda y_B}{1+\lambda}$
由于点 $A, B$ 在曲线 $\mathcal{C}$ 上，所以$T(A,A) = T(B,B) = 0$。

\textbf{证明插入律：}用定比点差法：
\begin{align*}
&T(P,A) = T(\frac{A+\lambda B}{1+\lambda},A)=\frac{1}{1+\lambda} T(A,A) +\frac{\lambda}{1+\lambda} T(B,A)=\frac{\lambda}{1+\lambda} T(B,A)=\beta T(A,B)\\
&T(P,B) =T(\frac{A+\lambda B}{1+\lambda},B)=\frac{1}{1+\lambda} T(A,B) +\frac{\lambda}{1+\lambda} T(B,B)=\frac{1}{1+\lambda} T(A,B)=\alpha T(A,B)\\
&\implies\text{加:}T(P,A)+T(P,B)=\frac{\lambda}{1+\lambda} T(B,A)+\frac{1}{1+\lambda} T(A,B)= T(A,B)
\end{align*}

\textbf{证明调和律（相乘法）：}
将 $T(P,A)$ 与 $T(P,B)$ 两式相乘，并用$T(A,A)=T(B,B)=0$配凑:
\begin{align*}
T(P,A) \cdot T(P,B) &= \left( \frac{\lambda}{1+\lambda} T(A,B) \right) \cdot \left( \frac{1}{1+\lambda} T(A,B) \right) = \frac{\lambda\big[T(A,B)\big]^2}{(1+\lambda)^2} =\dfrac12\frac{2\lambda T(A,B)}{(1+\lambda)^2} T(A,B)\\
&=\frac{T(A,B)}2\left(\frac{1}{(1+\lambda)^2} T(A,A) + \frac{2\lambda}{(1+\lambda)^2} T(A,B) + \frac{\lambda^2}{(1+\lambda)^2} T(B,B)\right)\\
&=\frac12T(A,B)\cdot T(P,P)
\end{align*}
为什么称之为“调和”律？结合插入律 $T(A,B) = T(P,A) + T(P,B)$，将调和律等式中的 $T(A,B)$ 替换，可化简为：
$$ T(P,P) = \frac{2 T(P,A) T(P,B)}{T(P,A) + T(P,B)} $$
下面证明扩展的定比律：
\begin{align*}
&T(P,A) = T(\frac{A+\lambda B}{1+\lambda},A)=\frac{1}{1+\lambda} T(A,A) +\frac{\lambda}{1+\lambda} T(B,A)=\frac{\lambda}{1+\lambda} T(B,A)=\beta T(A,B)\\
&T(P,B) =T(\frac{A+\lambda B}{1+\lambda},B)=\frac{1}{1+\lambda} T(A,B) +\frac{\lambda}{1+\lambda} T(B,B)=\frac{1}{1+\lambda} T(A,B)=\alpha T(A,B)\\
&\implies\lambda = \frac{T(P,A)}{T(P,B)}=\frac{T(P,A)\cdot T(P,B)}{T(P,B)^2}=\frac{T(P,P)\cdot T(A,B)}{2T(P,B)^2}=\frac{2T(P,A)^2}{T(P,P)\cdot T(A,B)}\\
&\implies\alpha=\frac{T(P,B)}{T(A,B)}=\frac{T(P,B)\cdot T(P,A)}{T(A,B)\cdot T(P,A)}=\frac{\frac12T(A,B)\cdot T(P,P)}{T(A,B)\cdot T(P,A)}=\frac12\frac{T(P,P)}{T(P,A)}\\
&\implies\beta=\frac{T(P,A)}{T(A,B)}=\frac{T(P,A)\cdot T(P,B)}{T(A,B)\cdot T(P,B)}=\frac{\frac12T(A,B)\cdot T(P,P)}{T(A,B)\cdot T(P,B)}=\frac12\frac{T(P,P)}{T(P,B)}
\end{align*}

\begin{conclusion}{差分律（中点共轭律）}{}
已知二次曲线 $C$ 上有两点 $A, B$。直线 $AB$ 上一动点 $P$ 满足 $\overrightarrow{AP} = \lambda \overrightarrow{PB}$。则：
$$ E(P, \overrightarrow{BA}) = T(P,A) - T(P,B) = \frac{\lambda - 1}{\lambda + 1} T(A,B) $$
\textbf{推论（秒杀点差法）：}当 $P$ 为弦 $AB$ 的中点时，必然成立 $E(P, \overrightarrow{BA}) = 0$。
\end{conclusion}

\textbf{证明差分律：}
将 $T(P,A)$ 与 $T(P,B)$ 两式相减，对比得证：
\begin{align*}
T(P,A) - T(P,B) &= \frac{\lambda}{1+\lambda} T(A,B) - \frac{1}{1+\lambda} T(A,B) = \frac{\lambda - 1}{\lambda + 1} T(A,B)\\
T(P,A) - T(P,B) &= \big(E(P,A) - 1\big) - \big(E(P,B) - 1\big) = E(P,A) - E(P,B)=E(P, \overrightarrow{BA})
\end{align*}
\textit{【批注：差分律是解析几何中“点差法”的算子化降维表达。当动点 $P$ 移动到弦 $AB$ 中点时，定比 $\lambda = 1$，等式右侧直接归零，得到 $E(P, \overrightarrow{BA}) = 0$。将其展开为坐标形式即：
$$ \frac{x_P(x_A-x_B)}{a^2} + \frac{y_P(y_A-y_B)}{b^2} = 0 \implies k_{OP} \cdot k_{AB} = -\frac{b^2}{a^2} $$
我们仅用一行算子减法，就直接跨越了传统解题中设坐标、代入方程、作差、因式分解等极其繁琐的步骤，得到了大名鼎鼎的中点弦定理（垂径定理）！】}

\begin{conclusion}{非对称分式的内积坍缩定理}{}
椭圆上两点 $A, B$ 坐标为$(x_1,y_1),(x_2,y_2)$，中点为$M$，弦 $AB$ 对应的极点（即过 $A,B$ 的切线交点）为 $P$
$$ \frac{x_1y_1+x_2y_2}{x_1y_2+x_2y_1} = E(A,B) = 2E(M,M) - 1 = \frac{1 - T(P,P)}{1 + T(P,P)}  $$
\end{conclusion}
利用平方差公式推导：$ \frac{x_1y_1+x_2y_2}{x_1y_2+x_2y_1} = E(A,B)$
\begin{align*}
\frac{x_1y_1+x_2y_2}{x_1y_2+x_2y_1} & = \frac{(x_1y_1+x_2y_2) \cdot W(A,B)}{(x_1y_2+x_2y_1) \cdot W(A,B)} = \frac{(x_1y_1+x_2y_2)(x_1y_2-x_2y_1)}{x_1^2 y_2^2 - x_2^2 y_1^2} \\
&=\frac{x_1^2 y_1 y_2 - x_1 x_2 y_1^2 + x_1 x_2 y_2^2 - x_2^2 y_1 y_2}{ x_1^2 \left[ b^2 \left(1 - \frac{x_2^2}{a^2}\right) \right] - x_2^2 \left[ b^2 \left(1 - \frac{x_1^2}{a^2}\right) \right] }=\frac{y_1y_2(x_1^2 - x_2^2) + x_1x_2(y_2^2 - y_1^2)}{b^2 (x_1^2 - x_2^2) }\\
&=\frac{ b^2 (x_1^2 - x_2^2) E(A,B) }{ b^2 (x_1^2 - x_2^2) } = E(A,B)
\end{align*}
设中点 $M = \frac{A+B}{2}$。对其施加内积算子并利用双线性分配律完全展开：
\begin{align*} E(M,M) &= E\left(\frac{A+B}{2}, \frac{A+B}{2}\right) = \frac{1}{4} \big[ E(A,A) + 2E(A,B) + E(B,B) \big] =\frac{1 + E(A,B)}{2} \\
    \implies E(A,B) &= 2E(M,M) - 1\end{align*}
也可以反过来写，反正就是凑完全平方公式。

\textbf{严格证明最后一个等号：}假设$\overrightarrow{OA}$ 与 $\overrightarrow{OB}$ 不共线，它们可作为平面的一组基底。因此可设：
$ P = \alpha A + \beta B $，由于点 $A, B$ 在曲线上，$P$在$A,B$的极线上，故
\begin{align*}
1=E(P, A) &= E(\alpha A + \beta B, A) = \alpha E(A,A) + \beta E(B,A) \implies \alpha + \beta E(A,B) = 1 \\
1=E(P, B) &= E(\alpha A + \beta B, B) = \alpha E(A,B) + \beta E(B,B) \implies \alpha E(A,B) + \beta = 1\\
\text{相减：}&(\alpha - \beta) - (\alpha - \beta) E(A,B) = 0 \implies (\alpha - \beta)\big[1 - E(A,B)\big] = 0
\end{align*}
对于椭圆上两个不同的点 $A \neq B$，必有 $E(A-B, A-B)= 2 - 2E(A,B) > 0 > 0$。故$ \alpha = \beta $解得精准的伸缩系数：
$$ \alpha = \beta = \frac{1}{1 + E(A,B)} \implies P = \frac{A + B}{1 + E(A,B)} $$
接下来，计算极点 $P$ 的自内积 $E(P,P)$，继续使用算子的分配律展开：
\begin{align*}
E(P,P) &= E\left( \frac{A+B}{1+E(A,B)}, \frac{A+B}{1+E(A,B)} \right) = \frac{E(A,A) + E(B,B) + 2E(A,B)}{\big[1+E(A,B)\big]^2} \\
&= \frac{2 + 2E(A,B)}{\big[1+E(A,B)\big]^2} = \frac{2\big[1 + E(A,B)\big]}{\big[1+E(A,B)\big]^2} = \frac{2}{1 + E(A,B)}
\end{align*}
反解出 $E(A,B)$ 的表达式：
$$ 1 + E(A,B) = \frac{2}{E(P,P)} \implies E(A,B) = \frac{2}{E(P,P)} - 1= \frac{1 - T(P,P)}{1 + T(P,P)} $$

\begin{conclusion}{降维结论}{}
椭圆上两点$(x_1,y_1),(x_2,y_2)$，非椭圆上一点$P(m,n)$，则
\[(x_1y_2+x_2y_1)\left(\dfrac{m^2}{a^2}+\dfrac{n^2}{b^2}\right)=m(y_1+y_2) + n(x_1+x_2) - 2mn \]
\end{conclusion}
直线 $AB$ 上的动点 $P(m,n)$ 是 $A$ 和 $B$ 的仿射组合。设其仿射系数为 $\alpha, \beta$，即 $P = \alpha A + \beta B$（此时 $m = \alpha x_1 + \beta x_2, n = \alpha y_1 + \beta y_2$，且必定满足 $\alpha+\beta=1$）。
\begin{align*}
&(x_1y_2+x_2y_1)\left(\dfrac{m^2}{a^2}+\dfrac{n^2}{b^2}\right) \\
&= (x_1y_2+x_2y_1) E(P, P) \\
&= (x_1y_2+x_2y_1) E(\alpha A + \beta B, \alpha A + \beta B)= (x_1y_2+x_2y_1) \big[ \alpha^2 E(A,A) + \beta^2 E(B,B) + 2\alpha\beta E(A,B) \big] \\
&= (x_1y_2+x_2y_1) \big[ \alpha^2 + \beta^2 + 2\alpha\beta E(A,B) \big] = (\alpha^2 + \beta^2)(x_1y_2+x_2y_1) + 2\alpha\beta (x_1y_1 + x_2y_2) \\
&= 1 \cdot (x_1y_2+x_2y_1) - 2\alpha\beta (x_1y_2+x_2y_1) + 2\alpha(1-\alpha) x_1y_1 + 2\beta(1-\beta) x_2y_2\\
&= (\alpha+\beta)(x_1y_2+x_2y_1) + 2\alpha x_1y_1 - 2\alpha^2 x_1y_1 + 2\beta x_2y_2 - 2\beta^2 x_2y_2 - 2\alpha\beta(x_1y_2+x_2y_1)  \\
&= \big[ \alpha x_1y_2 + \beta x_1y_2 + \alpha x_2y_1 + \beta x_2y_1 + 2\alpha x_1y_1 + 2\beta x_2y_2 \big] - 2 \big( \alpha^2 x_1y_1 + \beta^2 x_2y_2 + \alpha\beta x_1y_2 + \alpha\beta x_2y_1 \big)  \\
&= \big[ \alpha x_1(y_1+y_2) + \alpha y_1(x_1+x_2) + \beta x_2(y_1+y_2) + \beta y_2(x_1+x_2) \big] - 2 \big( \alpha x_1 + \beta x_2 \big)\big( \alpha y_1 + \beta y_2 \big) \\
&= \big[ (\alpha x_1 + \beta x_2)(y_1+y_2) + (\alpha y_1 + \beta y_2)(x_1+x_2) \big] - 2 x_P y_P  \\
&= m(y_1+y_2) + n(x_1+x_2) - 2mn
\end{align*}









\textit{【上帝视角验证：参数方程的终极统一】\\
为什么极点控制律长得如此像三角万能代换公式 $\cos\Delta\theta = \frac{1-\tan^2(\Delta\theta/2)}{1+\tan^2(\Delta\theta/2)}$？\\
我们启用椭圆的离心角参数方程进行降维验证：设 $A(a\cos\alpha, b\sin\alpha), B(a\cos\beta, b\sin\beta)$。\\
考察分子（和差化积）：
$$ x_1y_1 + x_2y_2 = ab\cos\alpha\sin\alpha + ab\cos\beta\sin\beta = \frac{1}{2}ab(\sin 2\alpha + \sin 2\beta) = ab\sin(\alpha+\beta)\cos(\alpha-\beta) $$
考察分母（两角和正弦）：
$$ x_1y_2 + x_2y_1 = ab\cos\alpha\sin\beta + ab\cos\beta\sin\alpha = ab\sin(\alpha+\beta) $$
两式直接相除，极度非对称的 $ab\sin(\alpha+\beta)$ 瞬间约去，原分式竟然等价于极其纯粹的 $\cos(\alpha-\beta)$！\\
而这恰好就是两点间的内积算子：
$$ E(A,B) = \frac{a\cos\alpha \cdot a\cos\beta}{a^2} + \frac{b\sin\alpha \cdot b\sin\beta}{b^2} = \cos\alpha\cos\beta + \sin\alpha\sin\beta = \cos(\alpha-\beta) $$
而极点 $P$ 的极化算子 $T(P,P)$，其代数本质正是半角正切的平方 $\tan^2\frac{\alpha-\beta}{2}$。算子代数用最纯粹的多项式外壳，完美同构了超越函数的三角代换法则。几何与代数在这里达到了令人战栗的和谐统一！}


\begin{conclusion}{直线参数方程}{}
设有一条以定点 $P(x_0, y_0)$ 为基点，以向量 $\vec{v} = (v_x, v_y)$ 为方向向量的射线（或直线），其参数方程可表示为 $X = P + t\vec{v}$。将该点 $X$ 代入二次曲线的极化算子中，其结果是一个关于参数 $t$ 的标准一元二次多项式：
$$ T(X,X) = E(\vec{v},\vec{v}) t^2 + 2T(P, \vec{v}) t + T(P,P) $$
其中，我们在定义域上将 $T(P, \vec{v})$ 严格定义为 $E(P, \vec{v}) = \frac{x_0 v_x}{a^2} + \frac{y_0 v_y}{b^2}$。韦达定理：
\[ t_1 + t_2 = \frac{-2T(P,\vec{v})}{E(\vec{v},\vec{v})}, \quad t_1 t_2 = \frac{T(P,P)}{E(\vec{v},\vec{v})} \]
\end{conclusion}
\vspace{-20pt}
\begin{align*}
T(X,X)& = E(P + t\vec{v}, P + t\vec{v}) - 1 = E(P, P+t\vec{v}) + E(t\vec{v}, P+t\vec{v})-1\\
&= E(P, P+t\vec{v}) +t E(\vec{v}, P+t\vec{v})-1= (E(P,P) + tE(P,\vec{v})) + t(E(\vec{v},P) + tE(\vec{v},\vec{v}))-1 \\
&=E(P,P) + 2tE(P,\vec{v}) + t^2E(\vec{v},\vec{v})-1\\
&=T(P,P)+2tE(P,\vec{v}) + t^2E(\vec{v},\vec{v})
\end{align*}
\textit{【批注：这说明当坐标系原点发生平移时，二次曲线方程被算子系统直接降维成了一个无需手动代入 $(kx+m)^2$ 繁琐展开的参数多项式。】}

\begin{conclusion}{斜率的外积探测公式}{}
定义水平方向单位向量 $\vec{i} = (1, 0)$ 和竖直方向单位向量 $\vec{j} = (0, 1)$。
对于平面上任意一条以向量 $\vec{v}$ 为方向的直线，其斜率 $k$ 完全等价于外积算子 $W$ 的比值：
$$ k = \frac{W(\vec{i}, \vec{v})}{W(\vec{v}, \vec{j})} $$
\end{conclusion}
\textbf{【极简证明】：}
设 $\vec{v} = (v_x, v_y)$。根据外积算子的定义 $W(M,N) = x_M y_N - x_N y_M$：
用 $\vec{i}$ 探测：$W(\vec{i}, \vec{v}) = 1 \cdot v_y - 0 \cdot v_x = v_y$（完美提取了纵向分量 $\Delta y$）。
用 $\vec{j}$ 探测：$W(\vec{v}, \vec{j}) = v_x \cdot 1 - v_y \cdot 0 = v_x$（完美提取了横向分量 $\Delta x$）。
两者相除，立刻得到 $k = \frac{v_y}{v_x}$。

\textit{【实战降维意义】：当题目让你证明 $k_1 + k_2 = 0$ 或者 $k_1 = \lambda k_2$ 时，我们再也不用去写繁杂的分子分母坐标。因为 $W$ 算子就像乘法一样满足完美的分配律，所有的“斜率加减”都可以直接在算子内部进行向量的合并！这彻底消灭了通分带来的满屏代数风暴。}

\begin{conclusion}{算子拉格朗日恒等式}{}
对于平面上任意两个向量 $\vec{u}, \vec{v}$，其欧氏外积 $W$ 与椭圆内积 $E$ 永远满足以下绝对恒等关联：
$$ \big[W(\vec{u},\vec{v})\big]^2 = a^2 b^2 \Big[ E(\vec{u},\vec{u})E(\vec{v},\vec{v}) - \big[E(\vec{u},\vec{v})\big]^2 \Big] $$
\end{conclusion}
设 $\vec{u}=(x_1, y_1), \vec{v}=(x_2, y_2)$。
左侧外积的平方：$W(\vec{u},\vec{v})^2 = (x_1 y_2 - x_2 y_1)^2 = x_1^2 y_2^2 + x_2^2 y_1^2 - 2x_1 x_2 y_1 y_2$。
计算内积的平方并同乘 $a^2b^2$：$a^2b^2 E(\vec{u},\vec{v})^2 = a^2b^2 \left(\frac{x_1 x_2}{a^2} + \frac{y_1 y_2}{b^2}\right)^2 = \frac{b^2}{a^2}x_1^2 x_2^2 + \frac{a^2}{b^2}y_1^2 y_2^2 + 2x_1 x_2 y_1 y_2$。
将这两式\textbf{相加}，交叉项完全抵消：
\begin{align*}
W(\vec{u},\vec{v})^2 + a^2b^2 E(\vec{u},\vec{v})^2 &= x_1^2 y_2^2 + x_2^2 y_1^2 + \frac{b^2}{a^2}x_1^2 x_2^2 + \frac{a^2}{b^2}y_1^2 y_2^2 \\
&= x_1^2 \left( y_2^2 + \frac{b^2}{a^2}x_2^2 \right) + y_1^2 \left( x_2^2 + \frac{a^2}{b^2}y_2^2 \right)
\end{align*}
观察括号内部，直接提取常数化为椭圆内积算子：
$y_2^2 + \frac{b^2}{a^2}x_2^2 = b^2 \left( \frac{x_2^2}{a^2} + \frac{y_2^2}{b^2} \right) = b^2 E(\vec{v},\vec{v})$。同理，后半部分的括号等于 $a^2 E(\vec{v},\vec{v})$。
整体合并：
$$ x_1^2 \big[b^2 E(\vec{v},\vec{v})\big] + y_1^2 \big[a^2 E(\vec{v},\vec{v})\big] = a^2 b^2 E(\vec{v},\vec{v}) \left(\frac{x_1^2}{a^2} + \frac{y_1^2}{b^2}\right) = a^2 b^2 E(\vec{v},\vec{v}) E(\vec{u},\vec{u}) $$
移项即得证！这把开启面积降维打击的钥匙已经握在我们手中。

\begin{conclusion}{万能算子弦长公式}{}
已知直线过定点 $P$，且方向向量为 $\vec{v}$，交二次曲线于 $A, B$ 两点。则弦长 $|AB|$ 仅取决于算子 $T, E, I$ 在基点 $P$ 与方向 $\vec{v}$ 上的取值：
$$ |AB| = \frac{2\sqrt{\big[E(P, \vec{v})\big]^2 - E(\vec{v}, \vec{v})T(P, P)}}{E(\vec{v}, \vec{v})} \sqrt{I(\vec{v}, \vec{v})} = 2 \frac{\sqrt{E(\vec{v},\vec{v}) - \frac{\big[W(P,\vec{v})\big]^2}{a^2 b^2}}}{E(\vec{v},\vec{v})} \cdot \sqrt{I(\vec{v},\vec{v})} $$
特别地，当 $P$ 是弦 $AB$ 的\textbf{中点}时，公式瞬间坍缩为：$|AB| = 2\sqrt{\frac{-T(P,P)}{E(\vec{v}, \vec{v})}} \sqrt{I(\vec{v}, \vec{v})}$。
\end{conclusion}
直线上动点 $X = P + t\vec{v}$ 代入曲线极化算子得到：
$$ E(\vec{v},\vec{v})t^2 + 2E(P, \vec{v})t + T(P,P) = 0 $$
设交点对应的参数为 $t_1, t_2$。弦长本质上是参数的差值乘以方向向量的实际欧氏长度：$|AB| = |t_1 - t_2| \cdot |\vec{v}|$。
由一元二次方程求根差公式 $|t_1 - t_2| = \frac{\sqrt{\Delta}}{a}$，直接算出：
$$ |t_1 - t_2| = \frac{\sqrt{4\big[E(P, \vec{v})\big]^2 - 4E(\vec{v}, \vec{v})T(P, P)}}{E(\vec{v}, \vec{v})} $$
利用 $T(P,P) = E(P,P) - 1$ 将根号内的式子展开，并代入刚刚证明的《算子拉格朗日恒等式》：
\begin{align*}
\frac{\Delta}{4} &= E(P,\vec{v})^2 - \big(E(P,P) - 1\big)E(\vec{v},\vec{v}) = E(\vec{v},\vec{v}) - \Big[ E(P,P)E(\vec{v},\vec{v}) - E(P,\vec{v})^2 \Big] \\
&= E(\vec{v},\vec{v}) - \frac{\big[W(P,\vec{v})\big]^2}{a^2 b^2}
\end{align*}
而向量 $\vec{v}$ 的欧氏长度恰好就是欧氏内积算子开根号：$|\vec{v}| = \sqrt{I(\vec{v}, \vec{v})}$。相乘即得目标公式！

\subsection{第四步：面积的绝对坍缩（原点与内接三角形）}

\begin{conclusion}{原点三角形 $OAB$ 面积公式}{}
已知椭圆上有两点 $A, B$。则由原点 $O$ 与 $A,B$ 构成的三角形面积，完全由这两点的极化算子唯一决定：
$$ S_{\triangle OAB} = \frac{1}{2} ab \sqrt{-T(A,B) \big(T(A,B)+2\big)} $$
\end{conclusion}

\textbf{【面积坍缩推导】：}
三角形面积等于外积绝对值的一半：$S_{\triangle OAB} = \frac{1}{2} |W(A,B)|$。
既然 $A, B$ 都在椭圆上，那么把它们代入极化算子必定等于 0：$T(A,A)=0 \implies E(A,A) = 1$。同理 $E(B,B) = 1$。
代入刚证的拉格朗日恒等式：
$$ \big[W(A,B)\big]^2 = a^2 b^2 \big(1 \times 1 - E(A,B)^2\big) = a^2 b^2 \big(1 - E(A,B)^2\big) $$
我们把内积算子 $E$ 换回更基础的极化算子 $T$，即代入 $E(A,B) = T(A,B) + 1$：
$$ 1 - \big(T(A,B)+1\big)^2 = 1 - \big(T(A,B)^2 + 2T(A,B) + 1\big) = -T(A,B)\big(T(A,B)+2\big) $$
开根号并乘以 $1/2$，\textbf{公式瞬间成型！面积被极化算子单向锁死，没有任何求坐标的死算！}

\vspace{0.5cm}

接下来，迎接算子理论在度量几何上最璀璨的明珠：\textbf{椭圆内接三角形面积公式}。
它将展现出与古希腊“海伦公式”完全同构的绝美对称！

\begin{conclusion}{内接三角形的“算子海伦公式”}{}
已知椭圆内接三角形 $\triangle ABC$。
我们只需算出这三条边的\textbf{极化常数}（因为内接弦的夹角余弦小于 1，所以相互极化均为负值，加负号转为正）：
$$ t_1 = -T(B,C), \quad t_2 = -T(A,C), \quad t_3 = -T(A,B) $$
则该三角形的面积，完完全全由这三个算子常数决定，呈现出绝对对称的代数多项式：
$$ S_{\triangle ABC} = \frac{1}{2} a b \sqrt{ 2t_1 t_2 + 2t_2 t_3 + 2t_3 t_1 - t_1^2 - t_2^2 - t_3^2 } $$
\textit{（神级补充：由于 $A,B,C$ 同在一个椭圆上，必定满足三角恒等式约束，上述根号内的长多项式在代数上\textbf{严格恒等于 $2t_1 t_2 t_3$}！因此，面积迎来了不可思议的终极坍缩：$S_{\triangle ABC} = \frac{\sqrt{2}}{2} ab \sqrt{t_1 t_2 t_3}$）。}
\end{conclusion}

\textbf{【体验算子分配律的极致暴力美学推导】：}
内接三角形的面积为两条边向量外积的一半：$S_{\triangle ABC} = \frac{1}{2} \big| W(B-A, C-A) \big|$。
设向量 $\vec{u} = B-A, \vec{v} = C-A$。我们再次调用拉格朗日恒等式：
$$ \big[W(\vec{u}, \vec{v})\big]^2 = a^2 b^2 \Big[ E(\vec{u}, \vec{u}) E(\vec{v}, \vec{v}) - \big[E(\vec{u}, \vec{v})\big]^2 \Big] $$
我们利用算子 $E$ 像初中多项式乘法一样的分配律，将这三个量完全拆开。记住点在椭圆上 $E(A,A)=E(B,B)=E(C,C)=1$：
1. $E(\vec{u}, \vec{u}) = E(B-A, B-A) = E(B,B) - 2E(A,B) + E(A,A) = 2 - 2E(A,B)$。
由于 $E(A,B) = T(A,B)+1$，代入后瞬间坍缩：$E(\vec{u}, \vec{u}) = -2T(A,B) = 2t_3$。
2. 完全同理，右边：$E(\vec{v}, \vec{v}) = E(C-A, C-A) = -2T(A,C) = 2t_2$。
3. 交叉项展开：$E(\vec{u}, \vec{v}) = E(B-A, C-A) = E(B,C) - E(A,B) - E(A,C) + E(A,A)$。
全部转化为 $T$ 算子：$(T(B,C)+1) - (T(A,B)+1) - (T(A,C)+1) + 1 = T(B,C) - T(A,B) - T(A,C)$。
将其用我们定义的极化常数 $t$ 替换：$E(\vec{u}, \vec{v}) = -t_1 + t_3 + t_2$。

将这三个极其简洁的算子结果，直接塞回拉格朗日恒等式中：
\begin{align*}
\frac{W^2}{a^2 b^2} &= (2t_3)(2t_2) - (t_3 + t_2 - t_1)^2 \\
&= 4t_2 t_3 - (t_3^2 + t_2^2 + t_1^2 + 2t_2 t_3 - 2t_1 t_3 - 2t_1 t_2) \\
&= 2t_1 t_2 + 2t_2 t_3 + 2t_3 t_1 - t_1^2 - t_2^2 - t_3^2
\end{align*}
开根号并乘上 $\frac{1}{2}$，\textbf{完美的算子海伦公式跃然纸上！}

弦长、面积、斜率——这三个曾经逼迫我们退化回暴力死算的“度量域毒瘤”，在今天被极化算子 $(E, T, W, I)$ 彻底净化。我们没有写任何一个 $\sqrt{1+k^2}$，也没有去解任何一个判别式 $\Delta$。至此，从射影空间的点线相交，到度量空间的长度面积，算子大一统体系完成了对高中解析几何无死角的终极统治！

\section{圆锥曲线经典性质的证明}

\begin{definition}{极点与极线}{}
在极化算子体系中，对于二次曲线内（或外）的任意定点 $P$，定义满足方程 $T(P, X) = 0$ 的所有动点 $X$ 构成的直线，为点 $P$ 对应的\textbf{极线（Polar Line）}。
\end{definition}

\begin{conclusion}{完全四边形极点极线定理}{}
已知二次曲线 $\mathcal{C}$ 内（或外）有一点 $P$。过 $P$ 作两条弦 $AB$ 与 $CD$。连接 $AD, BC$ 交于点 $Q$；连接 $AC, BD$ 交于点 $R$。
则交点 $Q$ 和 $R$ 必然都在点 $P$ 的极线上，即必然成立：
$$ T(P, Q) = 0 \quad \text{且} \quad T(P, R) = 0 $$
\end{conclusion}
设 $P = \alpha_1 A + \beta_1 B = \alpha_2 C + \beta_2 D$，其中$\alpha_1+\beta_1 = \alpha_2+\beta_2 = 1$，我们要寻找直线 $AD$ 与 $BC$ 的交点 $Q$，只需将含 $A, D$ 的项移至等号同侧，含 $B, C$ 的项移至另一侧：
$$ \alpha_1 A - \beta_2 D = \alpha_2 C - \beta_1 B $$
等式两侧的系数和显然相等，设为 $k$。将两边同除以 $k$ （假设直线不平行即 $k \neq 0$），使系数和为1：
$$ Q = \frac{\alpha_1 A - \beta_2 D}{k} = \frac{\alpha_2 C - \beta_1 B}{k} $$
\textit{（左侧系数和为 1，表明点在 $AD$ 直线上；右侧表明在 $BC$ 直线上，故该点坐标必定是交点 $Q$）}现在，对交点 $Q$ 施加极化算子 $T(P, \cdot)$。根据算子关于仿射组合的分配律展开：
\begin{align*}
T(P,Q) &= T\left(P, \frac{\alpha_1 A - \beta_2 D}{k}\right) = \frac{1}{k} \big[ \alpha_1 T(P,A) - \beta_2 T(P,D) \big]\\
&= \frac{1}{k} \big[\frac12T(P,P)-\frac12 T(P,P) \big]=0,~~~\text{定比律}
\end{align*}
交点 $Q$ 必定满足 $T(P, Q)=0$，即位于点 $P$ 的极线上！\textbf{同理对称证明点 $R$}:若要寻找 $AC$ 与 $BD$ 的交点 $R$，只需将原式重新按 $A,C$ 和 $B,D$ 移项：
$$ \alpha_1 A - \alpha_2 C = \beta_2 D - \beta_1 B $$
同除以系数和 $m = \alpha_1 - \alpha_2 = \beta_2 - \beta_1$ 得到交点 $R$ 的精确仿射组合，再次施加极化算子：
\begin{align*}
T(P,R) &= T\left(P, \frac{\alpha_1 A - \alpha_2 C}{m}\right) = \frac{1}{m} \big[ \alpha_1 T(P,A) - \alpha_2 T(P,C) \big] \\
&= \frac{1}{m} \left[ \frac{1}{2} T(P,P) - \frac{1}{2} T(P,P) \right] = 0
\end{align*}
交点 $R$ 也同样完美满足 $T(P, R) = 0$ 位于极线上，定理彻底得证。

\begin{conclusion}{坎迪定理（Candy's Theorem）与 蝴蝶定理}{}
    已知二次曲线 $\mathcal{C}$ 上有弦 $PQ$，直线 $PQ$ 上有一动点 $M$。过 $M$ 作两弦 $AB$ 和 $CD$。连结 $AD, BC$ 分别交直线 $PQ$ 于点 $X, Y$。则关于有向线段必有比例式：$$ \frac{1}{MX} + \frac{1}{MY} = \frac{1}{MP} + \frac{1}{MQ} $$\textbf{推论（蝴蝶定理）：}当 $M$ 为弦 $PQ$ 中点时，$M$ 必然也是 $XY$ 的中点。
\end{conclusion}
以 $M$ 为基点，设直线 $PQ$ 的方向向量为 $\vec{v}$，弦 $AB, CD$ 的方向向量分别为 $\vec{u}, \vec{w}$。
由二次曲线上的\textbf{直线参数方程}，若$X = P + t\vec{v}$，交点 $P, Q$ 对应的参数 $t_P, t_Q$ 是关于 $t$ 的方程 $T(X,X)=0$ 的两根，由韦达定理取倒数和：$$ \frac{1}{t_P} + \frac{1}{t_Q} = \frac{t_P+t_Q}{t_Pt_Q} = \frac{-2E(M, \vec{v})}{T(M,M)} $$同理，设 $A, B$ 参数为 $t_A, t_B$，$C, D$ 参数为 $t_C, t_D$，必然有：$$ \frac{1}{t_A} + \frac{1}{t_B} = \frac{-2E(M, \vec{u})}{T(M,M)}, \quad \frac{1}{t_C} + \frac{1}{t_D} = \frac{-2E(M, \vec{w})}{T(M,M)} $$

点 $X$ 在 $PQ$ 上，设其参数为 $x$ 即 $X = M + x\vec{v}$。同时 $A = M+t_A\vec{u}$，$D = M+t_D\vec{w}$。由于 $X, A, D$ 三点共线，由向量叉乘（算子 $W$）代表的有向面积归零性质，即 $W(\overrightarrow{AX}, \overrightarrow{AD}) = 0$。利用 $W$ 的双线性与 $W(\vec{u},\vec{u})=0$ 展开：$$ W(x\vec{v} - t_A\vec{u}, \ t_D\vec{w} - t_A\vec{u}) = x t_D W(\vec{v}, \vec{w}) - x t_A W(\vec{v}, \vec{u}) - t_A t_D W(\vec{u}, \vec{w}) = 0 $$两边同除以 $x \cdot t_A \cdot t_D \cdot W(\vec{u}, \vec{w})$，并利用反对称性 $W(\vec{u}, \vec{v}) = -W(\vec{v}, \vec{u})$ 移项分离出 $1/x$：$$ \frac{1}{x} = \frac{W(\vec{v}, \vec{w})}{W(\vec{u}, \vec{w})} \frac{1}{t_A} + \frac{W(\vec{u}, \vec{v})}{W(\vec{u}, \vec{w})} \frac{1}{t_D} $$完全对称地，交点 $Y = BC \cap PQ$ 对应参数 $y$，由 $Y(y\vec{v}), B(t_B\vec{u}), C(t_C\vec{w})$ 共线可得：$$ \frac{1}{y} = \frac{W(\vec{v}, \vec{w})}{W(\vec{u}, \vec{w})} \frac{1}{t_B} + \frac{W(\vec{u}, \vec{v})}{W(\vec{u}, \vec{w})} \frac{1}{t_C} $$

将两式相加，并代入第1步的极化倒数和公式。此时，借助内积算子 $E(M, \cdot)$ 完美的线性分配律，我们将外积标量直接“吸入”算子内部，触发一场极度舒适的二维平面基底坍缩：
\begin{align*}
\frac{1}{x} + \frac{1}{y} &= \frac{W(\vec{v}, \vec{w})}{W(\vec{u}, \vec{w})}\left(\frac{1}{t_A} + \frac{1}{t_B}\right) + \frac{W(\vec{u}, \vec{v})}{W(\vec{u}, \vec{w})}\left(\frac{1}{t_C} + \frac{1}{t_D}\right) \\
&= \frac{W(\vec{v}, \vec{w})}{W(\vec{u}, \vec{w})}\left[ \frac{-2E(M, \vec{u})}{T(M,M)} \right] + \frac{W(\vec{u}, \vec{v})}{W(\vec{u}, \vec{w})}\left[ \frac{-2E(M, \vec{w})}{T(M,M)} \right] \\
&= \frac{-2}{T(M,M) \cdot W(\vec{u}, \vec{w})} \Big[ W(\vec{v}, \vec{w})E(M, \vec{u}) + W(\vec{u}, \vec{v})E(M, \vec{w}) \Big] \\
&= \frac{-2}{T(M,M) \cdot W(\vec{u}, \vec{w})} E\Big(M, \; W(\vec{v}, \vec{w})\vec{u} + W(\vec{u}, \vec{v})\vec{w} \Big) \quad \text{(逆向调用算子 } E \text{ 的线性分配律)} \\
&= \frac{-2}{T(M,M) \cdot W(\vec{u}, \vec{w})} E\Big(M, \; W(\vec{u}, \vec{w})\vec{v} \Big) \quad \text{(神来之笔：触发二维基底折叠恒等式)} \\
&= \frac{-2 W(\vec{u}, \vec{w})}{T(M,M) \cdot W(\vec{u}, \vec{w})} E(M, \vec{v}) \quad \text{(算子标量齐次性，提取常数)} \\
&= \frac{-2E(M, \vec{v})}{T(M,M)} \quad \text{(外积面积因子瞬间约去)} \\
&= \frac{1}{t_P} + \frac{1}{t_Q} \quad \text{(完美闭环：回归弦 } PQ \text{ 的极化倒数和)}
\end{align*}

\textit{【批注：推导中触发的底衬替换，基于一个极度纯粹的线性代数事实——在二维平面中，不平行的两方向向量 $\vec{u}, \vec{w}$ 必然构成一组基底。任意向量 $\vec{v}$ 均可唯一分解为 $\vec{v} = \alpha\vec{u} + \beta\vec{w}$。等式两边分别与 $\vec{w}$ 和 $\vec{u}$ 作外积 $W$，利用自身外积归零的反对称性，立得面积投影系数 $\alpha = \frac{W(\vec{v}, \vec{w})}{W(\vec{u}, \vec{w})}, \beta = \frac{W(\vec{u}, \vec{v})}{W(\vec{u}, \vec{w})}$。算子 $E$ 惊人的线性空间包容度，让这场基于面积比例的克莱姆法则（Cramer's Rule）分解，在代数推演的胃里被毫无痕迹地完美消化！】}

因为参数均对应于同一基准向量 $\vec{v}$ 所在的同一直线，参数倒数和等价于有向线段倒数和，得到：
$$ \frac{1}{MX} + \frac{1}{MY} = \frac{1}{MP} + \frac{1}{MQ} $$

    \item \textbf{切线方程}：曲线上点 $A$ 处的切线方程退化为：
    $$ T(X,A) = 0 $$
    \item \textbf{极线方程（切点弦）}：曲线外一点 $P$ 对应的极线方程为：
    $$ T(X,P) = 0 $$
\end{enumerate}
\end{conclusion}

\textbf{证明：}
1. \textbf{割线}：由前文定比分点的仿射性质，若动点 $X$ 在直线 $AB$ 上，存在实数 $\alpha, \beta$ 满足 $\alpha+\beta=1$ 使 $X = \alpha A + \beta B$。对极化算子运用线性分配律：
$$ T(X,A) + T(X,B) = \alpha T(A,A) + \beta T(B,A) + \alpha T(A,B) + \beta T(B,B) $$
因 $A,B$ 在曲线上，$T(A,A)=T(B,B)=0$，且 $T$ 满足交换律，故上式等于 $(\alpha+\beta)T(A,B) = T(A,B)$。这说明割线上的任意点必满足该一次算子方程，两点定线，故割线得证。
2. \textbf{切线}：利用极限微元思想。令割线上的动点 $B \to A$，上述割线方程直接取极限：
$$ T(X,A) + T(X,A) - T(A,A) = 2T(X,A) - 0 = 0 \implies T(X,A) = 0 $$
3. \textbf{极线}：从极点 $P$ 引两条切线分别切于 $A,B$。由切线方程，切线 $PA$ 满足 $T(X,A)=0$，且 $P$ 在其上，故代入有 $T(P,A)=0$。利用算子交换律，等价反转为 $T(A,P)=0$。同理，由切线 $PB$ 得 $T(B,P)=0$。这说明切点 $A, B$ 的坐标同时满足以 $X$ 为变量的一次线性算子方程 $T(X,P)=0$。故切点弦方程即为 $T(X,P)=0$。
\textit{【批注：传统几何中极其复杂的极点极线对称性理论，在这里仅仅是算子交换律 $T(P,A)=T(A,P)$ 这三个字符对调的一次等价复述！不可谓不震撼。】}

\subsection{核心同构：经典二次曲线系的“乐高拼装”}

基于上文的直线泛函，我们可以像拼装积木一样，将经典的二次曲线系写成极简的算子组合。

\begin{conclusion}{四大经典交点曲线系}{}
二次曲线 $\mathcal{C}: T(X,X)=0$ 的所有相交簇均可通过算子的恒等组合一键生成：
\begin{enumerate}
    \item \textbf{四交点系}：过 $\mathcal{C}$ 上 $A, B, C, D$ 四点的二次曲线系：
    $$ T(X,X) + \lambda \cdot L_{AB}(X) \cdot L_{CD}(X) = 0 $$
    \item \textbf{三交一切系}：在 $A$ 点相切，且过 $C, D$ 的曲线系（即令四交点系中 $B \to A$）：
    $$ T(X,X) + \lambda \cdot T(X,A) \cdot L_{CD}(X) = 0 $$
    \item \textbf{双切线系}：与 $\mathcal{C}$ 在 $A, B$ 两点发生双重相切的曲线系（设切点弦极点为 $P$）：
    $$ T(X,X) + \lambda \big[T(X,P)\big]^2 = 0 $$
    \item \textbf{超密切系}：在 $A$ 点发生四阶接触的曲线系（令双切系中 $P \to A$）：
    $$ T(X,X) + \lambda \big[T(X,A)\big]^2 = 0 $$
\end{enumerate}
\end{conclusion}

\textbf{高阶相切的代数秒杀证明：}
对于四交点系，直接将 $A,B,C,D$ 代入，由于 $T(A,A)=0$ 且 $L_{AB}(A)=0$，方程 $0+\lambda \cdot 0 = 0$ 显然恒成立。
此处重点展示算子理论如何秒杀几何上极难证明的\textbf{双切线系}的高阶相切性：
设过切点 $A$ 的直线上有动点 $X = A + t\vec{v}$，我们研究该射线与新曲线系 $F(X) = T(X,X) + \lambda [T(X,P)]^2 = 0$ 的相交参数 $t$。
利用已证的《直线参数方程》，展开原曲线部分：
$$ T(A+t\vec{v}, A+t\vec{v}) = T(A,A) + 2t E(A,\vec{v}) + t^2 E(\vec{v},\vec{v}) $$
对于极线算子部分，利用平移分配律：
$$ T(P, A+t\vec{v}) = E(P, A+t\vec{v}) - 1 = \big(E(P,A) - 1\big) + tE(P,\vec{v}) = T(P,A) + tE(P,\vec{v}) $$
关键截断发生！因为 $P$ 为切点弦的极点，点 $A$ 在极线上必有 $T(P,A)=0$。极线部分平方后直接坍缩为纯二次项：$\lambda \big[0 + t E(P,\vec{v})\big]^2 = \lambda t^2 [E(P,\vec{v})]^2$。
合并参数多项式：
$$ 2t E(A,\vec{v}) + t^2 \left( E(\vec{v},\vec{v}) + \lambda [E(P,\vec{v})]^2 \right) = 0 $$
若该直线为新曲线系在 $A$ 处的切线，该参数方程必须有重根 $t=0$，即一次项系数必须严格为 $0$：
$$ 2E(A,\vec{v}) = 0 \implies E(A,\vec{v}) = 0 \implies T(X, A) = 0 $$
这正是原曲线在 $A$ 处的切线！新曲线系完美继承了原二次曲线的切向。
\textit{【批注：未用任何微积分求导，而是利用极线算子固有的截断条件 $T(P,A)=0$ 瞬间扫清交叉项。这就是算子代数的绝对统治力。】}

\subsection{绝对降维：退化曲线系与蒙日圆的代数坍缩}

\begin{conclusion}{外切线对联合算子方程（退化曲线系）}{}
从二次曲线外一点 $P$ 作原曲线的两条切线，这两条相交切线本身构成一个退化的二次曲线。其算子方程为极其纯粹对称的：
$$ \big[T(P,X)\big]^2 = T(P,P) \cdot T(X,X) $$
\textit{（注：此即传统解析几何中计算量极大的齐次大招 $S_1^2 = S_{11} S_0$ 或 $T_1^2 = S_1 S$）}
\end{conclusion}

\textbf{纯算子流无坐标推导：}
设过极点 $P$ 的射线上的动点为 $Y = P + t\vec{v}$。为了代数归一，直接定义方向向量为 $\vec{v} = X - P$（其中 $X$ 为直线上任意固定点）。由《直线参数方程》：
$$ T(Y,Y) = E(\vec{v},\vec{v}) t^2 + 2E(P,\vec{v}) t + T(P,P) = 0 $$
若该射线恰好就是切线，则其与原曲线只有唯一交点，参数方程必定满足判别式 $\Delta = 0$：
$$ 4\big[E(P,\vec{v})\big]^2 - 4E(\vec{v},\vec{v})T(P,P) = 0 \implies \big[E(P, \vec{v})\big]^2 = E(\vec{v},\vec{v}) \cdot T(P,P) $$
利用内积算子 $E$ 的双线性分配法则，以及恒等代换 $E(M,N) = T(M,N)+1$，将 $\vec{v} = X - P$ 展开化回纯 $T$ 算子：
\begin{align*}
E(P, \vec{v}) &= E(P, X-P) = E(P,X) - E(P,P) \\
&= \big(T(P,X)+1\big) - \big(T(P,P)+1\big) = T(P,X) - T(P,P) \\
E(\vec{v}, \vec{v}) &= E(X-P, X-P) = E(X,X) - 2E(P,X) + E(P,P) \\
&= \big(T(X,X)+1\big) - 2\big(T(P,X)+1\big) + \big(T(P,P)+1\big) = T(X,X) - 2T(P,X) + T(P,P)
\end{align*}
将拆分后的两项代回 $\Delta=0$ 的方程：
$$ \big[T(P,X) - T(P,P)\big]^2 = \big[T(X,X) - 2T(P,X) + T(P,P)\big] \cdot T(P,P) $$
直接展开等式的左侧与右侧：
左侧 $= [T(P,X)]^2 - 2T(P,X)T(P,P) + [T(P,P)]^2$
右侧 $= T(X,X)T(P,P) - 2T(P,X)T(P,P) + [T(P,P)]^2$
令人震颤的代数消隐出现了——等式两端极其繁杂的后两项竟然\textbf{完全等价并同时湮灭抵消}！迷雾散去，最终只留下几何上最本源的恒等式：
$$ \big[T(P,X)\big]^2 = T(P,P) \cdot T(X,X) $$

\begin{conclusion}{蒙日圆（准圆）轨迹的系数对冲}{}
互相垂直的外切线交点 $P$ 的轨迹（蒙日圆）方程必然为 $x_P^2 + y_P^2 = a^2 + b^2$。
\end{conclusion}
\textbf{证明：}
互相垂直的切线，意味着在外切线对的方程 $\big[T(P,X)\big]^2 - T(P,P)T(X,X) = 0$ 中，关于变量 $X(x,y)$ 的二次齐次项 $x^2$ 与 $y^2$ 的总系数之和必定为零。
观察等式左侧 $\big[T(P,X)\big]^2$ 展开式，提取二次项 $\big(\frac{x_P x}{a^2} + \frac{y_P y}{b^2}\big)^2$，其 $x^2$ 系数为 $\frac{x_P^2}{a^4}$，$y^2$ 系数为 $\frac{y_P^2}{b^4}$。
观察等式右侧 $T(P,P)T(X,X)$ 展开式，其二次部分来源于 $T(P,P)\big(\frac{x^2}{a^2} + \frac{y^2}{b^2}\big)$，故 $x^2$ 系数为 $\frac{T(P,P)}{a^2}$，$y^2$ 系数为 $\frac{T(P,P)}{b^2}$。
两部分相减，系数之和归零：
$$ \left( \frac{x_P^2}{a^4} - \frac{T(P,P)}{a^2} \right) + \left( \frac{y_P^2}{b^4} - \frac{T(P,P)}{b^2} \right) = 0 \implies \frac{x_P^2}{a^4} + \frac{y_P^2}{b^4} = T(P,P)\left(\frac{1}{a^2} + \frac{1}{b^2}\right) $$
将常数 $T(P,P) = \frac{x_P^2}{a^2} + \frac{y_P^2}{b^2} - 1$ 完整代入右侧并展开：
$$ \frac{x_P^2}{a^4} + \frac{y_P^2}{b^4} = \frac{x_P^2}{a^4} + \frac{x_P^2}{a^2 b^2} + \frac{y_P^2}{a^2 b^2} + \frac{y_P^2}{b^4} - \left(\frac{1}{a^2} + \frac{1}{b^2}\right) $$
等式两端最高次幂分数项 $\frac{x_P^2}{a^4}$ 和 $\frac{y_P^2}{b^4}$ 再次精妙地完全抵消！同乘 $a^2b^2$ 得到：
$$ 0 = x_P^2 + y_P^2 - (b^2 + a^2) \implies x_P^2 + y_P^2 = a^2 + b^2 $$
\textit{【全局总结：在这套“比值代换法”与极化算子体系下，整个“二次曲线系”理论被彻底兼并！我们再也不需要去硬凑 $k_1 k_2 = -1$ 或者死算极其庞杂的多项式配凑。所有的点线面几何条件，都被纯粹地内化在了算子的分配律、交换律与线性张成空间之中！这宣告了算子理论在处理解析几何时无懈可击的降维统治力。】}

\section{第三部分：算子化体系的应用与实战}

本章将挑选三道典型的高考压轴题，严格使用“四段式”（题干翻译$\Rightarrow$算子作用$\Rightarrow$代数计算$\Rightarrow$结论输出）展现算子的统治力。

\subsection{实战一：定点弦斜率调整问题（昊天原创题）}
\textbf{题目}：过点 $M(2,1)$ 的直线交椭圆 $\mathcal{C}: \frac{x^2}{2} + y^2 = 1$ 于 $A, B$ 两点。点 $P(0,3)$，直线 $PA$ 交直线 $l: x+y=1$ 于点 $Q$。记直线 $QA$ 和 $QB$ 的斜率分别为 $k_1, k_2$。求证：$\frac{1}{k_1+1} - \frac{1}{k_2+1}$ 为定值。

\textbf{步骤1：题干翻译} \\
将椭圆方程转化为极化算子：$T(U,V) = \frac{x_U x_V}{2} + y_U y_V - 1$。
代入定点 $M(2,1)$ 求极化函数：$T(M, X) = \frac{2 x_X}{2} + 1 \cdot y_X - 1 = x_X + y_X - 1$。
\textit{题干几何条件神级翻译}：由于直线 $l$ 的方程为 $x+y-1=0$，这恰好等价于方程 $T(M, X) = 0$。因为动点 $Q$ 在直线 $l$ 上，故条件被翻译为绝对代数约束：$T(M,Q) = 0$。

\textbf{步骤2：算子作用} \\
目标是转化非对称的斜率表达式 $k_1+1 = k_{QA}+1$。
根据斜率定义 $k_{QA} = \frac{y_A - y_Q}{x_A - x_Q}$。代入表达式中并通分：
$$ k_{QA} + 1 = \frac{y_A - y_Q + x_A - x_Q}{x_A - x_Q} = \frac{(x_A + y_A - 1) - (x_Q + y_Q - 1)}{x_A - x_Q} $$
利用步骤1的翻译结论：
$$ k_{QA} + 1 = \frac{T(M,A) - T(M,Q)}{x_A - x_Q} $$
由于 $T(M,Q) = 0$，上式坍缩为 $\frac{T(M,A)}{x_A - x_Q}$。
同理，$k_{QB} + 1 = \frac{T(M,B)}{x_B - x_Q}$。
所求目标变为算子形式：
$$ \text{目标} = \frac{x_A - x_Q}{T(M,A)} - \frac{x_B - x_Q}{T(M,B)} $$

\textbf{步骤3：代数计算} \\
因为 $P, Q, A$ 三点共线，设 $Q = \alpha P + \beta A$（$\alpha+\beta=1$）。
将 $Q$ 代入 $T(M,Q) = 0$ 中，应用分配律（第一部分性质2）：
$$ T(M, \alpha P + \beta A) = \alpha T(M,P) + \beta T(M,A) = 0 $$
代入 $P(0,3)$ 到算子得 $T(M,P) = 0 + 3 - 1 = 2$。
得到方程：$2\alpha + \beta T(M,A) = 0$。
解得 $\alpha = \frac{-T(M,A)}{2 - T(M,A)}$，$\beta = \frac{2}{2 - T(M,A)}$。
计算横坐标分子 $x_A - x_Q = x_A - (\alpha x_P + \beta x_A)$。因为 $x_P = 0$，所以 $x_A - x_Q = x_A(1-\beta) = \alpha x_A$。
代入目标式第一项：
$$ \frac{\alpha x_A}{T(M,A)} = \frac{ \frac{-T(M,A)}{2 - T(M,A)} x_A }{T(M,A)} = \frac{-x_A}{2 - T(M,A)} $$
代入第二项所需 $x_Q = \beta x_A = \frac{2x_A}{2-T(M,A)}$。目标式通分：
$$ \text{目标} = \frac{-x_A}{2 - T(M,A)} - \frac{x_B - \frac{2x_A}{2 - T(M,A)}}{T(M,B)} = \frac{-x_A T(M,B) - 2x_B + x_B T(M,A) + 2x_A}{T(M,B)(2 - T(M,A))} $$
分析分子核心项：$x_B T(M,A) - x_A T(M,B) = x_B(x_A+y_A-1) - x_A(x_B+y_B-1) = x_B y_A - x_A y_B + x_A - x_B$。
由于 $A, M, B$ 共线，应用外积算子 \textbf{【性质1绝对清零】}：
$W(\overrightarrow{MA}, \overrightarrow{MB}) = 0 \Rightarrow (x_A-2)(y_B-1) - (y_A-1)(x_B-2) = 0$
展开移项得到：$x_B y_A - x_A y_B = x_B - x_A + 2y_A - 2y_B$。
将此外积结果代入分子，分子变为：
$$ (x_B - x_A + 2y_A - 2y_B) + x_A - x_B - 2x_B + 2x_A = 2x_A + 2y_A - 2x_B - 2y_B = 2(T(M,A) - T(M,B)) $$
利用 \textbf{【定理1的第3点：调和极化律】} $T(M,M)T(A,B) = 2T(M,A)T(M,B)$。由于 $T(M,M) = \frac{4}{2}+1-1 = 2$，代入变形：
$\frac{1}{T(M,A)} + \frac{1}{T(M,B)} = \frac{2}{T(M,M)} = 1 \Rightarrow T(M,A) + T(M,B) = T(M,A)T(M,B)$。
代入分母 $T(M,B)(2 - T(M,A)) = 2T(M,B) - T(M,A)T(M,B) = 2T(M,B) - (T(M,A) + T(M,B)) = -(T(M,A) - T(M,B))$。

\textbf{步骤4：结论输出} \\
目标式 $= \frac{2(T(M,A) - T(M,B))}{-(T(M,A) - T(M,B))} = -2$。原命题得证。

\subsection{实战二：极化参数解决阿波罗尼斯圆问题}
\textbf{题目}：已知抛物线 $\mathcal{C}: y^2 = 2x$，点 $P(1,2)$。若过点 $P$ 的动弦 $l$ 交抛物线于 $A, B$。问是否存在异于 $P$ 的点 $Q$ 使得 $\frac{PA}{PB} = \frac{QA}{QB}$ 恒成立？

\textbf{步骤1：题干翻译} \\
抛物线的极化算子定义为 $T(U,V) = y_U y_V - (x_U + x_V)$。自身极化 $T(P,P) = 2^2 - 2(1) = 2$。
题干要求：$\frac{PA^2}{PB^2} = \frac{QA^2}{QB^2}$ 恒成立。设未知向量 $\vec{u} = Q - P$，需通过算子找到 $\vec{u}$ 的值。

\textbf{步骤2：算子作用} \\
设直线 $l$ 方向向量为 $\vec{v} = (v_x, v_y)$，参数方程 $X = P + r\vec{v}$。
应用 \textbf{【定理2：坐标平移下的算子行为】}，将 $X$ 代入 $T(X,X) = 0$：
$$ T(\vec{v},\vec{v}) r^2 + 2T(P, \vec{v}) r + T(P,P) = 0 $$
由一元二次方程根与系数关系得到参数倒数和：
$$ \frac{1}{r_A} + \frac{1}{r_B} = -\frac{2T(P, \vec{v})}{T(P,P)} $$

\textbf{步骤3：代数计算} \\
利用欧氏内积展开距离平方：$QA^2 = |Q-A|^2 = |\vec{u} - r_A\vec{v}|^2 = |\vec{u}|^2 - 2r_A I(\vec{u}, \vec{v}) + r_A^2 |\vec{v}|^2$。
代入题干条件并除以各自的分母 $r_A^2$ 与 $r_B^2$：
$$ \frac{|\vec{u}|^2}{r_A^2} - \frac{2I(\vec{u}, \vec{v})}{r_A} + |\vec{v}|^2 = \frac{|\vec{u}|^2}{r_B^2} - \frac{2I(\vec{u}, \vec{v})}{r_B} + |\vec{v}|^2 $$
两边同时减去 $|\vec{v}|^2$ 并移项：
$$ |\vec{u}|^2 \left(\frac{1}{r_A^2} - \frac{1}{r_B^2}\right) - 2I(\vec{u}, \vec{v}) \left(\frac{1}{r_A} - \frac{1}{r_B}\right) = 0 $$
由于 $A, B$ 是相异两点，$r_A \neq r_B$，两边同时除以 $(\frac{1}{r_A} - \frac{1}{r_B})$：
$$ |\vec{u}|^2 \left(\frac{1}{r_A} + \frac{1}{r_B}\right) - 2I(\vec{u}, \vec{v}) = 0 $$
代入步骤2得到的倒数和：
$$ |\vec{u}|^2 \left(-\frac{2T(P, \vec{v})}{T(P,P)}\right) - 2I(\vec{u}, \vec{v}) = 0 $$
两边同除以 $-2$。根据算子定义 $T(P, \vec{v}) = 2v_y - v_x = I( (-1, 2), \vec{v} )$。令特征向量 $\nabla P = (-1, 2)$。
方程变为纯内积的恒等式：
$$ I \left( \frac{|\vec{u}|^2}{T(P,P)} \nabla P + \vec{u}, \vec{v} \right) = 0 $$
由于对任意弦方向 $\vec{v}$ 恒成立，括号内向量必为零向量：
$$ \vec{u} = -\frac{|\vec{u}|^2}{2} (-1, 2) $$

\textbf{步骤4：结论输出} \\
设 $\vec{u} = m(-1, 2)$，则 $|\vec{u}|^2 = 5m^2$。
代入得 $m(-1,2) = -\frac{5m^2}{2}(-1,2)$。解得 $m = -\frac{2}{5}$（舍去 $m=0$ 因为 $Q \neq P$）。
$\vec{u} = (\frac{2}{5}, -\frac{4}{5})$。
$Q = P + \vec{u} = (1,2) + (\frac{2}{5}, -\frac{4}{5}) = (\frac{7}{5}, \frac{6}{5})$。存在。

\subsection{实战三：双切线交点轨迹问题}
\textbf{题目}：已知圆 $C_1: x^2+y^2-3x+1=0$。点 $A(2,-1)$ 在椭圆 $C_2: x^2+4y^2-8=0$ 上且在 $C_1$ 上。过 $A$ 的直线交两曲线于 $B,C$。设 $C_1$ 在 $B$ 的切线与 $C_2$ 在 $C$ 的切线交于 $D$。求 $D$ 的轨迹方程。

\textbf{步骤1：题干翻译} \\
建立双算子系统：$T_1(U,V) = x_U x_V + y_U y_V - \frac{3}{2}(x_U+x_V) + 1$；$T_2(U,V) = x_U x_V + 4y_U y_V - 8$。
几何条件“$D$ 是 $B$ 点处关于 $C_1$ 的切线上的点”翻译为：$T_1(D, B) = 0$。
同理，条件翻译为：$T_2(D, C) = 0$。

\textbf{步骤2：算子作用} \\
应用 \textbf{【定理2：坐标平移下的算子行为】}。设直线参数方程 $X = A + t\vec{v}$，$\vec{v}=(m,1)$。
代入圆 $T_1(A+t\vec{v}, A+t\vec{v}) = 0$，因 $A$ 在圆上 $T_1(A,A)=0$，解得 $B$ 的参数 $t_B = -\frac{2T_1(A,\vec{v})}{T_1(\vec{v},\vec{v})}$。
代入切线方程 $T_1(D, A + t_B\vec{v}) = 0$，利用第一部分的【分配律】：
$$ T_1(D, A) + t_B T_1(D, \vec{v}) = 0 \Rightarrow T_1(D, A) \cdot T_1(\vec{v}, \vec{v}) - 2T_1(A, \vec{v}) \cdot T_1(D, \vec{v}) = 0 $$
同理作用于 $C_2$ 得到 $C$ 的参数方程：
$$ T_2(D, A) \cdot T_2(\vec{v}, \vec{v}) - 2T_2(A, \vec{v}) \cdot T_2(D, \vec{v}) = 0 $$

\textbf{步骤3：代数计算} \\
设 $D(x,y)$。计算各项算子值：
$T_1(\vec{v},\vec{v}) = m^2+1$；$T_1(A,\vec{v}) = 2m-1-\frac{3}{2}(m) = \frac{1}{2}m-1$。
$T_1(D,A) = 2x-y-\frac{3}{2}(x+2)+1 = \frac{1}{2}x-y-2$；$T_1(D,\vec{v}) = mx+y-\frac{3}{2}m$。
代入化简得圆切线展开：$m^2(x+2y+1) - 2m(2x-y-3) - (x+2y-4) = 0$。
同理，椭圆切线展开：$m^2(x+2y+4) - 4m(x-2y) - 4(x+2y-4) = 0$。
令 $S = x+2y$，利用加减消元法，将上式乘 $4$ 减下式，得 $3S m^2 - 12(x-2)m = 0 \Rightarrow m = \frac{4(x-2)}{S}$。
代回原式：$(\frac{4x-8}{S})^2(S+1) - 2\frac{4x-8}{S}(2x-y-3) - (S-4) = 0$。

\textbf{步骤4：结论输出} \\
两边同乘 $S^2$ 展开，化简后得到唯一有效解 $S = 4$。
所以点 $D$ 的轨迹方程为 $x + 2y - 4 = 0$。

\textbf{【题目】} 已知抛物线 $C_1: y^2 = 4cx\ (c > 0)$ 和双曲线 $C_2: \frac{x^2}{a^2} - \frac{y^2}{b^2} = 1\ (a > 0, b > 0)$ 共用一个焦点 $F(c, 0)$。设直线 $l$ 是 $C_1$ 和 $C_2$ 的公切线，$l$ 与 $C_1$ 相切于点 $A$，与 $C_2$ 相切于点 $B$。已知 $C_1$ 经过点 $P(1, 2)$。
\begin{enumerate}
    \item 求 $c$ 的值；
    \item 求公切线 $l$ 的方程（用参数 $a$ 表示）；
    \item 求 $\triangle ABF$ 的外接圆半径 $R$ 的最小值。
\end{enumerate}

\textbf{【算子降维暴力拆解】：}

\textbf{第一步：基础参数归一与算子建系} \\
将点 $P(1,2)$ 代入抛物线 $C_1: y^2 = 4cx$，直接得到 $2^2 = 4c \times 1 \implies c = 1$。（第 1 问秒杀）
因此两曲线共用焦点 $F(1, 0)$，抛物线方程为 $y^2 = 4x$。对于双曲线 $C_2$，根据焦点定义必然满足核心约束：
$$ a^2 + b^2 = c^2 = 1 \implies b^2 = 1 - a^2 \quad (0 < a < 1) $$
写出抛物线和双曲线各自的极化算子（二次型）：
\begin{itemize}
    \item $C_1$ 的极化算子：$T_1(U, V) = y_U y_V - 2(x_U + x_V)$
    \item $C_2$ 的极化算子：$T_2(U, V) = \frac{x_U x_V}{a^2} - \frac{y_U y_V}{b^2} - 1$
\end{itemize}

$$ -2x + y_A y - 2x_A = 0 \implies -\frac{1}{x_A}x + \frac{y_A}{2x_A}y - 1 = 0 $$
