\chapter{算子代换法}
\section{算子的定义与算术性质}
\begin{definition}{算子 $E$}{}
    针对椭圆方程，定义输入为任意两点 $M(x_M, y_M)$ 与 $N(x_N, y_N)$ 的二元函数（输出为实数）：
$$ E(M, N) = \frac{x_M x_N}{a^2} + \frac{y_M y_N}{b^2} $$
\end{definition}
显然，$E(M,N)=E(N,M)$满足交换律，下证$E$满足分配律。设存在任意三点 $A(x_A, y_A), B(x_B, y_B), C(x_C, y_C)$。设点 $P$ 是 $A$ 和 $B$ 的仿射组合，即存在实数 $\alpha, \beta$，使得 $x_P = \alpha x_A + \beta x_B$ 且 $y_P = \alpha y_A + \beta y_B$。
\begin{align*}
E(P, C) &= E( \alpha A + \beta B, C) \\
&=\frac{(\alpha x_A + \beta x_B)x_C}{a^2} + \frac{(\alpha y_A + \beta y_B)y_C}{b^2}  = \frac{\alpha x_A x_C}{a^2} + \frac{\beta x_B x_C}{a^2} + \frac{\alpha y_A y_C}{b^2} + \frac{\beta y_B y_C}{b^2} \\
&= \alpha \left( \frac{x_A x_C}{a^2} + \frac{y_A y_C}{b^2} \right) + \beta \left( \frac{x_B x_C}{a^2} + \frac{y_B y_C}{b^2} \right) =\alpha E(A,C) + \beta E(B,C) 
\end{align*}
又由于交换律，另外一种情况不证自明。
\begin{definition}{算子 $T$ }{}
定义包含曲线边界信息的极化算子，当 $M$ 与 $N$ 重合且位于椭圆上时，$T(M,M)=0$。
$$ T(M, N) = E(M, N) - 1 = \frac{x_M x_N}{a^2} + \frac{y_M y_N}{b^2} - 1 $$
\end{definition}
显然，$T(M,N)=T(N,M)$满足交换律，下证当$P,M,N$共线时，算子$T$满足分配律：设点 $P$ 是 $A$ 和 $B$ 的仿射组合，即存在实数 $\alpha, \beta$，使得 $x_P = \alpha x_A + \beta x_B$ 且 $y_P = \alpha y_A + \beta y_B$，且$\alpha+\beta=1$。
\begin{align*}
  T(P,C) &= E(P,C) - 1  =E( \alpha  A + \beta B,C) - 1  \\
  &=\alpha E(A,C) + \beta E(B,C) - 1 =\alpha E(A,C) + \beta E(B,C)-\alpha - \beta\\
  &= \alpha E(A,C) - \alpha + \beta E(B,C) - \beta= \alpha T(A,C) + \beta T(B,C)
\end{align*}
\begin{definition}{外积算子 $W$}{}
对于以原点为起点的两个位置向量 $\overrightarrow{OM} = (x_M, y_M)$ 与 $\overrightarrow{ON} = (x_N, y_N)$，定义：
$$ W(M, N) = x_M y_N - x_N y_M $$
\end{definition}
\begin{itemize}
\item $W(M,N) = x_M y_N - x_N y_M = -(x_N y_M - x_M y_N) = -W(N,M)$（根据提取公因式 $-1$）。
\item 由 $W(M,M) = -W(M,M)$，两边同时加上 $W(M,M)$ 得到 $2W(M,M) = 0 \Rightarrow W(M,M) = 0$。
\item 由 $W(M,M) = -W(M,M)$，两边同时加上 $W(M,M)$ 得到 $2W(M,M) = 0 \Rightarrow W(M,M) = 0$。
\item 分配律：设存在任意三点（或位置向量） $A(x_A, y_A), B(x_B, y_B), C(x_C, y_C)$。设 $P$ 是 $A$ 和 $B$ 的线性组合，即存在实数 $\alpha, \beta$，使得 $x_P = \alpha x_A + \beta x_B$ 且 $y_P = \alpha y_A + \beta y_B$。对于第一变元，代入算子定义并展开：
\begin{align*}W(P, C) &= W(\alpha A + \beta B, C)= (\alpha x_A + \beta x_B)y_C - x_C(\alpha y_A + \beta y_B) \\&= \alpha x_A y_C + \beta x_B y_C - \alpha x_C y_A - \beta x_C y_B \\&= \alpha (x_A y_C - x_C y_A) + \beta (x_B y_C - x_C y_B) \\&= \alpha W(A, C) + \beta W(B, C)\end{align*}
\item 进一步，结合已证的反对称性 $W(M,N) = -W(N,M)$，可直接推导算子 $W$ 对第二变元也同样满足分配律（无需再次代入繁琐的坐标）：
\begin{align*}W(C, P) &= W(C, \alpha A + \beta B) = -W(\alpha A + \beta B, C)= -\big(\alpha W(A, C) + \beta W(B, C)\big) \\&= \alpha \big(-W(A, C)\big) + \beta \big(-W(B, C)\big) = \alpha W(C, A) + \beta W(C, B)\end{align*}
\end{itemize}

\begin{definition}{内积算子 $I$}{}
$$ I(M, N) = x_M x_N + y_M y_N $$
\end{definition}
\begin{itemize}
    \item 对称性：$I(M,N) = I(N,M)$，即内积算子满足交换律。
    \item 对第一变元的线性性：对于任意实数 $\alpha,\beta$ 及点 $A,B,C$，有
    $$ I(\alpha A + \beta B, C) = \alpha I(A,C) + \beta I(B,C). $$
    \item 对第二变元的线性性：结合对称性与第一变元的线性性，可得
    $$ I(C, \alpha A + \beta B) = \alpha I(C,A) + \beta I(C,B). $$
    \item 自内积与模长的关系：$I(M,M) = x_M^2 + y_M^2 = |OM|^2$，即点 $M$ 到原点距离的平方。
    \item 与标准内积的联系：$I(M,N) = \overrightarrow{OM} \cdot \overrightarrow{ON}$，即欧几里得空间中的点积。
\end{itemize}

\begin{definition}{直线算子$F$}{}
    对于三点共线的$A(x_A,y_A),B(x_B,y_B),X(x,y)$，直线算子
    \[F_{AB}(X)=y(x_A-x_B)+x(y_B-y_A)-(x_Ay_B-x_By_A)\]
\end{definition}
\begin{itemize}
    \item 对称性：$F_{AB}(X)$ 关于前两个变元是反对称的，即
    $$ F_{AB}(X) = -F(B,A,X). $$
    特别地，$F(A,A,X)=0$ 恒成立。
    \item 退化性：$F_{AB}(X)=0$ 当且仅当点 $X$ 位于直线 $AB$ 上。且 $F(A,B,A)=F(A,B,B)=0$。
    \item 三线性：将点视为向量（坐标），算子 $F$ 对每个变元都是线性的。即对任意实数 $\alpha,\beta$ 及任意点（或向量）$A,A',B,B',X,X'$，有
    \begin{align*}
        F(\alpha A+\beta A',\,B,\,X) &= \alpha F_{AB}(X) + \beta F(A',B,X), \\
        F(A,\,\alpha B+\beta B',\,X) &= \alpha F_{AB}(X) + \beta F(A,B',X), \\
        F(A,\,B,\,\alpha X+\beta X') &= \alpha F_{AB}(X) + \beta F(A,B,X').
    \end{align*}
    \item 与外积算子 $W$ 的关系：
    $$ F_{AB}(X) = y(x_A-x_B) + x(y_B-y_A) - W(A,B), $$
    其中 $W(A,B)=x_Ay_B-x_By_A$。更紧凑地，利用三阶行列式可写为
    $$ F_{AB}(X) = -\begin{vmatrix}
        x & y & 1 \\
        x_A & y_A & 1 \\
        x_B & y_B & 1
    \end{vmatrix}. $$
    \item 轮换性质：由行列式的完全反对称性，有
    $$ F_{AB}(X) = F(B,X,A) = F(X,A,B) = -F(B,A,X) = -F(A,X,B) = -F(X,B,A). $$
\end{itemize}

\section{核心结论推导证明}

\begin{conclusion}{定比律，插入律，调和律}{}
已知二次曲线 $C$ 上有两点 $A, B$。直线 $AB$ 上有一动点 $P$ 满足 $\overrightarrow{AP} = \lambda \overrightarrow{PB}$。则必然成立：
\begin{enumerate}
    \item 比例生成：
    \[\lambda = \frac{T(P,A)}{T(P,B)},~~~\alpha=\frac{T(P,B)}{T(A,B)}=\frac12\frac{T(P,P)}{T(P,A)},~~~\beta=\frac{T(P,A)}{T(A,B)}=\frac12\frac{T(P,P)}{T(P,B)}\]
    \item 插入拆分：$T(A,B) = T(A,P) + T(P,B)$
    \item 调和极化：$T(P,P) \cdot T(A,B) = 2 T(P,A) \cdot T(P,B)\Longleftrightarrow T(P,P) = \frac{2 T(P,A) T(P,B)}{T(P,A) + T(P,B)}$
\end{enumerate}
\end{conclusion}
定比分点有$ x_P = \frac{x_A+\lambda x_B}{1+\lambda},y_P=\frac{y_A+\lambda y_B}{1+\lambda}$
由于点 $A, B$ 在曲线 $\mathcal{C}$ 上，所以$T(A,A) = T(B,B) = 0$。\\
\textbf{证明插入律：}用定比点差法：
\begin{align*}
&T(P,A) = T(\frac{A+\lambda B}{1+\lambda},A)=\frac{1}{1+\lambda} T(A,A) +\frac{\lambda}{1+\lambda} T(B,A)=\frac{\lambda}{1+\lambda} T(B,A)=\beta T(A,B)\\
&T(P,B) =T(\frac{A+\lambda B}{1+\lambda},B)=\frac{1}{1+\lambda} T(A,B) +\frac{\lambda}{1+\lambda} T(B,B)=\frac{1}{1+\lambda} T(A,B)=\alpha T(A,B)\\
&\implies\text{加:}T(P,A)+T(P,B)=\frac{\lambda}{1+\lambda} T(B,A)+\frac{1}{1+\lambda} T(A,B)= T(A,B)
\end{align*}

\textbf{证明调和律（相乘法）：}
将 $T(P,A)$ 与 $T(P,B)$ 两式相乘，并用$T(A,A)=T(B,B)=0$配凑:
\begin{align*}
T(P,A) \cdot T(P,B) &= \left( \frac{\lambda}{1+\lambda} T(A,B) \right) \cdot \left( \frac{1}{1+\lambda} T(A,B) \right) = \frac{\lambda\big[T(A,B)\big]^2}{(1+\lambda)^2} =\dfrac12\frac{2\lambda T(A,B)}{(1+\lambda)^2} T(A,B)\\
&=\frac{T(A,B)}2\left(\frac{1}{(1+\lambda)^2} T(A,A) + \frac{2\lambda}{(1+\lambda)^2} T(A,B) + \frac{\lambda^2}{(1+\lambda)^2} T(B,B)\right)\\
&=\frac12T(A,B)\cdot T(P,P)
\end{align*}
为什么称之为“调和”律？结合插入律 $T(A,B) = T(P,A) + T(P,B)$，将调和律等式中的 $T(A,B)$ 替换，可化简为：
$$ T(P,P) = \frac{2 T(P,A) T(P,B)}{T(P,A) + T(P,B)} $$
下面证明扩展的定比律：
\begin{align*}
&T(P,A) = T(\frac{A+\lambda B}{1+\lambda},A)=\frac{1}{1+\lambda} T(A,A) +\frac{\lambda}{1+\lambda} T(B,A)=\frac{\lambda}{1+\lambda} T(B,A)=\beta T(A,B)\\
&T(P,B) =T(\frac{A+\lambda B}{1+\lambda},B)=\frac{1}{1+\lambda} T(A,B) +\frac{\lambda}{1+\lambda} T(B,B)=\frac{1}{1+\lambda} T(A,B)=\alpha T(A,B)\\
&\implies\lambda = \frac{T(P,A)}{T(P,B)}=\frac{T(P,A)\cdot T(P,B)}{T(P,B)^2}=\frac{T(P,P)\cdot T(A,B)}{2T(P,B)^2}=\frac{2T(P,A)^2}{T(P,P)\cdot T(A,B)}\\
&\implies\alpha=\frac{T(P,B)}{T(A,B)}=\frac{T(P,B)\cdot T(P,A)}{T(A,B)\cdot T(P,A)}=\frac{\frac12T(A,B)\cdot T(P,P)}{T(A,B)\cdot T(P,A)}=\frac12\frac{T(P,P)}{T(P,A)}\\
&\implies\beta=\frac{T(P,A)}{T(A,B)}=\frac{T(P,A)\cdot T(P,B)}{T(A,B)\cdot T(P,B)}=\frac{\frac12T(A,B)\cdot T(P,P)}{T(A,B)\cdot T(P,B)}=\frac12\frac{T(P,P)}{T(P,B)}
\end{align*}

\begin{conclusion}{差分律（中点共轭律）}{}
已知二次曲线 $C$ 上有两点 $A, B$。直线 $AB$ 上一动点 $P$ 满足 $\overrightarrow{AP} = \lambda \overrightarrow{PB}$。则：
$$ E(P, \overrightarrow{BA}) = T(P,A) - T(P,B) = \frac{\lambda - 1}{\lambda + 1} T(A,B) $$
\textbf{点差法：}当 $P$ 为弦 $AB$ 的中点时，必然成立 $E(P, \overrightarrow{BA}) = 0$。将其展开为坐标形式即：
$$ \frac{x_P(x_A-x_B)}{a^2} + \frac{y_P(y_A-y_B)}{b^2} = 0 \implies k_{OP} \cdot k_{AB} = -\frac{b^2}{a^2} $$
\end{conclusion}

\textbf{证明差分律：}
将 $T(P,A)$ 与 $T(P,B)$ 两式相减，对比得证：
\begin{align*}
T(P,A) - T(P,B) &= \frac{\lambda}{1+\lambda} T(A,B) - \frac{1}{1+\lambda} T(A,B) = \frac{\lambda - 1}{\lambda + 1} T(A,B)\\
T(P,A) - T(P,B) &= \big(E(P,A) - 1\big) - \big(E(P,B) - 1\big) = E(P,A) - E(P,B)=E(P, \overrightarrow{BA})
\end{align*}

\begin{conclusion}{非对称分式的内积坍缩定理}{}
椭圆上两点 $A, B$ 坐标为$(x_1,y_1),(x_2,y_2)$，中点为$M$，弦 $AB$ 对应的极点（即过 $A,B$ 的切线交点）为 $P$
$$ \frac{x_1y_1+x_2y_2}{x_1y_2+x_2y_1} = E(A,B) = 2E(M,M) - 1 = \frac{1 - T(P,P)}{1 + T(P,P)}  $$
\end{conclusion}
利用平方差公式推导：$ \frac{x_1y_1+x_2y_2}{x_1y_2+x_2y_1} = E(A,B)$
\begin{align*}
\frac{x_1y_1+x_2y_2}{x_1y_2+x_2y_1} & = \frac{(x_1y_1+x_2y_2) \cdot W(A,B)}{(x_1y_2+x_2y_1) \cdot W(A,B)} = \frac{(x_1y_1+x_2y_2)(x_1y_2-x_2y_1)}{x_1^2 y_2^2 - x_2^2 y_1^2} \\
&=\frac{x_1^2 y_1 y_2 - x_1 x_2 y_1^2 + x_1 x_2 y_2^2 - x_2^2 y_1 y_2}{ x_1^2 \left[ b^2 \left(1 - \frac{x_2^2}{a^2}\right) \right] - x_2^2 \left[ b^2 \left(1 - \frac{x_1^2}{a^2}\right) \right] }=\frac{y_1y_2(x_1^2 - x_2^2) + x_1x_2(y_2^2 - y_1^2)}{b^2 (x_1^2 - x_2^2) }\\
&=\frac{ b^2 (x_1^2 - x_2^2) E(A,B) }{ b^2 (x_1^2 - x_2^2) } = E(A,B)
\end{align*}
设中点 $M = \frac{A+B}{2}$。对其施加内积算子并利用双线性分配律完全展开：
\begin{align*} E(M,M) &= E\left(\frac{A+B}{2}, \frac{A+B}{2}\right) = \frac{1}{4} \big[ E(A,A) + 2E(A,B) + E(B,B) \big] =\frac{1 + E(A,B)}{2} \\
    \implies E(A,B) &= 2E(M,M) - 1\end{align*}
也可以反过来写，反正就是凑完全平方公式。

\textbf{严格证明最后一个等号：}假设$\overrightarrow{OA}$ 与 $\overrightarrow{OB}$ 不共线，它们可作为平面的一组基底。因此可设：
$ P = \alpha A + \beta B $，由于点 $A, B$ 在曲线上，$P$在$A,B$的极线上，故
\begin{align*}
1=E(P, A) &= E(\alpha A + \beta B, A) = \alpha E(A,A) + \beta E(B,A) \implies \alpha + \beta E(A,B) = 1 \\
1=E(P, B) &= E(\alpha A + \beta B, B) = \alpha E(A,B) + \beta E(B,B) \implies \alpha E(A,B) + \beta = 1\\
\text{相减：}&(\alpha - \beta) - (\alpha - \beta) E(A,B) = 0 \implies (\alpha - \beta)\big[1 - E(A,B)\big] = 0
\end{align*}
对于椭圆上两个不同的点 $A \neq B$，必有 $E(A-B, A-B)= 2 - 2E(A,B) > 0 > 0$。故$ \alpha = \beta $解得精准的伸缩系数：
$$ \alpha = \beta = \frac{1}{1 + E(A,B)} \implies P = \frac{A + B}{1 + E(A,B)} $$
接下来，计算极点 $P$ 的自内积 $E(P,P)$，继续使用算子的分配律展开：
\begin{align*}
E(P,P) &= E\left( \frac{A+B}{1+E(A,B)}, \frac{A+B}{1+E(A,B)} \right) = \frac{E(A,A) + E(B,B) + 2E(A,B)}{\big[1+E(A,B)\big]^2} \\
&= \frac{2 + 2E(A,B)}{\big[1+E(A,B)\big]^2} = \frac{2\big[1 + E(A,B)\big]}{\big[1+E(A,B)\big]^2} = \frac{2}{1 + E(A,B)}
\end{align*}
反解出 $E(A,B)$ 的表达式：
$$ 1 + E(A,B) = \frac{2}{E(P,P)} \implies E(A,B) = \frac{2}{E(P,P)} - 1= \frac{1 - T(P,P)}{1 + T(P,P)} $$

\textit{【上帝视角验证：参数方程的终极统一】\\
为什么极点控制律长得如此像三角万能代换公式 $\cos\Delta\theta = \frac{1-\tan^2(\Delta\theta/2)}{1+\tan^2(\Delta\theta/2)}$？\\
我们启用椭圆的离心角参数方程进行降维验证：设 $A(a\cos\alpha, b\sin\alpha), B(a\cos\beta, b\sin\beta)$。\\
考察分子（和差化积）：
$$ x_1y_1 + x_2y_2 = ab\cos\alpha\sin\alpha + ab\cos\beta\sin\beta = \frac{1}{2}ab(\sin 2\alpha + \sin 2\beta) = ab\sin(\alpha+\beta)\cos(\alpha-\beta) $$
考察分母（两角和正弦）：
$$ x_1y_2 + x_2y_1 = ab\cos\alpha\sin\beta + ab\cos\beta\sin\alpha = ab\sin(\alpha+\beta) $$
两式直接相除，极度非对称的 $ab\sin(\alpha+\beta)$ 瞬间约去，原分式竟然等价于极其纯粹的 $\cos(\alpha-\beta)$！\\
而这恰好就是两点间的内积算子：
$$ E(A,B) = \frac{a\cos\alpha \cdot a\cos\beta}{a^2} + \frac{b\sin\alpha \cdot b\sin\beta}{b^2} = \cos\alpha\cos\beta + \sin\alpha\sin\beta = \cos(\alpha-\beta) $$
而极点 $P$ 的极化算子 $T(P,P)$，其代数本质正是半角正切的平方 $\tan^2\frac{\alpha-\beta}{2}$。算子代数用最纯粹的多项式外壳，完美同构了超越函数的三角代换法则。几何与代数在这里达到了令人战栗的和谐统一！}

\begin{conclusion}{降维结论}{}
椭圆上两点$(x_1,y_1),(x_2,y_2)$，非椭圆上一点$P(m,n)$，则
\[(x_1y_2+x_2y_1)\left(\dfrac{m^2}{a^2}+\dfrac{n^2}{b^2}\right)=m(y_1+y_2) + n(x_1+x_2) - 2mn \]
\end{conclusion}
直线 $AB$ 上的动点 $P(m,n)$ 是 $A$ 和 $B$ 的仿射组合。设其仿射系数为 $\alpha, \beta$，即 $P = \alpha A + \beta B$（此时 $m = \alpha x_1 + \beta x_2, n = \alpha y_1 + \beta y_2$，且必定满足 $\alpha+\beta=1$）。
\begin{align*}
&(x_1y_2+x_2y_1)\left(\dfrac{m^2}{a^2}+\dfrac{n^2}{b^2}\right) \\
&= (x_1y_2+x_2y_1) E(P, P) \\
&= (x_1y_2+x_2y_1) E(\alpha A + \beta B, \alpha A + \beta B)= (x_1y_2+x_2y_1) \big[ \alpha^2 E(A,A) + \beta^2 E(B,B) + 2\alpha\beta E(A,B) \big] \\
&= (x_1y_2+x_2y_1) \big[ \alpha^2 + \beta^2 + 2\alpha\beta E(A,B) \big] = (\alpha^2 + \beta^2)(x_1y_2+x_2y_1) + 2\alpha\beta (x_1y_1 + x_2y_2) \\
&= 1 \cdot (x_1y_2+x_2y_1) - 2\alpha\beta (x_1y_2+x_2y_1) + 2\alpha(1-\alpha) x_1y_1 + 2\beta(1-\beta) x_2y_2\\
&= (\alpha+\beta)(x_1y_2+x_2y_1) + 2\alpha x_1y_1 - 2\alpha^2 x_1y_1 + 2\beta x_2y_2 - 2\beta^2 x_2y_2 - 2\alpha\beta(x_1y_2+x_2y_1)  \\
&= \big[ \alpha x_1y_2 + \beta x_1y_2 + \alpha x_2y_1 + \beta x_2y_1 + 2\alpha x_1y_1 + 2\beta x_2y_2 \big] - 2 \big( \alpha^2 x_1y_1 + \beta^2 x_2y_2 + \alpha\beta x_1y_2 + \alpha\beta x_2y_1 \big)  \\
&= \big[ \alpha x_1(y_1+y_2) + \alpha y_1(x_1+x_2) + \beta x_2(y_1+y_2) + \beta y_2(x_1+x_2) \big] - 2 \big( \alpha x_1 + \beta x_2 \big)\big( \alpha y_1 + \beta y_2 \big) \\
&= \big[ (\alpha x_1 + \beta x_2)(y_1+y_2) + (\alpha y_1 + \beta y_2)(x_1+x_2) \big] - 2 x_P y_P  \\
&= m(y_1+y_2) + n(x_1+x_2) - 2mn
\end{align*}


\begin{conclusion}{直线参数方程}{}
设有一条以定点 $P(x_0, y_0)$ 为基点，以向量 $\vec{v} = (v_x, v_y)$ 为方向向量的射线（或直线），其参数方程可表示为 $X = P + t\vec{v}$。将该点 $X$ 代入二次曲线的极化算子中，其结果是一个关于参数 $t$ 的标准一元二次多项式：
$$ T(X,X) = E(\vec{v},\vec{v}) t^2 + 2T(P, \vec{v}) t + T(P,P) $$
其中，我们在定义域上将 $T(P, \vec{v})$ 严格定义为 $E(P, \vec{v}) = \frac{x_0 v_x}{a^2} + \frac{y_0 v_y}{b^2}$。韦达定理：
\[ t_1 + t_2 = \frac{-2T(P,\vec{v})}{E(\vec{v},\vec{v})}, \quad t_1 t_2 = \frac{T(P,P)}{E(\vec{v},\vec{v})} \]
\end{conclusion}
\vspace{-20pt}
\begin{align*}
T(X,X)& = E(P + t\vec{v}, P + t\vec{v}) - 1 = E(P, P+t\vec{v}) + E(t\vec{v}, P+t\vec{v})-1\\
&= E(P, P+t\vec{v}) +t E(\vec{v}, P+t\vec{v})-1= (E(P,P) + tE(P,\vec{v})) + t(E(\vec{v},P) + tE(\vec{v},\vec{v}))-1 \\
&=E(P,P) + 2tE(P,\vec{v}) + t^2E(\vec{v},\vec{v})-1\\
&=T(P,P)+2tE(P,\vec{v}) + t^2E(\vec{v},\vec{v})
\end{align*}
\textit{【批注：这说明当坐标系原点发生平移时，二次曲线方程被算子系统直接降维成了一个无需手动代入 $(kx+m)^2$ 繁琐展开的参数多项式。】}

\begin{conclusion}{斜率的外积探测公式}{}
定义水平方向单位向量 $\vec{i} = (1, 0)$ 和竖直方向单位向量 $\vec{j} = (0, 1)$。
对于平面上任意一条以向量 $\vec{v}$ 为方向的直线，其斜率 $k$ 完全等价于外积算子 $W$ 的比值：
$$ k = \frac{W(\vec{i}, \vec{v})}{W(\vec{v}, \vec{j})} $$
\end{conclusion}
\textbf{【极简证明】：}
设 $\vec{v} = (v_x, v_y)$。根据外积算子的定义 $W(M,N) = x_M y_N - x_N y_M$：
用 $\vec{i}$ 探测：$W(\vec{i}, \vec{v}) = 1 \cdot v_y - 0 \cdot v_x = v_y$（完美提取了纵向分量 $\Delta y$）。
用 $\vec{j}$ 探测：$W(\vec{v}, \vec{j}) = v_x \cdot 1 - v_y \cdot 0 = v_x$（完美提取了横向分量 $\Delta x$）。
两者相除，立刻得到 $k = \frac{v_y}{v_x}$。

\textit{【实战降维意义】：当题目让你证明 $k_1 + k_2 = 0$ 或者 $k_1 = \lambda k_2$ 时，我们再也不用去写繁杂的分子分母坐标。因为 $W$ 算子就像乘法一样满足完美的分配律，所有的“斜率加减”都可以直接在算子内部进行向量的合并！这彻底消灭了通分带来的满屏代数风暴。}

\begin{conclusion}{调和线束与合分比的算子化本源}{}
对于任意定点 $P$ 与共线的三点 $A, B, C$，传统解法中极其冗长的斜率关系公式，被外积算子 $W$ 及其分配律绝对降维：
$$ \frac{k_{PA}-k_{PC}}{k_{PB}-k_{PC}} \cdot \frac{x_A-x_P}{x_B-x_P} = -\frac{T(C,A)}{T(C,B)} = \frac{\overrightarrow{AC}}{\overrightarrow{BC}} $$
\end{conclusion}

\textbf{纯算子化绝对降维证明：}
对于直线上任意两点，其与定点 $P$ 的斜率差，通分后分子必然产生欧氏外积算子 $W$：
$$ k_{PA} - k_{PC} = \frac{y_A-y_P}{x_A-x_P} - \frac{y_C-y_P}{x_C-x_P} = \frac{W(\overrightarrow{PC}, \overrightarrow{PA})}{(x_A-x_P)(x_C-x_P)} $$
将两组斜率差相除，横坐标差 $(x_C-x_P)$ 瞬间对消，左侧斜率比值直接化简为外积之比：
$$ \frac{k_{PA}-k_{PC}}{k_{PB}-k_{PC}} = \frac{W(\overrightarrow{PC}, \overrightarrow{PA})}{W(\overrightarrow{PC}, \overrightarrow{PB})} \cdot \frac{x_B-x_P}{x_A-x_P} $$
因为 $A, B, C$ 共线，设定比仿射组合 $C = \frac{A + \lambda B}{1+\lambda}$，其中定比 $\lambda = \frac{\overrightarrow{AC}}{\overrightarrow{CB}} = \frac{T(C,A)}{T(C,B)}$（根据第二部分极化算子定比律）。
将其代入外积算子，应用\textbf{双线性分配律}与\textbf{反对称性}（$W(\vec{u},\vec{u})=0$）：
\begin{align*}
W(\overrightarrow{PC}, \overrightarrow{PA}) &= W\left(\frac{\overrightarrow{PA} + \lambda \overrightarrow{PB}}{1+\lambda}, \overrightarrow{PA}\right) = -\frac{\lambda}{1+\lambda} W(\overrightarrow{PA}, \overrightarrow{PB}) \\
W(\overrightarrow{PC}, \overrightarrow{PB}) &= W\left(\frac{\overrightarrow{PA} + \lambda \overrightarrow{PB}}{1+\lambda}, \overrightarrow{PB}\right) = \frac{1}{1+\lambda} W(\overrightarrow{PA}, \overrightarrow{PB})
\end{align*}
将上述两式相除，外积项 $W(\overrightarrow{PA}, \overrightarrow{PB})$ 完美抵消，比例直接坍缩为 $-\lambda$：
$$ \frac{W(\overrightarrow{PC}, \overrightarrow{PA})}{W(\overrightarrow{PC}, \overrightarrow{PB})} = -\lambda = -\frac{T(C,A)}{T(C,B)} \implies\frac{k_{PA}-k_{PC}}{k_{PB}-k_{PC}} \frac{x_A-x_P}{x_B-x_P} = -\lambda = \frac{\overrightarrow{AC}}{\overrightarrow{BC}} $$

\begin{conclusion}{斜率调整法}{}
椭圆$\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$上有$P(x_0,y_0),A(x_1,y_1),B(x_2,y_2)$三个点，令$k_1=k_{PA},k_2=k_{PB}$，那么
    $$\frac{a^2}{b^2} k_1 k_2 - \frac{x_1 + x_2}{y_1 - y_2}(k_1 - k_2) + 1 = 0 $$
\end{conclusion}
已知椭圆上有三个定点 $P(x,y)$、$A(x_1,y_1)$、$B(x_2,y_2)$。弦 $PA$ 和 $PB$ 的斜率 $k_1, k_2$ 。定义两个方向向量：$\vec{u} = A - P$，所以 $k_1 = \frac{u_y}{u_x}$$\vec{v} = B - P$，所以 $k_2 = \frac{v_y}{v_x}$我试着把这两个斜率相减：
$$k_1 - k_2 = \frac{u_y}{u_x} - \frac{v_y}{v_x} = \frac{u_y v_x - v_y u_x}{u_x v_x}=\frac{-W(\vec{u}, \vec{v})}{u_x v_x}$$
为了凑出 $\frac{u_y v_y}{u_x v_x}$ 也就是 $k_1 k_2$，直接给内积算子提取公因式 $u_x v_x$：
$$E(\vec{u}, \vec{v}) = u_x v_x \left( \frac{1}{a^2} + \frac{1}{b^2} \frac{u_y v_y}{u_x v_x} \right) = u_x v_x \left( \frac{1}{a^2} + \frac{k_1 k_2}{b^2} \right)\implies a^2 E(\vec{u}, \vec{v}) = u_x v_x \left( 1 + \frac{a^2}{b^2} k_1 k_2 \right)$$
两个结果相除：$$\frac{\frac{a^2}{b^2} k_1 k_2 + 1}{k_1 - k_2} = \frac{a^2 E(\vec{u}, \vec{v})}{-W(\vec{u}, \vec{v})}$$
\begin{itemize}
\item 引入 $A, B$ 的中点 $M = \frac{A+B}{2}$，这意味 $A+B = 2M$。
\item 先拆解分子 $E(\vec{u}, \vec{v})$：
\begin{align*}
E(A-P, B-P) &= E(A,B)+ E(P,P) - (E(A,P) + E(B,P)) \\
&= E(A,B)+ E(P,P) -E(A+B, P)=E(A,B)+ E(P,P)-2E(M, P)\\
&= (2E(M,M) - 1) - 2E(M,P) + 1 =  {2E(M, M-P)}
\end{align*}
\item 再拆解分母 $W(\vec{u}, \vec{v})$：同样用分配律展开：
\begin{align*}
W(A-P, B-P) &= W(A,B) - W(A,P) - W(P,B) + W(P,P)\\
&= W(A,B) - W(A,P) + W(B,P)=W(A,B)-W(A-B, P)\\
&=W\left(M + \frac{A-B}{2},M - \frac{A-B}{2}\right)- W(A-B, P) \\
&=  W(A-B, M) - W(A-B, P) =W(A-B, M-P)
\end{align*}
\end{itemize}
现在我们把化简好的分子分母放回我们的比例中：
$$\frac{a^2 E(\vec{u}, \vec{v})}{-W(\vec{u}, \vec{v})} = \frac{2a^2 E(M, M-P)}{-W(A-B, M-P)}$$
把分子写成坐标形式：
$$2a^2 \left( \frac{x_M (x_M - x_P)}{a^2} + \frac{y_M (y_M - y_P)}{b^2} \right) = 2 \left( x_M (x_M - x_P) + \frac{a^2}{b^2}y_M (y_M - y_P) \right)$$
核心卡点： 我们怎么处理 $x_M$？别忘了 $A,B$ 在椭圆上，根据椭圆“点差法”（也就是极化共轭差分律）：$E(M, A-B) = 0 \implies \frac{x_M (x_A - x_B)}{a^2} + \frac{y_M (y_A - y_B)}{b^2} = 0$移项得到一个代换神器：$x_M = - \frac{a^2}{b^2} y_M \frac{y_A - y_B}{x_A - x_B}$。我们把这个 $x_M$ 仅仅代入到分子第一项的前面那个 $x_M$ 中：
\begin{align*}
\text{分子} &= 2 \left( {- \frac{a^2}{b^2} y_M \frac{y_A - y_B}{x_A - x_B}} (x_M - x_P) + \frac{a^2}{b^2} y_M (y_M - y_P) \right)\\
&=2 \frac{a^2}{b^2} \frac{y_M}{x_A - x_B} \Big[ -(y_A - y_B)(x_M - x_P) + (x_A - x_B)(y_M - y_P) \Big]\\
&= \frac{2a^2 y_M}{b^2 (x_A - x_B)} W(A-B, M-P)
\end{align*}
所以
$$\frac{ \frac{2a^2 y_M}{b^2 (x_A - x_B)} W(A-B, M-P) }{ -W(A-B, M-P) } = - \frac{2a^2 y_M}{b^2 (x_A - x_B)}$$
把 $2y_M = y_A + y_B$ 代回上面的结果：$$\text{右侧} = - \frac{a^2(y_A + y_B)}{b^2(x_A - x_B)}$$
最后一次借用点差法 $\frac{x_A + x_B}{a^2}(x_A - x_B) + \frac{y_A + y_B}{b^2}(y_A - y_B) = 0$，移项可得：$$- \frac{a^2(y_A + y_B)}{b^2(x_A - x_B)} = \frac{x_A + x_B}{y_A - y_B} = \frac{x_1 + x_2}{y_1 - y_2}$$现在，我们把第三步的等式左边，和刚刚化简到底的右边连起来：$$\frac{\frac{a^2}{b^2} k_1 k_2 + 1}{k_1 - k_2} = \frac{x_1 + x_2}{y_1 - y_2}\implies\frac{a^2}{b^2} k_1 k_2 - \frac{x_1 + x_2}{y_1 - y_2}(k_1 - k_2) + 1 = 0 $$

\begin{conclusion}{算子拉格朗日恒等式}{}
对于平面上任意两个向量 $\vec{u}, \vec{v}$，其欧氏外积 $W$ 与椭圆内积 $E$ 永远满足以下绝对恒等关联：
$$ \big[W(\vec{u},\vec{v})\big]^2 = a^2 b^2 \Big[ E(\vec{u},\vec{u})E(\vec{v},\vec{v}) - \big[E(\vec{u},\vec{v})\big]^2 \Big] $$
\end{conclusion}
设 $\vec{u}=(x_1, y_1), \vec{v}=(x_2, y_2)$。
左侧外积的平方：$W(\vec{u},\vec{v})^2 = (x_1 y_2 - x_2 y_1)^2 = x_1^2 y_2^2 + x_2^2 y_1^2 - 2x_1 x_2 y_1 y_2$。
计算内积的平方并同乘 $a^2b^2$：$a^2b^2 E(\vec{u},\vec{v})^2 = a^2b^2 \left(\frac{x_1 x_2}{a^2} + \frac{y_1 y_2}{b^2}\right)^2 = \frac{b^2}{a^2}x_1^2 x_2^2 + \frac{a^2}{b^2}y_1^2 y_2^2 + 2x_1 x_2 y_1 y_2$。
将这两式\textbf{相加}，交叉项完全抵消：
\begin{align*}
W(\vec{u},\vec{v})^2 + a^2b^2 E(\vec{u},\vec{v})^2 &= x_1^2 y_2^2 + x_2^2 y_1^2 + \frac{b^2}{a^2}x_1^2 x_2^2 + \frac{a^2}{b^2}y_1^2 y_2^2 \\
&= x_1^2 \left( y_2^2 + \frac{b^2}{a^2}x_2^2 \right) + y_1^2 \left( x_2^2 + \frac{a^2}{b^2}y_2^2 \right)
\end{align*}
观察括号内部，直接提取常数化为椭圆内积算子：
$y_2^2 + \frac{b^2}{a^2}x_2^2 = b^2 \left( \frac{x_2^2}{a^2} + \frac{y_2^2}{b^2} \right) = b^2 E(\vec{v},\vec{v})$。同理，后半部分的括号等于 $a^2 E(\vec{v},\vec{v})$。
整体合并：
$$ x_1^2 \big[b^2 E(\vec{v},\vec{v})\big] + y_1^2 \big[a^2 E(\vec{v},\vec{v})\big] = a^2 b^2 E(\vec{v},\vec{v}) \left(\frac{x_1^2}{a^2} + \frac{y_1^2}{b^2}\right) = a^2 b^2 E(\vec{v},\vec{v}) E(\vec{u},\vec{u}) $$
移项即得证！这把开启面积降维打击的钥匙已经握在我们手中。

\begin{conclusion}{万能算子弦长公式}{}
已知直线过定点 $P$，且方向向量为 $\vec{v}$，交二次曲线于 $A, B$ 两点。则弦长 $|AB|$ 仅取决于算子 $T, E, I$ 在基点 $P$ 与方向 $\vec{v}$ 上的取值：
$$ |AB| = \frac{2\sqrt{\big[E(P, \vec{v})\big]^2 - E(\vec{v}, \vec{v})T(P, P)}}{E(\vec{v}, \vec{v})} \sqrt{I(\vec{v}, \vec{v})} = 2 \frac{\sqrt{E(\vec{v},\vec{v}) - \frac{\big[W(P,\vec{v})\big]^2}{a^2 b^2}}}{E(\vec{v},\vec{v})} \cdot \sqrt{I(\vec{v},\vec{v})} $$
特别地，当 $P$ 是弦 $AB$ 的\textbf{中点}时，公式瞬间坍缩为：$|AB| = 2\sqrt{\frac{-T(P,P)}{E(\vec{v}, \vec{v})}} \sqrt{I(\vec{v}, \vec{v})}$。
\end{conclusion}
直线上动点 $X = P + t\vec{v}$ 代入曲线极化算子得到：
$$ E(\vec{v},\vec{v})t^2 + 2E(P, \vec{v})t + T(P,P) = 0 $$
设交点对应的参数为 $t_1, t_2$。弦长本质上是参数的差值乘以方向向量的实际欧氏长度：$|AB| = |t_1 - t_2| \cdot |\vec{v}|$。
由一元二次方程求根差公式 $|t_1 - t_2| = \frac{\sqrt{\Delta}}{a}$，直接算出：
$$ |t_1 - t_2| = \frac{\sqrt{4\big[E(P, \vec{v})\big]^2 - 4E(\vec{v}, \vec{v})T(P, P)}}{E(\vec{v}, \vec{v})} $$
利用 $T(P,P) = E(P,P) - 1$ 将根号内的式子展开，并代入刚刚证明的《算子拉格朗日恒等式》：
\begin{align*}
\frac{\Delta}{4} &= E(P,\vec{v})^2 - \big(E(P,P) - 1\big)E(\vec{v},\vec{v}) = E(\vec{v},\vec{v}) - \Big[ E(P,P)E(\vec{v},\vec{v}) - E(P,\vec{v})^2 \Big] \\
&= E(\vec{v},\vec{v}) - \frac{\big[W(P,\vec{v})\big]^2}{a^2 b^2}
\end{align*}
而向量 $\vec{v}$ 的欧氏长度恰好就是欧氏内积算子开根号：$|\vec{v}| = \sqrt{I(\vec{v}, \vec{v})}$。相乘即得目标公式！

\begin{conclusion}{原点三角形 $OAB$ 面积公式}{}
已知椭圆上有两点 $A, B$。则由原点 $O$ 与 $A,B$ 构成的三角形面积，完全由这两点的极化算子唯一决定：
$$ S_{\triangle OAB} = \frac{1}{2} ab \sqrt{-T(A,B) \big(T(A,B)+2\big)} $$
\end{conclusion}
三角形面积等于外积绝对值的一半：$S_{\triangle OAB} = \frac{1}{2} |W(A,B)|$。
既然 $A, B$ 都在椭圆上，那么把它们代入极化算子必定等于 0：$T(A,A)=0 \implies E(A,A) = 1$。同理 $E(B,B) = 1$。
代入刚证的拉格朗日恒等式：
$$ \big[W(A,B)\big]^2 = a^2 b^2 \big(1 \times 1 - E(A,B)^2\big) = a^2 b^2 \big(1 - E(A,B)^2\big) $$
我们把内积算子 $E$ 换回更基础的极化算子 $T$，即代入 $E(A,B) = T(A,B) + 1$：
$$ 1 - \big(T(A,B)+1\big)^2 = 1 - \big(T(A,B)^2 + 2T(A,B) + 1\big) = -T(A,B)\big(T(A,B)+2\big) $$
开根号并乘以 $1/2$，得证。

\begin{conclusion}{内接三角形的“算子海伦公式”}{}
已知椭圆内接三角形 $\triangle ABC$。
我们只需算出这三条边的\textbf{极化常数}（因为内接弦的夹角余弦小于 1，所以相互极化均为负值，加负号转为正）：
$$ t_1 = -T(B,C), \quad t_2 = -T(A,C), \quad t_3 = -T(A,B) $$
则该三角形的面积，完完全全由这三个算子常数决定，呈现出绝对对称的代数多项式：
$$ S_{\triangle ABC} = \frac{1}{2} a b \sqrt{ 2t_1 t_2 + 2t_2 t_3 + 2t_3 t_1 - t_1^2 - t_2^2 - t_3^2 } $$
由于 $A,B,C$ 同在一个椭圆上，必定满足三角恒等式约束，上述根号内的长多项式在代数上等于 $2t_1 t_2 t_3$。因此，$S_{\triangle ABC} = \frac{\sqrt{2}}{2} ab \sqrt{t_1 t_2 t_3}$）。
\end{conclusion}
内接三角形的面积为两条边向量外积的一半：$S_{\triangle ABC} = \frac{1}{2} \big| W(B-A, C-A) \big|$。
设向量 $\vec{u} = B-A, \vec{v} = C-A$。我们再次调用拉格朗日恒等式：
$$ \big[W(\vec{u}, \vec{v})\big]^2 = a^2 b^2 \Big[ E(\vec{u}, \vec{u}) E(\vec{v}, \vec{v}) - \big[E(\vec{u}, \vec{v})\big]^2 \Big] $$
将这三个量完全拆开。记住点在椭圆上 $E(A,A)=E(B,B)=E(C,C)=1$：
\begin{itemize}
\item $E(\vec{u}, \vec{u}) = E(B-A, B-A) = E(B,B) - 2E(A,B) + E(A,A) = 2 - 2E(A,B)$。由于 $E(A,B) = T(A,B)+1$，代入后瞬间坍缩：$E(\vec{u}, \vec{u}) = -2T(A,B) = 2t_3$。
\item 完全同理，右边：$E(\vec{v}, \vec{v}) = E(C-A, C-A) = -2T(A,C) = 2t_2$。
\item 交叉项展开：$E(\vec{u}, \vec{v}) = E(B-A, C-A) = E(B,C) - E(A,B) - E(A,C) + E(A,A)$。全部转化为 $T$ 算子：$(T(B,C)+1) - (T(A,B)+1) - (T(A,C)+1) + 1 = T(B,C) - T(A,B) - T(A,C)$。将其用我们定义的极化常数 $t$ 替换：$E(\vec{u}, \vec{v}) = -t_1 + t_3 + t_2$。
\end{itemize}
将这三个极其简洁的算子结果，直接塞回拉格朗日恒等式中：
\begin{align*}
\frac{W^2}{a^2 b^2} &= (2t_3)(2t_2) - (t_3 + t_2 - t_1)^2 \\
&= 4t_2 t_3 - (t_3^2 + t_2^2 + t_1^2 + 2t_2 t_3 - 2t_1 t_3 - 2t_1 t_2) \\
&= 2t_1 t_2 + 2t_2 t_3 + 2t_3 t_1 - t_1^2 - t_2^2 - t_3^2
\end{align*}
开根号并乘上 $\frac{1}{2}$，\textbf{完美的算子海伦公式跃然纸上！}

\section{圆锥曲线经典性质的证明}

\begin{definition}{极点与极线}{}
在极化算子体系中，对于二次曲线内（或外）的任意定点 $P$，定义满足方程 $T(P, X) = 0$ 的所有动点 $X$ 构成的直线，为点 $P$ 对应的\textbf{极线（Polar Line）}。
\end{definition}

\begin{conclusion}{完全四边形极点极线定理}{}
已知二次曲线 $\mathcal{C}$ 内（或外）有一点 $P$。过 $P$ 作两条弦 $AB$ 与 $CD$。连接 $AD, BC$ 交于点 $Q$；连接 $AC, BD$ 交于点 $R$。
则交点 $Q$ 和 $R$ 必然都在点 $P$ 的极线上，即必然成立：
$$ T(P, Q) = 0 \quad \text{且} \quad T(P, R) = 0 $$
\end{conclusion}
设 $P = \alpha_1 A + \beta_1 B = \alpha_2 C + \beta_2 D$，其中$\alpha_1+\beta_1 = \alpha_2+\beta_2 = 1$，我们要寻找直线 $AD$ 与 $BC$ 的交点 $Q$，只需将含 $A, D$ 的项移至等号同侧，含 $B, C$ 的项移至另一侧：
$$ \alpha_1 A - \beta_2 D = \alpha_2 C - \beta_1 B $$
等式两侧的系数和显然相等，设为 $k$。将两边同除以 $k$ （假设直线不平行即 $k \neq 0$），使系数和为1：
$$ Q = \frac{\alpha_1 A - \beta_2 D}{k} = \frac{\alpha_2 C - \beta_1 B}{k} $$
\textit{（左侧系数和为 1，表明点在 $AD$ 直线上；右侧表明在 $BC$ 直线上，故该点坐标必定是交点 $Q$）}现在，对交点 $Q$ 施加极化算子 $T(P, \cdot)$。根据算子关于仿射组合的分配律展开：
\begin{align*}
T(P,Q) &= T\left(P, \frac{\alpha_1 A - \beta_2 D}{k}\right) = \frac{1}{k} \big[ \alpha_1 T(P,A) - \beta_2 T(P,D) \big]\\
&= \frac{1}{k} \big[\frac12T(P,P)-\frac12 T(P,P) \big]=0,~~~\text{定比律}
\end{align*}
交点 $Q$ 必定满足 $T(P, Q)=0$，即位于点 $P$ 的极线上！\textbf{同理对称证明点 $R$}:若要寻找 $AC$ 与 $BD$ 的交点 $R$，只需将原式重新按 $A,C$ 和 $B,D$ 移项：
$$ \alpha_1 A - \alpha_2 C = \beta_2 D - \beta_1 B $$
同除以系数和 $m = \alpha_1 - \alpha_2 = \beta_2 - \beta_1$ 得到交点 $R$ 的精确仿射组合，再次施加极化算子：
\begin{align*}
T(P,R) &= T\left(P, \frac{\alpha_1 A - \alpha_2 C}{m}\right) = \frac{1}{m} \big[ \alpha_1 T(P,A) - \alpha_2 T(P,C) \big] \\
&= \frac{1}{m} \left[ \frac{1}{2} T(P,P) - \frac{1}{2} T(P,P) \right] = 0
\end{align*}
交点 $R$ 也同样完美满足 $T(P, R) = 0$ 位于极线上，定理彻底得证。

\begin{conclusion}{坎迪定理（Candy's Theorem）与 蝴蝶定理}{}
    已知二次曲线 $\mathcal{C}$ 上有弦 $PQ$，直线 $PQ$ 上有一动点 $M$。过 $M$ 作两弦 $AB$ 和 $CD$。连结 $AD, BC$ 分别交直线 $PQ$ 于点 $X, Y$。则关于有向线段必有比例式：$$ \frac{1}{MX} + \frac{1}{MY} = \frac{1}{MP} + \frac{1}{MQ} $$\textbf{推论（蝴蝶定理）：}当 $M$ 为弦 $PQ$ 中点时，$M$ 必然也是 $XY$ 的中点。
\end{conclusion}
以 $M$ 为基点，设直线 $PQ$ 的方向向量为 $\vec{v}$，弦 $AB, CD$ 的方向向量分别为 $\vec{u}, \vec{w}$。
由二次曲线上的\textbf{直线参数方程}，若$X = P + t\vec{v}$，交点 $P, Q$ 对应的参数 $t_P, t_Q$ 是关于 $t$ 的方程 $T(X,X)=0$ 的两根，由韦达定理取倒数和：$$ \frac{1}{t_P} + \frac{1}{t_Q} = \frac{t_P+t_Q}{t_Pt_Q} = \frac{-2E(M, \vec{v})}{T(M,M)} $$同理，设 $A, B$ 参数为 $t_A, t_B$，$C, D$ 参数为 $t_C, t_D$，必然有：$$ \frac{1}{t_A} + \frac{1}{t_B} = \frac{-2E(M, \vec{u})}{T(M,M)}, \quad \frac{1}{t_C} + \frac{1}{t_D} = \frac{-2E(M, \vec{w})}{T(M,M)} $$

点 $X$ 在 $PQ$ 上，设其参数为 $x$ 即 $X = M + x\vec{v}$。同时 $A = M+t_A\vec{u}$，$D = M+t_D\vec{w}$。由于 $X, A, D$ 三点共线，由向量叉乘（算子 $W$）代表的有向面积归零性质，即 $W(\overrightarrow{AX}, \overrightarrow{AD}) = 0$。利用 $W$ 的双线性与 $W(\vec{u},\vec{u})=0$ 展开：$$ W(x\vec{v} - t_A\vec{u}, \ t_D\vec{w} - t_A\vec{u}) = x t_D W(\vec{v}, \vec{w}) - x t_A W(\vec{v}, \vec{u}) - t_A t_D W(\vec{u}, \vec{w}) = 0 $$两边同除以 $x \cdot t_A \cdot t_D \cdot W(\vec{u}, \vec{w})$，并利用反对称性 $W(\vec{u}, \vec{v}) = -W(\vec{v}, \vec{u})$ 移项分离出 $1/x$：$$ \frac{1}{x} = \frac{W(\vec{v}, \vec{w})}{W(\vec{u}, \vec{w})} \frac{1}{t_A} + \frac{W(\vec{u}, \vec{v})}{W(\vec{u}, \vec{w})} \frac{1}{t_D} $$完全对称地，交点 $Y = BC \cap PQ$ 对应参数 $y$，由 $Y(y\vec{v}), B(t_B\vec{u}), C(t_C\vec{w})$ 共线可得：$$ \frac{1}{y} = \frac{W(\vec{v}, \vec{w})}{W(\vec{u}, \vec{w})} \frac{1}{t_B} + \frac{W(\vec{u}, \vec{v})}{W(\vec{u}, \vec{w})} \frac{1}{t_C} $$

将两式相加，并代入第1步的极化倒数和公式。此时，借助内积算子 $E(M, \cdot)$ 完美的线性分配律，我们将外积标量直接“吸入”算子内部，触发一场极度舒适的二维平面基底坍缩：
\begin{align*}
\frac{1}{x} + \frac{1}{y} &= \frac{W(\vec{v}, \vec{w})}{W(\vec{u}, \vec{w})}\left(\frac{1}{t_A} + \frac{1}{t_B}\right) + \frac{W(\vec{u}, \vec{v})}{W(\vec{u}, \vec{w})}\left(\frac{1}{t_C} + \frac{1}{t_D}\right) \\
&= \frac{W(\vec{v}, \vec{w})}{W(\vec{u}, \vec{w})}\left[ \frac{-2E(M, \vec{u})}{T(M,M)} \right] + \frac{W(\vec{u}, \vec{v})}{W(\vec{u}, \vec{w})}\left[ \frac{-2E(M, \vec{w})}{T(M,M)} \right] \\
&= \frac{-2}{T(M,M) \cdot W(\vec{u}, \vec{w})} \Big[ W(\vec{v}, \vec{w})E(M, \vec{u}) + W(\vec{u}, \vec{v})E(M, \vec{w}) \Big] \\
&= \frac{-2}{T(M,M) \cdot W(\vec{u}, \vec{w})} E\Big(M, \; W(\vec{v}, \vec{w})\vec{u} + W(\vec{u}, \vec{v})\vec{w} \Big) \quad \text{(逆向调用算子 } E \text{ 的线性分配律)} \\
&= \frac{-2}{T(M,M) \cdot W(\vec{u}, \vec{w})} E\Big(M, \; W(\vec{u}, \vec{w})\vec{v} \Big) \quad \text{(神来之笔：触发二维基底折叠恒等式)} \\
&= \frac{-2 W(\vec{u}, \vec{w})}{T(M,M) \cdot W(\vec{u}, \vec{w})} E(M, \vec{v}) \quad \text{(算子标量齐次性，提取常数)} \\
&= \frac{-2E(M, \vec{v})}{T(M,M)} \quad \text{(外积面积因子瞬间约去)} \\
&= \frac{1}{t_P} + \frac{1}{t_Q} \quad \text{(完美闭环：回归弦 } PQ \text{ 的极化倒数和)}
\end{align*}
因为参数均对应于同一基准向量 $\vec{v}$ 所在的同一直线，参数倒数和等价于有向线段倒数和，得到：
$$ \frac{1}{MX} + \frac{1}{MY} = \frac{1}{MP} + \frac{1}{MQ} $$
\textbf{蝴蝶定理}：当 $M$ 为弦 $PQ$ 中点时，由极化\textbf{差分律（中点共轭律推论）}立得 $E(M, \vec{v}) = 0$。代入上述等式的中心枢纽，右侧瞬间归零，即 $\frac{1}{x} + \frac{1}{y} = 0 \implies x = -y$，直接证明 $M$ 也是 $XY$ 的中点。

\begin{conclusion}{帕斯卡定理（Pascal's Theorem）}{}
    圆锥曲线内接六边形 $A_1A_2A_3A_4A_5A_6$ 的三对对边交点 $P=A_1A_2 \cap A_4A_5$，$Q=A_2A_3 \cap A_5A_6$，$R=A_3A_4 \cap A_6A_1$ 必然共线。
\end{conclusion}
\textbf{详细证明过程：}
\textbf{1. 详细推导“算子相交弦定理”（卡诺定理）}
\textit{【补充理由：必须解释清楚，为什么有向线段的乘积比，可以直接变成极化算子 $T$ 的比？】}
设直线 $\mathcal{L}$ 交二次曲线于 $A, B$ 两点，该直线上有任意两个不同的定点 $K$ 和 $M$。因为 $K, M, A, B$ 四点严格共线，我们在建立参数方程时，可以为这条直线指定同一个方向向量 $\vec{v}$：\begin{itemize}
    \item \textbf{以 $K$ 为基点建系：} 直线上任意一点可表示为 $X = K + t_K\vec{v}$。由前文推导的直线参数多项式韦达定理，交点 $A, B$ 的参数乘积为：$t_{KA} \cdot t_{KB} = \frac{T(K, K)}{E(\vec{v}, \vec{v})}$。此时对应的有向线段可表示为 $\overrightarrow{KA} = t_{KA}\vec{v}$，$\overrightarrow{KB} = t_{KB}\vec{v}$。
    \item \textbf{以 $M$ 为基点建系：} 维持共线的方向向量 $\vec{v}$ 绝对不变！直线上动点表示为 $X = M + t_M\vec{v}$。同理，交点 $A, B$ 的参数乘积为：$t_{MA} \cdot t_{MB} = \frac{T(M, M)}{E(\vec{v}, \vec{v})}$。对应的有向线段为 $\overrightarrow{MA} = t_{MA}\vec{v}$，$\overrightarrow{MB} = t_{MB}\vec{v}$。
\end{itemize}
现在计算共线有向线段乘积的纯量比值。由于方向向量 $\vec{v}$ 完全相同，向量的比值直接等价于参数的比值（$\vec{v}$ 在上下作比时会被完全约掉）：
$$ \frac{\overrightarrow{KA} \cdot \overrightarrow{KB}}{\overrightarrow{MA} \cdot \overrightarrow{MB}} = \frac{t_{KA}\vec{v} \cdot t_{KB}\vec{v}}{t_{MA}\vec{v} \cdot t_{MB}\vec{v}} = \frac{t_{KA} \cdot t_{KB}}{t_{MA} \cdot t_{MB}} = \frac{\frac{T(K, K)}{E(\vec{v}, \vec{v})}}{\frac{T(M, M)}{E(\vec{v}, \vec{v})}} = \frac{T(K,K)}{T(M,M)} $$
\textit{【这就证明了：同一直线上测量的线段乘积比，与直线的倾斜方向 $E(\vec{v}, \vec{v})$ 毫无关系，分母被完美对消，只取决于两基点自身极化算子的比值！】}\\
\textbf{2. 严格构造梅涅劳斯拓扑框架（理清点线归属）}\textit{【补充理由：必须明确写出截线到底是怎么截三角形的，交点为什么在那条边上。】}选取内接六边形中互不相邻的三条边所在的直线，延长交出参考三角形 $\triangle KLM$：
\begin{itemize}

\textbf{对于边 $KL$ 组：}同理，分母 $\overrightarrow{A_3K} \cdot \overrightarrow{A_4K} = (-\overrightarrow{KA_3}) \cdot (-\overrightarrow{KA_4}) = \overrightarrow{KA_3} \cdot \overrightarrow{KA_4}$。\item \textbf{对于边 $LM$ 组：}同理，分母 $\overrightarrow{A_5L} \cdot \overrightarrow{A_6L} = (-\overrightarrow{LA_5}) \cdot (-\overrightarrow{LA_6}) = \overrightarrow{LA_5} \cdot A_2 \implies$ 比例化为 $\frac{T(K,K)}{T(M,M)}$\item 边 $KL$ 上，基点为 $L,K$，曲线交点为 $A_3, A_4 \implies$ 比例化为 $\frac{T(L,L)}{T(K,K)}$\item 边 $LM$ 上，基点为 $M,L$，曲线交点为 $A_5, A_6 \implies$ 比例化为 $-1$。根据\textbf{梅涅劳斯定理的逆定理}，这三个截点 $P, Q, R$ 必定共线于同一直线（即帕斯卡线）。\textbf{帕斯卡定理证毕！}

\begin{conclusion}{布列安桑定理（Brianchon's Theorem）}{}圆锥曲线外切六边形 $V_1V_2V_3V_4V_5V_6$ 的三条主对角线 $V_1V_4, V_2V_5, V_3V_6$ 必然共点。\end{conclusion}\textbf{证明：}此定理为帕斯卡定理的射影对偶。在算子体系下，波利亚对偶（Pole-Polar Duality）不再是抽象几何直观，而是 $T$ 算子的代数属性。

\textbf{切线化为极线方程}
当直线与曲线相切于点 $A$ 时，代入参数多项式方程 $T(X,X)=0$ 意味着根的判别式退化（重根 $t=0$），故必须有 $E(A, \vec{v})=0$。因此过切点 $A$ 的切线上任意动点 $X=A+t\vec{v}$ 均满足 $T(A, X) = T(A, A+t\vec{v}) = T(A,A) + tE(A,\vec{v}) = 0$。
我们将方程 $T(P, X) = 0$ 广义定义为任意点 $P$ 的\textbf{极线}。
基于算子的对称交换律 $T(P, Q) = T(Q, P)$，立得极化对偶律：\textbf{点 $P$ 在点 $Q$ 的极线上 $\iff$ 点 $Q$ 在点 $P$ 的极线上}。

\textbf{顶点的极线即切点弦}
设外切六边形各边与曲线的切点分别为 $A_1, A_2, A_3, A_4, A_5, A_6$（它们构成了内接六边形）。
外切顶点 $V_1$ 是切线 $A_1$ 与切线 $A_6$ 的交点。
因为 $V_1$ 在 $A_1$ 的极线上，由对偶律有 $T(A_1, V_1)=0 \implies T(V_1, A_1)=0$；同理 $T(V_1, A_6)=0$。
由于极线方程 $T(V_1, X)=0$ 是关于 $X$ 的线性方程，这意味着\textbf{穿过两个切点 $A_1, A_6$ 的割线（切点弦）正是极点 $V_1$ 的极线}！同理，切点弦 $A_3A_4$ 是极点 $V_4$ 的极线。

\textbf{对角线对偶为帕斯卡交点}
考察主对角线 $V_1 V_4$，它连接了极点 $V_1$ 与 $V_4$。
设弦 $A_1A_6$ 与 $A_3A_4$ 相交于点 $P'$。由于 $P'$ 同时存在于 $V_1$ 和 $V_4$ 的极线上，反用对偶律推导知，$V_1$ 和 $V_4$ 也必然在 $P'$ 的极线上。
因此，\textbf{主对角线 $V_1V_4$ 整体正是交点 $P'$ 的极线}！

椭圆$\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$上一点$P$引出弦$PA,PB$，斜率分别为$k_1,k_2$，如果将$PA:y-y_0=k_1(x-x_0)$，$PB:y-y_0=k_2(x-x_0)$相乘，可以得到双直线方程，若选取关于原点对称的两个齐次原点$P,-P$，分别进行齐次化，将两个方程相加，得到：
\begin{align*}
    AB:&k_1k_2(x - x_0)^2 - (k_1+k_2)(x - x_0)(y - y_0) + (y - y_0)^2=0\\
    AB:&k_1k_2\frac{a^2}{b^2}(y + y_0)^2+ (k_1+k_2)(x + x_0)(y + y_0) + \frac{b^2}{a^2}(x + x_0)^2=0\\
\implies AB:&k_1k_2\left(x^2+\frac{a^2}{b^2}y^2+x_0^2+\frac{a^2}{b^2}y_0^2-2x_0x+2\frac{a^2}{b^2}y_0y\right)\\
&+(k_1+k_2)\left[(x - x_0+2x_0)(y - y_0+2y_0)-(x - x_0)(y - y_0)\right]\\
&+\left(\frac{b^2}{a^2}x^2+y^2+\frac{b^2}{a^2}x_0^2+y_0^2+2\frac{b^2}{a^2}x_0x-2y_0y\right)=0
\end{align*}
注意，这里的$x,y$分别指的是$x_A,x_B,y_A,y_B$，所以$\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$，那么这些二次项都可以消掉，得到一条直线方程，唯一地由$A,B$两点确定，所以这条直线就是$AB$的方程：
\begin{align*}
\Longleftrightarrow AB:&k_1k_2\left(2a^2-2x_0x+2y_0y\right)+(k_1+k_2)(2y_0x+2x_0y)+2b^2+2\frac{b^2}{a^2}x_0x-2y_0y=0\\
\Longleftrightarrow AB:&k_1k_2\left(a^2-x_0x+y_0y\right)+(k_1+k_2)(y_0x+x_0y)+b^2+\frac{b^2}{a^2}x_0x-y_0y=0\\
\Longleftrightarrow AB:&\left( \frac{b^2}{a^2} x_0 - k_1k_2 x_0 + (k_1+k_2) y_0 \right) x + \left( k_1k_2 \frac{a^2}{b^2} y_0 - y_0 + (k_1+k_2) x_0 \right) y + k_1k_2 a^2 + b^2 = 0
\end{align*}
对于双曲线 $\frac{x^2}{a^2}-\frac{y^2}{b^2}=1$ 上一点 $P(x_0,y_0)$ 引出弦 $PA,PB$，斜率分别为 $k_1,k_2$，将 $PA:y-y_0=k_1(x-x_0)$，$PB:y-y_0=k_2(x-x_0)$ 相乘，得到双直线方程。选取关于原点对称的两点 $P$ 和 $-P$ 分别进行齐次化，将两个方程相加，得到：
\begin{align*}
AB:&\ k_1k_2(x - x_0)^2 - (k_1+k_2)(x - x_0)(y - y_0) + (y - y_0)^2=0\\
AB:&\ -k_1k_2\frac{a^2}{b^2}(y + y_0)^2+ (k_1+k_2)(x + x_0)(y + y_0) - \frac{b^2}{a^2}(x + x_0)^2=0\\
\implies AB:&\ k_1k_2\left[x^2-\frac{a^2}{b^2}y^2+x_0^2-\frac{a^2}{b^2}y_0^2-2x_0x-2\frac{a^2}{b^2}y_0y\right]\\
&\quad +(k_1+k_2)\big[(x - x_0+2x_0)(y - y_0+2y_0)-(x - x_0)(y - y_0)\big]\\
&\quad +\left[-\frac{b^2}{a^2}x^2+y^2-\frac{b^2}{a^2}x_0^2+y_0^2-2\frac{b^2}{a^2}x_0x-2y_0y\right]=0\\
\Longleftrightarrow AB:&\ k_1k_2\left[2a^2-2x_0x-2\frac{a^2}{b^2}y_0y\right]+(k_1+k_2)(2y_0x+2x_0y)-2b^2-2\frac{b^2}{a^2}x_0x-2y_0y=0\\
\Longleftrightarrow AB:&\ k_1k_2\left[a^2-x_0x-\frac{a^2}{b^2}y_0y\right]+(k_1+k_2)(y_0x+x_0y)-b^2-\frac{b^2}{a^2}x_0x-y_0y=0\\
\Leftrightarrow k_1k_2 a^2 - b^2 &\ + \left( -\frac{b^2}{a^2} x_0  - k_1k_2 x_0 + (k_1+k_2) y_0 \right) x + \left( -k_1k_2 \frac{a^2}{b^2} y_0 - y_0 + (k_1+k_2) x_0 \right) y  = 0
\end{align*}
对于抛物线情形，我们可以选取$P$点和$x$轴上的无穷远点进行齐次化联立，对于无穷远点发出的两条直线，在有限空间里看就是平行于 $x$ 轴的水平线。因此，从 $P_\infty$ 连向 $A$ 和 $B$ 的线，就是直线 $y = y_A$ 和 $y = y_B$，即
\begin{align*}
(y-y_A)(y-y_B)=0&\Leftrightarrow \left[  (y + y_0) - (y_A + y_0) \right] \left[ (y + y_0) - (y_B + y_0)\right] = 0\\
&\Leftrightarrow \left[  (y + y_0) - \frac{2p}{k_1} \right] \left[ (y + y_0) - \frac{2p}{k_2}\right] = 0\\
&\Leftrightarrow k_1k_2(y + y_0)^2 - 2p(k_1+k_2)(y + y_0) + 4p^2 = 0
\end{align*}
利用$y^2=2px$消去二次项得到
\begin{align*}
\Longleftrightarrow AB:&k_1k_2\left(2px+2px_0+2y_0y\right)-2p(k_1+k_2)(y+y_0)+4p^2=0\\
\Longleftrightarrow AB:&k_1k_2\left(px+px_0+y_0y\right)-p(k_1+k_2)(y+y_0)+2p^2=0\\
\Longleftrightarrow AB:&p k_1k_2 x + \left( k_1k_2 y_0 - p(k_1+k_2) \right) y + p k_1k_2 x_0 - p(k_1+k_2) y_0 + 2p^2 = 0
\end{align*}
总结一下，弦$AB$的方程可以写成（记$P*$为$P$关于$x$轴的对称点）：
\begin{itemize}

    \item 抛物线：$ k_1k_2 \cdot T(P^*, X) + p (k_1+k_2) \cdot (y - y_{P^*}) - 2p^2 = 0 $
\end{itemize}

\begin{conclusion}{直线系统的极简算子表达}{}
已知二次曲线的方程为 ${C}: T(X,X)=0$。
\begin{enumerate}
    \item \textbf{割线方程$AB$}：$ F_{AB}(X) = T(X,A) + T(X,B) - T(A,B) = 0 $
    \item \textbf{点 $A$ 处的切线方程}：$ T(X,A) = 0 $
    \item \textbf{切点弦方程（极线）}：从曲线外一点 $P$ 引两条切线，切点为 $A,B$：$AB: T(X,P) = 0 $
    \item \textbf{中点弦方程（秒杀点差法）}：过曲线内一点 $P$ 作一条弦，若 $P$ 恰好是这条弦的中点，则该弦方程:$ T(X,P) = T(P,P) $
\end{enumerate}
\end{conclusion}
\begin{itemize}
\item 直线上任意动点 $X$ 都必定是 $A, B$ 的仿射组合，即 $X = \alpha A + \beta B$，且必定满足恒等约束 $\alpha + \beta = 1$。
我们对 $X$ 分别用 $A$ 和 $B$ 进行极化“探测”，利用分配律展开：
\begin{align*}
    T(X,A) &= T(\alpha A + \beta B, A) = \alpha T(A,A) + \beta T(B,A) = \beta T(A,B) \\
    T(X,B) &= T(\alpha A + \beta B, B) = \alpha T(A,B) + \beta T(B,B) = \alpha T(A,B)
\end{align*}
为了利用 $\alpha + \beta = 1$ 这个仿射不变量消去参数，我们将两式直接相加：
$$ T(X,A) + T(X,B) = (\alpha + \beta) T(A,B) = 1 \cdot T(A,B) $$
移项即得 $T(X,A) + T(X,B) - T(A,B) = 0$。
\item 切线:想象割线上的点 $B$ 慢慢靠近点 $A$，直到重合。上面的割线方程就变成了 $T(X,A) + T(X,A) - T(A,A) = 0$。因为 $T(A,A)=0$，方程化简为 $2T(X,A) = 0 \implies T(X,A) = 0$。顺理成章！
\item 切点弦:切线 $PA$ 的方程是 $T(X,A)=0$。因为外面的点 $P$ 在这条切线上，把 $X=P$ 代入必然成立，得到 $T(P,A)=0$。根据算子的乘法交换律，这就等价于 $T(A,P)=0$！同理，从另一条切线 $PB$ 可以得到 $T(B,P)=0$。既然点 $A$ 和点 $B$ 的坐标都完美满足方程 $T(X,P)=0$，且它是个关于 $X$ 的一次方程，那它当然就是切点弦 $AB$ 的方程！
\item 中点弦:设直线上任意一点 $X = P + \vec{v}$，即方向向量 $\vec{v} = X - P$。因为 $P$ 是中点，交点参数方程的两根之和必为 $0$（韦达定理），这就要求一次项系数为 $0$，即 $E(P, \vec{v}) = 0$。
代入展开：$E(P, X-P) = 0 \implies E(P,X) - E(P,P) = 0$。
把内积 $E$ 换成极化 $T$（利用恒等式 $E = T + 1$），两边的 $1$ 直接抵消，瞬间得到 $T(X,P) = T(P,P)$。满篇平方差因式分解的“点差法”被一行秒杀！
\end{itemize}

\begin{conclusion}{四大经典交点曲线系一键生成}{}
设原二次曲线为 $\mathcal{C}: T(X,X)=0$。
\begin{enumerate}
    \item \textbf{四交点系}：过 $C$ 上 $A, B, C, D$ 四点的新二次曲线族：$$ T(X,X) + \lambda \cdot \underbrace{F_{AB}(X)}_{\text{割线}AB} \cdot \underbrace{F_{CD}(X)}_{\text{割线}CD} = 0 $$
    \item \textbf{三交一切系}：在 $A$ 点相切，且过 $C, D$ 的曲线系（让上面公式中的 $B \to A$）：$$ T(X,X) + \lambda \cdot \underbrace{T(X,A)}_{\text{A的切线}} \cdot \underbrace{F_{CD}(X)}_{\text{割线}CD}  = 0 $$
    \item \textbf{双切线系}：与 $C$ 在 $A, B$ 两点发生双重相切的曲线系（设切点弦 $AB$ 对应的极点为 $P$）：$$ T(X,X) + \lambda \cdot \big[T(X,P)\big]^2 = 0 $$
    \item \textbf{超密切系}：在 $A$ 点发生四阶接触（极限相切）的曲线系（让双切系中 $P \to A$）：$$ T(X,X) + \lambda \cdot \big[T(X,A)\big]^2 = 0 $$
\end{enumerate}
\end{conclusion}

\begin{example}{}{}
椭圆外一点$P$向椭圆做两条切线，切线对的联合方程？
\end{example}
\begin{solution}
设点 $X$ 为从点 $P$ 引出的切线上异于 $P$ 的任意一点。此时直线 $PX$ 与椭圆有且仅有一个交点。
设直线 $PX$ 上的动点可表示为 $Y = P + t\vec{v}$，其中方向向量为 $\vec{v} = X - P$。
将点 $Y$ 的坐标代入椭圆极化方程 $T(Y,Y) = 0$ 中，利用双线性分配律展开可得
\[ T(P + t\vec{v}, P + t\vec{v}) = E(\vec{v},\vec{v})t^2 + 2E(P,\vec{v})t + T(P,P) = 0. \]
由于直线 $PX$ 为椭圆的切线，该关于 $t$ 的一元二次方程必然存在两个相等的实数根。因此，其判别式 $\Delta$ 必须为零，即
\[ \Delta = 4\big[E(P,\vec{v})\big]^2 - 4E(\vec{v},\vec{v})T(P,P) = 0. \]
化简可得 $\big[E(P,\vec{v})\big]^2 = E(\vec{v},\vec{v})T(P,P)$。
将方向向量 $\vec{v} = X - P$ 代入内积算子中展开。对于一次项，有
\[ E(P, \vec{v}) = E(P, X - P) = E(P,X) - E(P,P) = T(P,X) - T(P,P). \]
对于二次项，有
\[ E(\vec{v}, \vec{v}) = E(X - P, X - P) = E(X,X) - 2E(P,X) + E(P,P) = T(X,X) - 2T(P,X) + T(P,P). \]
将上述展开式代回判别式为零的等式中，得到
\[ \big[T(P,X) - T(P,P)\big]^2 = \big[T(X,X) - 2T(P,X) + T(P,P)\big] T(P,P). \]
将等式两边完全展开，可得
\[ T(P,X)^2 - 2T(P,P)T(P,X) + T(P,P)^2 = T(P,P)T(X,X) - 2T(P,P)T(P,X) + T(P,P)^2. \]
消去等式两边相同的同类项 $-2T(P,P)T(P,X)$ 与 $T(P,P)^2$，最终得到切线对的联合算子方程为
\[ T(P,X)^2 - T(P,P)T(X,X) = 0. \]
\end{solution}

\begin{example}{}{}
过椭圆上一点$P(x_0,y_0)$引弦$PA,PB$，试用$x_0,y_0,F(P,A,X),F(P,B,X)$表示弦$AB$方程。（注：$F(A,B,X):y(x_A-x_B)+x(y_B-y_A)-(x_Ay_B-x_By_A)$）
\end{example}
\begin{solution}
已知椭圆方程为 $T(X,X) = 0$。由于点 $P(x_0,y_0)$ 在椭圆上，其极化自内积满足 $T(P,P) = 0$。
过点 $P$ 的两条割线方程可分别表示为 $F(P,A,X)=0$ 与 $F(P,B,X)=0$。
考虑经过点 $P, P$（即在点 $P$ 处与椭圆相切）以及点 $A, B$ 的二次曲线系。根据四交点曲线系原理，该曲线系必定可由原椭圆方程与退化线对 $PA \cup PB$ 的方程线性组合而成。同时，该曲线系也包含了由点 $P$ 处的切线 $T(P,X)=0$ 与割线 $L_{AB}(X)=0$ 构成的退化二次曲线。因此，必然存在唯一实常数 $\mu$，使得如下多项式恒等式成立：
\[ T(X,X) - \mu F(P,A,X)F(P,B,X) \equiv T(P,X) L_{AB}(X). \]
为求解常数 $\mu$，在切线 $T(P,X)=0$ 上选取一点 $Q$，使得 $Q \neq P$。由椭圆在点 $P$ 处的切线方向向量为 $\vec{t} = (a^2 y_0, -b^2 x_0)$，可构造切线上一点
\[ Q = P + \vec{t} = (x_0 + a^2 y_0, y_0 - b^2 x_0). \]
将点 $X = Q$ 代入等式。由于点 $Q$ 位于切线 $T(P,X)=0$ 上，必有 $T(P,Q) = 0$，等式右侧归零：
\[ T(Q,Q) - \mu F(P,A,Q)F(P,B,Q) = 0. \]
对于 $T(Q,Q)$，展开可得
\[ T(Q,Q) = T(P+\vec{t}, P+\vec{t}) = T(P,P) + 2E(P,\vec{t}) + E(\vec{t},\vec{t}). \]
由于 $T(P,P)=0$ 且 $\vec{t}$ 为切线方向向量满足 $E(P,\vec{t})=0$，可直接计算得 $T(Q,Q) = E(\vec{t},\vec{t}) = a^2b^2$。
将此结果代入，由于点 $Q$ 不在割线 $PA$ 与 $PB$ 上，有 $F(P,A,Q)F(P,B,Q) \neq 0$，解得
\[ \mu = \frac{a^2b^2}{F(P,A,Q)F(P,B,Q)}. \]
将求得的常数 $\mu$ 代回原恒等式，并隔离出未知函数 $L_{AB}(X)$，得到弦 $AB$ 方程为
\[ L_{AB}(X) = \frac{F(P,A,Q) F(P,B,Q) T(X,X) - a^2b^2 F(P,A,X) F(P,B,X)}{F(P,A,Q) F(P,B,Q) T(P,X)} = 0. \]
已知由四交点曲线系确定的多项式恒等式为：
\[ T(X,X) - \mu F(P,A,X)F(P,B,X) \equiv T(P,X) L_{AB}(X) \]
由于除式（切线）是一次多项式，被除式是二次多项式，商式必定是一次多项式。我们直接设绝对坐标系下商式所代表的直线 $AB$ 的算子为 $L_{AB}(X) = Ax + By + C$。
在绝对坐标系下展开等式左右两端，我们只提取“首项”（纯二次项 $x^2, y^2$）和“尾项”（常数项）进行比对，直接跳过所有繁琐的一次项与交叉项：
等式左端（被除式）：
\begin{align*}
    \text{左端} &= \left(\frac{x^2}{a^2} + \frac{y^2}{b^2} - 1\right) - \mu \big[y - y_0 - k_1(x-x_0)\big] \big[y - y_0 - k_2(x-x_0)\big] \\
    &= \left(\frac{1}{a^2} - \mu k_1k_2\right) x^2 + \cdots xy + \left(\frac{1}{b^2} - \mu\right) y^2 + \cdots x + \cdots y \\
    &\quad - 1 - \mu(k_1x_0 - y_0)(k_2x_0 - y_0)
\end{align*}
等式右端（除式 $\times$ 商式）：
\begin{align*}
    \text{右端} &= \left(\frac{x_0}{a^2}x + \frac{y_0}{b^2}y - 1\right) (Ax + By + C) \\
    &= \frac{x_0}{a^2}A x^2 + \cdots xy + \frac{y_0}{b^2}B y^2 + \cdots x + \cdots y - C
\end{align*}
既然多项式恒等，同类项系数必然绝对相等。我们直接“掐头去尾”比对系数，并顺势提出公因数 $\mu$：
\begin{align*}
    [x^2] \text{ 项}: \quad \frac{x_0}{a^2}A &= \frac{1}{a^2} - \mu k_1k_2 \implies A = \frac{1 - \mu a^2 k_1k_2}{x_0} = \frac{\mu}{x_0}\left(\frac{1}{\mu} - a^2 k_1k_2\right)\\
    [y^2] \text{ 项}: \quad \frac{y_0}{b^2}B &= \frac{1}{b^2} - \mu \implies B = \frac{1 - \mu b^2}{y_0} = \frac{\mu}{y_0}\left(\frac{1}{\mu} - b^2\right)\\
    \text{常数项}: \quad -C &= - 1 - \mu(k_1x_0 - y_0)(k_2x_0 - y_0) \implies C = \mu\left[ \frac{1}{\mu} + (k_1x_0 - y_0)(k_2x_0 - y_0) \right]
\end{align*}
将已知的逆常数 $\frac{1}{\mu} = k_1k_2\frac{a^2}{b^2}y_0^2 + (k_1+k_2)x_0y_0 + \frac{b^2}{a^2}x_0^2$ 代入中括号内，并利用点 $P$ 在椭圆上的天然降次条件 $\frac{x_0^2}{a^2} + \frac{y_0^2}{b^2} = 1$ 进行代数对消：
对于系数 $A$：
\begin{align*}
    A &= \frac{\mu}{x_0} \left[ k_1k_2\frac{a^2}{b^2}y_0^2 + (k_1+k_2)x_0y_0 + \frac{b^2}{a^2}x_0^2 - a^2 k_1k_2 \right] \\
    &= \frac{\mu}{x_0} \left[ k_1k_2 a^2 \left(\frac{y_0^2}{b^2} - 1\right) + (k_1+k_2)x_0y_0 + \frac{b^2}{a^2}x_0^2 \right] \\
    &= \frac{\mu}{x_0} \left[ k_1k_2 a^2 \left(-\frac{x_0^2}{a^2}\right) + (k_1+k_2)x_0y_0 + \frac{b^2}{a^2}x_0^2 \right] \\
    &= \mu \left( \frac{b^2}{a^2}x_0 - k_1k_2 x_0 + (k_1+k_2)y_0 \right)
\end{align*}
对于系数 $B$：
\begin{align*}
    B &= \frac{\mu}{y_0} \left[ k_1k_2\frac{a^2}{b^2}y_0^2 + (k_1+k_2)x_0y_0 + \frac{b^2}{a^2}x_0^2 - b^2 \right] \\
    &= \frac{\mu}{y_0} \left[ k_1k_2\frac{a^2}{b^2}y_0^2 + (k_1+k_2)x_0y_0 + b^2 \left(\frac{x_0^2}{a^2} - 1\right) \right] \\
    &= \frac{\mu}{y_0} \left[ k_1k_2\frac{a^2}{b^2}y_0^2 + (k_1+k_2)x_0y_0 + b^2 \left(-\frac{y_0^2}{b^2}\right) \right] \\
    &= \mu \left( k_1k_2\frac{a^2}{b^2}y_0 - y_0 + (k_1+k_2)x_0 \right)
\end{align*}
对于常数项 $C$（注意此处的交叉项绝妙对消）：
\begin{align*}
    C &= \mu \left[ \left( k_1k_2\frac{a^2}{b^2}y_0^2 + (k_1+k_2)x_0y_0 + \frac{b^2}{a^2}x_0^2 \right) + (k_1x_0 - y_0)(k_2x_0 - y_0) \right] \\
    &= \mu \left[ k_1k_2\frac{a^2}{b^2}y_0^2 + (k_1+k_2)x_0y_0 + \frac{b^2}{a^2}x_0^2 + k_1k_2x_0^2 - (k_1+k_2)x_0y_0 + y_0^2 \right] \\
    &= \mu \left[ k_1k_2 a^2 \left( \frac{y_0^2}{b^2} + \frac{x_0^2}{a^2} \right) + b^2 \left( \frac{x_0^2}{a^2} + \frac{y_0^2}{b^2} \right) \right] \\
    &= \mu \left( k_1k_2 a^2 + b^2 \right)
\end{align*}
将求得的 $A, B, C$ 拼接在一起，得到商式 $L_{AB}(X)$：
\begin{align*}
    L_{AB}(X) = \mu\left[\left( \frac{b^2}{a^2} x_0 - k_1k_2 x_0 + (k_1+k_2) y_0 \right) x + \left( k_1k_2 \frac{a^2}{b^2} y_0 - y_0 + (k_1+k_2) x_0 \right) y + k_1k_2 a^2 + b^2\right]
\end{align*}
令弦 $AB$ 的方程 $L_{AB}(X) = 0$。由于对于给定的非退化二次曲线及割线，有 $\mu \neq 0$，等式两边可以直接除以非零常数 $\mu$，一气呵成地得到绝对坐标系下的直线 $AB$ 表达式：
\begin{align*}
    \Longleftrightarrow AB:&\left( \frac{b^2}{a^2} x_0 - k_1k_2 x_0 + (k_1+k_2) y_0 \right) x + \left( k_1k_2 \frac{a^2}{b^2} y_0 - y_0 + (k_1+k_2) x_0 \right) y + k_1k_2 a^2 + b^2 = 0
\end{align*}
\end{solution}


\begin{example}{}{}
过非椭圆一点$P(x_0,y_0)$引割线$PAB,PCD$，试用$x_0,y_0,F_{AB}(X),F_{CD}(X),T(P,X)$表示割线$AC,BD,AD,BC$的方程。
\end{example}
\begin{solution}
设过极点 $P$ 的割线 $PAB$ 和 $PCD$ 的算子方程分别为 $F_{AB}(X) = 0$ 与 $F_{CD}(X) = 0$。
选取平面内相交于点 $P$ 的三条不重合直线对应的算子 $T(P,X), F_{AB}(X)$ 与 $F_{CD}(X)$ 作为一次线性函数空间的基底。
由于对角割线 $AC$ 不经过点 $P$，其算子方程可唯一地表示为
\[ F_{AC}(X) = T(P,X) + \mu_1 F_{AB}(X) + \mu_2 F_{CD}(X) = 0. \]
首先，将点 $X = A$ 代入该基底方程。因点 $A$ 位于割线 $PAB$ 上，满足 $F(A,B,A) = 0$。方程化简为 $T(P,A) + \mu_2 F(C,D,A) = 0$，解得 $\mu_2 = -\frac{T(P,A)}{F(C,D,A)}$。
其次，将点 $X = C$ 代入基底方程。因点 $C$ 位于割线 $PCD$ 上，满足 $F(C,D,C) = 0$。方程化简为 $T(P,C) + \mu_1 F(A,B,C) = 0$，解得 $\mu_1 = -\frac{T(P,C)}{F(A,B,C)}$。
令 $k_1 = \frac{T(P,C)}{F(A,B,C)}$ 且 $k_2 = \frac{T(P,A)}{F(C,D,A)}$，则割线 $AC$ 的方程为
\[ F_{AC}(X) \equiv T(P,X) - k_1 F_{AB}(X) - k_2 F_{CD}(X) = 0. \]
接下来探讨直线 $BD$ 与直线 $AC$ 的透射映射关系。点 $B$ 为直线 $PA$ 与椭圆的交点，故点 $B$ 可表示为 $P$ 与 $A$ 的仿射组合。设 $B = \frac{P + \lambda A}{1+\lambda}$。
由于点 $B$ 位于椭圆上，必须满足 $T(B,B) = 0$。利用算子的双线性分配律展开可得
\[ \frac{T(P,P) + 2\lambda T(P,A) + \lambda^2 T(A,A)}{(1+\lambda)^2} = 0. \]
已知点 $A$ 在椭圆上，有 $T(A,A) = 0$。由于 $B \neq A$，必有 $\lambda \neq 0$，等式化简为 $T(P,P) + 2\lambda T(P,A) = 0$，解得 $\lambda = -\frac{T(P,P)}{2T(P,A)}$。
将其代入仿射组合公式并同乘分母，得到点 $B$ 的齐次坐标形式为 $B \propto 2T(P,A)P - T(P,P)A$。
设直线 $BD$ 的方程为一次算子 $F_{BD}(X) = 0$。将点 $B$ 的坐标代入方程中，利用一次算子的线性分配律，可得
\[ F_{BD}\big( 2T(P,A)P - T(P,P)A \big) = 2T(P,A)F_{BD}(P) - T(P,P)F_{BD}(A) = 0. \]
根据极化算子的对称性 $T(P,A)=T(A,P)$，上式表明点 $A$ 满足如下关于动点 $X$ 的一次线性方程
\[ 2F_{BD}(P)T(P,X) - T(P,P)F_{BD}(X) = 0. \]
同理，点 $D$ 为直线 $PC$ 与椭圆的交点，可推知点 $D$ 的齐次坐标满足 $D \propto 2T(P,C)P - T(P,P)C$。将其代入 $F_{BD}(D) = 0$，同理可证点 $C$ 亦满足上述一次线性方程。
由于点 $A$ 与点 $C$ 均满足该方程，且两点不重合，则该方程必定等价于过这两点的直线 $AC$ 的方程 $F_{AC}(X) = 0$。
利用几何对称性，将该透射映射关系反向应用，直线 $BD$ 的方程可由直线 $AC$ 的拉回映射得出，即
\[ F_{BD}(X) \propto 2F_{AC}(P)T(P,X) - T(P,P)F_{AC}(X) = 0. \]
计算常数 $F_{AC}(P)$ 的值。将点 $P$ 代入 $F_{AC}(X)$ 的表达式中，由于点 $P$ 同时位于直线 $AB$ 与 $CD$ 上，满足 $F(A,B,P) = 0$ 且 $F(C,D,P) = 0$，代入后可得
\[ F_{AC}(P) = T(P,P) - 0 - 0 = T(P,P). \]
将 $F_{AC}(P) = T(P,P)$ 与 $F_{AC}(X)$ 的完整表达式代入 $F_{BD}(X)$ 的恒等式中，有
\[ F_{BD}(X) \propto 2T(P,P)T(P,X) - T(P,P) \left[ T(P,X) - k_1 F_{AB}(X) - k_2 F_{CD}(X) \right] = 0. \]
由于点 $P$ 位于椭圆外，满足极化自内积 $T(P,P) \neq 0$。将等式两边同除以 $T(P,P)$ 并合并同类项，得到
\[ 2T(P,X) - T(P,X) + k_1 F_{AB}(X) + k_2 F_{CD}(X) = 0. \]
最终得出对角割线 $BD$ 的方程为
\[ F_{BD}(X) \equiv T(P,X) + k_1 F_{AB}(X) + k_2 F_{CD}(X) = 0. \]
接下来，同理推导对角割线 $AD$ 与 $BC$ 的方程。
由于割线 $AD$ 不经过点 $P$，其算子方程同样可由这三个一次算子构成的基底线性表出：
\[ F_{AD}(X) = T(P,X) + \mu_1 F_{AB}(X) + \mu_2 F_{CD}(X) = 0. \]
首先，将点 $X = A$ 代入该方程。因点 $A$ 位于割线 $PAB$ 上，满足 $F(A,B,A) = 0$。方程化简为 $T(P,A) + \mu_2 F(C,D,A) = 0$，解得
\[ \mu_2 = -\frac{T(P,A)}{F(C,D,A)} = -k_2. \]
其次，将点 $X = D$ 代入。因点 $D$ 位于割线 $PCD$ 上，满足 $F(C,D,D) = 0$。方程化简为 $T(P,D) + \mu_1 F(A,B,D) = 0$，解得
\[ \mu_1 = -\frac{T(P,D)}{F(A,B,D)}. \]
为了计算 $\mu_1$，利用前文求点 $B$ 坐标时完全相同的透射映射原理：点 $D$ 为直线 $PC$ 与椭圆的另一交点，其代数结构与点 $B$ 完全对称，故点 $D$ 的齐次坐标必然满足 $D \propto 2T(P,C)P - T(P,P)C$。
将其分别代入 $T(P,X)$ 与 $F_{AB}(X)$ 算子中，利用一次算子的双线性分配律展开：
\begin{align*}
T(P,D) &\propto 2T(P,C)T(P,P) - T(P,P)T(P,C) = T(P,P)T(P,C). \\
F(A,B,D) &\propto 2T(P,C)F(A,B,P) - T(P,P)F(A,B,C).
\end{align*}
因为极点 $P$ 位于割线 $PAB$ 上，显然满足方程 $F(A,B,P) = 0$，故：
\[ F(A,B,D) \propto -T(P,P)F(A,B,C). \]
将两式相除，非零的极化自内积常数 $T(P,P)$ 在分子分母完美对消并翻转负号：
\[ \mu_1 = -\frac{T(P,D)}{F(A,B,D)} = -\frac{T(P,P)T(P,C)}{-T(P,P)F(A,B,C)} = \frac{T(P,C)}{F(A,B,C)} = k_1. \]
将求得的系数代回，瞬间得到割线 $AD$ 的纯算子方程为：
\[ F_{AD}(X) \equiv T(P,X) + k_1 F_{AB}(X) - k_2 F_{CD}(X) = 0. \]
最后，求解割线 $BC$。利用绝对对称的代数结构，设 $F_{BC}(X) = T(P,X) + \nu_1 F_{AB}(X) + \nu_2 F_{CD}(X) = 0$。
将点 $X = C$ 代入，由 $F(C,D,C) = 0$ 解得 $\nu_1 = -\frac{T(P,C)}{F(A,B,C)} = -k_1$。\\
将点 $X = B$ 代入，由 $F(A,B,B) = 0$ 解得 $\nu_2 = -\frac{T(P,B)}{F(C,D,B)}$。\\
将前文求出的点 $B$ 的齐次坐标 $B \propto 2T(P,A)P - T(P,P)A$ 再次代入，并利用点 $P$ 在割线 $PCD$ 上满足 $F(C,D,P) = 0$ 的性质展开：
\begin{align*}
T(P,B) &\propto 2T(P,A)T(P,P) - T(P,P)T(P,A) = T(P,P)T(P,A) \\
F(C,D,B) &\propto 2T(P,A)\underbrace{F(C,D,P)}_{=0} - T(P,P)F(C,D,A) = -T(P,P)F(C,D,A)
\end{align*}
两式相除，同样发生 $T(P,P)$ 的对消与负号翻转：
\[ \nu_2 = -\frac{T(P,B)}{F(C,D,B)} = -\frac{T(P,P)T(P,A)}{-T(P,P)F(C,D,A)} = \frac{T(P,A)}{F(C,D,A)} = k_2. \]
因此，得到割线 $BC$ 的算子方程为：
\[ F_{BC}(X) \equiv T(P,X) - k_1 F_{AB}(X) + k_2 F_{CD}(X) = 0. \]
\textbf{【四线大一统与完全四边形定理的终极降维】}：至此，由极点 $P$ 引出的完整相交割线系统（即圆锥曲线内接四边形 $ABCD$），其四条连线（对角线与对边）的算子方程被高度对称地统摄为：
\begin{align*}
    AC:&\quad T(P,X) - k_1 F_{AB}(X) - k_2 F_{CD}(X) = 0 \\
    BD:&\quad T(P,X) + k_1 F_{AB}(X) + k_2 F_{CD}(X) = 0 \\
    AD:&\quad T(P,X) + k_1 F_{AB}(X) - k_2 F_{CD}(X) = 0 \\
    BC:&\quad T(P,X) - k_1 F_{AB}(X) + k_2 F_{CD}(X) = 0 
\end{align*}
仔细观察这四组极其纯粹的线性组合方程，我们在不借助任何复杂几何定理与坐标强算的情况下，只需对它们进行最简单的代数相加：
\begin{enumerate}
    \item 将对角线 $AC$ 与 $BD$ 的方程相加，一次算子 $F_{AB}(X)$ 与 $F_{CD}(X)$ 瞬间发生完美的正负代数对消：
    \[ F_{AC}(X) + F_{BD}(X) = 2T(P,X) = 0 \implies T(P,X) = 0. \]
    这意味着对角线交点 $Q = AC \cap BD$ 必定同时满足上述两个方程，代入和式即得 $T(P,Q) = 0$。交点 $Q$ 绝对地钉死在基点 $P$ 的极线上！
    
    \item 同理，将侧边弦 $AD$ 与 $BC$ 的方程相加：
    \[ F_{AD}(X) + F_{BC}(X) = 2T(P,X) = 0 \implies T(P,X) = 0. \]
    这意味着对边交点 $R = AD \cap BC$ 同样满足 $T(P,R) = 0$，即交点 $R$ 也绝对钉死在点 $P$ 的极线上！
\end{enumerate}
\end{solution}

\begin{example}{蝴蝶定理}{}
已知椭圆$\frac{x^2}{a^2}+\frac{y^2}{b^2}=1$，点$P$是非椭圆上一点，$A(x_1,y_1),B(x_2,y_2)$在椭圆上，连接$AP,BP$交椭圆于$C,D$，求$CD$的方程。
\end{example}
\begin{solution}
设点 $C$ 为直线 $AP$ 与椭圆的交点。由于 $P, A, C$ 三点共线，可设点 $C$ 为 $P$ 与 $A$ 的仿射组合 $C = \frac{P + \lambda A}{1+\lambda}$。
由于点 $C$ 位于椭圆上，其极化自内积必须为零，即 $T(C,C) = 0$。利用双线性分配律展开，可得
\[ \frac{T(P,P) + 2\lambda T(P,A) + \lambda^2 T(A,A)}{(1+\lambda)^2} = 0. \]
已知点 $A$ 在椭圆上，有 $T(A,A) = 0$。由于 $C \neq A$，必有 $\lambda \neq 0$。等式化简为 $T(P,P) + 2\lambda T(P,A) = 0$，解得常数 $\lambda = -\frac{T(P,P)}{2T(P,A)}$。
将 $\lambda$ 的值代回仿射组合公式并同乘分母化简，可得点 $C$ 的齐次坐标形式为
\[ C \propto 2T(P,A)P - T(P,P)A. \]
同理，由于点 $B, P, D$ 共线且 $B, D$ 在椭圆上，通过完全对称的步骤可求得点 $D$ 的齐次坐标为
\[ D \propto 2T(P,B)P - T(P,P)B. \]
设待求弦 $CD$ 的算子方程为一次线性方程 $L_{CD}(X) = 0$。
将点 $C$ 的齐次坐标代入该方程，利用一次算子的线性分配律，可得
\[ L_{CD}(C) = L_{CD}\big(2T(P,A)P - T(P,P)A\big) = 2T(P,A)L_{CD}(P) - T(P,P)L_{CD}(A) = 0. \]
根据极化算子的对称性 $T(P,A) = T(A,P)$，整理上式可得点 $A$ 满足如下关于动点 $X$ 的一次线性方程：
\[ 2L_{CD}(P)T(P,X) - T(P,P)L_{CD}(X) = 0. \]
同理，将点 $D$ 的齐次坐标代入 $L_{CD}(D) = 0$ 中，展开可得
\[ 2T(P,B)L_{CD}(P) - T(P,P)L_{CD}(B) = 0. \]
这表明点 $B$ 亦满足上述同一一次线性方程。
由于直线 $AB$ 上的两个不同点 $A$ 和 $B$ 均满足该方程，且该方程为一次线性方程，因此该方程必定等价于过 $A, B$ 两点的直线 $AB$ 的算子方程。由此可建立起弦对射的拉回恒等式：
\[ L_{AB}(X) \propto 2L_{CD}(P)T(P,X) - T(P,P)L_{CD}(X). \]
利用点集透射映射的对合对称性，将上述推导反向应用，直线 $CD$ 的方程可由直线 $AB$ 的算子函数唯一表示为：
\[ L_{CD}(X) \propto 2L_{AB}(P)T(P,X) - T(P,P)L_{AB}(X) = 0. \]
设已知弦 $AB$ 的方程为面积算子 $L_{AB}(X) = F_{AB}(X) = 0$。将其函数表达式以及其在定点 $P$ 处的取值 $L_{AB}(P) = F(A,B,P)$ 代入上述映射方程中，可得：
\[ 2 F(A,B,P) T(P,X) - T(P,P) F_{AB}(X) = 0. \]
移项整理，最终得出弦 $CD$ 的方程为
\[ T(P,P) F_{AB}(X) = 2 T(P,X) F(A,B,P). \]
\end{solution}

\begin{example}{}{}
已知 \(A,B,C,D\) 为椭圆 \(\frac{x^2}{4} + \frac{y^2}{3} = 1\) 上四点，\(AC \cap BD = Q, Q(1,1)\)，且 \(\overrightarrow{AQ} = 2\overrightarrow{QC}, \overrightarrow{BQ} = 2\overrightarrow{QD}\)，求直线 \(CD，BC\) 方程；
\end{example}
\begin{solution}
对于已知二次曲线椭圆 $\frac{x^2}{4} + \frac{y^2}{3} = 1$，定义其内积算子 $E(M, N) = \frac{x_M x_N}{4} + \frac{y_M y_N}{3}$ 与外积算子 $W(M, N) = x_M y_N - x_N y_M$。椭圆方程等价于 $E(X, X) = 1$。
对于定点 $Q(1,1)$，代入算子定义可得其内积常数为 $E(Q, Q) = \frac{1}{4} + \frac{1}{3} = \frac{7}{12}$。\\
（1）首先求解直线 $CD$ 的方程。由已知条件 $\overrightarrow{AQ} = 2\overrightarrow{QC}$ 可知 $A = 3Q - 2C$。代入 $E(A,A) = 1$可得
\begin{align*}
E(3Q - 2C, 3Q - 2C) &= 9E(Q,Q) - 12E(Q,2C) + 4E(C,C)\\
&= 9E(Q,Q) - 24E(Q,C) + 4 = 1 \\
\implies& 8E(Q,C) - 3E(Q,Q) - 1 = 0
\end{align*}
同理点 $D$ 同样满足 $8E(Q,D) - 3E(Q,Q) - 1 = 0$。
由于点 $C$ 与点 $D$ 为椭圆上相异两点，且均满足关于动点 $X$ 的一次线性方程，故直线 $CD$ 的方程即为
\[ 8E(Q,X) - 3E(Q,Q) - 1 = 0 \]
将 $E(Q,Q) = \frac{7}{12}$ 代入，得$16E(Q,X) - 11 = 0$。展开 $E(Q,X) = \frac{x}{4} + \frac{y}{3}$，可得 $16\left(\frac{x}{4} + \frac{y}{3}\right) - 11 = 0$。即$12x + 16y - 33 = 0$。\\
（2）其次求解直线 $BC$ 的方程。对此我们必须先求$AB$方程。由前述 $A = 3Q - 2C$ 移项可得 $C = \frac{3}{2}Q - \frac{1}{2}A$。将其代入 $E(C,C) = 1$ 展开，并注意$E(A,A) = 1$
\[  9E(Q,Q) - 6E(Q,A) + E(A,A) = 4\Leftrightarrow 2E(Q,A) - 3E(Q,Q) + 1 = 0 \]
同理，由 $D = \frac{3}{2}Q - \frac{1}{2}B$ 结合 $E(D,D) = 1$ 可得 $2E(Q,B) - 3E(Q,Q) + 1 = 0$。因此 $AB$ 的方程为 $2E(Q,X) - 3E(Q,Q) + 1 = 0$。代入 $E(Q,Q) = \frac{7}{12}$ 化简得 $8E(Q,X) - 3 = 0$。\\
（3）引入参数 $\lambda$，过椭圆上 $A, B, C, D$ 四点的所有二次曲线
\begin{align*}
& E(X,X) - 1 + \lambda \big( 8E(Q,X) - 3 \big)\big( 16E(Q,X) - 11 \big) = 0\\
\Longleftrightarrow & E(X,X) - 1 + \lambda \left( 128E(Q,X)^2 - 136E(Q,X) + 33 \right) = 0
\end{align*}
根据拉格朗日恒等式，内积算子与外积算子满足绝对代数关系
\[ W(Q,X)^2 = a^2b^2 \Big( E(Q,Q)E(X,X) - E(Q,X)^2 \Big)=12 \Big( E(Q,Q)E(X,X) - E(Q,X)^2 \Big)\]
代入 $E(Q,Q) = \frac{7}{12}$，得到
\[ W(Q,X)^2 = 12 \left( \frac{7}{12}E(X,X) - E(Q,X)^2 \right) = 7E(X,X) - 12E(Q,X)^2. \]
由此反解出纯二次项 $E(X,X)$：
\[ E(X,X) = \frac{12}{7}E(Q,X)^2 + \frac{1}{7}W(Q,X)^2. \]
将其代入四交点曲线系方程中，合并关于 $E(Q,X)$ 的同类项，方程可转化为
\[ \left( \frac{12}{7} + 128\lambda \right) E(Q,X)^2 - 136\lambda E(Q,X) + (33\lambda - 1) + \frac{1}{7}W(Q,X)^2 = 0\]
当该曲线系退化为交叉直线对 $AD \cup BC$ 时，方程在实数域内必然可分解为两个一次多项式的乘积。由于上述方程中不存在 $E(Q,X)$ 与 $W(Q,X)$ 的交叉项，关于 $E(Q,X)$ 的二次三项式部分必须构成完全平方式的相反数，即要求其判别式 $\Delta$ 必须为 $0$：
\[ \Delta = (-136\lambda)^2 - 4 \left( \frac{12}{7} + 128\lambda \right) (33\lambda - 1) = 0. \]
方程两边同除以 $4$ 并展开乘积，得
\[ 4624\lambda^2 - \left( 4224\lambda^2 + \frac{396}{7}\lambda - 128\lambda - \frac{12}{7} \right) = 0. \]
化简合并同类项可得 $400\lambda^2 + \frac{500}{7}\lambda + \frac{12}{7} = 0$，等式两边同乘 $\frac{7}{4}$ 化为 $700\lambda^2 + 125\lambda + 3 = 0$。
解此一元二次方程，其判别式为 $125^2 - 4 \times 700 \times 3 = 15625 - 8400 = 7225$。求得两根分别为 $\lambda_1 = -\frac{1}{35}$ 与 $\lambda_2 = -\frac{3}{20}$。
考察该曲线系，其中亦包含由对角线 $AC$ 与 $BD$ 构成的退化曲线。由于两对角线相交于点 $Q$，当 $X=Q$ 时，具有外积 $W(Q,Q) = 0$ 且内积 $E(Q,Q) = \frac{7}{12}$。将 $\lambda = -\frac{3}{20}$ 代入该二次多项式中验证可知，其在 $E(Q,X) = \frac{7}{12}$ 时取得恒等极值 $0$，由此判定 $\lambda = -\frac{3}{20}$ 对应对角线线系 $AC \cup BD$。因此，交叉相交直线对 $AD \cup BC$ 对应的代数系参数唯一确定为 $\lambda_1 = -\frac{1}{35}$。
将 $\lambda = -\frac{1}{35}$ 代回关于 $E(Q,X)$ 的多项式各项系数中：
二次项系数化为 $\frac{12}{7} + 128\left(-\frac{1}{35}\right) = -\frac{68}{35}$；
一次项系数化为 $-136\left(-\frac{1}{35}\right) = \frac{136}{35}$；
常数项化为 $33\left(-\frac{1}{35}\right) - 1 = -\frac{68}{35}$。
提取公因式 $-\frac{68}{35}$，该多项式配方形式化为 $-\frac{68}{35}\left(E(Q,X)^2 - 2E(Q,X) + 1\right) = -\frac{68}{35}(E(Q,X) - 1)^2$。
联合算子方程由此得以极简表达为
\[ -\frac{68}{35}(E(Q,X) - 1)^2 + \frac{1}{7}W(Q,X)^2 = 0. \]
等式两边同乘 $35$ 得 $5W(Q,X)^2 - 68(E(Q,X) - 1)^2 = 0$，运用平方差公式分解为两个一次项：
\[ \sqrt{5}W(Q,X) \pm 2\sqrt{17}(E(Q,X) - 1) = 0. \]
代入坐标基底下外积算子 $W(Q,X) = y - x$ 以及内积算子项 $E(Q,X) - 1 = \frac{x}{4} + \frac{y}{3} - 1 = \frac{3x+4y-12}{12}$，可得
\[ \sqrt{5}(y - x) \pm 2\sqrt{17}\left(\frac{3x+4y-12}{12}\right) = 0. \]
两边同乘 $6$ 化简为 $6\sqrt{5}(y - x) \pm \sqrt{17}(3x+4y-12) = 0$。合并变量 $x$ 与 $y$ 的系数，整理得
\[ (3\sqrt{17} \mp 6\sqrt{5})x + (4\sqrt{17} \pm 6\sqrt{5})y - 12\sqrt{17} = 0. \]
基于椭圆对称性及定比分点比例的等价性，该退化曲线对表示直线 $AD$ 与 $BC$。由此可得直线 $BC$ 的解析式。
\end{solution}

\begin{example}{}{}
    已知 \(P\) 为椭圆 \(C: \frac{x^2}{4} + \frac{y^2}{3} = 1\) 上一动点，过点 \(P\) 作圆 \(O: x^2 + y^2 = 1\) 的两条切线分别交椭圆 \(C\) 于 \(A,B\) 两点，求证：直线 \(OP\) 与直线 \(AB\) 的斜率之积为定值；
\end{example}
\begin{solution}
首先，设点 $P(x_0, y_0)$ 满足方程 $\frac{x_0^2}{4} + \frac{y_0^2}{3} = 1$。
对于圆 $O: x^2 + y^2 = 1$，定义其极化算子为 $T_O(M, N) = x_M x_N + y_M y_N - 1$。
根据切线极化算子等式，过点 $P$ 向圆 $O$ 作出的两条切线 $PA, PB$ 的联合二次多项式方程可表示为
\[ T_O(P, X)^2 - T_O(P, P)T_O(X, X)=(x_0 x + y_0 y - 1)^2 - (x_0^2 + y_0^2 - 1)(x^2 + y^2 - 1) = 0\]
将上述方程的左侧进行完全展开，
\begin{align*}
& (x_0^2 x^2 + y_0^2 y^2 + 1 + 2x_0 y_0 xy - 2x_0 x - 2y_0 y) \\
& - (x_0^2 x^2 + x_0^2 y^2 - x_0^2 + y_0^2 x^2 + y_0^2 y^2 - y_0^2 - x^2 - y^2 + 1) = 0.
\end{align*}
合并同类项，化简可得退化二次多项式方程
\[ (1 - y_0^2)x^2 + (1 - x_0^2)y^2 + 2x_0 y_0 xy - 2x_0 x - 2y_0 y + x_0^2 + y_0^2 = 0. \]
存在实数 $\lambda$，使得对平面内所有坐标 $(x, y)$，如下多项式恒等式成立：
\begin{align*}
& (1 - y_0^2)x^2 + (1 - x_0^2)y^2 + 2x_0 y_0 xy - 2x_0 x - 2y_0 y + x_0^2 + y_0^2 + \lambda\left(\frac{x^2}{4} + \frac{y^2}{3} - 1\right) \\
& \equiv \left(\frac{x_0}{4} x + \frac{y_0}{3} y - 1\right)(Ax + By + C_0).
\end{align*}
进一步，展开该恒等式的右侧，
\[ \frac{A x_0}{4}x^2 + \frac{B y_0}{3}y^2 + \left(\frac{B x_0}{4} + \frac{A y_0}{3}\right)xy + \left(\frac{C_0 x_0}{4} - A\right)x + \left(\frac{C_0 y_0}{3} - B\right)y - C_0. \]
通过比较恒等式两侧同次项对应的系数，可求解参数 $A, B, C_0$。
\begin{itemize}
\item 对于常数项，左侧为 $x_0^2 + y_0^2 - \lambda$，右侧为 $-C_0$，得到 $\lambda = x_0^2 + y_0^2 + C_0$。
\item 对于 $x$ 的一次项，左侧为 $-2x_0$，右侧为 $\frac{C_0 x_0}{4} - A$，解得 $A = x_0\left(\frac{C_0}{4} + 2\right)$。
\item 对于 $y$ 的一次项，左侧为 $-2y_0$，右侧为 $\frac{C_0 y_0}{3} - B$，解得 $B = y_0\left(\frac{C_0}{3} + 2\right)$。
\item 对于 $xy$ 的交叉项，左侧为 $2x_0 y_0$，右侧为 $\frac{B x_0}{4} + \frac{A y_0}{3}$。将所求得的 $A$ 与 $B$ 的表达式代入，可得
\begin{align*}
2x_0 y_0 &= \frac{x_0}{4} y_0 \left(\frac{C_0}{3} + 2\right) + \frac{y_0}{3} x_0 \left(\frac{C_0}{4} + 2\right) \\
&= x_0 y_0 \left( \frac{C_0}{12} + \frac{1}{2} + \frac{C_0}{12} + \frac{2}{3} \right) \\
&= x_0 y_0 \left( \frac{C_0}{6} + \frac{7}{6} \right).
\end{align*}
当 $x_0 y_0 \neq 0$ 时，等式两端约去非零因子 $x_0 y_0$，得到
\[ 2 = \frac{C_0}{6} + \frac{7}{6}. \]
解得 $C_0 = 5$。
\item 对于 $x_0 y_0 = 0$ 的情形，由 $\frac{x_0^2}{4} + \frac{y_0^2}{3} = 1$ 可知，当 $x_0 = 0$ 时 $y_0^2 = 3$。比较 $x^2$ 项系数，左侧为 $1 - y_0^2 + \frac{\lambda}{4} = -2 + \frac{3 + C_0}{4}$，右侧为 $0$，解得 $C_0 = 5$。当 $y_0 = 0$ 时 $x_0^2 = 4$。比较 $y^2$ 项系数，左侧为 $1 - x_0^2 + \frac{\lambda}{3} = -3 + \frac{4 + C_0}{3}$，右侧为 $0$，同样解得 $C_0 = 5$。综上所述，常数 $C_0 = 5$ 恒成立。
将 $C_0 = 5$ 代回 $A, B$ 的表达式，分别求得 $A = \frac{13}{4} x_0$ 与 $B = \frac{11}{3} y_0$。
\end{itemize}
因此，割线 $AB$ 的直线方程为 $\frac{13}{4} x_0 x + \frac{11}{3} y_0 y + 5 = 0$。将等号两侧同乘 $12$，可化为一般式
\[ 39x_0 x + 44y_0 y + 60 = 0. \]
基于该直线方程，可得直线 $AB$ 的斜率为 $k_{AB} = -\frac{39x_0}{44y_0}$。
已知直线 $OP$ 的斜率为 $k_{OP} = \frac{y_0}{x_0}$。将两斜率相乘，计算可得
\[ k_{OP} \cdot k_{AB} = \frac{y_0}{x_0} \cdot \left(-\frac{39x_0}{44y_0}\right) = -\frac{39}{44}. \]
该乘积为不依赖于点 $P$ 的常数，故直线 $OP$ 与直线 $AB$ 的斜率之积为定值，命题得证。
\end{solution}


\begin{example}{}{}
已知 \(A,B,C,D\) 为椭圆 \(E: \frac{x^2}{2} + y^2 = 1\) 上四点，直线 \(AC, BD\) 均过点 \(F_2(1,0)\)，
\begin{enumerate}
\item 若直线 \(AB\) 斜率为 1，求证：直线 \(CD\) 过定点；
\item 若直线 \(AB\) 过点 \(F_1(-1,0)\)，求证：\(\frac{k_{AB}}{k_{CD}}\) 为定值；
\end{enumerate}
\end{example}
\begin{solution}
对于椭圆 $E: \frac{x^2}{2} + y^2 = 1$，定义其极化算子为 $T(M, N) = \frac{x_M x_N}{2} + y_M y_N - 1$。点 $A, B, C, D$ 均在椭圆上，满足$T(A,A) = T(B,B) = T(C,C) = T(D,D) = 0$。记定点 $F_2(1,0)$ 为 $P$，已知直线 $AC$ 与 $BD$ 均过点 $P$。
由于点 $A, P, C$ 共线，点 $C$ 可表示为 $P$ 与 $A$ 的仿射组合，设 $C = \frac{P + \lambda A}{1+\lambda}$。将其代入$T(C,C) = 0$展开，可得
\begin{align*}
\frac{1}{(1+\lambda)^2} \left[ T(P,P) + 2\lambda T(P,A) + \lambda^2 T(A,A) \right] = 0\implies \lambda = -\frac{T(P,P)}{2T(P,A)}
\end{align*}
设直线 $CD$ 的方程为一次线性多项式方程 $L_{CD}(X) = 0$。将点 $C$ 的坐标表达式代入，由一次函数对组合点的线性分配律可得
\begin{align*}
L_{CD}(C) = L_{CD}\left(\frac{P + \lambda A}{1+\lambda}\right) = \frac{L_{CD}(P) + \lambda L_{CD}(A)}{1+\lambda} = 0
\end{align*}
即 $L_{CD}(P) + \lambda L_{CD}(A) = 0$。将 $\lambda$ 的表达式代入，可得
\begin{align*}
L_{CD}(P) - \frac{T(P,P)}{2T(P,A)} L_{CD}(A) = 0.
\end{align*}
等式两边同乘 $2T(P,A)$，并利用极化算子的交换律 $T(P,A) = T(A,P)$，可将其重写为
\begin{align*}
2L_{CD}(P)T(P,A) - T(P,P)L_{CD}(A) = 0.
\end{align*}
这表明，点 $A$ 的坐标满足关于变元 $X$ 的一次线性多项式方程
\begin{align*}
2L_{CD}(P)T(P,X) - T(P,P)L_{CD}(X) = 0.
\end{align*}
同理，由于点 $B, P, D$ 共线且 $B, D$ 均在椭圆上，由完全对称的推导过程可得点 $B$ 亦满足上述同一个关于 $X$ 的一次线性方程。
由于该方程为关于 $X$ 的一次线性方程，且直线 $AB$ 上的两相异点 $A, B$ 均满足该方程，故该方程即为直线 $AB$ 的算子方程 $L_{AB}(X) = 0$。
因此，存在非零常数 $\mu$，使得对于平面内任意一点 $X$，恒有
\begin{align*}
L_{AB}(X) = \mu \left[ 2L_{CD}(P)T(P,X) - T(P,P)L_{CD}(X) \right].
\end{align*}
令 $X = P$ 代入该恒等式，可得
\begin{align*}
L_{AB}(P) &= \mu \left[ 2L_{CD}(P)T(P,P) - T(P,P)L_{CD}(P) \right]= \mu T(P,P) L_{CD}(P).
\end{align*}
由此解得 $\mu L_{CD}(P) = \frac{L_{AB}(P)}{T(P,P)}$。
将此等量整体代回恒等式，分离出包含 $L_{CD}(X)$ 的项，得到
\begin{align*}
L_{AB}(X) = 2\frac{L_{AB}(P)}{T(P,P)} T(P,X) - \mu T(P,P) L_{CD}(X).
\end{align*}
移项可得
\begin{align*}
\mu T(P,P) L_{CD}(X) = 2\frac{L_{AB}(P)}{T(P,P)} T(P,X) - L_{AB}(X).
\end{align*}
等式两边同乘非零常数 $T(P,P)$，可得直线 $CD$ 的算子方程等价表示为
\begin{align*}
2L_{AB}(P) T(P,X) - T(P,P) L_{AB}(X) = 0.
\end{align*}
对于基点 $P(1,0)$，代入极化算子计算可得
\begin{align*}
T(P,P) &= \frac{1 \times 1}{2} + 0 - 1 = -\frac{1}{2}, \\
T(P,X) &= \frac{1 \cdot x}{2} + 0 \cdot y - 1 = \frac{x-2}{2}.
\end{align*}
将其代入所推导的直线 $CD$ 等效方程中，可得
\begin{align*}
2L_{AB}(1,0)\left(\frac{x-2}{2}\right) - \left(-\frac{1}{2}\right) L_{AB}(X) = 0.
\end{align*}
化简得
\begin{align*}
L_{AB}(1,0)(x - 2) + \frac{1}{2} L_{AB}(X) = 0.
\end{align*}
方程两边同乘 $2$，得到联系两直线方程的核心恒等关系：
\begin{align*}
2L_{AB}(1,0)(x - 2) + L_{AB}(X) = 0.
\end{align*}
当直线 $AB$ 的斜率为 $1$ 时，设直线 $AB$ 的方程为 $L_{AB}(X) = x - y + m = 0$。
将定点 $P(1,0)$ 代入，可得常数项 $L_{AB}(1,0) = 1 - 0 + m = m + 1$。
将 $L_{AB}(1,0)$ 与 $L_{AB}(X)$ 代入核心恒等关系中，得到
\begin{align*}
2(m+1)(x - 2) + (x - y + m) = 0\implies(3x - y - 4) + m(2x - 3) = 0.
\end{align*}
若该直线方程对于满足条件的任意实数 $m$ 均成立，则必有
\begin{align*}
\begin{cases}
2x - 3 = 0 \\
3x - y - 4 = 0
\end{cases}
\end{align*}
由第一式解得 $x = \frac{3}{2}$，代入第二式可得 $3\left(\frac{3}{2}\right) - y - 4 = 0$，解得 $y = \frac{1}{2}$。
由此可得，直线 $CD$ 恒过定点 $\left(\frac{3}{2}, \frac{1}{2}\right)$。
当直线 $AB$ 过点 $F_1(-1,0)$ 时，设其方程为 $L_{AB}(X) = k(x + 1) - y = 0$，即 $kx - y + k = 0$。
将定点 $P(1,0)$ 代入，可得常数项 $L_{AB}(1,0) = k(1) - 0 + k = 2k$。
将其代入直线 $CD$ 的核心恒等关系中，得到
\begin{align*}
2(2k)(x - 2) + (kx - y + k) = 0.
\end{align*}
展开并化简得
\begin{align*}
4kx - 8k + kx - y + k = 0,
\end{align*}
即
\begin{align*}
5kx - y - 7k = 0.
\end{align*}
由此可知，直线 $CD$ 的方程化为 $y = 5kx - 7k$，其斜率为 $k_{CD} = 5k$。
结合直线 $AB$ 的斜率 $k_{AB} = k$，当 $k \neq 0$ 时，可得两者斜率的比值为
\begin{align*}
\frac{k_{AB}}{k_{CD}} = \frac{k}{5k} = \frac{1}{5}.
\end{align*}
综上所述，比值 $\frac{k_{AB}}{k_{CD}}$ 为定值 $\frac{1}{5}$。
\end{solution}

\begin{example}{}{}
已知 \(R,S\) 是 \(x\) 轴上两定点，过 \(S\) 的直线交椭圆 \(\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1\) 于 \(A,B\) 两点，分别延长 \(AR,BR\) 再次交椭圆于 \(D,C\) 两点。求证：直线 \(CD\) 过定点；
\end{example}
\begin{solution}
定义针对椭圆 $\frac{x^2}{a^2} + \frac{y^2}{b^2} = 1$ 的极化算子为 $T(M, N) = \frac{x_M x_N}{a^2} + \frac{y_M y_N}{b^2} - 1$。
对于该椭圆上的任意一点 $X$，恒有极化自内积方程 $T(X, X) = 0$ 成立。
已知直线 $AB$ 经过定点 $S(s, 0)$，设其方程为关于横纵坐标的一次多项式算子方程 $L_{AB}(X) = 0$。由于点 $S$ 位于直线 $AB$ 上，必定满足等式 $L_{AB}(S) = 0$。\\
根据题意，点 $R, A, D$ 三点共线。可将点 $A$ 表示为定点 $R$ 与点 $D$ 的仿射组合。设存在实数 $\lambda$，使得：
\[ A = \frac{R + \lambda D}{1+\lambda} \]
由于点 $A$ 位于椭圆上，将其仿射组合表达式代入椭圆的极化自内积算子中，并利用双线性分配律展开：
\[ T\left(\frac{R + \lambda D}{1+\lambda}, \frac{R + \lambda D}{1+\lambda}\right) = \frac{T(R, R) + 2\lambda T(R, D) + \lambda^2 T(D, D)}{(1+\lambda)^2} = 0. \]
已知点 $D$ 亦位于椭圆上，满足 $T(D, D) = 0$。代入上式，分子部分化简为：
\[ T(R, R) + 2\lambda T(R, D) = 0. \]
由于直线 $AR$ 与椭圆交于相异两点 $A, D$，故 $\lambda \neq 0$ 且 $T(R, D) \neq 0$。解得仿射参数为：
\[ \lambda = -\frac{T(R, R)}{2T(R, D)}. \]
将其代回点 $A$ 的仿射组合形式中，分子与分母同乘 $2T(R, D)$，得到点 $A$ 的算子化齐次坐标表示：
\[ A = \frac{2T(R, D)R - T(R, R)D}{2T(R, D) - T(R, R)}. \]
由于点 $A$ 位于直线 $AB$ 上，将上述坐标表达式代入直线 $AB$ 的算子方程 $L_{AB}(A) = 0$ 中。利用一次线性算子 $L_{AB}(X)$ 保持仿射组合不变的性质，将其恒等展开：
\[ \frac{2T(R, D)L_{AB}(R) - T(R, R)L_{AB}(D)}{2T(R, D) - T(R, R)} = 0. \]
等式两端同乘非零分母，得到点 $D$ 所满足的一个一阶线性关系式：
\[ 2L_{AB}(R)T(R, D) - T(R, R)L_{AB}(D) = 0. \]
另一方面，点 $R, B, C$ 三点同样共线，点 $B, C$ 均位于椭圆上，且点 $B$ 位于直线 $AB$ 上。通过完全对称的代数推导过程，点 $C$ 必定满足与之结构一致的约束条件：
\[ 2L_{AB}(R)T(R, C) - T(R, R)L_{AB}(C) = 0. \]
上述两式表明，椭圆上的相异两点 $C$ 与 $D$ 均满足关于动点 $X$ 的同一个一次线性方程：
\[ 2L_{AB}(R)T(R, X) - T(R, R)L_{AB}(X) = 0. \]
因平面内两相异点唯一确定一条直线，该等式即为经过点 $C, D$ 的直线 $CD$ 的算子方程，记为：
\[ L_{CD}(X) = 2L_{AB}(R)T(R, X) - T(R, R)L_{AB}(X) = 0. \]
为求证直线 $CD$ 恒过定点，即探求一固定点 $T$，使得无论满足 $L_{AB}(S) = 0$ 的直线的算子 $L_{AB}$ 如何变化，恒有 $L_{CD}(T) = 0$ 成立。
已知定点 $R(r, 0)$ 与 $S(s, 0)$ 均位于 $x$ 轴上。假设待求定点 $T$ 亦位于 $x$ 轴上，则点 $T$ 可表示为 $R$ 与 $S$ 的仿射组合。
设 $T = \alpha S + \beta R$，其中 $\alpha + \beta = 1$。
将点 $T$ 的表达式代入直线 $AB$ 的算子方程中，由线性分配性质展开得：
\[ L_{AB}(T) = \alpha L_{AB}(S) + \beta L_{AB}(R). \]
代入条件 $L_{AB}(S) = 0$，上式化简为 $L_{AB}(T) = \beta L_{AB}(R)$。
同时，将点 $T$ 代入极化算子中展开：
\[ T(R, T) = \alpha T(R, S) + \beta T(R, R). \]
将 $L_{AB}(T)$ 与 $T(R, T)$ 共同代入直线 $CD$ 的方程 $L_{CD}(T) = 0$ 中：
\[ 2L_{AB}(R)\big[\alpha T(R, S) + \beta T(R, R)\big] - T(R, R)\big[\beta L_{AB}(R)\big] = 0. \]
对于一般情形下不经过点 $R$ 的割线 $AB$，有 $L_{AB}(R) \neq 0$。将等式两端同除以 $L_{AB}(R)$，提取出不含动直线参数的独立等式：
\[ 2\alpha T(R, S) + 2\beta T(R, R) - \beta T(R, R) = 0 \implies 2\alpha T(R, S) + \beta T(R, R) = 0. \]
结合 $\alpha + \beta = 1$，将 $\alpha = 1 - \beta$ 代入得到：
\[ 2(1 - \beta) T(R, S) + \beta T(R, R) = 0 \implies \beta \big[2T(R, S) - T(R, R)\big] = 2T(R, S). \]
解得对应的仿射参数：
\[ \beta = \frac{2T(R, S)}{2T(R, S) - T(R, R)}, \quad \alpha = 1 - \beta = \frac{-T(R, R)}{2T(R, S) - T(R, R)}. \]
由于参数 $\alpha$ 与 $\beta$ 仅由定点 $R, S$ 的位置参数与椭圆的极化自内积常数决定，完全独立于直线 $AB$ 的变动状态，因此点 $T$ 为一固定点。\\
进一步计算定点 $T$ 的具体坐标。将 $S(s, 0)$ 与 $R(r, 0)$ 代入极化算子：
\[ T(R, S) = \frac{rs}{a^2} - 1, \quad T(R, R) = \frac{r^2}{a^2} - 1. \]
计算参数的分母部分：
\[ 2T(R, S) - T(R, R) = 2\left(\frac{rs}{a^2} - 1\right) - \left(\frac{r^2}{a^2} - 1\right) = \frac{2rs - r^2 - a^2}{a^2}. \]
由于 $R, S$ 的纵坐标均为 $0$，点 $T$ 的纵坐标 $y_T = \alpha \cdot 0 + \beta \cdot 0 = 0$。点 $T$ 的横坐标 $x_T$ 计算如下：
\begin{align*}
x_T &= \alpha s + \beta r = \frac{-s T(R, R) + 2r T(R, S)}{2T(R, S) - T(R, R)} \\
&= \frac{-s\left(\frac{r^2 - a^2}{a^2}\right) + 2r\left(\frac{rs - a^2}{a^2}\right)}{\frac{2rs - r^2 - a^2}{a^2}} \\
&= \frac{-sr^2 + sa^2 + 2r^2s - 2ra^2}{2rs - r^2 - a^2} \\
&= \frac{r^2s + a^2s - 2a^2r}{2rs - r^2 - a^2}.
\end{align*}
将分子与分母同时乘 $-1$，整理得：
\[ x_T = \frac{2a^2r - a^2s - r^2s}{a^2 + r^2 - 2rs}. \]
由此可得，在 $a^2 + r^2 - 2rs \neq 0$ 时，直线 $CD$ 恒经过 $x$ 轴上的定点 $\left( \frac{2a^2r - a^2s - r^2s}{a^2 + r^2 - 2rs}, 0 \right)$。结论成立。\\
最后，计算并验证直线 $CD$ 与 $AB$ 的斜率比值。
设直线 $AB$ 的方程为 $x = my + s$，其斜率为 $k_{AB} = \frac{1}{m}$。该直线的算子函数等价表示为 $L_{AB}(X) = x_X - m y_X - s$。
代入定点 $R(r, 0)$ 得到 $L_{AB}(R) = r - s$。对于动点 $X(x, y)$，有 $L_{AB}(X) = x - my - s$。
将其代入直线 $CD$ 的全局算子方程中：
\[ 2(r - s)\left(\frac{rx}{a^2} - 1\right) - \left(\frac{r^2 - a^2}{a^2}\right)(x - my - s) = 0. \]
等号两端同乘 $a^2$ 展开：
\[ 2(r - s)(rx - a^2) - (r^2 - a^2)(x - my - s) = 0. \]
提取并合并关于 $x$ 与 $y$ 的同类项系数。对于 $x$ 的系数：
\[ 2r(r - s) - (r^2 - a^2) = 2r^2 - 2rs - r^2 + a^2 = r^2 - 2rs + a^2. \]
对于 $y$ 的系数：
\[ -(r^2 - a^2)(-m) = m(r^2 - a^2). \]
由此得到直线 $CD$ 的一般方程为 $(r^2 - 2rs + a^2)x + m(r^2 - a^2)y + C = 0$（常数项记为 $C$）。其斜率 $k_{CD}$ 为：
\[ k_{CD} = -\frac{r^2 - 2rs + a^2}{m(r^2 - a^2)} = \frac{a^2 + r^2 - 2rs}{m(a^2 - r^2)}. \]
两直线斜率相除，参数 $m$ 自然消去，得到两者的斜率比值为：
\[ \frac{k_{CD}}{k_{AB}} = \frac{\frac{a^2 + r^2 - 2rs}{m(a^2 - r^2)}}{\frac{1}{m}} = \frac{a^2 + r^2 - 2rs}{a^2 - r^2}. \]
综上，各项结论均严密成立。
\end{solution}


\begin{conclusion}{弦对原点张角的“齐次化”联合方程}{}
    已知不经过原点的直线交椭圆于 $A, B$ 两点。设这条直线对应的极点为 $M$，也就是直线方程可以写成标准形式：$E(M,X) = 1$。那么，连接原点与交点的两条射线 $OA, OB$ 构成了一个“退化曲线”（两相交直线），它的方程只需一步即可写出：$$ E(X,X) - \big[E(M,X)\big]^2 = 0 $$
\end{conclusion}
原理解释极度简单：
\begin{itemize}
\item 验证交点： 把交点 $A$ 的坐标代入看看。因为 $A$ 在椭圆上，所以 $E(A,A)=1$；因为 $A$ 在直线上，所以 $E(M,A)=1$。代入原式：$1 - 1^2 = 0$。完美成立！说明方程确实经过交点 $A, B$。
\item 验证过原点： 我们来看看算子展开后的特征。椭圆内积 $E(X,X) = \frac{x^2}{a^2} + \frac{y^2}{b^2}$ 是一个纯粹的二次项（没有一次项和常数项）。而 $\big[E(M,X)\big]^2 = \big(\frac{x_M \cdot x}{a^2} + \frac{y_M \cdot y}{b^2}\big)^2$，展开后也全部是纯粹的二次项！既然整个方程展开后每一项都是二次的（形如 $C_1 x^2+ C_2 xy+ C_3 y^2=0$），在解析几何中，这必定代表一组“交于原点的相交直线”！
\end{itemize}
如果题目要求 $OA \perp OB$，根据平面几何常识，只要这个方程里 $x^2$ 的总系数加上 $y^2$ 的总系数等于 $0$ 即可。我们根本不需要联立方程求坐标，只需盯着 $E(X,X) - \big[E(M,X)\big]^2 = 0$ 提取系数：
\begin{itemize}
\item $E(X,X)$ 提供：$x^2$ 的系数是 $\frac{1}{a^2}$，$y^2$ 的系数是 $\frac{1}{b^2}$。
\item $\big[E(M,X)\big]^2$ 提供：$x^2$ 的系数是 $\frac{x_M^2}{a^4}$，$y^2$ 的系数是 $\frac{y_M^2}{b^4}$。两者相减等于 $0$，立刻得到极其优美的结论：$$ \left(\frac{1}{a^2} - \frac{x_M^2}{a^4}\right) + \left(\frac{1}{b^2} - \frac{y_M^2}{b^4}\right) = 0 \implies \frac{1}{a^2} + \frac{1}{b^2} = \frac{x_M^2}{a^4} + \frac{y_M^2}{b^4} $$有了极点 $M$ 的坐标，这类题目直接一步登天，过程之简洁让人叹为观止。
\end{itemize}
解析几何中最痛苦的计算之一，就是求“曲线外一点 $P$ 引出的两条切线构成的总方程”（传统方法要硬算并死记大招公式 $S_1^2 = S_{11}S_0$）。
现在，我将为你展示整个算子理论中最令人起鸡皮疙瘩的推导——用初中生的完全平方公式，两行写出外切线对的方程！
\begin{conclusion}{外切线对联合方程与蒙日圆}{}
    从二次曲线外一点 $P$ 作原曲线的两条切线，这两条切线（退化为交叉直线的二次曲线），其算子方程为极度对称的：$$ \big[T(P,X)\big]^2 = T(P,P) \cdot T(X,X) $$\textbf{推论（蒙日圆）：} 若这两条切线互相垂直，点 $P$ 的轨迹必然是一个圆 $x_P^2 + y_P^2 = a^2 + b^2$。
\end{conclusion}
设 $X$ 是这“两条切线”上的任意一个动点。连结 $P$ 和 $X$，线段 $PX$ 上的任意一点 $Y$ 都可以用定比分点表示为：$$ Y = \frac{P + \lambda X}{1+\lambda} $$既然直线 $PX$ 就是切线，意味着这条直线与椭圆有且仅有一个交点（即唯一的切点）！我们将点 $Y$ 代入椭圆的内积算子方程 $E(Y,Y) = 1$ 中，利用类似初中多项式乘法的分配律展开：$$ E\left(\frac{P+\lambda X}{1+\lambda}, \frac{P+\lambda X}{1+\lambda}\right) = \frac{E(P,P) + 2\lambda E(P,X) + \lambda^2 E(X,X)}{(1+\lambda)^2} = 1 $$把分母 $(1+\lambda)^2 = 1 + 2\lambda + \lambda^2$ 乘到右边，然后整体移项到左边。
这里的神来之笔出现了！我们将移项后的每一项巧妙地“结对子”：$$ \big(E(P,P) - 1\big) + 2\lambda \big(E(P,X) - 1\big) + \lambda^2 \big(E(X,X) - 1\big) = 0 $$
回忆定义，极化算子 $T$ 本质就是内积算子减去 1（即 $T = E - 1$）！原方程瞬间变成了关于 $\lambda$ 的绝美一元二次方程：
$$ T(P,P) + 2\lambda T(P,X) + \lambda^2 T(X,X) = 0 $$
因为直线 $PX$ 是切线，所以这个关于 $\lambda$ 的一元二次方程，必须有且仅有一个解（即重根）！一元二次方程有重根，意味着判别式 $\Delta = 0$！
$$ \Delta = \big[2T(P,X)\big]^2 - 4 \cdot T(P,P) \cdot T(X,X) = 0 $$把 4 约掉，移项，直接结束战斗：$$ \big[T(P,X)\big]^2 = T(P,P) \cdot T(X,X) $$(全程没有求导，没有死算斜率通分，仅仅利用了定比分点展开，然后让判别式 $\Delta=0$，就轻松得到了传统解析几何里最难推导的切线对大招！)\\
\textbf{巅峰应用：不设切线秒杀蒙日圆（准圆）轨迹}\\
互相垂直的切线对，意味着在它的联合方程中展开后，$x^2$ 的系数与 $y^2$ 的系数之和必须等于 $0$。
我们直接看刚刚证出来的切线对公式：$\big[T(P,X)\big]^2 = T(P,P) \cdot T(X,X)$。
提取等号两边关于动点 $X(x,y)$ 的 $x^2$ 和 $y^2$ 系数：
\begin{itemize}
\item 左边展开来自于 $\big(\frac{x_P x}{a^2} + \frac{y_P y}{b^2} - 1\big)^2$，$x^2$ 系数为 $\frac{x_P^2}{a^4}$，$y^2$ 系数为 $\frac{y_P^2}{b^4}$。
\item 右边展开来自于 $T(P,P) \cdot \big(\frac{x^2}{a^2} + \frac{y^2}{b^2} - 1\big)$，$x^2$ 系数为 $T(P,P)\cdot\frac{1}{a^2}$，$y^2$ 系数为 $T(P,P)\cdot\frac{1}{b^2}$。
\end{itemize}
令两边的系数之差相加等于 0：
$$ \left( \frac{x_P^2}{a^4} - \frac{T(P,P)}{a^2} \right) + \left( \frac{y_P^2}{b^4} - \frac{T(P,P)}{b^2} \right) = 0 \implies \frac{x_P^2}{a^4} + \frac{y_P^2}{b^4} = T(P,P)\left(\frac{1}{a^2} + \frac{1}{b^2}\right) $$
把 $P$ 点的自身极化能量 $T(P,P) = \frac{x_P^2}{a^2} + \frac{y_P^2}{b^2} - 1$ 完整代入右侧，像初中乘法一样乘进去拆开：
$$ \frac{x_P^2}{a^4} + \frac{y_P^2}{b^4} = {\left(\frac{x_P^2}{a^4}\right)} + \frac{x_P^2}{a^2 b^2} + \frac{y_P^2}{a^2 b^2} + {\left(\frac{y_P^2}{b^4}\right)} - \left(\frac{1}{a^2} + \frac{1}{b^2}\right) $$
看到了吗？最复杂的最高次分数项 $\frac{x_P^2}{a^4}$ 和 $\frac{y_P^2}{b^4}$ 再次在等号两边奇迹般地完美抵消（加粗部分）！
方程瞬间精简为：$$ 0 = \frac{x_P^2 + y_P^2}{a^2 b^2} - \frac{a^2+b^2}{a^2 b^2} $$两边同乘 $a^2b^2$，得到极简的最终轨迹方程：$$ x_P^2 + y_P^2 = a^2 + b^2 $$


\section{解释“参数方程法”}
所谓的“万能代换”，根本不需要引入 $\tan\frac{\theta}{2}$ 等超越函数。它的几何本质仅仅是：\textbf{以椭圆左顶点 $S(-a, 0)$ 为基点，任作一条方向向量为 $\vec{v} = (a, bt)$ 的割线。}这里的参数 $t$，仅仅是方向向量的“斜率微元”。

将射线上动点 $X = S + \tau \vec{v}$ 代入曲线极化多项式 $T(X,X)=0$ 中。由于基点 $S$ 在椭圆上，必有 $T(S,S)=0$（即一元二次方程的常数项归零），方程被绝对截断，直接解出唯一的非零微元 $\tau$：$$ \tau = \frac{-2E(S,\vec{v})}{E(\vec{v},\vec{v})} = \frac{-2(-1)}{1+t^2} = \frac{2}{1+t^2} $$将其代回 $X = S + \tau \vec{v}$，得到有理参数坐标：$$ X(t) = \left( -a + a\frac{2}{1+t^2}, \ 0 + bt\frac{2}{1+t^2} \right) = \left( a\frac{1-t^2}{1+t^2}, b\frac{2t}{1+t^2} \right) $$
设极点坐标为 $P_{12}(x_0, y_0)$。根据极点与极线的对偶关系，既然 $P_{12}$ 是弦 $M_1M_2$ 对应的极点，那么点 $M_1, M_2$ 必定都在点 $P_{12}$ 的极线上。
极线方程全宇宙统一为内积算子形式：$E(P_{12}, X) = 1$。
我们将任意参数动点 $X(t)$ 的坐标代入这条极线方程：
$$ \frac{x_0}{a}\left(\frac{1-t^2}{1+t^2}\right) + \frac{y_0}{b}\left(\frac{2t}{1+t^2}\right) = 1 $$两边同乘 $(1+t^2)$，并按 $t$ 的降幂合并同类项，我们得到了一个关于 $t$ 的标准一元二次方程：$$ \left(1 + \frac{x_0}{a}\right)t^2 - \left(\frac{2y_0}{b}\right)t + \left(1 - \frac{x_0}{a}\right) = 0 $$
因为 $M_1(t_1), M_2(t_2)$ 都在这条极线上，这意味着$t_1, t_2$ 正好是这个一元二次方程的两个根。韦达定理：
\begin{align*}
    t_1 t_2 &= \frac{1 - x_0/a}{1 + x_0/a} \implies t_1 t_2 \left(1 + \frac{x_0}{a}\right) = 1 - \frac{x_0}{a} \implies {x_0 = a\frac{1-t_1t_2}{1+t_1t_2}} \\
    t_1 + t_2 &= \frac{2y_0/b}{1 + x_0/a} \implies \text{将} 1+\frac{x_0}{a} = \frac{2}{1+t_1t_2} \text{ 代入解纵坐标：} {y_0 = b\frac{t_1+t_2}{1+t_1t_2}}
\end{align*}
把上面刚求出的 $x_0, y_0$ 放进极线方程 $E(P_{12}, X) = 1$：
$$ \frac{1-t_1t_2}{1+t_1t_2} \frac{x}{a} + \frac{t_1+t_2}{1+t_1t_2} \frac{y}{b} = 1 $$
在“等效判别式”流派中，大家都在死记硬背直线 $Ax+By=1$ 的判别常数 $\Delta_0 = a^2A^2+b^2B^2-1 = C^2$。但在算子系统中，弦方程 $\frac{x_0}{a^2}x + \frac{y_0}{b^2}y = 1$ 与 $Ax+By=1$ 的系数完全对应：$x_0 = a^2A, y_0 = b^2B$。这说明，所谓的 $\Delta_0$，其实就是\textbf{把极点 $P_{12}$ 代入极化算子后得到的自身极化值 $T(P_{12},P_{12})$}！$$ \Delta_0 = a^2\left(\frac{x_0}{a^2}\right)^2 + b^2\left(\frac{y_0}{b^2}\right)^2 - 1 = \frac{x_0^2}{a^2} + \frac{y_0^2}{b^2} - 1 = T(P_{12},P_{12}) $$我们把 $x_0, y_0$ 代入算一算这个极化值究竟是多少：
\begin{align*}
T(P_{12}, P_{12}) &= \left(\frac{1-t_1t_2}{1+t_1t_2}\right)^2 + \left(\frac{t_1+t_2}{1+t_1t_2}\right)^2 - 1 \\
&= \frac{(1 - 2t_1t_2 + t_1^2t_2^2) + (t_1^2 + 2t_1t_2 + t_2^2) - (1+2t_1t_2+t_1^2t_2^2)}{(1+t_1t_2)^2} \\
&= \frac{t_1^2 + t_2^2 - 2t_1t_2}{(1+t_1t_2)^2} = \left(\frac{t_1-t_2}{1+t_1t_2}\right)^2 \equiv C^2
\end{align*}
等效判别式 $\Delta_0$（或叫缩放常数 $C^2$），在代数上等于 $\left(\frac{t_1-t_2}{1+t_1t_2}\right)^2$。而在几何本质上，它代表了极点 $P_{12}$ 相对于椭圆的极化位置能量 $T(P_{12},P_{12})$.
