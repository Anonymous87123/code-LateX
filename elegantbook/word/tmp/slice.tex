\chapter{POS, POST, PON, POUND}

\begin{itemize}
  \item 来自拉丁语的 pos, post, pon, pound 意为 to put, to place。同义词根有来自拉丁语的 loc, sert, stal 和来自希腊语的 thes / thet / them, top 等。
\end{itemize}

\begin{word}[*]{apposite}{'æpəzit}
  \wordbreakdown{放在近处的 ap<ad (= to, near) + pos (= put)}
  \wordsense{a. 适当的，合适的}
  \wordfamily{appositeness 适当，合适}
  \wordfamily{appositely 适当地，合适地}
  \wordphrase{apposite behavior}
  \wordexample{Her jokes and smiles were not apposite behavior for her father's funeral.（她在她爸爸的葬礼上开玩笑和微笑不合适。）}
\end{word}



\begin{word}[**]{depose}{di'pauz}
  \wordbreakdown{将某人分开放 de (= from, away) + pos (= put)}
  \wordfamily{deposition（废黜）；apposite（适当的）；exposition（博览会）}
  \wordsense{vt. 罢免，废黜；【律】宣誓作证}
  \wordphrase{A popular liberation force}
  \wordexample{A popular liberation force deposed the monarch and took power.（一支民众解放势力罢黜国王夺取了政权。）}
\end{word}
