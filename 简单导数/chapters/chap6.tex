\chapter{均值不等式链}

\section{引言与记号}
不失一般性，设二元正数 $x, y > 0$ 且 $x \neq y$（后续证明中常假设 $0 < x < y$）。定义如下均值：
\begin{align*}
    A &= \frac{x+y}{2} \quad \text{(算术平均, } M_1 \text{)} \\
    G &= \sqrt{xy} \quad \text{(几何平均, } M_0 \text{)} \\
    H &= \frac{2xy}{x+y} \quad \text{(调和平均, } M_{-1} \text{)} \\
    L &= \frac{x-y}{\ln x - \ln y} \quad \text{(对数平均)} \\
    I &= \frac{1}{e}\left(\frac{y^y}{x^x}\right)^{\frac{1}{y-x}} \quad \text{(指数平均)} \\
    M_p &= \left(\frac{x^p+y^p}{2}\right)^{\frac{1}{p}} \quad \text{(幂平均, 且 } \lim_{p\to 0}M_p = G \text{)}
\end{align*}

这里采用双曲参数化。令 $t = \frac{1}{2}\ln\left(\frac{y}{x}\right) > 0$，即 $y = G e^t,\ x = G e^{-t}$。原式除以 $G$ 后，各项均值写成
\[
A = \cosh t, \quad G = 1, \quad L = \frac{\sinh t}{t}, \quad I = \exp(t\coth t - 1), \quad M_{1/2} = \frac{\cosh t + 1}{2}
\]

\section{基础部分与幂平均性质}

\begin{example}
证明基础均值不等式链：$M_1 \ge I \ge L \ge M_0 \ge M_{-1}$。
\end{example}
\begin{solution}
由于对称性，设 $0 < x < y$，令 $t = \frac{y}{x} > 1$。对 $A \ge I$，两边取对数后化为
\[
\ln(1+t) - \ln 2 \ge \frac{t\ln t}{t-1} - 1.
\]
构造
\[
f(t) = [\ln(1+t)-\ln 2+1](t-1) - t\ln t,
\]
则
\[
f'(t) = \frac{2t}{t+1} - \ln\left(\frac{2t}{t+1}\right) - 1.
\]
由基本不等式 $k-1 \ge \ln k$ 可知 $f'(t) \ge 0$，因而 $f(t) \ge f(1) = 0$，于是 $A \ge I$。

对 $I \ge L$，取对数后化为
\[
f(t) = \frac{t\ln t}{t-1} - \ln(t-1) + \ln(\ln t) - 1 \ge 0.
\]
此时
\[
f'(t) = \frac{(t-1)^2 - t\ln^2 t}{t(t-1)^2\ln t}.
\]
记分子为 $g(t) = (t-1)^2 - t\ln^2 t$，则
\[
g''(t) = \frac{2(t-1-\ln t)}{t} > 0.
\]
因此 $g'(t)$ 递增，且 $g'(t) > g'(1) = 0$，从而 $g(t) > g(1) = 0$。于是 $f'(t) > 0$，进而 $f(t) \ge 0$，故 $I \ge L$。

不等式 $L \ge G$ 等价于 $\frac{t-1}{\ln t} \ge \sqrt{t}$。构造 $f(t) = t - 1 - \sqrt{t}\ln t$，有
\[
f'(t) = \frac{\sqrt{t}-1-\ln\sqrt{t}}{\sqrt{t}} \ge 0,
\]
于是结论成立。最后，
\[
\sqrt{xy} \ge \frac{2xy}{x+y} \iff x+y \ge 2\sqrt{xy} \iff (\sqrt{x}-\sqrt{y})^2 \ge 0,
\]
故 $G \ge H$，从而得到 $M_1 \ge I \ge L \ge M_0 \ge M_{-1}$。
\end{solution}

\begin{example}
证明幂平均的递增性：若 $p > q$，则 $M_p \ge M_q$。
\end{example}
\begin{solution}
考察函数 $f(x) = x^{\beta/\alpha} (x>0)$，其二阶导数为 $f''(x) = \frac{\beta}{\alpha}\left(\frac{\beta}{\alpha}-1\right)x^{\frac{\beta}{\alpha}-2}$。
当 $0<\alpha<\beta$ 时，$f''(x) > 0$。由 Jensen 不等式的下凸性质：
$f\left(\frac{x_1+x_2+\cdots+x_n}{n}\right) \le \frac{f(x_1)+f(x_2)+\cdots+f(x_n)}{n}$。
取 $x_i = a_i^\alpha$，两边同时进行 $\frac{1}{\beta}$ 次乘方运算，便得到所需不等式。对于 $\alpha<\beta<0$ 等其余区间，也可通过凹凸性引起的不等号方向变化完成证明。
\end{solution}

\begin{example}
证明幂平均的混合放缩性质：当 $p, q \ge 0, pq \le 0, \alpha, \beta > 0$ 时，$\frac{\alpha M_p + \beta M_q}{\alpha+\beta} \ge M_{\frac{\alpha p + \beta q}{\alpha+\beta}}$。
\end{example}
\begin{solution}
令 $t = \frac{\alpha}{\alpha+\beta}$，则不等式等价于 $tM_p + (1-t)M_q \ge M_{tp+(1-t)q}$。这是由于在 $p,q \ge 0, pq \le 0$ 的条件下，幂平均 $M_p$ 具有凹函数性质所致。
以此类推，取特殊点可得 $\frac{M_p+M_q}{2} \le M_{\frac{p+q}{2}}$ 等界限。
\end{solution}

\section{混合平均不等式}

\begin{example}
证明：(1) $2G+A \ge 3L$ \qquad (2) $2A+H \ge 3L$
\end{example}
\begin{solution}
利用最佳系数逼近定理，已知 $L \le M_{1/3}$，于是
\[
2G+A = 2M_0 + M_1 \ge 3M_{1/3} \ge 3L,\qquad
2A+H = 2M_1 + M_{-1} \ge 3M_{1/3} \ge 3L.
\]
前一式若改用微分法，可令 $x=t^2 \ge 1$，构造
\[
f = \ln x - \frac{3(x-1)}{\frac{1+x}{2} + 2\sqrt{x}},
\]
求导后提出共同因子，得到
\[
f_1'(t) = \frac{(t-1)^4}{t(t^2+4t+1)^2} \ge 0,
\]
便可得到同样的结论。后一式也可沿相同思路，通过连续求导逐步化简对数项得到。
\end{solution}

\begin{example}
证明乘法与加法界限：(1) $AG \le LI$ \qquad (2) $A+G \ge L+I$
\end{example}
\begin{solution}
利用双曲参数化代入各项，
\[
AG \le LI \iff \cosh t \le \frac{\sinh t}{t} \exp(t\coth t - 1).
\]
又因为 $\cosh t = \frac{\sinh t}{t}(t\coth t)$，再用基本不等式 $u \le \exp(u-1)$（取 $u = t\coth t$），便得 $AG \le LI$。

另一方面，
\[
A+G \ge L+I \iff \cosh t + 1 \ge \frac{\sinh t}{t} + \exp(t\coth t - 1).
\]
构造函数 $f(t) = \ln\left(\cosh t + 1 - \frac{\sinh t}{t}\right) - t\coth t + 1$。将其求导并展开麦克劳林级数后，级数各项的系数
\[
C_n = \frac{3}{4}(9^{n+1}-1) - 2(n+1)(2n+3)(4^n-2n+1)
\]
在 $n \ge 3$ 时恒正，因此 $f'(t) > 0$。结合 $f(0)=0$，便有 $A+G \ge L+I$。
\end{solution}

\begin{example}
证明：(1) $2A+G \le 3I$ \quad (2) $A+L \le 2I$ \quad (3) $G+I \ge 2L$ \quad (4) $AL \le I^2$
\end{example}
\begin{solution}
这些结论都可由“最佳系数问题”的结论和单调性放缩推出。具体地，
\[
\frac{2A+G}{3} = He_1 \le M_{2/3} \le I,\qquad
A+L \le M_1 + M_{1/3} \le 2M_{2/3} \le 2I,
\]
\[
G+I \ge M_0 + M_{2/3} \ge 2M_{1/3} \ge 2L,\qquad
AL \le M_1 M_{1/3} \le M_{2/3}^2 \le I^2.
\]
四式分别对应所求的不等式。
\end{solution}

\section{若干补充不等式}

\begin{example}
高阶放缩补充：证明 $AG^2 \le L^3$。
\end{example}
\begin{solution}
采用双曲参数化，等价于证明 $\cosh t \le \left(\frac{\sinh t}{t}\right)^3 \iff t^3 \cosh t \le \sinh^3 t$。
两边分别展开为泰勒级数：
左侧 $= \sum_{n=1}^{\infty} \frac{1}{(2n-2)!} t^{2n+1}$，
右侧利用三倍角公式 $= \sum_{n=1}^{\infty} \frac{3^{2n+1}-3}{4(2n+1)!} t^{2n+1}$。
对比 $t^{2n+1}$ 的系数：$n=1$ 与 $n=2$ 时两边系数精准相等（1 与 1/2）。
当 $n \ge 3$ 时，需证 $3^{2n+1}-3 > 4(2n+1)(2n)(2n-1)$。左端随 $n$ 指数增长，右端仅为三次多项式，因此该不等式对一切 $n \ge 3$ 成立，系数之差均为正，不等式严格成立。
\end{solution}

\begin{example}
关于 $M_{1/2}$ 的极值界定：证明 $L+I \le 2M_{1/2}$ 与 $LI \le M_{1/2}^2$。
\end{example}
\begin{solution}
由定义展开，$M_{1/2} = \left(\frac{x^{1/2}+y^{1/2}}{2}\right)^2 = \frac{x+y+2\sqrt{xy}}{4} = \frac{A+G}{2}$，即 $2M_{1/2} = A+G$。
不等式 $L+I \le 2M_{1/2}$ 等价于 $L+I \le A+G$，而这在本文第 4 节中已证明，因此可以沿用该结论。
对于乘积形式，根据基础 AM-GM 不等式，有 $\sqrt{LI} \le \frac{L+I}{2}$。结合已证的 $\frac{L+I}{2} \le M_{1/2}$，连起来平方即得 $LI \le M_{1/2}^2$。
\end{solution}

\begin{example}
连续分式极值：证明 $\frac{1}{3} < \frac{I-L}{A-G} < \frac{2}{e} < \frac{I+L}{A+G}$。
\end{example}
\begin{solution}
左端不等式等价于 $3I - 3L > A-G$。由已知 $2A+G \le 3I$ 得 $3I \ge 2A+G$，又由 $2G+A \ge 3L$ 得 $-3L \ge -A-2G$。两式相加即有 $3I - 3L \ge A-G$，而 $x \neq y$ 时不等号严格，故
\[
\frac{I-L}{A-G} > \frac{1}{3}.
\]

右端不等式等价于 $e(I+L) > 2A+2G$。其中 $eL > 2G$ 由 $L>G$ 与 $e>2$ 立即得到；对 $eI > 2A$，将双曲形式代入后等价于 $e^{t\coth t} > 2\cosh t$。两边取对数并令 $u = e^{-2t} \in (0,1)$，可化为
\[
-\frac{u\ln u}{1-u} > \ln(1+u).
\]
对应辅助函数的二阶导数恒小于 $0$，且两端取值都为 $0$，因此区间内部恒为正，于是
\[
\frac{I+L}{A+G} > \frac{2}{e}.
\]

中间不等式等价于 $e(I-L) < 2(A-G)$。构造
\[
H(t) = 2(\cosh t - 1) - e\exp(t\coth t - 1) + e\frac{\sinh t}{t}.
\]
在 $t=0$ 附近作泰勒展开，
\[
H(t) = \left(1-\frac{e}{6}\right)t^2 + \left(\frac{1}{12}-\frac{e}{40}\right)t^4 + \cdots .
\]
由展开式可见，$H(t)$ 在 $t>0$ 时恒为正，因此
\[
\frac{I-L}{A-G} < \frac{2}{e}.
\]
综上，$\frac{1}{3} < \frac{I-L}{A-G} < \frac{2}{e} < \frac{I+L}{A+G}$。
\end{solution}

\begin{example}
在附加条件 $0 < x < y \le e$ 下，证明不对称幂链
\[
G^A < L^I < I^L < A^G.
\]
\end{example}
\begin{solution}
原命题若不加范围限制并不成立。例如取 $x=10,\ y=1$，则
\[
A^G \approx 219.4 < G^A \approx 562.3.
\]
因此必须加入 \(0<x<y\le e\) 这一条件。此时有
\[
0 < G < L < I < A \le e.
\]

关系 $L^I < I^L$ 可由函数 $f(z) = \frac{\ln z}{z}$ 的单调性推出。其导数
\[
f'(z) = \frac{1-\ln z}{z^2}
\]
在区间 \((0,e]\) 上严格为正，因此 \(f\) 单调递增。由 \(L<I\le e\) 可得
\[
\frac{\ln L}{L}<\frac{\ln I}{I},
\]
从而 \(L^I<I^L\)。

而 $I^L < A^G$ 在取对数后等价于
\[
L \ln I < G \ln A.
\]
除以 \(G\) 后写成
\[
L_0 \ln(I_0 G) < \ln(A_0 G),
\]
其中下标 \(0\) 表示 \(x=Ge^v,\ y=Ge^{-v}\) 时对应的双曲均值形式。因为限制了 $x,y \le e$，故必有 $\ln G + v \le 1 \implies \ln G \le 1-v$，于是原不等式转化为极值界限
\[
(1-v)(L_0 - 1) < \ln A_0 - L_0 \ln I_0.
\]
移项后构造辅助函数
\[
D(v) = \ln A_0 - L_0 \ln I_0 - (1-v)(L_0 - 1).
\]
代入
\[
A_0 = \cosh v,\qquad L_0 = \frac{\sinh v}{v},\qquad \ln I_0 = v\coth v - 1,
\]
可化为
\begin{align*}
D(v) &= \ln \cosh v - \frac{\sinh v}{v}(v\coth v - 1) - (1-v)\left(\frac{\sinh v}{v} - 1\right) \\
&= \ln \cosh v - \cosh v + \sinh v + 1 - v.
\end{align*}
再求导得
\[
D'(v) = \tanh v - \sinh v + \cosh v - 1
= \tanh v + e^{-v} - 1
= \frac{e^{-v}(1-e^{-v})^2}{1+e^{-2v}} > 0.
\]
因为 \(D(0)=0\) 且 \(D'(v)>0\)，所以 \(D(v)>0\)，从而 \(I^L<A^G\)。另一侧不等式 \(G^A<L^I\) 同理，于是得到
\[
G^A < L^I < I^L < A^G.
\]
\end{solution}
