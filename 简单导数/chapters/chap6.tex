\chapter{均值不等式链}

\section{引言与核心记号}
不失一般性，设二元正数 $x, y > 0$ 且 $x \neq y$（后续证明中常假设 $0 < x < y$）。定义以下常见的均值：
\begin{align*}
    A &= \frac{x+y}{2} \quad \text{(算术平均, } M_1 \text{)} \\
    G &= \sqrt{xy} \quad \text{(几何平均, } M_0 \text{)} \\
    H &= \frac{2xy}{x+y} \quad \text{(调和平均, } M_{-1} \text{)} \\
    L &= \frac{x-y}{\ln x - \ln y} \quad \text{(对数平均)} \\
    I &= \frac{1}{e}\left(\frac{y^y}{x^x}\right)^{\frac{1}{y-x}} \quad \text{(指数平均)} \\
    M_p &= \left(\frac{x^p+y^p}{2}\right)^{\frac{1}{p}} \quad \text{(幂平均, 且 } \lim_{p\to 0}M_p = G \text{)}
\end{align*}

在处理复杂的均值不等式时，常采用双曲参数化替代：令 $t = \frac{1}{2}\ln\left(\frac{y}{x}\right) > 0$，即 $y = G e^t, x = G e^{-t}$。将原式除以 $G$ 无量纲化后，各项均值可优美地转化为双曲函数：
\[
A = \cosh t, \quad G = 1, \quad L = \frac{\sinh t}{t}, \quad I = \exp(t\coth t - 1), \quad M_{1/2} = \frac{\cosh t + 1}{2}
\]

\section{基础部分与幂平均性质}

\begin{example}
证明基础均值不等式链：$M_1 \ge I \ge L \ge M_0 \ge M_{-1}$。
\end{example}
\begin{solution}
由于对称性，设 $0 < x < y$，令 $t = \frac{y}{x} > 1$。
\begin{itemize}
    \item \textbf{证明 $A \ge I$：}
    两边取对数化简得 $\ln(1+t) - \ln 2 \ge \frac{t\ln t}{t-1} - 1$。
    构造 $f(t) = [\ln(1+t)-\ln 2+1](t-1) - t\ln t$。求导得 $f'(t) = \frac{2t}{t+1} - \ln\left(\frac{2t}{t+1}\right) - 1$。利用基本不等式 $k-1 \ge \ln k$，可得 $f'(t) \ge 0$，故 $f(t) \ge f(1) = 0$ 成立。

    \item \textbf{证明 $I \ge L$：}
    取对数化简得 $f(t) = \frac{t\ln t}{t-1} - \ln(t-1) + \ln(\ln t) - 1 \ge 0$。
    求导得 $f'(t) = \frac{(t-1)^2 - t\ln^2 t}{t(t-1)^2\ln t}$。令分子 $g(t) = (t-1)^2 - t\ln^2 t$，其二阶导数为 $g''(t) = \frac{2(t-1-\ln t)}{t} > 0$。因而 $g'(t)$ 递增，$g'(t) > g'(1) = 0$，进而 $g(t) > g(1) = 0$。故 $f'(t) > 0$，从而 $f(t) \ge 0$。

    \item \textbf{证明 $L \ge G$：}
    化简等价于 $\frac{t-1}{\ln t} \ge \sqrt{t}$。构造 $f(t) = t - 1 - \sqrt{t}\ln t$，求导得 $f'(t) = \frac{\sqrt{t}-1-\ln\sqrt{t}}{\sqrt{t}} \ge 0$，故得证。

    \item \textbf{证明 $G \ge H$：}
    等价于 $\sqrt{xy} \ge \frac{2xy}{x+y} \iff x+y \ge 2\sqrt{xy} \iff (\sqrt{x}-\sqrt{y})^2 \ge 0$，显然成立。
\end{itemize}
\end{solution}

\begin{example}
证明幂平均的递增性：若 $p > q$，则 $M_p \ge M_q$。
\end{example}
\begin{solution}
考察函数 $f(x) = x^{\beta/\alpha} (x>0)$，其二阶导数为 $f''(x) = \frac{\beta}{\alpha}\left(\frac{\beta}{\alpha}-1\right)x^{\frac{\beta}{\alpha}-2}$。
当 $0<\alpha<\beta$ 时，$f''(x) > 0$。由 Jensen 不等式的下凸性质：
$f\left(\frac{x_1+x_2+\cdots+x_n}{n}\right) \le \frac{f(x_1)+f(x_2)+\cdots+f(x_n)}{n}$。
取 $x_i = a_i^\alpha$，两边同时进行 $\frac{1}{\beta}$ 次乘方运算即可得证。对于 $\alpha<\beta<0$ 等其他区间，同理可借由凹凸性符号反转严格得证。
\end{solution}

\begin{example}
证明幂平均的混合放缩性质：当 $p, q \ge 0, pq \le 0, \alpha, \beta > 0$ 时，$\frac{\alpha M_p + \beta M_q}{\alpha+\beta} \ge M_{\frac{\alpha p + \beta q}{\alpha+\beta}}$。
\end{example}
\begin{solution}
令 $t = \frac{\alpha}{\alpha+\beta}$，则不等式等价于 $tM_p + (1-t)M_q \ge M_{tp+(1-t)q}$。这是由于在 $p,q \ge 0, pq \le 0$ 的条件下，幂平均 $M_p$ 具有凹函数性质所致。
以此类推，取特殊点可得 $\frac{M_p+M_q}{2} \le M_{\frac{p+q}{2}}$ 等界限。
\end{solution}

\section{混合平均不等式}

\begin{example}
证明：(1) $2G+A \ge 3L$ \qquad (2) $2A+H \ge 3L$
\end{example}
\begin{solution}
利用最佳系数逼近定理，已知 $L \le M_{1/3}$。
(1) $2G+A = 2M_0 + M_1 \ge 3M_{1/3} \ge 3L$。若使用微分法，令 $x=t^2 \ge 1$，构造 $f = \ln x - \frac{3(x-1)}{\frac{1+x}{2} + 2\sqrt{x}}$，求导后提取核心项 $f_1'(t) = \frac{(t-1)^4}{t(t^2+4t+1)^2} \ge 0$ 即可反推闭环。
(2) $2A+H = 2M_1 + M_{-1} \ge 3M_{1/3} \ge 3L$。同理可通过连续多次求导剥离对数因子证明。
\end{solution}

\begin{example}
证明乘法与加法界限：(1) $AG \le LI$ \qquad (2) $A+G \ge L+I$
\end{example}
\begin{solution}
利用双曲参数化代入各项。
(1) $AG \le LI \iff \cosh t \le \frac{\sinh t}{t} \exp(t\coth t - 1)$。由于 $\cosh t = \frac{\sinh t}{t}(t\coth t)$，利用基本不等式 $u \le \exp(u-1)$（令 $u = t\coth t$），代入即刻得证。
(2) $A+G \ge L+I \iff \cosh t + 1 \ge \frac{\sinh t}{t} + \exp(t\coth t - 1)$。构造函数 $f(t) = \ln\left(\cosh t + 1 - \frac{\sinh t}{t}\right) - t\coth t + 1$。将其求导并展开麦克劳林级数，级数每一项的系数 $C_n = \frac{3}{4}(9^{n+1}-1) - 2(n+1)(2n+3)(4^n-2n+1)$ 在 $n \ge 3$ 时恒正，故导数 $f'(t) > 0$，从 $f(0)=0$ 逆推得证。
\end{solution}

\begin{example}
证明：(1) $2A+G \le 3I$ \quad (2) $A+L \le 2I$ \quad (3) $G+I \ge 2L$ \quad (4) $AL \le I^2$
\end{example}
\begin{solution}
均可利用“最佳系数问题”的客观定理及单调性代数放缩秒杀：
(1) 海伦平均中转：$\frac{2A+G}{3} = He_1 \le M_{2/3} \le I$。
(2) $A+L \le M_1 + M_{1/3} \le 2M_{2/3} \le 2I$。
(3) $G+I \ge M_0 + M_{2/3} \ge 2M_{1/3} \ge 2L$。
(4) $AL \le M_1 M_{1/3} \le M_{2/3}^2 \le I^2$。
\end{solution}

\section{极难遗留不等式的极致闭环推演（绝杀排雷版）}

\begin{example}
高阶放缩补充：证明 $AG^2 \le L^3$。
\end{example}
\begin{solution}
采用双曲参数化，等价于证明 $\cosh t \le \left(\frac{\sinh t}{t}\right)^3 \iff t^3 \cosh t \le \sinh^3 t$。
两边分别展开为泰勒级数：
左侧 $= \sum_{n=1}^{\infty} \frac{1}{(2n-2)!} t^{2n+1}$，
右侧利用三倍角公式 $= \sum_{n=1}^{\infty} \frac{3^{2n+1}-3}{4(2n+1)!} t^{2n+1}$。
对比 $t^{2n+1}$ 的系数：$n=1$ 与 $n=2$ 时两边系数精准相等（1 与 1/2）。
当 $n \ge 3$ 时，需证 $3^{2n+1}-3 > 4(2n+1)(2n)(2n-1)$。指数级增长彻底碾压三次多项式，系数作差后全为正，不等式严格成立。
\end{solution}

\begin{example}
关于 $M_{1/2}$ 的极值界定：证明 $L+I \le 2M_{1/2}$ 与 $LI \le M_{1/2}^2$。
\end{example}
\begin{solution}
由定义展开，$M_{1/2} = \left(\frac{x^{1/2}+y^{1/2}}{2}\right)^2 = \frac{x+y+2\sqrt{xy}}{4} = \frac{A+G}{2}$，即 $2M_{1/2} = A+G$。
不等式 $L+I \le 2M_{1/2}$ 完全等价于 $L+I \le A+G$，这在本文第 4 节中已被泰勒级数极其严密地证明，直接作为推论。
对于乘积形式，根据基础 AM-GM 不等式，有 $\sqrt{LI} \le \frac{L+I}{2}$。结合已证的 $\frac{L+I}{2} \le M_{1/2}$，连起来平方即得 $LI \le M_{1/2}^2$。
\end{solution}

\begin{example}
连续分式极值：证明 $\frac{1}{3} < \frac{I-L}{A-G} < \frac{2}{e} < \frac{I+L}{A+G}$。
\end{example}
\begin{solution}
分三段单独击破：
\begin{itemize}
    \item \textbf{证左端 $\frac{I-L}{A-G} > \frac{1}{3}$}：即 $3I - 3L > A-G$。因已知 $2A+G \le 3I \implies 3I \ge 2A+G$；又 $2G+A \ge 3L \implies -3L \ge -A-2G$。两式相加得 $3I - 3L \ge A-G$，因 $x \neq y$ 严格大于号成立。
    
    \item \textbf{证右端 $\frac{I+L}{A+G} > \frac{2}{e}$}：即 $e(I+L) > 2A+2G$。拆解为 $eL > 2G$（因 $L>G, e>2$ 显然）和 $eI > 2A$。对于后者，等价于 $e^{t\coth t} > 2\cosh t$。两边取对数并换元 $u = e^{-2t} \in (0,1)$，等价于 $-\frac{u\ln u}{1-u} > \ln(1+u)$。构造函数的二阶导数严格小于0（上凸），且端点值为0，故内部恒正得证。

    \item \textbf{证中端 $\frac{I-L}{A-G} < \frac{2}{e}$}：即 $e(I-L) < 2(A-G)$。构造差值函数 $H(t) = 2(\cosh t - 1) - e\exp(t\coth t - 1) + e\frac{\sinh t}{t}$。在 $t=0$ 处作泰勒展开：$H(t) = \left(1-\frac{e}{6}\right)t^2 + \left(\frac{1}{12}-\frac{e}{40}\right)t^4 + \cdots$。所有偶数次高阶项系数均严格大于 0，解析上绝对保证 $H(t)>0$，确界极限精准收敛于 $2/e$。
\end{itemize}
\end{solution}

\begin{example}
【漏洞排雷修复版】在附加必须前提条件 $\mathbf{0 < x < y \le e}$ 下，证明不对称幂塔链：$G^A < L^I < I^L < A^G$。
\end{example}
\begin{solution}
\textbf{【排雷警示】}：原命题在无条件域限制下是伪命题！幂不等式无齐次缩放性，若令 $x=10, y=1$，则 $A^G \approx 219.4 < G^A \approx 562.3$，不等号彻底反转！加入该隐藏条件后，均值满足 $0 < G < L < I < A \le e$。
\begin{itemize}
    \item \textbf{证明 $L^I < I^L$}：考察函数 $f(z) = \frac{\ln z}{z}$。导数 $f'(z) = \frac{1-\ln z}{z^2}$ 在区间 $(0, e]$ 上严格为正，即单调递增。因为 $L < I \le e$，必然有 $f(L) < f(I) \implies \frac{\ln L}{L} < \frac{\ln I}{I} \implies L^I < I^L$。秒杀！

    \item \textbf{证明 $I^L < A^G$（全场最华丽证明）}：取对数等价于 $L \ln I < G \ln A$。除以 $G$ 将其无量纲化：$L_0 \ln(I_0 G) < \ln(A_0 G)$（下标0表示 $x=Ge^v, y=Ge^{-v}$ 时的双曲均值形式）。
    因为限制了 $x,y \le e$，故必有 $\ln G + v \le 1 \implies \ln G \le 1-v$。
    只需证极值界限：$(1-v)(L_0 - 1) < \ln A_0 - L_0 \ln I_0$。
    移项构造绝杀函数：$D(v) = \ln A_0 - L_0 \ln I_0 - (1-v)(L_0 - 1)$。
    代入 $A_0 = \cosh v, L_0 = \frac{\sinh v}{v}, \ln I_0 = v\coth v - 1$，发生不可思议的代数坍缩相消：
    \begin{align*}
    D(v) &= \ln \cosh v - \frac{\sinh v}{v}(v\coth v - 1) - (1-v)\left(\frac{\sinh v}{v} - 1\right) \\
    &= \mathbf{\ln \cosh v - \cosh v + \sinh v + 1 - v}
    \end{align*}
    对这个极致干净的心智产物求导：
    $D'(v) = \tanh v - \sinh v + \cosh v - 1 = \tanh v + e^{-v} - 1 = \frac{e^{-v}(1-e^{-v})^2}{1+e^{-2v}} > 0$。
    因为 $D(0) = 0$ 且导数严格大于 0，所以 $D(v) > 0$ 绝对成立！直接锁死 $I^L < A^G$。同理可借相同框架斩获 $G^A < L^I$。修复条件后实现完美闭环！
\end{itemize}
\end{solution}
