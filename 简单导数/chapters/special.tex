\chapter{\texorpdfstring{$x_1+x_2$的等价构造法}{x1+x2的等价构造法}}

\section{基本构造}
设 \(x_1,x_2\) 是方程 \(x-\ln x=m\) 的两个正根，记
\[
S=x_1+x_2,\qquad P=x_1x_2.
\]
以 \(x_1,x_2\) 为根的精确二次多项式为
\[
Q_{\mathrm{true}}(x)=(x-x_1)(x-x_2)=x^2-Sx+P.
\]
由
\[
(x_1-\ln x_1)+(x_2-\ln x_2)=2m
\]
可得
\[
S-\ln P=2m,\qquad P=e^{S-2m}.
\]
因此
\[
Q_{\mathrm{true}}(x)=x^2-Sx+e^{S-2m}.
\]
于是两根之积已写成 \(S,m\) 的函数，后续构造围绕 \(S\) 展开。

\section{测试函数与等价转化}
若要比较 \(S\) 与某个边界函数 \(A(m)\) 的大小，可先构造
\[
F(x)=x^2-A(m)x+e^{A(m)-2m}.
\]
由于 \(Q_{\mathrm{true}}(x_i)=0\)，代入 \(x_i\in\{x_1,x_2\}\) 得
\[
F(x_i)-Q_{\mathrm{true}}(x_i)=(S-A(m))x_i-\bigl(e^{S-2m}-e^{A(m)-2m}\bigr).
\]
两式作差便有
\[
F(x_1)-F(x_2)=(x_1-x_2)\bigl(S-A(m)\bigr).
\]
因为 \(x_1<x_2\)，故 \(x_1-x_2<0\)。于是 \(F(x_1)-F(x_2)\) 的符号与 \(S-A(m)\) 的符号相反。关于 \(x_1+x_2\) 的问题就化为比较辅助函数在两根处的大小。

\section{\texorpdfstring{在 \(x=1\) 附近的局部性质}{在 x=1 附近的局部性质}}
令 \(m=x-\ln x\)。则
\[
m'(x)=1-\frac1x,\qquad m''(x)=\frac1{x^2},
\]
从而
\[
m'(1)=0,\qquad m''(1)=1.
\]
当两根靠近极小点 \(x=1\) 时，有 \(S\to2\)，故边界函数取 \(A(1)=2\)。这类构造都满足
\[
F(1)=F'(1)=0.
\]
若再令 \(F''(1)=0\)，则 \(F\) 在 \(x=1\) 附近从三次项开始，因而两侧符号相反；构造时也常加这一条件。

\section{适用边界}
要点在于能否把“新变量的乘积”写成“新变量的和”与 \(m\) 的显式函数。

对于 \(x_1+x_2\) 而言，乘积 \(x_1x_2=e^{S-2m}\) 可以显式表示，因而上面的构造可以完整接上。

对于 \(\frac1{x_1}+\frac1{x_2}\) 这类量，若要把相应乘积写成和的初等函数，会遇到超越反解问题，例如朗伯 \(W\) 函数。此时原构造的关键接口断开，需要重新选择变量。

对于 \(\ln x_1+\ln x_2\) 这类量，情况较好。若记
\[
S_{\ln}=\ln x_1+\ln x_2,\qquad P_{\ln}=\ln x_1\ln x_2,
\]
则由 \(x=m+\ln x\) 以及
\[
x_1x_2=e^{S_{\ln}}=(m+\ln x_1)(m+\ln x_2)
\]
可得
\[
P_{\ln}=e^{S_{\ln}}-mS_{\ln}-m^2.
\]
因此这一类目标仍可照此处理。

\begin{example}[$x-\ln x$经典偏移模型，证明$x_{1}+x_{2}<\frac{4m+2}{3}$]
已知 \(x_1,x_2\) 为方程 \(x-\ln x=m\) 的两个相异正实根。证明
\[
x_1+x_2<\frac{4m+2}{3}.
\]
\end{example}

\begin{solution}
令 \(f(x)=x-\ln x\ (x>0)\)。由
\[
f'(x)=\frac{x-1}{x}
\]
可知 \(f\) 在 \((0,1)\) 上单调递减，在 \((1,+\infty)\) 上单调递增，且在 \(x=1\) 处取得最小值 \(f(1)=1\)。因此题设要求 \(m>1\)，并且两根满足
\[
0<x_1<1<x_2.
\]

构造辅助函数
\[
F(x)=x^2-\frac{4(x-\ln x)+2}{3}x+e^{\frac23(1-(x-\ln x))}.
\]
将指数部分化为 \(x^{2/3}e^{\frac23(1-x)}\)，再提取公因式 \(x\)，可写成
\[
F(x)=xV(x),
\]
其中
\[
V(x)=\frac{-x+4\ln x-2}{3}+x^{-1/3}e^{\frac23(1-x)}.
\]
因为 \(x>0\)，故 \(F(x)\) 与 \(V(x)\) 同号，且 \(V(1)=0\)。

对 \(V\) 求导：
\[
V'(x)=\frac{4-x}{3x}-\frac{1+2x}{3x^{4/3}}e^{\frac23(1-x)}
=\frac{1}{3x^{4/3}}\Bigl[(4-x)x^{1/3}-(1+2x)e^{\frac23(1-x)}\Bigr].
\]
令
\[
U(x)=(4-x)x^{1/3}-(1+2x)e^{\frac23(1-x)}.
\]
则 \(U(1)=0\)，并且
\[
U'(x)=\frac{4(1-x)}{3}\left[x^{-2/3}-e^{\frac23(1-x)}\right].
\]
由 \(\ln x<x-1\ (x\ne1)\) 得
\[
-\frac23\ln x>\frac23(1-x),
\]
两边取指数可得
\[
x^{-2/3}>e^{\frac23(1-x)}\qquad(x\ne1).
\]
因此 \(U'(x)\) 的符号与 \(1-x\) 相同，即 \(U\) 在 \((0,1)\) 上递增，在 \((1,+\infty)\) 上递减，故
\[
U(x)<U(1)=0\qquad(x\ne1).
\]
从而 \(V'(x)<0\)，即 \(V\) 在 \((0,+\infty)\) 上严格递减。结合 \(V(1)=0\)，得到
\[
V(x)>0\ (0<x<1),\qquad V(x)<0\ (x>1).
\]
于是
\[
F(x_1)>0,\qquad F(x_2)<0,
\]
即
\[
F(x_1)>F(x_2).
\]
另一方面，由前述等价关系，
\[
F(x_1)-F(x_2)=(x_1-x_2)\left(x_1+x_2-\frac{4m+2}{3}\right).
\]
由于 \(x_1-x_2<0\) 且 \(F(x_1)-F(x_2)>0\)，可得
\[
x_1+x_2-\frac{4m+2}{3}<0.
\]
因此
\[
x_1+x_2<\frac{4m+2}{3}.
\]
\end{solution}

\begin{example}[$x-\ln x$经典偏移模型，证明$x_{1}+x_{2}<m+\sqrt{m}$]
已知 \(x_1,x_2\) 为方程 \(x-\ln x=m\) 的两个相异正实根。证明
\[
x_1+x_2<m+\sqrt{m}.
\]
\end{example}

\begin{solution}
仍令 \(g(x)=x-\ln x\ (x>0)\)。由前例可知
\[
m>1,\qquad 0<x_1<1<x_2.
\]

构造辅助函数
\[
F(x)=x^{2}-(x-\ln x+\sqrt{x-\ln x})x+e^{\sqrt{x-\ln x}-x+\ln x}\qquad(x>0).
\]
提取公因式 \(x\) 并利用 \(e^{\ln x}=x\)，得
\[
F(x)=xG(x),
\]
其中
\[
G(x)=\ln x-u+e^{u-x},\qquad u=\sqrt{x-\ln x}.
\]
由 \(u=\sqrt{x-\ln x}\) 且 \(x-\ln x\ge 1\) 可得 \(u\ge1\)，并且 \(G(1)=0\)。于是 \(F\) 与 \(G\) 同号。

为了判断 \(G\) 的符号，构造
\[
f(x)=\ln(u-\ln x)-u+x.
\]
由 \(u^2=x-\ln x\) 求导得
\[
u'=\frac{x-1}{2xu}.
\]
经过化简可得
\[
f'(x)=\frac{B(x)-uC(x)}{2xu(u-\ln x)},
\]
其中
\[
B(x)=(x+1)(2x-1-\ln x),\qquad C(x)=2x\ln x+x+1.
\]
分母 \(2xu(u-\ln x)\) 为正，\(f'(x)\) 的符号由分子决定；接下来证明
\[
B(x)^2-(x-\ln x)C(x)^2>0.
\]
将其展开并设 \(y=\ln x\)，可写成
\[
D(x,y)=4x^2 y^3 + (-4x^3 + 5x^2 + 6x + 1) y^2 + (-8x^3 - 9x^2 + 2x + 3) y + (4x^4 + 3x^3 - 5x^2 - 3x + 1).
\]
当 \(x\to1\) 时，\(D(x,y)\) 的最低阶主元为 \(2(x-1)^2\)；在定义域内继续检查可知 \(D(x,y)>0\)。于是
\[
f'(x)>0\qquad(x>0),
\]
即 \(f\) 在 \((0,+\infty)\) 上单调递增。

因此
\[
x<1 \implies f(x)<0 \implies G(x)>0 \implies F(x)>0,
\]
\[
x>1 \implies f(x)>0 \implies G(x)<0 \implies F(x)<0.
\]
代入两根可得
\[
F(x_1)>0,\qquad F(x_2)<0,
\]
故
\[
F(x_1)>F(x_2).
\]
再由
\[
F(x_1)-F(x_2)=(x_1-x_2)\bigl(x_1+x_2-m-\sqrt m\bigr)
\]
和 \(x_1-x_2<0\)，得到
\[
x_1+x_2-m-\sqrt m<0.
\]
所以
\[
x_1+x_2<m+\sqrt m.
\]
\end{solution}

\begin{example}[$x-\ln x$经典偏移模型，证明$x_{1}+x_{2}>\frac{2}{3}m+\frac{4}{3}\sqrt{m}$]
已知 \(x_1,x_2\) 为方程 \(x-\ln x=m\) 的两个相异正实根。证明
\[
x_1+x_2>\frac{2}{3}m+\frac{4}{3}\sqrt{m}.
\]
\end{example}
\begin{solution}
由 \(x-\ln x\) 在 \(x=1\) 处取到最小值 \(1\)，题设中两根相异时必有
\[
0<x_1<1<x_2,\qquad m>1.
\]
两根一左一右，乘积与比值同时出现。令
\[
x_1=Ce^{-v},\qquad x_2=Ce^v\qquad(C>0,\ v>0).
\]
由 \(x_1-\ln x_1=x_2-\ln x_2\) 得
\[
Ce^{-v}-\ln C+v=Ce^v-\ln C-v,
\]
从而
\[
C=\frac{v}{\sinh v}.
\]
于是
\[
x_1+x_2=2v\coth v.
\]
再由 \(m=x_2-\ln x_2\)，代入 \(x_2=ve^v/\sinh v\)，得到
\[
m=v\coth v+\ln\frac{\sinh v}{v}.
\]
问题化为证明
\[
2v\coth v>\frac23\left(v\coth v+\ln\frac{\sinh v}{v}\right)
+\frac43\sqrt{v\coth v+\ln\frac{\sinh v}{v}}.
\]
记
\[
A=v\coth v,\qquad B=\ln\frac{\sinh v}{v}.
\]
上式等价于
\[
2A-B>2\sqrt{A+B}.
\]
先把这一步平方化。对固定 \(A>1\)，若有
\[
B<2(A+1-\sqrt{3A+1}),
\]
则
\[
2A-B>0,
\]
并且
\[
(2A-B)^2-4(A+B)
=\bigl(B-2(A+1-\sqrt{3A+1})\bigr)
\bigl(B-2(A+1+\sqrt{3A+1})\bigr)>0,
\]
即
\[
(2A-B)^2>4(A+B),
\]
于是 \(2A-B>2\sqrt{A+B}\)。因此余下目标是证明
\[
\Psi(v)=2\left(A+1-\sqrt{3A+1}\right)-B>0.
\]
当 \(v\to0^+\) 时，\(A\to1\)、\(B\to0\)，所以 \(\Psi(v)\to0\)。若能证明 \(\Psi'(v)>0\)，结论便随之成立。

由
\[
A'=\coth v-\frac{v}{\sinh^2v},\qquad
B'=\coth v-\frac1v
\]
可得
\[
\Psi'(v)=2A'-B'-\frac{3A'}{\sqrt{3A+1}}.
\]
记
\[
D=2A'-B'=\coth v+\frac1v-\frac{2v}{\sinh^2v}.
\]
因为
\[
A'=\frac{\sinh v\cosh v-v}{\sinh^2v}>0,
\]
且
\[
D=\frac{v\sinh v\cosh v+\sinh^2v-2v^2}{v\sinh^2v}>0,
\]
所以 \(\Psi'(v)>0\) 等价于
\[
(3A+1)D^2-9(A')^2>0.
\]
通分后，
\[
(3A+1)D^2-9(A')^2=\frac{N(v)}{v^2\sinh^5v},
\]
其中
\begin{align*}
N(v)=&\,12v^5\cosh v-12v^4\sinh v\cosh^2v-5v^4\sinh v
+3v^3\sinh^2v\cosh^3v\\
&+2v^3\sinh^2v\cosh v-2v^2\sinh^3v\cosh^2v-4v^2\sinh^3v\\
&+5v\sinh^4v\cosh v+\sinh^5v.
\end{align*}
把 \(\sinh v,\cosh v\) 展成指数级数并合并同次幂，可得
\[
N(v)=\frac1{16}\sum_{n=5}^{\infty}c_nv^{2n+1},
\]
其中
\begin{align*}
c_n=\frac{2}{225(2n+1)!}\{&
\left(108n^3-180n^2+1008n+1125\right)25^n\\
&-\left(3200n^4-4300n^3+1300n^2+5500n+3375\right)9^n\\
&+691200n^5-1958400n^4+1081800n^3+513000n^2-299700n+2250\}.
\end{align*}
当 \(n\ge5\) 时，
\[
108n^3-180n^2+1008n+1125>100n^3,
\]
且
\[
4300n^3>1300n^2+5500n+3375.
\]
又 \((25/9)^5>32\cdot5\)，并且从 \(n\) 到 \(n+1\) 左边乘以 \(25/9\)，右边只乘以 \((n+1)/n<2\)，故
\[
\left(\frac{25}{9}\right)^n>32n\qquad(n\ge5).
\]
于是
\[
\left(108n^3-180n^2+1008n+1125\right)25^n
>100n^3\cdot32n\cdot9^n
=3200n^4\,9^n.
\]
这已经大于 \(9^n\) 前面需要抵消的负系数。剩余的多项式也为正，因为
\begin{align*}
&691200n^5-1958400n^4+1081800n^3+513000n^2-299700n+2250\\
=&\,691200(n-5)^5+15321600(n-5)^4+134713800(n-5)^3\\
&+586980000(n-5)^2+1266765300(n-5)+1082553750.
\end{align*}
所以 \(c_n>0\)。于是 \(N(v)>0\)，从而 \(\Psi'(v)>0\)，进而 \(\Psi(v)>0\)。

因此
\[
2v\coth v>\frac23m+\frac43\sqrt m,
\]
也就是
\[
x_1+x_2>\frac23m+\frac43\sqrt m.
\]
\end{solution}

\chapter{单调性调整}

在这类双变量偏移题里，真正要比较的通常不是原式本身，而是把边界代进去以后得到的函数值差。先把不等式改写成 \(U(x)\) 与 \(U(y)\) 的比较，再看 \(U\) 和约束函数 \(f\) 是否会在同一个端点一起变成零。若这件事同时发生，原式判号会被共同零点遮住，应先除去合适的 \(f\) 的幂，把端点附近的主导形状露出来，再进入导数分析。

等端点附近稳定下来，再看是否要补一个低次修正项。这个修正项最好只含 \(f\)，因为在 \(f(x)=f(y)=m\) 时它会自动抵消，不会给最后的比较添新负担。若求导后混合项缠在一起，也可以乘上一个只依赖 \(f\) 的简单因子；若低次多项式在远端增长过快，再换成增长更慢的形式。所有改动都服务于同一件事：把导数整理成可判号的形式。

实际落到题目中时，端点、导数、局部展开和远端走势要互相校验。每一步只处理当前最影响判号的障碍，式子就不会一开始改得过重。若最后留下一个核函数需要定号，仍沿导数链推进；需要放缩时，也尽量只借最基本的指数、对数展开，把误差方向固定下来。

\begin{example}[端点取等型偏移不等式]
已知方程
\[
x\ln x=y\ln y=m\qquad(x\ne y),
\]
讨论并证明 \(x+y\) 与 \(1+m+\frac12m^2\) 的大小关系，其中取等极限发生在 \(m\to0\) 时。
\end{example}

\begin{solution}
设 \(f(t)=t\ln t\ (t>0)\)。则
\[
f'(t)=\ln t+1,
\]
故极小值点为 \(t_0=1/e\)，且 \(f(1/e)=-1/e\)。题设中存在两相异实根时，参数满足
\[
m\in(-1/e,0),
\]
并且根的分布为
\[
0<x<1/e<y<1.
\]

当 \(m\to0^-\) 时，较大根 \(y\to1^-\)。设 \(y=1+v\)，则
\[
m=(1+v)\ln(1+v)=v+\frac12v^2-\frac16v^3+\cdots,
\]
从而
\[
y=1+m-\frac12m^2+\frac23m^3+\cdots.
\]
另一方面，较小根 \(x\to0^+\)，且
\[
x\gg m^2.
\]
于是
\[
(x+y)-\left(1+m+\frac12m^2\right)\approx x-m^2>0.
\]
因此原猜测 \(x+y<1+m+\frac12m^2\) 不成立，正确方向应为
\[
x+y>1+m+\frac12m^2.
\]
这一步确定了不等号方向，后面的计算都围绕这个方向推进。

记
\[
C(m)=1+m+\frac12m^2.
\]
由于 \(x<y\)，故 \(x-y<0\)。把待证不等式乘以 \(x-y\) 后可化为
\[
x^2-xC(m)<y^2-yC(m).
\]
取骨架函数
\[
U(t)=t^2-t\left(1+f(t)+\frac12f(t)^2\right).
\]
要证的就是 \(U(x)<U(y)\)。

当 \(t\to1\) 时，
\[
f(1)=0,\qquad U(1)=0.
\]
这里两个量同时归零，原式比较会被共同小量干扰，于是先除去共同的小量：
\[
\phi(t)=\frac{U(t)}{f(t)^2}.
\]
在 \(t=1+v\) 附近，\(f(1+v)\sim v\)，而 \(U(1+v)\sim -v^2\)，故 \(\phi(t)\) 在端点附近有有限极限。

局部展开可见 \(\phi'(1)=-1/3<0\)，说明端点附近还留着一点偏斜，再补一个只依赖 \(f\) 的一次项：
\[
\mathcal F(t)=\phi(t)+\frac12f(t).
\]
因为 \(f(x)=f(y)=m\)，这一修正项在比较 \(\mathcal F(x)\) 与 \(\mathcal F(y)\) 时会自动抵消。

把 \(\phi(t)\) 展开可得
\[
\mathcal F(t)=\frac{t-1}{t\ln^2 t}-\frac{1}{\ln t}-\frac{t}{2}+\frac12t\ln t.
\]

目标已经转化为判定 \(\mathcal F\) 在 \((0,1)\) 上的单调性。求导得
\[
\mathcal F'(t)=\frac{(t+1)\ln t-2t+2+\frac12 t^2\ln^4 t}{t^2\ln^3 t}.
\]
对 \(t\in(0,1)\)，分母恒为负。要得到 \(\mathcal F'(t)>0\)，分子必须为负，因此转为证明
\[
H(t)=(t+1)\ln t-2t+2+\frac12 t^2\ln^4 t<0.
\]
令 \(t=e^{-s}\ (s>0)\)，则
\[
H(e^{-s})=-s(e^{-s}+1)-2e^{-s}+2+\frac12s^4e^{-2s}.
\]
两边同乘 \(e^s\)，记
\[
W(s)=e^s(2-s)-s-2+\frac12s^4e^{-s}.
\]
证明 \(W(s)<0\) 便可完成这一段判号。

这里
\[
W(0)=0,\qquad W'(0)=0,
\]
并且
\[
W''(s)<0
\]
可由下列放缩推出：
\[
e^{2s}>1+2s+2s^2+\frac43s^3,
\]
故
\[
e^{2s}-\left(6s-4s^2+\frac12s^3\right)
>1-4s+6s^2+\frac56s^3
=6\left(s-\frac13\right)^2+\frac13+\frac56s^3>0.
\]
于是
\[
W''(s)<0 \implies W'(s)<0 \implies W(s)<0,
\]
从而 \(H(t)<0\)，即
\[
\mathcal F'(t)>0\qquad(0<t<1).
\]

由于 \(0<x<y<1\) 且 \(\mathcal F\) 在 \((0,1)\) 上递增，故
\[
\mathcal F(x)<\mathcal F(y).
\]
代入定义，
\[
\frac{U(x)}{f(x)^2}+\frac12f(x)<\frac{U(y)}{f(y)^2}+\frac12f(y).
\]
由 \(f(x)=f(y)=m\) 可约去修正项，得到
\[
\frac{U(x)}{m^2}<\frac{U(y)}{m^2}.
\]
因 \(m^2>0\)，故
\[
U(x)<U(y).
\]
这等价于
\[
x^2-xC(m)<y^2-yC(m),
\]
即
\[
(x-y)(x+y)<(x-y)C(m).
\]
再由 \(x-y<0\)，可得
\[
x+y>C(m)=1+m+\frac12m^2.
\]
因此
\[
x+y>1+m+\frac12m^2.
\]
\end{solution}
