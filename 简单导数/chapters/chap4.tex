\chapter{恒成立}
\section{不等式题（对应高一基本不等式模块）}
\begin{example}[数海漫游考前100题]
    若 $x - 3\sqrt{y} = \sqrt{x - 3y}$，求 $x$ 的取值范围.
\end{example}
\begin{solution}
令 $u = \sqrt{x - 3y}, \quad v = \sqrt{y}$。由算术平方根的定义，需有
\[
u \geqslant 0,\qquad v \geqslant 0.
\]
代入后可得
\[
u^2 + 3v^2 = (x-3y)+3y = x,
\qquad
x-3v=u,
\]
从而
\[
x=u+3v,\qquad u^2-u+3v^2-3v=0.
\]
配方得
\[
\left(u - \frac{1}{2}\right)^2 + 3\left(v - \frac{1}{2}\right)^2 = 1.
\]
因此，问题转化为：在区域 $u \geqslant 0,\ v \geqslant 0$ 内，求线性函数 $x=u+3v$ 在上述椭圆上的取值范围。

取参数表示
\[
u = \frac{1}{2} + \cos\theta,\qquad
v = \frac{1}{2} + \frac{1}{\sqrt{3}}\sin\theta,
\]
则
\[
x = u+3v = 2 + 2\cos\left(\theta - \frac{\pi}{3}\right).
\]
再由约束 $u \geqslant 0,\ v \geqslant 0$ 得
\[
\cos\theta \geqslant -\frac{1}{2}
\implies
\theta \in \left[-\frac{2\pi}{3}, \frac{2\pi}{3}\right],
\]
\[
\sin\theta \geqslant -\frac{\sqrt{3}}{2}
\implies
\theta \in \left[-\pi, -\frac{2\pi}{3}\right] \cup \left[-\frac{\pi}{3}, \pi\right].
\]
两者交集为
\[
\theta \in \left\{-\frac{2\pi}{3}\right\} \cup \left[-\frac{\pi}{3}, \frac{2\pi}{3}\right].
\]

当 $\theta = -\frac{2\pi}{3}$ 时，$u=v=0$，故 $x=0$。

当 $\theta \in \left[-\frac{\pi}{3}, \frac{2\pi}{3}\right]$ 时，
\[
\theta-\frac{\pi}{3}\in\left[-\frac{2\pi}{3}, \frac{\pi}{3}\right],
\]
故
\[
\cos\left(\theta-\frac{\pi}{3}\right)\in\left[-\frac{1}{2},1\right].
\]
于是
\[
1 = 2+2\left(-\frac{1}{2}\right) \leqslant x \leqslant 2+2\cdot 1 = 4.
\]

综上，
\[
x \in \{0\} \cup [1,4].
\]
\end{solution}

\section{幂指对恒成立问题}

\begin{example}[群题]
    对于任意正整数n，证明\[\sqrt[n]{n!} + \sqrt[n+2]{(n+2)!} < 2\sqrt[n+1]{(n+1)!}\]
\end{example}
\begin{proof}
注意到$\sqrt[n]{n!}$是$f(x) = \left[\Gamma(x+1)\right]^{\frac1x}$（其中 $\Gamma(x)$ 为伽马函数）当 $x = n$（正整数）时的特殊形式，所以考虑证明对于任意正数$x$有\[f(x) + f(x+2) < 2f(x+1)\]也就是证明 $f(x)$ 在 $x > 0$ 上满足 $f''(x) < 0$即可，考虑到$f(x)$不好处理，于是引入$g(x)=\ln f(x)=\dfrac{\ln \Gamma(x+1)}{x}$，则
\[f(x) = e^{g(x)}~~~f'(x) = g'(x) e^{g(x)}~~~f''(x) = \left[g''(x) + (g'(x))^2\right] e^{g(x)}\]
因为 $e^{g(x)} > 0$，所以 $f''(x) < 0$ 等价于证明：$$g''(x) + (g'(x))^2 < 0 \iff (g'(x))^2 < -g''(x)$$
而$g(x)$好处理正是因为可以展开：
\[g(x)=\ln \Gamma(x+1) = -\gamma x + \lim_{n \to \infty} \sum_{k=1}^n \left[ \frac{x}{k} - \ln\left(1 + \frac{x}{k}\right) \right]\]
\begin{note}
由$\Gamma(x)$函数的反常积分定义
\begin{align*}
\Gamma(x) =& \int_0^\infty t^{x-1} e^{-t} \mathrm dt= \lim_{n \to \infty} \int_0^n t^{x-1} \left(1 - \frac{t}{n}\right)^n \mathrm dt=\int_0^n t^{x-1} \left(1 - \frac{t}{n}\right)^n \mathrm dt \\=& \int_0^1 (nu)^{x-1} (1-u)^n \cdot n \mathrm du= n^x \int_0^1 u^{x-1} (1-u)^n \mathrm du\end{align*}
分部积分
$$I_n = \left[ \frac{u^x}{x} (1-u)^n \right]_0^1 - \int_0^1 \frac{u^x}{x} \cdot \left( -n(1-u)^{n-1} \right) \mathrm du= \frac{n}{x} \int_0^1 u^x (1-u)^{n-1} \mathrm du = \frac{n}{x} I_{n-1}$$
递推：$$I_n = \frac{n}{x} \cdot \frac{n-1}{x+1} \cdot \frac{n-2}{x+2} \cdots \frac{1}{x+n-1} \int_0^1 u^{x+n-1} \mathrm du=\frac{n!}{x(x+1)(x+2)\cdots(x+n)}$$
那么$$\Gamma(x) = \lim_{n \to \infty} n^x I_n = \lim_{n \to \infty} \frac{n! \cdot n^x}{x(x+1)(x+2)\cdots(x+n)}$$
利用 $\Gamma(x+1) = x\Gamma(x)$：
$$\Gamma(x+1) = \lim_{n \to \infty} \frac{n! \cdot n^x}{(x+1)(x+2)\cdots(x+n)}= \lim_{n \to \infty} \frac{n^x}{\frac{x+1}{1} \cdot \frac{x+2}{2} \cdots \frac{x+n}{n}} = \lim_{n \to \infty} \frac{n^x}{\prod_{k=1}^n \left(1 + \frac{x}{k}\right)}$$
所以$\displaystyle\ln \Gamma(x+1) = \lim_{n \to \infty} \left[ x \ln n - \sum_{k=1}^n \ln\left(1 + \frac{x}{k}\right) \right]$，利用$\displaystyle x \ln n = x \sum_{k=1}^n \frac{1}{k} - \gamma x + x \epsilon_n$得
\[\ln \Gamma(x+1) = \lim_{n \to \infty} \left[ \left( x \sum_{k=1}^n \frac{1}{k} - \gamma x + x \epsilon_n \right) - \sum_{k=1}^n \ln\left(1 + \frac{x}{k}\right) \right]= -\gamma x + \lim_{n \to \infty} \sum_{k=1}^n \left[ \frac{x}{k} - \ln\left(1 + \frac{x}{k}\right) \right]\]
当 $k \to \infty$ 时，泰勒展开 $\ln(1+t) = t - \frac{t^2}{2} + O(t^3)$，令 $t = \frac{x}{k}$：
$\frac{x}{k} - \ln\left(1 + \frac{x}{k}\right) = \frac{x}{k} - \left( \frac{x}{k} - \frac{x^2}{2k^2} + O\left(\frac{x^3}{k^3}\right) \right) = \frac{x^2}{2k^2} + O\left(\frac{x^3}{k^3}\right)$由于级数 $\sum_{k=1}^\infty \frac{1}{k^2}$ 是严格收敛的（$p$-级数，$p=2$），根据比较审敛法，该通项在 $x > -1$ 时绝对收敛且一致收敛。因此求和极限交换：\[\ln \Gamma(x+1) = -\gamma x + \sum_{k=1}^\infty \left[ \frac{x}{k} - \ln\left(1 + \frac{x}{k}\right) \right]\]
\end{note}
所以$$g'(x) = \sum_{k=1}^\infty \frac{1}{x^2} \left[ \ln\left(1 + \frac{x}{k}\right) - \frac{x}{x+k} \right],~g''(x) = \sum_{k=1}^\infty \left\{ -\frac{2}{x^3}\left[ \ln\left(1 + \frac{x}{k}\right) - \frac{x}{x+k} \right] + \frac{1}{x(x+k)^2} \right\}$$
不难发现 $g''(x)$ 与 $g'(x)$ 满足：$$g''(x) = -\frac{2}{x}g'(x) + \frac{1}{x}\sum_{k=1}^\infty \frac{1}{(x+k)^2}$$
借助这个式子消去二阶导：$$(g'(x))^2 < \frac{2}{x}g'(x) - \frac{1}{x}\sum_{k=1}^\infty \frac{1}{(x+k)^2} \iff \sum_{k=1}^\infty \frac{1}{(x+k)^2} < 2g'(x) - x(g'(x))^2$$
为避免算错，换元 $u(t) = \ln(1+t) - \frac{t}{1+t}$（其中 $t > 0$），则有：
$$g'(x) = \sum_{k=1}^\infty \frac{1}{x^2} u\left(\frac{x}{k}\right)$$考虑逐项比较（加强证明），即证对于所有 $t > 0$，均有：$$\frac{t^2}{(1+t)^2} \le 2u(t) - \frac{1}{t}u(t)^2\implies \ln(1+t) \ge \frac{t(2+t) - t\sqrt{1+t+t^2}}{1+t}= \frac{3t}{2+t+\sqrt{1+t+t^2}}$$注意到对 $t > 0$ 有 $\displaystyle\sqrt{1+t+t^2} > \sqrt{1+t+\frac{1}{4}t^2} = 1+\frac{1}{2}t$。将其代入分母放缩：$$\frac{3t}{2+t+\sqrt{1+t+t^2}} < \frac{3t}{2+t+1+\frac{1}{2}t} = \frac{2t}{2+t}$$而 $\ln(1+t) > \frac{2t}{2+t}$是显然的。得证。
\end{proof}



\begin{example}[函数列单调性证明题]
设函数列$\{f_n\}$满足$f_1(x)=x-\ln x$，$f_{n+1}(x)=f_1(f_n(x))$，函数列$\{g_n\}$满足$g_n(x)=\dfrac{e^{1-f_n(x)}}{x}$，函数列$\{h_n\}$满足$h_n(x)=g_n(g_n(x))-x$。证明：对于任意正整数$n\geq 2$，$h_n(x)$在$(1,+\infty)$上单调递增。
\end{example}
\begin{solution}
先由归纳可知：对任意正整数 \(n\)，\(f_n\) 在 \((0,1)\) 上严格递减，在 \((1,+\infty)\) 上严格递增，且
\[
f_n(x)\ge f_n(1)=1,
\]
等号只在 \(x=1\) 取到。

又因为 \(g_n(1)=1\)，可把
\[
h_n(x)=g_n(g_n(x))-x=\frac{e^{1-f_n(g_n(x))}}{g_n(x)}-x.
\]
再代入
\[
g_n(x)=\frac{e^{1-f_n(x)}}{x},
\]
便有
\[
h_n(x)=x\cdot e^{f_n(x)-f_n(g_n(x))}-x=x\,e^{p_n(x)}-x,
\]
其中
\[
p_n(x)=f_n(x)-f_n(g_n(x)).
\]
由于 \(f_n(1)=g_n(1)=1\)，故 \(p_n(1)=0\)。求导得
\[
h_n'(x)=e^{p_n(x)}(1+x\,p_n'(x))-1.
\]
因此目标归结为在 \(x>1\) 上证明
\[
p_n(x)>0,\qquad p_n'(x)>0
\]
这两式成立后，便有 \(h_n'(x)>0\)。

记
\[
q_n(x)=f_{n-1}(x)-f_{n-1}(g_n(x)).
\]
因为
\[
p_n(x)=f_1(f_{n-1}(x))-f_1(f_{n-1}(g_n(x))),
\]
求导后有
\[
p_n'(x)=f_{n-1}'(x)\Bigl(1-\frac{1}{f_{n-1}(x)}\Bigr)-f_{n-1}'(g_n(x))\,g_n'(x)\Bigl(1-\frac{1}{f_{n-1}(g_n(x))}\Bigr).
\]
若先证 \(q_n'(x)>0\)，则 \(q_n(x)>q_n(1)=0\)，也就是
\[
f_{n-1}(x)>f_{n-1}(g_n(x))\qquad (x>1).
\]
代回上式可得
\[
p_n'(x)>f_{n-1}'(x)\Bigl(1-\frac{1}{f_{n-1}(x)}\Bigr)-f_{n-1}'(g_n(x))\,g_n'(x)\Bigl(1-\frac{1}{f_{n-1}(x)}\Bigr)
\]
\[
p_n'(x)>q_n'(x)\Bigl(1-\frac{1}{f_{n-1}(x)}\Bigr)>0
\]
因为 \(f_{n-1}(x)>1\)。所以问题继续降到
\[
r_n(x)=f_1(x)-f_1(g_n(x)),
\]
最后归结为证明 \(r_n'(x)>0\)。

先看 \(n=2\)。这时
\[
r_2(x)=f_1(x)-f_1(g_2(x))=x-\ln x-\Bigl(g_2(x)-\ln g_2(x)\Bigr),
\]
其中
\[
g_2(x)=\frac{x-\ln x}{e^{x-1}}.
\]
整理得
\[
r_2'(x)=-\frac{1}{x}+\frac{1-x+\ln x}{x-\ln x}\cdot\frac{1}{x}.
\]
当 \(1<x<e\) 时，
\[
r_2'(x)>\frac{e^{1-x}+\ln x-1}{x}>\frac{s(1)}{x}=0,
\]
其中
\[
s(x)=e^{1-x}+\ln x-1,\qquad s'(x)=\frac{1}{x}-\frac{1}{e^{x-1}}\ge 0.
\]
所以 \(s(x)>s(1)=0\)。当 \(x\ge e\) 时，
\[
r_2'(x)>\frac{e^{1-x}+\ln x-1}{x-\ln x}\cdot\frac{1}{x}>0
\]
因为分子、分母都为正。故 \(r_2'(x)>0\) 在 \(x>1\) 上成立。

再做归纳。设 \(k\ge3\)，并假定 \(r_{k-1}'(x)>0\) 对 \(x>1\) 成立。对 \(n=k\)，有
\[
r_k(x)=f_1(x)-f_1(g_k(x)),\quad r_k'(x)=f_1'(x)-(f_1(g_k(x)))'.
\]
又因为
\[
g_k(x)=f_1(g_{k-1}(x)),
\]
所以
\[
r_k'(x)> (f_1(g_{k-1}(x)))' - (f_1(g_k(x)))'
\]
再代入 \(f_1'(y)=1-\frac1y\)，整理得
\[
r_k'(x)>-e^{1-f_{k-1}(x)-\ln x}f_{k-1}'(x)+\frac{f_{k-1}'(x)}{f_{k-1}(x)}
\]
\[
=f_{k-1}'(x)\Bigl(\frac{1}{f_{k-1}(x)}-\frac{1}{e^{f_{k-1}(x)-1}}\Bigr).
\]
因为 \(f_{k-1}(x)>1\) 且 \(f_{k-1}'(x)>0\)，括号内为正，故 \(r_k'(x)>0\)。

于是对任意 \(n\ge2\)，都有 \(r_n'(x)>0\)（\(x>1\)），进而 \(q_n'(x)>0\)、\(p_n'(x)>0\)、\(p_n(x)>0\)，最后得到
\[
h_n'(x)>0\qquad (x>1).
\]
结论得证。
\end{solution}

\begin{example}[虚调子]
证明不等式 $4x^2(1-x)^2 + x(1-x) \leqslant x^{1-x}\cdot(1-x)^x$ 在 $x \in [0, 1]$ 上成立
\end{example}
\begin{solution}
当 \(x=0\) 或 \(x=1\) 时两端都为 \(0\)。端点处理完后，余下讨论 \(x\in(0,1)\)。

原不等式可写成
\[
x(1-x)\bigl[4x(1-x)+1\bigr]\le \frac{x(1-x)}{x^x(1-x)^{1-x}}.
\]
因为 \(x(1-x)>0\)，等价于
\[
4x(1-x)+1\le \frac1{x^x(1-x)^{1-x}}.
\]
两边取对数，得到
\[
f(x)=x\ln x+(1-x)\ln(1-x)+\ln\bigl(1+4x(1-x)\bigr)\le0.
\]

式子关于 \(x=\dfrac12\) 对称，令
\[
x=\frac{1+u}{2},\qquad u\in(-1,1).
\]
则
\[
x(1-x)=\frac{1-u^2}{4},\qquad 1+4x(1-x)=2-u^2,
\]
从而
\[
F(u)=\frac{1+u}{2}\ln(1+u)+\frac{1-u}{2}\ln(1-u)-\ln2+\ln(2-u^2).
\]
这是偶函数，讨论 \(u\in[0,1)\) 后便覆盖整个区间。

求导得
\[
F'(u)=\frac12\ln\frac{1+u}{1-u}-\frac{2u}{2-u^2}.
\]
再记
\[
G(u)=2F'(u)=\ln\frac{1+u}{1-u}-\frac{4u}{2-u^2},
\]
则
\[
G'(u)=\frac{2u^2(3u^2-2)}{(1-u^2)(2-u^2)^2}.
\]
因此 \(G\) 在 \(\left(0,\sqrt{\frac23}\right)\) 上递减，在 \(\left(\sqrt{\frac23},1\right)\) 上递增。又
\[
G(0)=0,\qquad \lim_{u\to1^-}G(u)=+\infty,
\]
故存在唯一的 \(u_0\in\left(\sqrt{\frac23},1\right)\) 使 \(G(u_0)=0\)。于是 \(F\) 先减后增。

再看两端：
\[
F(0)=0,
\]
且由 \(\lim_{t\to0^+}t\ln t=0\) 可得
\[
\lim_{u\to1^-}F(u)=0.
\]
所以 \(F(u)\le0\) 对一切 \(u\in(-1,1)\) 都成立，也就等价于原不等式在 \(x\in[0,1]\) 上成立。
\end{solution}

\begin{example}[虚调子]
证明不等式：$4x^2/(x+1)^3 + x/(x+1) \le x^{1/(x+1)}$对任意$x\in(0,1)$恒成立
\end{example}
\begin{solution}
令
\[
s=\frac{x-1}{x+1}\in(-1,1),
\]
则
\[
x=\frac{1+s}{1-s},\qquad x+1=\frac{2}{1-s},\qquad \frac{1}{x+1}=\frac{1-s}{2}.
\]
代入后有
\[
\frac{x}{x+1}=\frac{1+s}{2},\qquad \frac{4x^2}{(x+1)^3}=\frac12(1+s)^2(1-s),
\]
\[
x^{\frac{1}{x+1}}=\left(\frac{1+s}{1-s}\right)^{\frac{1-s}{2}}.
\]
因此原不等式等价于
\[
(1+s)\left(1-\frac{s^2}{2}\right)\le \left(\frac{1+s}{1-s}\right)^{\frac{1-s}{2}}.
\]

两边都为正，取对数后记
\[
Q(s)=\frac{1-s}{2}\ln\frac{1+s}{1-s}-\ln(1+s)-\ln\left(1-\frac{s^2}{2}\right).
\]
原不等式化为证明 \(Q(s)\ge0\)。又 \(Q\) 是偶函数，讨论 \(s\in[0,1)\) 便覆盖 \(s\in(-1,1)\)。

求导得
\begin{align*}
Q'(s)
&= -\frac{1}{2}\ln\left(\frac{1+s}{1-s}\right) + \frac{1-s}{2}\left(\frac{1}{1+s}+\frac{1}{1-s}\right) - \frac{1}{1+s} + \frac{s}{1-\frac{s^2}{2}} \\
&=\frac{s^2(2-3s^2)}{(2-s^2)^2(1-s^2)}.
\end{align*}
于是 \(Q'\) 在 \(\left(0,\sqrt{\frac23}\right)\) 上为正，在 \(\left(\sqrt{\frac23},1\right)\) 上为负，所以 \(Q\) 先增后减。

再看两端：
\[
Q(0)=0.
\]
当 \(s\to1^-\) 时，令 \(t=1-s\to0^+\)，则
\[
\frac{1-s}{2}\ln\frac{1+s}{1-s}=\frac{t}{2}\ln\frac{2-t}{t}\to0,
\]
并且
\[
-\ln(1+s)-\ln\left(1-\frac{s^2}{2}\right)\to -\ln2-\ln\frac12=0.
\]
所以
\[
\lim_{s\to1^-}Q(s)=0.
\]

因此 \(Q(s)\ge0\) 对 \(s\in(-1,1)\) 恒成立，等号只在 \(s=0\) 取到。代回原变量便得 \(x=1\)。所以原不等式在 \(x\in(0,1)\) 上恒成立。
\end{solution}

\begin{example}
    对于一切$x\geq0$证明\[\ln(x+1)+\frac{x}{\sqrt{\frac{1}{2}x^2+1}}+\frac{1}{2}x^2-2x \geq 0\]
\end{example}
\begin{solution}
设
\[
f(x) = \ln(x+1) + \frac{x}{\sqrt{\frac{1}{2}x^2+1}} + \frac{1}{2}x^2 - 2x.
\]
原题即为证明 \(f(x)\ge0\) 对一切 \(x\ge0\) 成立。先记
\[
f(0)=0.
\]

求导得
\[
f'(x) = \frac{1}{x+1} + \frac{1}{\left(\frac{1}{2}x^2+1\right)^{\frac{3}{2}}} + x - 2.
\]

中间那一项需要一个只含 \(y\) 的下界。对任意 \(y\ge0\)，有
\[
(1+y)^{\frac32}\le 1+\frac32y+\frac38y^2.
\]
两边平方后，差为
\[
\frac18y^3+\frac{9}{64}y^4\ge0.
\]
令 \(y=\dfrac{x^2}{2}\)，便得
\[
\left(1+\frac{x^2}{2}\right)^{\frac32}\le 1+\frac34x^2+\frac{3}{32}x^4,
\]
从而
\[
\frac{1}{\left(\frac{1}{2}x^2+1\right)^{\frac{3}{2}}}
\ge
\frac{1}{1 + \frac{3}{4}x^2 + \frac{3}{32}x^4}
=
\frac{32}{3x^4 + 24x^2 + 32}.
\]

代回 \(f'(x)\)：
\[
f'(x)\ge \frac{1}{x+1} + x - 2 + \frac{32}{3x^4 + 24x^2 + 32}
= \frac{x^2 - x - 1}{x+1} + \frac{32}{3x^4 + 24x^2 + 32}.
\]
把右边通分，得到
\[
f'(x)\ge \frac{N(x)}{(x+1)(3x^4+24x^2+32)},
\]
其中
\[
N(x)=(x^2-x-1)(3x^4+24x^2+32)+32(x+1)
=x^2(3x^4-3x^3+21x^2-24x+8).
\]

记
\[
H(x)=3x^4-3x^3+21x^2-24x+8.
\]
配方可得
\[
H(x)=3\left(x^2-\frac12x\right)^2+\frac{81}{4}\left(x-\frac{16}{27}\right)^2+\frac89>0.
\]
因此对 \(x\ge0\)，有 \(N(x)\ge0\)，分母中的 \(x+1\) 与 \(3x^4+24x^2+32\) 也都为正，所以
\[
f'(x)\ge0.
\]
于是 \(f\) 在 \([0,+\infty)\) 上递增。再结合 \(f(0)=0\)，便有
\[
f(x)\ge f(0)=0\qquad (x\ge0).
\]
原不等式得证。
\end{solution}


\begin{example}
    证明函数 $f(x) = \frac{[\ln(x+1)]^x}{x^{\ln x}}$ 在 $x \in (0, +\infty)$ 上严格单调递增
\end{example}
\begin{solution}
因为 $f(x)>0$，改证
\[
g(x)=\ln f(x)=x \ln(\ln(x+1)) - (\ln x)^2
\]
严格递增，也就是证明 $g'(x)>0$。求导得：
$$g'(x) = \ln(\ln(x+1)) + \frac{x}{(x+1)\ln(x+1)} - \frac{2\ln x}{x}$$

令
\[
y = \ln(x+1)>0,\qquad x = e^y - 1.
\]
则
$$P(y) = g'(e^y - 1) = \ln y + \frac{1 - e^{-y}}{y} - \frac{2\ln(e^y - 1)}{e^y - 1}$$
要证 $g'(x) > 0$，即证对于所有 $y > 0$ 均有 $P(y) > 0$。

利用双曲正弦函数恒等式 $e^y - 1 = 2e^{y/2}\sinh(y/2) = y e^{y/2} \frac{\sinh(y/2)}{y/2}$，对其两边取自然对数得：
$$\ln(e^y - 1) = \ln y + \frac{y}{2} + \ln\left(\frac{\sinh(y/2)}{y/2}\right)$$
这里需要估计
\[
\ln\left(\frac{\sinh u}{u}\right) < \frac{u^2}{6}\qquad (u>0).
\]
设
\[
h(u) = \frac{u^2}{6} - \ln\left(\frac{\sinh u}{u}\right),
\]
则 $h(0^+)=0$，而
\[
h'(u) = \frac{1}{u}\left(1 + \frac{u^2}{3} - u \coth u\right).
\]
再用
\[
\coth u = \frac{1}{u} + \sum_{k=1}^\infty \frac{2u}{u^2 + k^2\pi^2},
\]
便有
\[
u \coth u = 1 + \sum_{k=1}^\infty \frac{2u^2}{u^2 + k^2\pi^2}
< 1 + \sum_{k=1}^\infty \frac{2u^2}{k^2\pi^2}
= 1 + \frac{u^2}{3}.
\]
所以 \(h'(u)>0\)，从而 \(h(u)>0\)。

把 \(u=y/2\) 代入刚得到的估计，便有 $\ln(e^y - 1) < \ln y + \frac{y}{2} + \frac{y^2}{24}$。在 $P(y)$ 中，$\ln(e^y-1)$ 前的系数为 $-\frac{2}{e^y-1}<0$，所以用这个上界替换它时，得到的是 $P(y)$ 的严格下界：
$$P(y) > \ln y + \frac{1 - e^{-y}}{y} - \frac{2(\ln y + y/2 + y^2/24)}{e^y - 1} = \frac{e^y - 3}{e^y - 1}\ln y + \frac{1 - e^{-y}}{y} - \frac{y + y^2/12}{e^y - 1} \equiv B(y)$$
因为 $\dfrac{e^y - 3}{e^y - 1}$ 在 $y = \ln 3$ 处变号，接下来改看
$$Q(y) = \ln y - \frac{y^2 e^y + \frac{1}{12}y^3 e^y - (e^y - 1)^2}{y e^y (e^y - 3)}$$
此时 \(B(y)>0\) 等价于：\(y>\ln3\) 时 \(Q(y)>0\)，而 \(0<y<\ln3\) 时 \(Q(y)<0\)。

对 $Q(y)$ 求导，得到：
$$Q'(y) = \frac{1}{y} - \frac{d}{dy}\left[ \frac{y^2 e^y + \frac{1}{12}y^3 e^y - (e^y - 1)^2}{y e^y (e^y - 3)} \right] = \frac{Z_1(y)}{y (e^y - 3)^2}$$
记分子为 $Z_1(y)$，展开得：
$$Z_1(y) = 2 - 2y + 3y^2 + \frac{1}{2}y^3 + (3y + 3)e^{-y} - \left(1 + y + y^2 - \frac{5}{6}y^3 - \frac{1}{12}y^4\right)e^y$$
因为 \(e^y>0\)，乘上去不改符号，记
\[
W(y) = e^y Z_1(y).
\]
于是 \(W,Z_1,Q'\) 同号：
$$W(y) = \left(2 - 2y + 3y^2 + \frac{1}{2}y^3\right)e^y + 3y + 3 - \left(1 + y + y^2 - \frac{5}{6}y^3 - \frac{1}{12}y^4\right)e^{2y}$$

对 $W(y)$ 进行连续求导以判断其单调性与根的分布。
一阶导数：
$$W'(y) = \left(\frac{1}{2}y^3 + \frac{9}{2}y^2 + 4y\right)e^y + 3 - \left(3 + 4y - \frac{1}{2}y^2 - 2y^3 - \frac{1}{6}y^4\right)e^{2y}$$
二阶导数：
$$W''(y) = \left(\frac{1}{2}y^3 + 6y^2 + 13y + 4\right)e^y - \left(10 + 7y - 7y^2 - \frac{14}{3}y^3 - \frac{1}{3}y^4\right)e^{2y}$$
提取指数因子，令 $W_2(y) = W''(y)e^{-y}$：
$$W_2(y) = \frac{1}{2}y^3 + 6y^2 + 13y + 4 - \left(10 + 7y - 7y^2 - \frac{14}{3}y^3 - \frac{1}{3}y^4\right)e^y$$
对 $W_2(y)$ 继续求导，记为 $W_3(y)$：
$$W_2'(y) = \frac{3}{2}y^2 + 12y + 13 - \left(17 - 7y - 21y^2 - 6y^3 - \frac{1}{3}y^4\right)e^y \equiv W_3(y)$$
再对 $W_3(y)$ 求导两次：
$$W_3'(y) = 3y + 12 - \left(10 - 49y - 39y^2 - \frac{22}{3}y^3 - \frac{1}{3}y^4\right)e^y$$
$$W_3''(y) = 3 + \left(39 + 127y + 61y^2 + \frac{26}{3}y^3 + \frac{1}{3}y^4\right)e^y$$

由于 \(y>0\) 时 \(W_3''(y)\) 里每一项都为正，所以 \(W_3''(y)>0\)。于是
\[
W_3'(y)\ \text{递增},\qquad W_3'(0)=2>0,
\]
从而 \(W_3'(y)>0\)，也就是 \(W_3(y)=W_2'(y)\) 递增。再由
\[
W_2'(0)=-4<0,\qquad W_2'(y)\to+\infty\ (y\to+\infty),
\]
可知 \(W_2'\) 只有一个零点 \(\alpha\)。因此 \(W_2\) 先减后增。又
\[
W_2(0)=-6<0,\qquad W_2(y)\to+\infty,
\]
故 \(W_2\) 也只有一个零点 \(\beta\)。于是 \(W''(y)=e^yW_2(y)\) 先负后正，结合 \(W'(0)=0\)，可知 \(W'\) 先减后增，只过零一次；进一步，\(W\) 先减后增，只在中间有一个极小值。

再看几个值：
\[
W(0) = 4 > 0,\qquad W(\ln 3) \approx -0.052 < 0,\qquad W(y)\to+\infty\ (y\to+\infty).
\]
所以 \(W\) 在 \((0,+\infty)\) 上恰有两个根：
\[
y_1 \in (0, \ln 3), \qquad y_2 \in (\ln 3, +\infty).
\]
因为 $Q'(y)$ 与 $W(y)$ 同号，可知 $Q(y)$ 的单调性分布为：在 $(0, y_1)$ 单调递增，在 $(y_1, \ln 3)$ 单调递减；在 $(\ln 3, y_2)$ 单调递减，在 $(y_2, +\infty)$ 单调递增。
结合渐近线行为 $Q(0^+) \to -\infty$ 和 $Q(\ln 3^-) \to -\infty$，可知区间 $(0, \ln 3)$ 上的极大值 $Q(y_1)$ 必定小于 $0$，故当 $y \in (0, \ln 3)$ 时 $Q(y) < 0$ 恒成立。
同理，结合 $Q(\ln 3^+) \to +\infty$ 和 $Q(+\infty) \to +\infty$，可知区间 $(\ln 3, +\infty)$ 上的极小值 $Q(y_2)$ 必定大于 $0$，故当 $y \in (\ln 3, +\infty)$ 时 $Q(y) > 0$ 恒成立。

于是 \(B(y)>0\) 在 \(y>0\) 上恒成立，也就得到 \(P(y)>0\)，从而 \(g'(x)>0\)。因此
\[
f(x) = \frac{[\ln(x+1)]^x}{x^{\ln x}}
\]
在 \((0,+\infty)\) 上严格单调递增。
\end{solution}

\begin{example}
给定 $a > 0$。对每个正整数 $n$，定义
$$f_n(x) = (a + n - 1)^x + (a + n)^x - (a + n + 1)^x.$$
记其零点为 $x_n$（即 $f_n(x_n) = 0$）。设数列 $\{x_n\}$ 的前 $n$ 项和为
$$
S_n = \sum_{j=1}^n x_j.$$
固定题目中的正整数 $n$，若
$$\begin{cases}
S_n + S_{3n} + S_{8n} + S_{12n} - S_{4n} - S_{6n} - S_{9n} - S_{11n} = m, \\
S_{2n} + S_{4n} + S_{7n} + S_{11n} - S_n - S_{5n} - S_{8n} - S_{10n} = s, \\
S_{3n} + S_{5n} + S_{10n} - S_{2n} - S_{7n} - S_{9n} = r,
\end{cases}$$
证明：$s^2 < mr$。
\end{example}
\begin{solution}
固定正整数 $n$，记 $b=a+n>1$。方程 $f_n(x)=0$ 等价于
\[
(b-1)^x+b^x=(b+1)^x.
\]
两边同除以 $(b+1)^x$，得
\[
\left(\frac{b-1}{b+1}\right)^x+\left(\frac{b}{b+1}\right)^x=1.
\]
于是考察
\[
g(x)=\left(\frac{b-1}{b+1}\right)^x+\left(\frac{b}{b+1}\right)^x.
\]
两个底数都在 \((0,1)\) 内，所以 \(g'(x)<0\)，即 \(g\) 在 \([0,\infty)\) 上严格递减。再结合
\[
g(0)=2>1,\qquad \lim_{x\to\infty}g(x)=0<1,
\]
便知零点 \(x_n\) 存在且唯一。
 
再记分块和
\[
A_k = S_{kn} - S_{(k-1)n} = \sum_{j=(k-1)n+1}^{kn} x_j\qquad (k=1,2,\dots,12).
\]
由定义
\[
S_{kn} = \sum_{i=1}^k A_i.
\]
把题设里的 \(m,s,r\) 全部改写成 \(A_k\) 后，得到
$$m=A_{12}-A_{2}-A_{3}-2A_{4}-A_{5}-A_{6}-A_{9},$$
$$s=A_{2}-A_{5}-A_{8}+A_{11},$$
$$r=A_{3}-A_{6}-A_{7}+A_{10}.$$
 
再记二阶差分
\[
\mu_k = -(A_{k+2}-2A_{k+1}+A_k).
\]
由离散二次求和公式，
\[
A_k = A_1 + (k-1)(A_2-A_1) - \sum_{j=1}^{k-2}(k-1-j)\mu_j\qquad (k \ge 2).
\]
代回 \(s,r\) 可得
$$-s=\sum_{k=1}^{10}S_{k}\mu_{k}, \quad -r=\sum_{k=1}^{10}R_{k}\mu_{k},$$
其中相应的常数系数序列为 $(S_{k})_{k=1}^{10}=(0,1,2,3,3,3,3,2,1,0)$， $(R_{k})_{k=1}^{10}=(0,0,1,2,3,3,2,1,0,0)$。
同理，
$$-m = \sum_{k=1}^{10}M_{k}\mu_{k} + 6A_1 + 15(A_{12}-A_{11}),$$
其中 $(M_{k})_{k=1}^{10}=(6,11,15,17,18,18,18,18,17,16)$。由于方程的根函数随参数单调递增，故序列 $\{x_j\}$ 递增，从而分块和序列 $\{A_k\}$ 单调递增，保证了仿射项 $6A_1+15(A_{12}-A_{11}) \ge 0$。由此得出不等式 $-m \ge \sum_{k=1}^{10}M_{k}\mu_{k}$。令 $M = \sum_{k=1}^{10}M_{k}\mu_{k}$， $S = -s$， $R = -r$。若能证明 $S^2 < MR$ 且 $R>0$，结合 $-m \ge M$ 便可推导出 $s^2 < mr$。
 
接着看 \(\mu_k\) 的结构。把根扩成连续函数 \(X(t)\)：
\[
(t-1)^{X(t)}+t^{X(t)}=(t+1)^{X(t)}\qquad (t>1).
\]
因为 \(X(t)\) 是 Bernstein 函数，所以由 Lévy--Khintchine 表示定理，
$$X(x)=c_{0}+c_{1}x+\int_{0}^{\infty}(1-e^{-xs})d\Pi(s).$$
固定步长 \(h>0\) 与起点 \(b>0\)，令 \(a_k = X(b+kh)\)。计算二阶差分得
$$-\Delta^{2}a_{k}=\int_{0}^{\infty}e^{-(b+kh)s}(1-e^{-hs})^{2}d\Pi(s).$$
令 \(t=e^{-hs}\in(0,1)\) 并把测度推前到 \([0,1]\)，便知 \(-\Delta^2 a_k\) 是 Hausdorff 矩序列。回到本题，固定分块长度 \(n\)，对每个块内偏移 \(j=1,\dots,n\) 定义
\[
a_k^{(j)} = X(a+j+(k-1)n).
\]
于是存在 \([0,1]\) 上的正测度 \(\nu_j\)，使得
\[
-\Delta^2 a_k^{(j)} = \int_0^1 t^{k-1}\, d\nu_j(t).
\]
又因为 \(A_k = \sum_{j=1}^n a_k^{(j)}\)，令 \(\nu = \sum_{j=1}^n \nu_j\)，便有
$$\mu_{k}=-\Delta^{2}A_{k}=\int_{0}^{1}t^{k-1}d\nu(t) \quad (k \ge 1).$$
 
于是 \(M,S,R\) 都可以写成关于 \(\nu\) 的积分：
$$M=\int_{0}^{1}M(t)d\nu(t), \quad S=\int_{0}^{1}S(t)d\nu(t), \quad R=\int_{0}^{1}R(t)d\nu(t),$$
其中被积函数对应的多项式分别为：
$$M(t)=6+11t+15t^{2}+17t^{3}+18t^{4}+18t^{5}+18t^{6}+18t^{7}+17t^{8}+16t^{9},$$
$$S(t)=t(t+1)(t^{2}-t+1)(t^{2}+t+1)^{2},$$
$$R(t)=t^{2}(t+1)(t^{2}+1)(t^{2}+t+1).$$
 
将 \(D(t)\) 展开并整理为
\[
D(t) = M(t)R(t) - S(t)^2,
\]
可分解为
$$D(t)=t^{2}(t+1)(t^{2}+t+1)P(t),$$
其中 $P(t)=15t^{11}+15t^{10}+30t^{9}+30t^{8}+29t^{7}+28t^{6}+27t^{5}+26t^{4}+23t^{3}+17t^{2}+9t+5$。当 $t \in (0,1)$ 时，由于 $P(t)$ 的各项系数均为正，故 $P(t)>0$，且 $t^2(t+1)(t^2+t+1)>0$（因 $t\in(0,1)$ 时各因子均为正）。因此对于任意 $t \in (0,1)$，恒有 $M(t)R(t)-S(t)^2>0$。
 
因此对每个 \(t\in(0,1)\)，矩阵
\[
\begin{pmatrix}M(t)&S(t)\\ S(t)&R(t)\end{pmatrix}
\]
都是严格正定的。于是可取向量函数
$$u(t)=(\sqrt{M(t)},0), \quad v(t)=\left(\frac{S(t)}{\sqrt{M(t)}},\sqrt{R(t)-\frac{S(t)^{2}}{M(t)}}\right),$$
其满足 $|u(t)|^2 = M(t)$， $|v(t)|^2 = R(t)$，且 $u(t) \cdot v(t) = S(t)$。由积分形式的 Cauchy-Schwarz 不等式，有：
$$S^{2}=\left(\int_{0}^{1}u(t)\cdot v(t)d\nu(t)\right)^{2} \le \left(\int_{0}^{1}|u(t)|^{2}d\nu(t)\right)\left(\int_{0}^{1}|v(t)|^{2}d\nu(t)\right)=MR.$$
若这里取等，则 \(u(t)\) 与 \(v(t)\) 在 \(\nu\) 下几乎处处共线，也就是要求
\[
M(t)R(t)-S(t)^2=0
\]
几乎处处成立；但这与上面的严格正性矛盾，所以
\[
S^2 < MR.
\]
最后代回 \(M \le -m\)、\(S=-s\)、\(R=-r\)，便得
\[
s^2 < mr.
\]
\end{solution}

\begin{example}[$f(x)=\sqrt{\frac{x^{a}}{a}}+\sqrt{\frac{a^{x}}{x}}\geq2$]
已知函数 $f(x)=\sqrt{\frac{x^{a}}{a}}+\sqrt{\frac{a^{x}}{x}}-2$ 其中 $a>0$, $x>0$. \\
(I)当 $a=1$ 时,讨论函数 $f(x)$ 的单调性;\\
(II)证明: $f(x)\ge0$
\end{example}
\begin{solution}
当 $a=1$ 时, 函数化简为 $f(x)=\sqrt{x}+\frac{1}{\sqrt{x}}-2$. 对其求导可得：
$$f^{\prime}(x)=\frac{1}{2\sqrt{x}}-\frac{1}{2\sqrt{x^{3}}}=\frac{x-1}{2\sqrt{x^{3}}}.$$
易知当 $x \in (0,1)$ 时, $f^{\prime}(x)<0$, 函数 $f(x)$ 单调递减；当 $x \in (1,+\infty)$ 时, $f^{\prime}(x)>0$, 函数 $f(x)$ 单调递增.

再证 $f(x)\ge0$。由均值不等式，
$$f(x)\ge2\sqrt{\sqrt{\frac{x^{a}}{a}}\cdot\sqrt{\frac{a^{x}}{x}}}-2=2\sqrt[4]{x^{a-1}a^{x-1}}-2.$$
当 $a>1$ 且 $x>1$, 或 $0<a<1$ 且 $0<x<1$ 时, 有 $x^{a-1}a^{x-1}>1$, 从而 $f(x)>0$。当 $a=1$ 时，由上面的单调性知 $f(x)\ge f(1)=0$。又原式关于 $a,x$ 对称，余下处理 $0<a<1<x$ 的情形。

令 $x=p^{2}$, $a=q^{2}$, 其中 $p>1>q>0$，则原不等式化为
\[\frac{p^{q^{2}}}{q}+\frac{q^{p^{2}}}{p}\ge2\]
先证
\[
p^{q^{2}}\ge1+q\left(\frac{p}{p+q-pq}-1\right).
\]
记
\[
g(p)=q^{2}\ln p-\ln\left[1+q\left(\frac{p}{p+q-pq}-1\right)\right],
\]
则
$$g^{\prime}(p)=\frac{q^{2}}{p}-\frac{q\frac{(p+q-pq)-p(1-q)}{(p+q-pq)^{2}}}{1+q\left(\frac{p}{p+q-pq}-1\right)}=\frac{q^{2}}{p}-\frac{q^{2}}{(p+q-pq)^{2}+(p+q-pq)(pq^{2}-q^{2})}$$
$$=\frac{q^{2}(1-q)(p-1)[q^{2}(p-1)+p(1-q)]}{p(p+q-pq)(pq^{2}-q^{2}-pq+p+q)}.$$
由 $p>1>q>0$ 可知各因子都为正，故 $g'(p)>0$，从而 $g(p)\ge g(1)=0$。

同理，再证
\[
q^{p^{2}}\ge1+p\left(\frac{q}{p+q-pq}-1\right).
\]
由于左边恒正，右边为正时才需要继续比较，即
\[
1+p\left(\frac{q}{p+q-pq}-1\right)>0,
\]
这等价于 $\frac{p^{2}-p}{p^{2}-p+1}<q<\frac{p}{p-1}$。记
\[
h(q)=p^{2}\ln q-\ln\left[1+p\left(\frac{q}{p+q-pq}-1\right)\right],
\]
则
$$h^{\prime}(q)=\frac{p^{2}(p-1)(1-q)[p^{2}(q-1)+q(1-p)]}{q(p+q-pq)(p^{2}q-p^{2}-pq+p+q)}.$$
由 $p>1>q>\frac{p^{2}-p}{p^{2}-p+1}$ 可知 $h'(q)<0$，于是 $h(q)\ge h(1)=0$。

于是
$$\frac{p^{q^{2}}}{q}+\frac{q^{p^{2}}}{p}-2\ge \frac{1}{q}\left[1+q\left(\frac{p}{p+q-pq}-1\right)\right] + \frac{1}{p}\left[1+p\left(\frac{q}{p+q-pq}-1\right)\right] - 2$$
$$=\frac{1}{p}+\frac{1}{q}+\frac{p+q}{p+q-pq}-4=\frac{p+q}{pq}+\frac{p+q}{p+q-pq}-4$$
$$=\frac{(p+q)(p+q-pq)+(p+q)pq-4pq(p+q-pq)}{pq(p+q-pq)}$$
$$=\frac{p^{2}+q^{2}+4p^{2}q^{2}+2pq-4p^{2}q-4pq^{2}}{pq(p+q-pq)}=\frac{(p+q-2pq)^{2}}{pq(p+q-pq)}\ge0.$$
原不等式得证。

同一结论也可由琴生不等式推出。$a\ne1$ 时，原不等式等价于
$$\frac{x}{x+a}x^{\frac{a-1}{2}}+\frac{a}{x+a}a^{\frac{x-1}{2}}\ge\frac{2\sqrt{xa}}{x+a}.$$
由琴生不等式，
$$\frac{x}{x+a}x^{\frac{a-1}{2}}+\frac{a}{x+a}a^{\frac{x-1}{2}}\ge e^{\frac{x(a-1)\ln x+a(x-1)\ln a}{2(x+a)}},$$
因此目标转化为证明
\[
\frac{x(a-1)\ln x+a(x-1)\ln a}{2(x+a)}\ge \ln\frac{2\sqrt{xa}}{x+a}.
\]
交换 $a,x$ 后不等式不变，故可设 $t=\frac{a}{x}\ge1$，上式等价于：
$$(x-1)t\ln(tx)+(tx-1)\ln x+2(t+1)\ln\frac{t+1}{2\sqrt{t}}\ge0.$$
记左边为 $G(x)$，则
\[
G'(x)=t\ln t+2t+2t\ln x-\frac{t+1}{x}.
\]
$G'(x)$ 单调递增，且
\[
G'(1)=t\ln t+t-1\ge0,\qquad G'\left(\frac1t\right)=t-t^2-t\ln t\le0,
\]
故存在唯一的 $x_0\in\left[\frac1t,1\right]$ 使 $G'(x_0)=0$。于是
$$G(x)\ge G(x_{0})=t+1-2tx_{0}-t\ln t+2(t+1)\ln\frac{t+1}{2\sqrt{tx_{0}}}.$$
由 $G^{\prime}(x_{0})=0$ 结合对数不等式放缩有 $2t\left(1+\ln(\sqrt{t}x_{0})-\frac{t+1}{2tx_{0}}\right)\ge2t\left(2-\frac{1}{\sqrt{t}x_{0}}-\frac{t+1}{2tx_{0}}\right)$, 可推得 $4tx_{0}\le t+2\sqrt{t+1}$.
将其代入 $G(x_{0})$ 的表达式中放缩得：
$$G(x_{0})\ge\frac{t+1-2\sqrt{t}}{2}-t\ln t+2(t+1)\ln\frac{t+1}{\sqrt{t}+1}.$$
令 $u=\sqrt t\ge1$，记
\[
H(u)=\frac{u^{2}+1-2u}{2}-2u^{2}\ln u+2(u^{2}+1)\ln\frac{u^{2}+1}{u+1}.
\]
则
\[
H'(u)=4u\left[\frac{u^{2}+2u-3}{4u(u+1)}+\ln\frac{u^{2}+1}{u^{2}+u}\right].
\]
记
\[
\varphi(u)=\frac{u^{2}+2u-3}{4u(u+1)}+\ln\frac{u^{2}+1}{u^{2}+u},
\]
则
$$\varphi^{\prime}(u)=\frac{(u-1)^{2}(3u^{2}+8u+3)}{4u^{2}(u^{2}+1)(u+1)^{2}}\ge0.$$
故 $\varphi(u)\ge\varphi(1)=0$，进而 $H'(u)\ge0$，于是 $H(u)\ge H(1)=0$，原不等式得证。
\end{solution}

\begin{solution}
（1）$f'(x)=ae^{ax}+be^{bx}-1,\ f''(x)=a^2e^{ax}+b^2e^{bx}>0$，所以 $f'(x)$ 在 $\mathbf R$ 上严格递增。又
\[
\lim_{x\to-\infty}f'(x)=-1<0,\qquad \lim_{x\to+\infty}f'(x)=+\infty,
\]
故存在唯一的 $x_1$ 使 $f'(x_1)=0$。于是 $f(x)$ 在 $(-\infty,x_1)$ 上递减，在 $(x_1,+\infty)$ 上递增，因此在 $\mathbf R$ 上不单调。\\
（2）（i）由（1）知，若 $f(x)$ 只有一个零点 $x_0$，则它必是最小值点，即 $x_0=x_1$。于是
\[
e^{ax_0}+e^{bx_0}=x_0,\qquad ae^{ax_0}+be^{bx_0}=1.
\]
令 $u=ax_0,\ v=bx_0$，则 $u,v>0$，且
\[
e^u+e^v=x_0,\qquad ue^u+ve^v=x_0.
\]
消去 $x_0$ 得
\[
(u-1)e^u+(v-1)e^v=0.
\]
构造函数 $h(t) = (t-1)e^t \ (t > 0)$，因 $h'(t) = te^t > 0$，故 $h(t)$ 在 $(0, +\infty)$ 上严格单调递增，且 $h(1) = 0$。设 $u\ge v$。由 $h(u) + h(v) = 0$ 可知 $0 < v \leqslant 1 \leqslant u$（当且仅当 $u=v=1$ 时取等号）。
由于 $u \geqslant 1$ 且 $v > 0$，故 $x_0 = e^u + e^v > e^1 + e^0 = e+1$。
另一方面，将 $x_0$ 视为关于 $v$ 的函数 $S(v) = e^u + e^v \ (0 < v \leqslant 1)$。对 $(u-1)e^u + (v-1)e^v = 0$ 两边关于 $v$ 求导，得 $u e^u \cdot u' + v e^v = 0$，即 $u' = -\frac{v e^v}{u e^u}$。从而 $S'(v) = e^u \cdot u' + e^v = e^v(1 - \frac{v}{u})$。因 $0 < v \leqslant 1 \leqslant u$，有 $\frac{v}{u} \leqslant 1$，即 $S'(v) \geqslant 0$，这说明 $S(v)$ 在 $(0, 1]$ 上单调递增。当 $v = 1$ 时（此时 $u = 1$）取得最大值，故 $x_0 \leqslant S(1) = 2e$。因此，$e+1 < x_0 \leqslant 2e$ 获证。\\
（ii）当 $a=(e+2)b$ 时，$u=(e+2)v$。代回
\[
(u-1)e^u+(v-1)e^v=0
\]
得
\[
((e+2)v-1)e^{(e+1)v}+v-1=0.
\]
记
\[
k(v)=((e+2)v-1)e^{(e+1)v}+v-1 \quad (v>0),
\]
则
\[
k'(v)=e^{(e+1)v}\bigl[(e+1)(e+2)v+1\bigr]+1>0.
\]
故 $k(v)$ 在 $(0,+\infty)$ 上严格递增，零点至多一个。代入
\[
k\!\left(\frac1{e+1}\right)=0,
\]
所以
\[
v=\frac1{e+1},\qquad u=\frac{e+2}{e+1}.
\]
代回 $x_0$ 的表达式，得 $x_0 = e^u + e^v = e^{\frac{e+2}{e+1}} + e^{\frac{1}{e+1}} = e^{\frac{1}{e+1}}(e + 1)$。
\end{solution}

\begin{example}[单调性与极值]
若$t>0$，证明: \[\e^{\frac{t+1+\ln t}{(t+1)^2}\mathrm{e}^{\frac{t\ln t-t-1}{t+1}}x}+\mathrm{e}^{\frac{t+1-t\ln t}{(t+1)^2}\mathrm{e}^{\frac{t\ln t-t-1}{t+1}}x}-x\geqslant0\]
\end{example}
\begin{solution}
当 $t=1$ 时等号成立。以下设 $t>0,\ t\neq1$。

记
\[
x_1 = \frac{\ln t}{t-1},\qquad x_2 = \frac{t\ln t}{t-1},
\]
并设
\[
u = \frac{\ln t}{t+1}=\frac{x_1(x_2-x_1)}{x_1+x_2}, \quad v = \frac{t\ln t}{t+1}=\frac{x_2(x_2-x_1)}{x_1+x_2}.
\]
于是
\[
u+v=\ln t,\qquad \frac vu=t.
\]
又有
\[
\frac{t\ln t-t-1}{t+1}=v-1,\qquad \frac1{t+1}=\frac{x_1}{x_1+x_2},
\]
所以
\[
\frac{t+1+\ln t}{(t+1)^2} = \frac{x_1}{x_1+x_2}(1+u),\qquad
\frac{t+1-t\ln t}{(t+1)^2} = \frac{x_1}{x_1+x_2}(1-v).
\]
令
\[
Y=\frac{x_1}{x_1+x_2}\mathrm e^{v-1}x,
\]
则
\[
x = \frac{x_1+x_2}{x_1}\mathrm e^{1-v}Y
= (t+1)\mathrm e^{1-v}Y
= (\mathrm e^{u+v}+1)\mathrm e^{1-v}Y
= (\mathrm e^{1+u}+\mathrm e^{1-v})Y.
\]
原不等式化为
\[
\mathrm{e}^{(1+u)Y} + \mathrm{e}^{(1-v)Y} \geqslant  (\mathrm{e}^{1+u} + \mathrm{e}^{1-v})Y.
\]
在 $Y=1$ 处分别对 $\mathrm e^{(1+u)Y}$ 与 $\mathrm e^{(1-v)Y}$ 作切线，有
\[
\mathrm e^{\alpha Y}\ge \mathrm e^\alpha+\alpha \mathrm e^\alpha(Y-1)\qquad (\alpha=1+u,\ 1-v).
\]
两式相加得
\[
\mathrm e^{(1+u)Y}+\mathrm e^{(1-v)Y}
\ge \mathrm e^{1+u}+\mathrm e^{1-v}+\bigl((1+u)\mathrm e^{1+u}+(1-v)\mathrm e^{1-v}\bigr)(Y-1).
\]
而
\[
u\mathrm e^u=\frac{\ln t}{t+1}\,t^{\frac1{t+1}}
=\frac{t\ln t}{t+1}\,t^{-\frac t{t+1}}=v\mathrm e^{-v},
\]
故
\[
(1+u)\mathrm e^{1+u}+(1-v)\mathrm e^{1-v}=\mathrm e^{1+u}+\mathrm e^{1-v}.
\]
于是
\[
\mathrm e^{(1+u)Y}+\mathrm e^{(1-v)Y}\ge \mathrm e^{1+u}+\mathrm e^{1-v}+(\mathrm e^{1+u}+\mathrm e^{1-v})(Y-1)
=(\mathrm e^{1+u}+\mathrm e^{1-v})Y,
\]
即得原不等式。
\end{solution}


\begin{example}[虚调子]
    对于所有的 $x > 0$，均有
$$e^{-x} + e^{-\frac{1}{x}} \leq 1 - \frac{3\left(1 - \frac{2}{e}\right)}{1 + x + \frac{1}{x}}$$
\end{example}
\begin{solution}
原不等式在代换 $x \leftrightarrow \frac{1}{x}$ 下不变，以下固定 $x \ge 1$。

记常数$c = 3\left(1 - \frac{2}{e}\right)$。当 $x\ge1$ 时，两边同乘 $x^2+x+1$，原式等价于
$$(x^2+x+1)\left(e^{-x} + e^{-\frac{1}{x}}\right) \leq x^2+x+1 - cx$$
记
$$P(x) = (x^2+x+1)\left(1 - e^{-x} - e^{-\frac{1}{x}}\right) - cx$$
目标变为证明 $P(x)\ge0\ (x\ge1)$。先算
$$P(1) = 3(1 - 2e^{-1}) - c = 3\left(1 - \frac{2}{e}\right) - 3\left(1 - \frac{2}{e}\right) = 0$$
若能证明 $P'(x)>0\ (x>1)$，就有 $P(x)\ge P(1)=0$。求导得
$$P'(x) = (2x+1)\left(1 - e^{-x} - e^{-\frac{1}{x}}\right) + (x^2+x+1)\left(e^{-x} - \frac{1}{x^2}e^{-\frac{1}{x}}\right) - c$$
整理为
$$P'(x) = 2x+1 - c + (x^2-x)e^{-x} - \left(2x+2+\frac{1}{x}+\frac{1}{x^2}\right)e^{-\frac{1}{x}}$$

令 $s=\frac1x\in(0,1)$，则
$$P'\left(\frac{1}{s}\right) = \frac{1}{s} \left[ 2 + s - cs + \left(\frac{1}{s} - 1\right)e^{-\frac{1}{s}} - (2+2s+s^2+s^3)e^{-s} \right]$$
代入 $c = 3 - \frac{6}{e}$ 得
$$2 + s - cs = 2 + s - \left(3 - \frac{6}{e}\right)s = 2(1-s) + \frac{6s}{e}$$
代回得
$$P'\left(\frac{1}{s}\right) = \frac{1}{s} \left[ 2(1-s) + \frac{6s}{e} + \frac{1-s}{s}e^{-\frac{1}{s}} - (2+2s+s^2+s^3)e^{-s} \right]$$

其中第三项恒正。余下只要验证
$$f(s) = 2(1-s) + \frac{6s}{e} - (2+2s+s^2+s^3)e^{-s}$$
在 $(0,1)$ 内恒正。由
$$f(0) = 2 - 2 = 0$$
$$f(1) = \frac{6}{e} - (2+2+1+1)e^{-1} = 0$$
再求导：
$$f'(s) = -2 + \frac{6}{e} - \left[ (2+2s+3s^2)e^{-s} - (2+2s+s^2+s^3)e^{-s} \right] = -2 + \frac{6}{e} + (s^3 - 2s^2)e^{-s}$$
$$f''(s) = (3s^2 - 4s)e^{-s} - (s^3 - 2s^2)e^{-s} = e^{-s} (-s^3 + 5s^2 - 4s)$$
$$f''(s) = -s(s^2 - 5s + 4)e^{-s} = -s(s-1)(s-4)e^{-s}$$
故 $f''(s)<0$，从而 $f'(s)$ 在 $[0,1]$ 上严格递减。又
$$f'(0) = -2 + \frac{6}{e} > 0$$
$$f'(1) = -2 + \frac{6}{e} - e^{-1} = -2 + \frac{5}{e} < 0$$
故 $f'$ 在 $(0,1)$ 内恰有一个零点，$f$ 先增后减。再结合 $f(0)=f(1)=0$，得到
\[
f(s)>0\qquad (0<s<1).
\]
这就给出 $P'(1/s)>0$，也就是 $P'(x)>0\ (x>1)$。因此 $P(x)\ge P(1)=0$ 对一切 $x\ge1$ 成立。再由对称性，原不等式对所有 $x>0$ 都成立，且等号只在 $x=1$ 处取得。
\end{solution}


\begin{example}
\[\e^x\geqslant\frac{3\sqrt{\e}}{4}x+\frac{\sqrt{5\e}}{4}\sqrt{x^2+1}\]
\end{example}
\begin{solution}
原命题等价于证明：
\[
1 \geqslant \mathrm{e}^{\frac{1}{2}-x} \left(\frac{3}{4}x + \frac{\sqrt{5}}{4}\sqrt{x^2+1}\right)
\]
由于不等式中含有 $\sqrt{\mathrm{e}}$（即 $\mathrm{e}^{\frac{1}{2}}$），取
\[
f(x) = \mathrm{e}^{\frac{1}{2}-x}\left(\frac{3}{4}x + \frac{\sqrt{5}}{4}\sqrt{x^{2}+1}\right)
\]
目标转化为证明：对于任意实数 $x$，恒有 $f(x) \leqslant 1$。对 $f(x)$ 求导，得：
\[
\begin{aligned}
f'(x) &= -\mathrm{e}^{\frac{1}{2}-x}\left(\frac{3}{4}x+\frac{\sqrt{5}}{4}\sqrt{x^2+1}\right) + \mathrm{e}^{\frac{1}{2}-x}\left(\frac{3}{4} + \frac{\sqrt{5}x}{4\sqrt{x^2+1}}\right) \\
&= \frac{\mathrm{e}^{\frac{1}{2}-x}}{4\sqrt{x^2+1}} \left( -3x\sqrt{x^2+1} - \sqrt{5}(x^2+1) + 3\sqrt{x^2+1} + \sqrt{5}x \right) \\
&= \frac{\mathrm{e}^{\frac{1}{2}-x}}{4\sqrt{x^2+1}} \left[ \sqrt{x^2+1}(3-3x) - \sqrt{5}(x^2-x+1) \right]
\end{aligned}
\]
令 $f'(x) = 0$，括号内部分必须为零，即 \(\sqrt{x^2+1}(3-3x) = \sqrt{5}(x^2-x+1)\)。右侧
\[
x^2-x+1=\left(x-\frac{1}{2}\right)^2 + \frac{3}{4}>0
\]
恒为正，因此左侧也必须为正，只能有 $3-3x>0$，也就是 $x<1$。在这个前提下，再把两边平方去根号：
\[
(x^2+1)(3-3x)^2 = 5(x^2-x+1)^2\Rightarrow 5(x^2-x+1)^2 - 9(1-x)^2(x^2+1) = 0
\]
展开整理后，可以得到一个关于 $x$ 的一元四次方程：
\[
4x^4 - 8x^3 + 3x^2 - 8x + 4 = (2x-1)(x-2)(2x^2+x+2) = 0
\]
解得 $x = \frac{1}{2}$ 或 $x = 2$（二次因式 $2x^2+x+2=0$ 无实根）。结合前面的条件 $x < 1$，平方带来的增根 $x=2$ 必须舍去，所以 $f'(x)$ 只有一个零点 $x=\frac{1}{2}$。又因为 $f'(x)$ 在该点左侧为正，在右侧为负，$f$ 先增后减，于是最大值只能在 $x=\frac{1}{2}$ 处取得：
\[
f\left(\frac{1}{2}\right) = \mathrm{e}^{0}\left(\frac{3}{4} \cdot \frac{1}{2} + \frac{\sqrt{5}}{4}\sqrt{\frac{1}{4}+1}\right) = \frac{3}{8} + \frac{\sqrt{5}}{4} \cdot \frac{\sqrt{5}}{2} = \frac{3}{8} + \frac{5}{8} = 1
\]
即对任意实数 $x$，都有 $f(x) \leqslant f\left(\frac{1}{2}\right) = 1$，也就是：
\[
\mathrm{e}^{\frac{1}{2}-x}\left(\frac{3}{4}x + \frac{\sqrt{5}}{4}\sqrt{x^{2}+1}\right) \leqslant 1
\Leftrightarrow\mathrm{e}^x \geqslant \frac{3\sqrt{\mathrm{e}}}{4}x + \frac{\sqrt{5\mathrm{e}}}{4}\sqrt{x^2+1}
\]
证毕。
\end{solution}

\begin{example}[（2008年江西压轴题）]

c^2=1-\dfrac{8}{ax+8}=1-\dfrac{8}{(\dfrac{1}{a^2}-1)(\dfrac{1}{b^2}-1)+8}\Rightarrow (1-a^2)(1-b^2)(1-c^2)=8a^2b^2c^2
\]

\[
1-a^2=(1-a)(1+a)>1-a\ge b+c>2bc,
\]

\[
1-a^2=(1-a)(1+a)<2(1-a)\le 2(b+c-1)<2bc,

\]

\frac{1}{\sqrt{1+x}}+\frac{1}{\sqrt{1+y}}>
\frac{1}{1+\frac{x}{2}}+\frac{1}{1+\frac{y}{2}}
\ge
\frac{1}{1+\frac{x}{2}}+\frac{1}{1+\frac{2}{x}}

得到。

\sqrt{t^2+8}>1+t
\iff
8>1+2t
\]

\]

\]
\end{solution}

\begin{solution}
将 $a$ 视作参数，对 $f(x)$ 求导：
\begin{align*}f^{\prime}(x)&=-\frac{1}{2(1+x)^{\frac{3}{2}}}+\frac{1}{2(1+\frac{8}{ax})^{\frac{3}{2}}}\cdot\frac{8}{ax^2}\\&=-\frac{1}{2(1+x)^\frac{3}{2}}+\frac{4\sqrt{a}}{\sqrt{x}(ax+8)^\frac{3}{2}}=\frac{8\sqrt{a}(1+x)^\frac{3}{2}-\sqrt{x}(ax+8)^\frac{3}{2}}{2\sqrt{x}(1+x)^\frac{3}{2}(ax+8)^\frac{3}{2}}\\&=\frac{64a(1+x)^3-x(ax+8)^3}{2\sqrt{x}(1+x)^{\frac{3}{2}}(ax+8)^{\frac{3}{2}}\left[8\sqrt{a}(1+x)^{\frac{3}{2}}+\sqrt{x}(ax+8)^{\frac{3}{2}}\right]}\\&=\frac{-a^3x^4-24a^2x^3+64ax^3+192ax+64a-512x}{2\sqrt{x}(1+x)^{\frac{3}{2}}(ax+8)^{\frac{3}{2}}\left[8\sqrt{a}(1+x)^{\frac{3}{2}}+\sqrt{x}(ax+8)^{\frac{3}{2}}\right]}\\
&=\frac{-a(a^2x^4+(24a-64)x^3-8(24-\frac{64}{a})x-64)}{2\sqrt{x}(1+x)^{\frac{3}{2}}(ax+8)^{\frac{3}{2}}\left[8\sqrt{a}(1+x)^{\frac{3}{2}}+\sqrt{x}(ax+8)^{\frac{3}{2}}\right]}\\
&=\frac{-a\bigg((ax^2-8)(ax^2+8)+(24-\frac{64}{a})(ax^3-8x)\bigg)}{2\sqrt{x}(1+x)^{\frac{3}{2}}(ax+8)^{\frac{3}{2}}\left[8\sqrt{a}(1+x)^{\frac{3}{2}}+\sqrt{x}(ax+8)^{\frac{3}{2}}\right]}\\
&=\frac{\left(ax^{2}-8 \right) \left(-x^{2}a^{2}-24xa-8a+64x \right)}{2\sqrt{x}(1+x)^{\frac{3}{2}}(ax+8)^{\frac{3}{2}}\left[8\sqrt{a}(1+x)^{\frac{3}{2}}+\sqrt{x}(ax+8)^{\frac{3}{2}}\right]}\\
&=\frac{(\sqrt{a}x-2\sqrt2)(\sqrt{a}x+2\sqrt2)(-a^2x^2+(64-24a)x-8a)}{2\sqrt{x}(1+x)^{\frac{3}{2}}(ax+8)^{\frac{3}{2}}\left[8\sqrt{a}(1+x)^{\frac{3}{2}}+\sqrt{x}(ax+8)^{\frac{3}{2}}\right]}
\end{align*}
上式的分母恒正，符号由分子决定。继续分解后，关键在二次因子
\[
-a^2x^2+(64-24a)x-8a
\]
的符号。其判别式为
\[\Delta = (64-24a)^2-32a^3=32(-a^3+18a^2-96a+128)=32(2-a)(a-8)^2\]
若约定 $a\geq 2$，则 $\Delta\leq 0$。又因该二次式的二次项系数为负，所以它恒小于等于 $0$。这一约定是可行的，因为根号下的三个量 $a,\ x,\ \dfrac{8}{ax}$ 的乘积为 $8$，三者在结构上对称，可不失一般性地按大小顺序安排，使 $a\geq 2$。

再考察一次因子 $\sqrt{a}x-2\sqrt2$ 的符号，得到
\[
f'(x)>0 \quad \text{在 } \left(0,\sqrt{\frac{8}{a}}\right),
\qquad
f'(x)<0 \quad \text{在 } \left(\sqrt{\frac{8}{a}},+\infty\right).
\]
因此 $f(x)$ 先增后减，最大值为
\[
f\!\left(\sqrt{\frac{8}{a}}\right)=\frac{2}{\sqrt{1+\sqrt{\frac{8}{a}}}}+\frac{1}{\sqrt{1+a}}.
\]
又当 $x\to 0^+$ 与 $x\to +\infty$ 时，都有
\[
f(x)\to 1+\frac{1}{\sqrt{1+a}}.
\]
故
\[
1+\frac{1}{\sqrt{1+a}}<f(x)<\frac{2}{\sqrt{1+\sqrt{\frac{8}{a}}}}+\frac{1}{\sqrt{1+a}}.
\]
最后还需证明
\[
g(a)=\frac{2}{\sqrt{1+\sqrt{\frac{8}{a}}}}+\frac{1}{\sqrt{1+a}}<2.
\]
求导得
\begin{align*}g'(a)&=-\frac{1}{\left(1+\sqrt{\frac{8}{a}}\right)^{\frac{3}{2}}}\cdot\frac{-2\sqrt2}{2a^{\frac{3}{2}}}-\frac{1}{2(1+a)^{\frac{3}{2}}}\\&=\frac{2\sqrt{2}(1+a)^{\frac{3}{2}}-a^{\frac{3}{2}}\left(1+\sqrt{\frac{8}{a}}\right)^{\frac{3}{2}}}{2a^{\frac{3}{2}}\left(1+\sqrt{\frac{8}{a}}\right)^{\frac{3}{2}}(1+a)^{\frac{3}{2}}}\\&=\frac{8(1+a)^3-\left(a+2\sqrt{2a}\right)^3}{2a^{\frac{3}{2}}\left(1+\sqrt{\frac{8}{a}}\right)^{\frac{3}{2}}(1+a)^{\frac{3}{2}}\left[2\sqrt{2}(1+a)^{\frac{3}{2}}+a^{\frac{3}{2}}\left(1+\sqrt{\frac{8}{a}}\right)^{\frac{3}{2}}\right]}\\&=\frac{(a-2\sqrt{2a}+2)\left[4(1+a)^2+2(1+a)(a+2\sqrt{2a})+(a+2\sqrt{2a})^2\right]}{2a^{\frac{3}{2}}\left(1+\sqrt{\frac{8}{a}}\right)^{\frac{3}{2}}(1+a)^{\frac{3}{2}}\left[2\sqrt{2}(1+a)^{\frac{3}{2}}+a^{\frac{3}{2}}\left(1+\sqrt{\frac{8}{a}}\right)^{\frac{3}{2}}\right]}\\&=\frac{(\sqrt{a}-\sqrt{2})^2\left[4(1+a)^2+2(1+a)(a+2\sqrt{2a})+(a+2\sqrt{2a})^2\right]}{2a^{\frac{3}{2}}\left(1+\sqrt{\frac{8}{a}}\right)^{\frac{3}{2}}(1+a)^{\frac{3}{2}}\left[2\sqrt{2}(1+a)^{\frac{3}{2}}+a^{\frac{3}{2}}\left(1+\sqrt{\frac{8}{a}}\right)^{\frac{3}{2}}\right]}\geq0\end{align*}
故 $g(a)$ 单调递增，且
\[
\lim_{a\to +\infty} g(a)=2.
\]
因此任意 $a>0$ 都有 $g(a)<2$，从而 $f(x)<2$。
\end{solution}

\section{三角函数恒成立问题}
\begin{example}[经典题]
求$f(x) = \sin x \sin 2x \sin 3x$的值域.
\end{example}
\begin{solution}
反复积化和差
\begin{align*}
    f(x) &= \frac{1}{2}(\sin 2x \cos 2x - \sin 2x \cos 4x)= \frac{1}{4}(\sin 2x + \sin 4x - \sin 6x)\\
    f'(x)&= \frac{1}{4}(2\cos 2x + 4\cos 4x - 6\cos 6x) = 0
\end{align*}
令 $v = \cos 2x$，用二倍角和三倍角公式（$\cos 4x = 2v^2 - 1$，$\cos 6x = 4v^3 - 3v$）代入$f'(x)=0$得到
\[6v^3 - 2v^2 - 5v + 1 = (v - 1)(6v^2 + 4v - 1) = 0\Leftrightarrow v =\frac{-2 \pm \sqrt{10}}{6}\]
此时$f(x) = \frac{1}{4}\sin 2x (1 + 2\cos 2x - (4\cos^2 2x - 1)) = \frac{1}{2}\sin 2x (1 + v - 2v^2)$，将$\sin 2x$换成$v$的式子，考虑平方：$$f(x)^2 = \frac{1}{4}\sin^2 2x (1 + v - 2v^2)^2 = \frac{1}{4}(1 - v^2)(1 + v - 2v^2)^2$$用$6v^2 = 1 - 4v \implies v^2 = \frac{1 - 4v}{6}$反复降次得到极值$ \frac{1039 + 680v}{972}$，此时$f(x)^2 = \frac{2437 + 340\sqrt{10}}{11664}$，开方得到值域\[\left[- \frac{34\sqrt{2} + 5\sqrt{5}}{108}, \frac{34\sqrt{2} + 5\sqrt{5}}{108}\right]\]
\end{solution}

\begin{example}[$\tan x < \frac{x(9\pi^2 - 4x^2)}{9(\pi^2 - 4x^2)}$]
\( x \in (0, \frac{\pi}{2}) \) 时证明：
\[
\tan x < \frac{x(9\pi^2 - 4x^2)}{9(\pi^2 - 4x^2)}
\]
\end{example}
\begin{solution}
原式等价于
\[
9(\pi^2-4x^2)\tan x - x(9\pi^2-4x^2) < 0
\Leftrightarrow
\left( \frac{1}{\tan x} - \frac{1}{x} + \frac{9}{8} \right)x^2 < \frac{9\pi^2}{32}\Leftrightarrow f(x)< \frac{9\pi^2}{32}
\]
接下来把目标压到 \(f\) 的单调性上：证明 \( f(x) \) 在 \( (0,\frac{\pi}{2}) \) 上递减，并且
\[
\lim_{x\to 0^+} f(x) = \frac{9\pi^2}{32}.
\]
求导得
\[
f'(x) = \frac{x(x-3\tan x)^2 - 4x^2(1+\tan^2 x)}{4(x-\tan x)^2}
\]
分母恒正，符号由分子决定。令 \( t = \tan x > 0 \)，则
\[
\ (x-3t)^2 < 4x^2(1+t^2)
\iff (9-4x^2)t^2 - 6xt - 3x^2 < 0.
\]
当 \( x \ge \frac{3}{2} \) 时，二次项系数 \(9-4x^2\le0\)，而一次项与常数项都为负，所以对一切 \(t>0\) 都有该二次式小于 \(0\)。余下区间为 \(0<x<\frac32\)，这时可用夹在中间的不等式
\[
\tan x < \frac{x(x^2-12)}{5x^2-12} < \frac{6x + x\sqrt{36+12(9-4x^2)}}{2(9-4x^2)}
\]
右端正是二次式的较大根，因此再验证
\[
\tan x(5x^2-12) < x(x^2-12) \iff (2x^2-21x+36)^2 < 3(3-x)(5x-12)^2
\]
而右减左为
\[
3(3-x)(5x-12)^2-(2x^2-21x+36)^2=x^3(9-4x)>0
\]
因为此时 \(0<x<\frac32\)。于是 \(f'(x)<0\)，原不等式成立。
\end{solution}

\begin{example}[$\frac{3}{4} \tan x + \frac{1}{\tan x} > \frac{\pi^{2}}{x(\pi^{2}-4x^{2})}$]
\( x \in \left(0, \frac{\pi}{2}\right) \) 时证明
\[
\frac{3}{4} \tan x + \frac{1}{\tan x} > \frac{\pi^{2}}{x(\pi^{2}-4x^{2})}.
\]
\end{example}
\begin{solution}
两边同乘正数 \(x(\pi^2-4x^2)\tan x\)，得
\[
\frac{3}{4} \tan^2 x + 1 > \frac{\pi^2 \tan x}{x(\pi^2-4x^2)}.
\]
再把 \(x\) 换成 \(\pi-x\)，便成
\[
\frac{x(7 + \cos x)}{2 \sin x} > \frac{4\pi^2}{\pi^2 - x^2}.
\]
由 \(\tan x>x\)，有
\[
\frac{2 \sin x}{x(7 + \cos x)} < \frac{2 \sin x}{5x + 3 \sin x}.
\]
因此化为
\[
\frac{2 \sin x}{5x + 3 \sin x} < \frac{1}{4} - \frac{x^2}{4\pi^2}.
\]
整理为
\[
\sin x < \frac{5(\pi^2 - x^2)}{5\pi^2 + 3x^2}.
\]
记
\[
f(x) = \frac{1}{\frac{1}{\frac{\sin x}{x} + 1} - \frac{1}{2}} - \frac{5\pi^2 - x^2}{2x^2}.
\]
这就是证 \(f(x)<0\)。求导得
\[
f'(x) = \frac{4(x \cos x - \sin x)}{(x - \sin x)^2} + \frac{5\pi^2}{x^3}.
\]
由 \(x-\sin x>0\) 且
\[
x-\sin x>\frac{x^3}{6}-\frac{x^5}{120},
\]
可得
\[
f'(x) > \frac{4(x \cos x - \sin x)}{(x - \sin x)^2} + \frac{48}{x^3}.
\]
因此转为验证
\[
x^3(x \cos x - \sin x) + 12(x - \sin x)^2 > 0 \quad (x \in (0,\pi)).
\]
再把它写成
\[
\frac{12(x - \sin x)^2}{x^3} > (x - \sin x)\left(2 - \frac{x^2}{10}\right),
\]
也就是
\[
h(x) = x \cos x - \sin x + (x - \sin x)\left(2 - \frac{x^2}{10}\right) > 0 \quad (x \in (0,\pi)).
\]
求导得
\[
h'(x) = \frac{1}{10} \bigl( (x^2 - 20)\cos x - 8x \sin x + 20 - 3x^2 \bigr).
\]
记
\[
u(x) = (x^2 - 20)\cos x - 8x \sin x + 20 - 3x^2,
\]
则 \(h'(x)=u(x)/10\)，并且
\[
u'(x) = -12x \cos^2 \frac{x}{2} - 2(x^2 - 12)\sin \frac{x}{2} \cos \frac{x}{2} = -2 \cos \frac{x}{2} \Bigl( 6x \cos \frac{x}{2} + (x^2 - 12)\sin \frac{x}{2} \Bigr).
\]
因为 \(\cos \frac{x}{2}>0\)，所以 \(u'(x)>0\) 等价于
\[
\tan \frac{x}{2} > \frac{6x}{12 - x^2} \quad (x \in (0,\pi)),
\]
这是常用不等式，所以 \(u(x)>u(0)=0\)，从而 \(h'(x)>0\)。再由 \(h(0)=0\) 得 \(h(x)>0\)，进而 \(f'(x)>0\)。
\[
\lim_{x \to 0^+} f(x) = 0,
\]
所以 \(f\) 在 \((0,\pi)\) 上递增，且 \(f(x)<0\)。原不等式得证。
\end{solution}

\begin{example}[$Y \leqslant \frac{x + 4 - \sqrt{2x - 1}}{4} L$]
记$L = \frac{x - 1}{\ln x}, Y = 1 + L \ln L$。证明：$Y \leqslant \frac{x + 4 - \sqrt{2x - 1}}{4} L$
\end{example}
\begin{solution}
记
\[
m=\frac{Y}{L}=\frac{\ln x}{x - 1} - \ln \frac{\ln x}{x - 1}.
\]
于是转化为
\[
m<\frac{x + 4 - \sqrt{2x - 1}}{4}.
\]
由
\[
m < \frac{(x + \sqrt{x} + 1)\ln x}{x - 1} - 2
\]
及
\[
\frac{(x + \sqrt{x} + 1)\ln x}{x - 1} < 3 + \frac{(x - 1)^2}{8x},
\]
接下来比较
\[
\frac{(x - 1)^2}{8x}\quad\text{和}\quad \frac{x - \sqrt{2x - 1}}{4}.
\]
而
\[
\frac{(x - 1)^2}{8x} < \frac{x - \sqrt{2x - 1}}{4}
\iff
\sqrt{2x - 1} < x - \frac{(x - 1)^2}{2x}=\frac{x^2+2x-(x-1)^2}{2x},
\]
两边平方并化简后不等式成立，因此原命题成立。
\end{solution}

\begin{example}[三角放缩]
    证明对于任意 $x > 0$，有：$$x + \frac{1}{x} + 1 \geqslant \cot\left(\frac{\pi}{2(x^2+x+1)}\right)\cot\left(\frac{\pi x^2}{2(x^2+x+1)}\right)$$
\end{example}
\begin{solution}
设
\[
A = \frac{\pi}{2(x^2+x+1)},\qquad B = \frac{\pi x^2}{2(x^2+x+1)},\qquad C = \frac{\pi x}{2(x^2+x+1)}.
\]
则 \(A,B,C\in(0,\frac{\pi}{2})\)，并且
\[
A+B+C=\frac{\pi}{2},\qquad C^2=AB.
\]

由
\[
\tan(A+B)=\tan\left(\frac{\pi}{2}-C\right)=\cot C
\]
得
\[
\tan C(\tan A+\tan B)=1-\tan A\tan B,
\]
从而
\[
\cot A\cot B=1+\tan C(\cot A+\cot B).
\]
又
\[
x+\frac1x+1=\frac{x^2+x+1}{x}=\frac{\pi}{2C}=\frac{A+B+C}{C}=\frac{A+B}{C}+1.
\]
所以原不等式等价于
\[
\frac{\cot A+\cot B}{A+B}\le \frac{\cot C}{C}.
\]
再代入 \(C=\sqrt{AB}\)，问题化为
\[
\frac{\cot A+\cot B}{A+B}\le \frac{\cot \sqrt{AB}}{\sqrt{AB}}.
\]

写成部分分式级数
\[
\cot z = \frac{1}{z} - \sum_{n=1}^\infty \frac{2z}{n^2\pi^2 - z^2},
\]
代入后逐项比较
\[
\frac{1}{A+B} \left( \frac{A}{a - A^2} + \frac{B}{a - B^2} \right) \ge \frac{1}{a - AB},
\qquad a=n^2\pi^2.
\]
分母都为正，交叉相乘得
\[
[A(a-B^2)+B(a-A^2)](a-AB)\ge (A+B)(a-A^2)(a-B^2).
\]
左边化为 \((A+B)(a-AB)^2\)，比较便归结为
\[
a^2-2aAB+A^2B^2\ge a^2-a(A^2+B^2)+A^2B^2,
\]
也就是
\[
a(A-B)^2\ge0.
\]
因此原不等式成立。等号当且仅当 \(A=B\)，即 \(x=1\)。
\end{solution}

\begin{example}[三角放缩（二）]
    证明对于任意 $x > 0$，有：$$ \cot\left(\frac{\pi}{2(x^2+x+1)}\right)\cot\left(\frac{\pi x^2}{2(x^2+x+1)}\right)\geqslant 3$$
\end{example}
\begin{solution}
设
\[
A=\frac{\pi}{2(x^2+x+1)},\qquad B=\frac{\pi x^2}{2(x^2+x+1)},\qquad C=\frac{\pi x}{2(x^2+x+1)}.
\]
则 \(A,B,C\in(0,\frac{\pi}{2})\)，并且
\[
A+B+C=\frac{\pi}{2},\qquad C=\sqrt{AB}.
\]
与上一题一样，
\[
\cot A\cot B=1+\tan C(\cot A+\cot B)=1+\frac{\sin C}{\sin A\sin B}.
\]
因此转为估计 \(\sin A\sin B\)。

令
\[
f(t)=\ln(\sin e^t),
\]
其中 \(e^t\in(0,\frac{\pi}{2})\)。则
\[
f'(t)=e^t\cot e^t,\qquad
f''(t)=\frac{\frac12 e^t\sin(2e^t)-e^{2t}}{\sin^2 e^t}.
\]
对任意 \(u=e^t>0\)，由 \(\sin(2u)<2u\) 可知 \(f''(t)<0\)，所以 \(f\) 是严格凹函数。用 Jensen 不等式得到
\[
\frac{f(\ln A)+f(\ln B)}{2}\le f\!\left(\frac{\ln A+\ln B}{2}\right)=f(\ln\sqrt{AB})=f(\ln C).
\]
也就是
\[
\sin A\sin B\le \sin^2 C.
\]
代回去便得
\[
\cot A\cot B
=1+\frac{\sin C}{\sin A\sin B}
\ge 1+\frac{\sin C}{\sin^2 C}
=1+\frac1{\sin C}.
\]
又因为 \(A+B+C=\frac{\pi}{2}\) 且 \(A,B>0\)，所以 \(0<C\le\frac{\pi}{6}\)，故
\[
\cot A\cot B\ge 1+\frac1{\sin(\pi/6)}=3.
\]
等号成立当且仅当 \(A=B\) 且 \(C=\frac{\pi}{6}\)，也就是 \(x=1\)。
\end{solution}

\begin{example}[$\frac{\cot A + \cot B}{A+B} \le \frac{\cot \sqrt{AB}}{\sqrt{AB}}$]
    证明对于任意 $A, B \in (0, \frac{\pi}{2})$，恒有 $\displaystyle\frac{\cot A + \cot B}{A+B} \le \frac{\cot \sqrt{AB}}{\sqrt{AB}}$。
\end{example}
\begin{solution}
交换 \(A,B\) 后不等式不变，设 \(A\ge B\)。令 \(x=\sqrt{AB}\)，则 \(B=\frac{x^2}{A}\)，并把
\[
f(A)=\frac{\cot A+\cot\!\left(\frac{x^2}{A}\right)}{A+\frac{x^2}{A}}
\]
看成固定 \(x\) 时关于 \(A\) 的函数。

求导并代回 \(B=\frac{x^2}{A}\) 后，
\[
f'(A)=\frac{\Delta(A,B)}{A(A+B)^2},
\]
其中
\[
\Delta(A,B)=(A+B)(B\csc^2B-A\csc^2A)-(A-B)(\cot A+\cot B).
\]
分母恒正，符号由 \(\Delta(A,B)\) 决定。

记
\[
R(y)=y\cot y,\qquad Q(y)=y\csc^2y-\cot y=-R'(y).
\]
那么
\[
B\csc^2B-A\csc^2A=Q(B)-Q(A)+\cot B-\cot A,
\]
从而
\[
\Delta(A,B)=(A+B)[Q(B)-Q(A)]-2[R(A)-R(B)].
\]
再用
\[
R(A)-R(B)=\int_B^A R'(t)\,dt=-\int_B^A Q(t)\,dt,
\]
可写成
\[
F(A,B):=-\Delta(A,B)=(A+B)[Q(A)-Q(B)]-2\int_B^A Q(t)\,dt.
\]
接下来证明 \(F(A,B)\ge0\)。

对 \(A\) 求偏导，
\[
\frac{\partial F}{\partial A}=(A+B)Q'(A)-Q(A)-Q(B).
\]
继续求导，
\[
\frac{\partial^2 F}{\partial A^2}=(A+B)Q''(A).
\]
所以关键是 \(Q''(A)\) 的符号。由
\[
Q'(y)=2\csc^2y(1-y\cot y),
\]
\[
Q''(y)=2\csc^2y\,[y-3\cot y+3y\cot^2y]
\]
因此归结为证明
\[
y-3\cot y+3y\cot^2y>0.
\]
令 \(u=2y\in(0,\pi)\)，上式分子化为
\[
z(u)=u+\frac{u}{2}\cos u-\frac32\sin u.
\]
继续求导，
\[
z'(u)=1-\cos u-\frac{u}{2}\sin u,\qquad
z''(u)=\frac12(\sin u-u\cos u).
\]
当 \(u\in(0,\frac{\pi}{2}]\) 时，由 \(\tan u>u\) 得 \(\sin u>u\cos u\)；当 \(u\in(\frac{\pi}{2},\pi)\) 时，\(\cos u<0\)，同样有 \(\sin u-u\cos u>0\)。因此 \(z''(u)>0\) 在 \((0,\pi)\) 上恒成立。再结合 \(z'(0)=0,\ z(0)=0\)，可知 \(z'(u)>0\)、\(z(u)>0\)，故 \(Q''(y)>0\)。

因此 \(\frac{\partial^2 F}{\partial A^2}>0\)，所以 \(\frac{\partial F}{\partial A}\) 随 \(A\) 递增。取
\[
G(B,B):=\left.\frac{\partial F}{\partial A}\right|_{A=B}=2(BQ'(B)-Q(B)).
\]
记 \(H(y)=yQ'(y)-Q(y)\)，则
\[
H'(y)=yQ''(y)>0.
\]
又由
\[
Q(y)=\frac{2y}{3}+O(y^3),\qquad Q'(y)=\frac{2}{3}+O(y^2)
\]
可知 \(H(y)\to0^+\) 当 \(y\to0^+\)。又因 \(H'(y)>0\)，便有 \(H(y)>0\)，也就是 \(G(B,B)>0\)，从而
\[
\frac{\partial F}{\partial A}>0\qquad(A\ge B),
\]
再结合 \(F(B,B)=0\)，便得 \(F(A,B)\ge0\)，从而 \(\Delta(A,B)\le0\)，也就是 \(f'(A)\le0\)。

因此在固定 \(x=\sqrt{AB}\) 时，\(f(A)\) 随 \(A\) 递减，最大值在 \(A=B=x\) 处取得：
\[
f(A)\le f(x)=\frac{\cot x+\cot x}{x+x}=\frac{\cot x}{x}.
\]
把 \(x=\sqrt{AB}\) 代回便得
\[
\frac{\cot A+\cot B}{A+B}\le \frac{\cot\sqrt{AB}}{\sqrt{AB}}.
\]
等号当且仅当 \(A=B\)。
\end{solution}

\begin{example}[三角函数]
    求函数 $f_a(x) = a^{\sin x} + a^{\cos x}$ 的取值范围，其中 $a > 1$.
\end{example}
\begin{solution}
对于函数 $f_a(x) = a^{\sin x} + a^{\cos x}$（$a > 1$），先求最小值。由均值不等式，
\[
a^{\sin x} + a^{\cos x} \geq 2\sqrt{a^{\sin x + \cos x}}.
\]
当且仅当 $a^{\sin x} = a^{\cos x}$，即 $\sin x = \cos x$ 时取等。由辅助角公式 $\sin x + \cos x = \sqrt{2}\sin\left(x + \frac{\pi}{4}\right)$，其最小值为 $-\sqrt{2}$。指数函数 $a^u$ 单调递增，故 $\sqrt{a^{\sin x + \cos x}}$ 的最小值为 $\sqrt{a^{-\sqrt{2}}} = a^{-1/\sqrt{2}}$。当 $x = 2k\pi - \frac{3\pi}{4}$（$k \in \mathbb{Z}$）时，$\sin x = \cos x = -\frac{\sqrt{2}}{2}$，上述等号同时成立，所以 $f_a(x)$ 的最小值为 $2a^{-1/\sqrt{2}}$。

再看最大值。令 $x = y + \frac{\pi}{4}$，则
\[
f_a(x) = a^{\frac{\sin y + \cos y}{\sqrt{2}}} + a^{\frac{\cos y - \sin y}{\sqrt{2}}}.
\]
记 $b = \frac{\ln a}{\sqrt{2}} > 0$，利用 $\cosh u = \frac{e^u + e^{-u}}{2}$，可写成
\[
f_a(x) = 2e^{b \cos y} \cosh(b \sin y).
\]
对 $\frac{f_a(x)}{2}$ 取对数得
\[
\ln\frac{f_a(x)}{2} = b \cos y + \ln \cosh(b \sin y).
\]
又因为 $\cos y \leq \sqrt{1 - \sin^2 y}$，故
\[
\ln\frac{f_a(x)}{2} \leq \sqrt{b^2 - (b\sin y)^2} + \ln \cosh(b \sin y).
\]

令 $t = b \sin y$，则 $t \in [-b, b]$。记
\[
g(t) = \sqrt{b^2 - t^2} + \ln \cosh t.
\]
$g$ 是偶函数，讨论 $t \in [0,b]$。有
\[
g'(t) = \tanh t - \frac{t}{\sqrt{b^2 - t^2}},\qquad
g''(t) = \operatorname{sech}^2 t - \frac{b^2}{(b^2 - t^2)^{3/2}}.
\]
其中两项都随 $t$ 递减，所以 $g''(t)$ 也递减。再看 $g''(0) = 1 - \frac{1}{b}$，便可按 \(b\) 与 \(1\) 的大小继续讨论。

若 $b \leq 1$（即 $1 < a \leq e^{\sqrt{2}}$），则 $g''(0) \leq 0$。由 $g''(t)$ 严格递减可知 $g''(t) < 0$ 对一切 $t \in (0, b]$ 成立，故 $g'(t)$ 在 $[0, b]$ 上严格递减。又 $g'(0) = 0$，故在 $t \in (0, b]$ 内 $g'(t) < 0$，函数 $g(t)$ 在 $[0, b]$ 上严格递减。此时最大值在 $t = 0$ 处取得，即 $g(0) = b$。对应 $x = 2k\pi + \frac{\pi}{4}$ 时，原函数取得最大值 $2e^b = 2a^{1/\sqrt{2}}$。

若 $b > 1$（即 $a > e^{\sqrt{2}}$），则 $g''(0) > 0$。由于 $g''(t)$ 严格递减且当 $t \to b$ 时 $g''(t) \to -\infty$，存在唯一的 $t_1 \in (0, b)$ 使得 $g''(t_1) = 0$。此时 $g'(t)$ 在 $[0, t_1]$ 上严格递增，在 $[t_1, b]$ 上严格递减。结合 $g'(0) = 0$ 且当 $t \to b$ 时 $g'(t) \to -\infty$，可知在 $(0, b)$ 内存在唯一的 $t_0$ 使得 $g'(t_0) = 0$。此时 $g(t)$ 在 $t_0$ 处取得极大值，亦为最大值。由于 $g(t_0) > g(0) = b$，此情形下最大值大于 $2a^{1/\sqrt{2}}$。

因此，当 $1 < a \leq e^{\sqrt{2}}$ 时（如 $a=3$），函数 $f_a(x)$ 的取值范围为 $[2a^{-1/\sqrt{2}}, 2a^{1/\sqrt{2}}]$；当 $a > e^{\sqrt{2}}$ 时，最小值为 $2a^{-1/\sqrt{2}}$，最大值为 $2\exp\left(\sqrt{b^2 - t_0^2} + \ln \cosh t_0\right)$，其中 $t_0$ 为方程 $\tanh t = \frac{t}{\sqrt{b^2 - t^2}}$ 在 $(0, b)$ 内的唯一根。
\end{solution}

\begin{example}[正弦和最大值]
    求$\displaystyle\min_{a\in\mathbb{Q}^+}\{\max_{x\in R}\{\sin x+\sin ax\}\}$.
\end{example}
\begin{solution}
把 $a$ 写成 $\frac{p}{q}$，其中 $p,q$ 为互素正整数。令
\[
f(x)= \sin(qx) + \sin(px),\qquad M(p,q) = \max_{x \in \mathbb{R}} (\sin px + \sin qx).
\]
原问题就是求 $\displaystyle\min_{(p,q)} M(p,q)$。由和差化积公式，
\[
f(x) = 2\sin\left(\frac{p+q}{2}x\right)\cos\left(\frac{p-q}{2}x\right).
\]

若 $p,q$ 都是奇数且 $p\equiv q\pmod 4$，则两数同余于 $1$ 或同余于 $3$ 模 $4$。前一种情形取 $x=\frac{\pi}{2}$，后一种情形取 $x=\frac{3\pi}{2}$，都可使
\[
\sin(px)=\sin(qx)=1.
\]
因此
\[
M(p,q)\ge 2.
\]

若 $p,q$ 一奇一偶，取
\[
x_k=\frac{4k+1}{p+q}\pi \qquad (k\in\mathbb Z).
\]
这时
\[
\sin\left(\frac{p+q}{2}x_k\right)=1,
\]
所以
\[
f(x_k) = 2\cos\left(\frac{p-q}{p+q} \cdot \frac{4k+1}{2}\pi\right).
\]
记 $S=p+q,\ D=p-q$。一奇一偶时，$S,D$ 都是奇数，且 \(\gcd(D,S)=1\)。于是
\[
\Delta=\frac{D(4k+1)}{2S}\pi-2m\pi
=\frac{\pi}{2S}\bigl[4Dk-4mS+D\bigr].
\]
由裴蜀定理，$Dk-Sm$ 可取遍所有整数，所以 $4(Dk-Sm)+D$ 可取遍所有模 $4$ 余 $D$ 的整数。由于 $D$ 是奇数，其中必有绝对值等于 $1$ 的一项，于是可取到
\[
|\Delta|_{\min}=\frac{\pi}{2S}=\frac{\pi}{2(p+q)}.
\]
故
\[
M(p,q)\ge 2\cos\left(\frac{\pi}{2(p+q)}\right)\ge 2\cos\frac{\pi}{6}=\sqrt3.
\]

剩下的情形是 $p,q$ 都为奇数但不同余于 $4$。这时一个是 \(4n+1\)，另一个是 \(4n+3\)，于是
\[
S=p+q=4S',\qquad D=p-q=2D',
\]
其中 \(D'\) 为奇数。仍取
\[
x_k=\frac{4k+1}{p+q}\pi.
\]
此时
\[
\Delta=\frac{\pi}{2S}\bigl[D(4k+1)-4mS\bigr]
=\frac{\pi}{4S'}\bigl[4D'k-8mS'+D'\bigr].
\]
因为 \(\gcd(D',2S')=1\)，同样能取到
\[
|\Delta|_{\min}=\frac{\pi}{4S'}=\frac{\pi}{p+q}.
\]
所以
\[
M(p,q)\ge 2\cos\left(\frac{\pi}{p+q}\right).
\]
若 $p+q \ge 8$，便有
\[
M(p,q)\ge 2\cos\left(\frac{\pi}{8}\right)=\sqrt{2+\sqrt{2}}> \sqrt3.
\]
因此想把最大值压到 $\sqrt3$ 以下，$p+q$ 就不可能达到 $8$。本类别里的唯一例外为 $p+q=4$，而这时互素正整数只能是
\[
\{p,q\}=\{1,3\}.
\]

因此，除了 $\{1,3\}$ 之外，其余所有 $(p,q)$ 都满足 $M(p,q)>\sqrt3$。全局最小值只能在 $\{1,3\}$ 处取得，对应
\[
a=3\quad\text{或}\quad a=\frac13.
\]
这时
\[
\sin x+\sin 3x=2\sin 2x\cos x=4\sin x\cos^2x.
\]
设 $t=\sin x\in[-1,1]$，则
\[
\sin x+\sin 3x=4t(1-t^2).
\]
其最大值在 $t=\frac{1}{\sqrt3}$ 处取得，等于
\[
\frac{8\sqrt3}{9}.
\]
\end{solution}

e^{2t_0+2a} \geqslant e^{x_0},
\]
即 $2t_0 + 2a \geqslant x_0$。又因为 $x\ln(x-2)$ 在 $(2,+\infty)$ 上单增，这等价于 $f(2t_0+2a) \geqslant f(x_0) = 0$。

由 $2at_0^2+t_0-1=0$ 得 $a = \frac{1-t_0}{2t_0^2}$，代入得 $2t_0+2a = \frac{2t_0^3 - t_0 + 1}{t_0^2}$。设 $A(t_0) = f(2t_0+2a) = (2t_0+2a) \ln(2t_0+2a-2) - \ln a$，需使得 $A(t_0) \geqslant 0$ 恒成立。对 $A(t_0)$ 求导可知 $A'(t_0) < 0$ 恒成立。当 $a = 1$ 时，对应 $t_0 = \frac{1}{2}$，$2t_0+2a = 3$，此时 $A(1/2) = 0$。为保证 $A(t_0) \geqslant 0$，必须满足 $t_0 \le \frac{1}{2}$。又因 $a = \frac{1-t_0}{2t_0^2}$ 在 $(0, 1)$ 上随 $t_0$ 单调递减，故 $t_0 \le \frac{1}{2}$ 等价于 $a \geqslant 1$。因此，$a$ 的取值范围为 $[1, +\infty)$。
\newpage
（2）原方程等价于 $W(x) = \frac{f(x) - g(x)}{x} = \ln(x-2) - \frac{\ln a}{x} - 2a - \frac{1}{\ln x - a} = 0$。对 $W(x)$ 求导：
$$W'(x) = \frac{1}{x-2} + \frac{\ln a}{x^2} + \frac{1}{x(\ln x - a)^2}$$
函数 $W(x)$ 的定义域为 $x>2$ 且 $x\ne e^a$。

当 $a > \ln 2$ 时，$e^a > 2$，定义域为 $(2, e^a) \cup (e^a, +\infty)$。考察 $W'(x)$ 的前两项通分得 $\frac{x^2 + (x-2)\ln a}{x^2(x-2)}$。设分子为 $p(x) = x^2 + (x-2)\ln a$，其对称轴为 $x = -\frac{\ln a}{2}$。由于 $a > \ln 2$，可得对称轴严格小于 $\frac{1}{2}$，故 $p(x)$ 在 $x > 2$ 上严格单调递增。又因 $p(2) = 4 > 0$，故对任意 $x > 2$ 均有 $p(x) > 0$。从而 $W'(x) > 0$ 恒成立，函数 $W(x)$ 在区间 $(2, e^a)$ 和 $(e^a, +\infty)$ 内均严格单调递增。在 $(2, e^a)$ 内，当 $x \to 2^+$ 时 $W(x) \to -\infty$；当 $x \to (e^a)^-$ 时 $W(x) \to +\infty$，存在 $1$ 个零点。在 $(e^a, +\infty)$ 内，当 $x \to (e^a)^+$ 时 $W(x) \to -\infty$；当 $x \to +\infty$ 时 $W(x) \to +\infty$，也存在 $1$ 个零点。故此时 $h(x)$ 存在 $2$ 个零点。

当 $0 < a \leqslant \ln 2$ 时，$e^a \leqslant 2$，此时均有 $\ln x > \ln 2 \geqslant a$，定义域为 $(2, +\infty)$。当 $x \to 2^+$ 时 $W(x) \to -\infty$；当 $x \to +\infty$ 时 $W(x) \to +\infty$。若存在极值点 $x_1$ 使得 $W'(x_1) = 0$，则有 $\frac{\ln a}{x_1} = -\frac{x_1}{x_1-2} - \frac{1}{(\ln x_1 - a)^2}$。将其代入 $W(x_1)$ 得：
$$W(x_1) = \ln(x_1-2) + \frac{x_1}{x_1-2} + \frac{1}{(\ln x_1 - a)^2} - \frac{1}{\ln x_1 - a} - 2a$$
由于 $x_1 - 2 > 0$，有 $\ln(x_1-2) + \frac{x_1-2+2}{x_1-2} = \ln(x_1-2) + \frac{2}{x_1-2} + 1 \geqslant \ln 2 + 2$。又设 $t = \ln x_1 - a > 0$，则 $\frac{1}{t^2} - \frac{1}{t} = \left(\frac{1}{t} - \frac{1}{2}\right)^2 - \frac{1}{4} \geqslant -\frac{1}{4}$。因此对于任意极值点，有 $W(x_1) \geqslant \ln 2 + 2 - \frac{1}{4} - 2a \geqslant \ln 2 + \frac{7}{4} - 2\ln 2 > 0$。这表明 $W(x)$ 的所有极小值均严格大于 $0$。由于 $x \to 2^+$ 时函数从 $-\infty$ 开始单调上升，在到达首个极小值的过程中穿过 $x$ 轴一次，此后函数值恒为正。故此时仅存在 $1$ 个零点。

因此，当 $0 < a \leqslant \ln 2$ 时，$h(x)$ 有 $1$ 个零点；当 $a > \ln 2$ 时，$h(x)$ 有 $2$ 个零点。
\end{solution}

\begin{example}[共零点问题]
    函数 $f(x) = \log_a(\ln(x + a))$, $g(x) = \ln(x + b)$ 共零点.\\
（1）求 $b - a$ 的值;（2）若 $f(x) \geq g(x)$ 在区间 $(-b, 2]$ 上恒成立, 求实数 $a$ 的取值范围.
\end{example}
\begin{solution}
（1）$f(x)=0$ 的唯一解是 $x=e-a$，$g(x)=0$ 的唯一解是 $x=1-b$。由共零点可得
\[
e-a=1-b,
\]
所以
\[
b-a=1-e.
\]
\\
（2）令 $t=x+a$，则 $t\in(e-1,a+2]$，原不等式化为
\[
H(t)=\frac{\ln(\ln t)}{\ln a}-\ln(t+1-e)\ge0.
\]
且
\[
H(e)=0.
\]

当 $0 < a < 1$ 时，$\ln a < 0$。若 $a \le e - 2$，则区间右端点 $a + 2 \le e$。此时对于任意 $t \in (e - 1, a + 2]$，均有 $t \le e$，则 $\ln(\ln t) \le 0$。由于 $\ln a < 0$，得 $\frac{\ln(\ln t)}{\ln a} \ge 0$。同时，$t \le e$ 导致 $t + 1 - e \le 1$，故 $\ln(t + 1 - e) \le 0$，即 $-\ln(t + 1 - e) \ge 0$。此时 $H(t)$ 为两个非负项之和，恒有 $H(t) \ge 0$。若 $e - 2 < a < 1$，则区间包含 $t > e$ 的部分，此时若取 $t = a + 2 > e$，则 $\ln(\ln t) > 0$ 且 $\ln(t + 1 - e) > 0$，导致 $H(a + 2) < 0$，不满足题意。故此种情况下 $a \in (0, e - 2]$。

当 $a > 1$ 时，区间 $(e - 1, a + 2]$ 包含内部点 $t = e$。若 $H(t) \ge 0$ 在该区间内恒成立且 $H(e) = 0$，则 $t = e$ 必为极小值点，所以 $H'(e) = 0$。对 $H(t)$ 求导得：
$$H'(t) = \frac{1}{t \ln t \ln a} - \frac{1}{t + 1 - e}$$
令 $H'(e)=\frac{1}{e\ln a}-1=0$，解得
\[
a=e^{\frac{1}{e}}.
\]
再验充分性。此时
\[
H'(t)=\frac{e(t+1-e)-t\ln t}{t\ln t(t+1-e)}.
\]
记分子为 $u(t)=e(t+1-e)-t\ln t$，则 $u'(t)=e-1-\ln t$。在 $t\in(e-1,a+2]$ 上，由于 $a=e^{1/e}<2$，所以 $t<4<e^{e-1}$，从而 $\ln t<e-1$，即 $u'(t)>0$。又 \(u(e)=0\)，故 \(u(t)\) 在 \(e\) 左侧为负，在右侧为正，于是 \(H'(t)\) 在 \((e-1,e)\) 上为负，在 \((e,a+2]\) 上为正。这样 \(H(t)\) 在 \(t=e\) 处取到最小值 \(0\)，从而 \(H(t)\ge0\)。

因此，$b - a = 1 - e$，实数 $a$ 的取值范围是 $(0, e - 2] \cup \{ e^{\frac{1}{e}} \}$。
\end{solution}

\begin{example}[零点个数题]
已知函数$f(x)=e^{ax}+\ln(x+1)-2\tan x-1,\quad x\in\left(-1,\frac\pi2\right).$\\
(1) 当 $a=0$ 时，讨论$f(x)$ 的零点个数.\\
(2) 证明：当 $a\leqslant1$ 时，$f(x)$ 有且仅有两个零点.
\end{example}
\begin{solution}
（1）当 $a=0$ 时，$f(x)=\ln(x+1)-2\tan x$，且 $f(0)=0$。
求导：$$f'(x) = \frac{\cos^2 x - 2x - 2}{(x+1)\cos^2 x}=\frac{h(x)}{(x+1)\cos^2 x},h'(x) = -2\sin x \cos x - 2 = -\sin(2x) - 2$$
由 $h'(x)\le -1<0$，知 $h$ 在 $\left(-1,\frac{\pi}{2}\right)$ 上递减。又 $h(0)=-1<0$，所以 $x>0$ 时 $h(x)<0$，即 $f'(x)<0$。因此 $f$ 在 $\left(0,\frac{\pi}{2}\right)$ 上递减，而 $f(0)=0$，故这段区间没有零点。

又 $h(-1^+)>0>h(0)$，故存在唯一的 $x_0\in(-1,0)$ 使 $h(x_0)=0$。于是 $f$ 在 $(-1,x_0)$ 上递增，在 $(x_0,0)$ 上递减。再结合 $f(-1^+)=-\infty$、$f(0)=0$ 和 $f(x_0)>0$，可知 $(-1,0)$ 内恰有一个零点。综上，当 $a=0$ 时，$f(x)$ 恰有两个零点。\\
（2）当 $a \leqslant 1$ 时，仍有 $f(0)=0$。对 $x>0$，由 $a\le1$ 可得 $e^{ax}\le e^x$，于是
\[
f(x)\le e^x+\ln(x+1)-2\tan x-1\triangleq f_1(x).
\]
在 $\left(0,\frac{\pi}{2}\right)$ 上，
\[
f_1'(x)=e^x+\frac{1}{x+1}-\frac{2}{\cos^2 x}
\le (1+x+x^2)+(1-x+x^2)-(2+2x^2)=0.
\]
故 $f_1$ 在 $\left(0,\frac{\pi}{2}\right)$ 上递减，而 $f_1(0)=0$，所以这段区间没有零点。

再看 $(-1,0)$。令
\[
y(x) = 1 + 2\tan x - \ln(x+1),\qquad
y'(x) = \frac{2}{\cos^2 x} - \frac{1}{x+1},\qquad
y''(x) = \frac{4\sin x}{\cos^3 x} + \frac{1}{(x+1)^2}.
\]
令 $t = -x$，则 $t \in (0, 1)$。此时
\begin{align*}
y''(x) &= \frac{1}{(1-t)^2} - \frac{4\sin t}{\cos^3 t}>\frac{1}{(1-t)^2}-\frac{4t}{(1-\frac{t^2}2)^3}\\
&=\dfrac{1 - 4t + \frac{13}{2}t^2 - 4t^3 + \frac{3}{4}t^4 - \frac{1}{8}t^6}{(1-t)^2(1-\frac{t^2}2)^3}
=\frac{(1-t)^4 + \frac{t^2}{8}(4 - 2t^2 - t^4)}{(1-t)^2(1-\frac{t^2}2)^3}>0.
\end{align*}
由泰勒中值定理，对任意 $x \in (-1, 0)$，存在 $c \in (x, 0)$ 使
\[
y(x) = y(0) + y'(0)x + \frac{y''(c)}{2}x^2= 1 + x + \frac{y''(c)}{2}x^2>0.
\]
因此在 $(-1,0)$ 上，方程 $f(x)=0$ 等价于
\[
e^{ax}=y(x),
\]
也等价于
\[
q(x)\triangleq \frac{\ln y(x)}{x}=a.
\]

先看 $\left(-\frac12,0\right)$。令
\[
F(x)=e^x+\ln(x+1)-2\tan x-1.
\]
对 \(F\) 求导得
\[
F'(x)=e^x+\frac{1}{x+1}-\frac{2}{\cos^2 x}<0\qquad \left(-\frac12<x<0\right).
\]
又 $F(0)=0$，故
\[
F(x)>0\qquad \left(-\frac12<x<0\right),
\]
即
\[
y(x)<e^x\qquad \left(-\frac12<x<0\right).
\]
于是
\[
q(x)=\frac{\ln y(x)}{x}>1\qquad \left(-\frac12<x<0\right).
\]
所以当 $a\le1$ 时，$f(x)=0$ 在这段区间内无解。

再看 $\left(-1,-\frac12\right]$。对 \(\ln y(x)\) 求二阶导并整理，得到
\[
(\ln y(x))''=\frac{y(x)y''(x)-y'(x)^2}{y(x)^2}>0,
\]
所以 $\ln y(x)$ 在 $\left(-1,-\frac12\right]$ 上严格凸。由 $\ln y(0)=0$ 可知
\[
q(x)=\frac{\ln y(x)-\ln y(0)}{x-0}
\]
是这段区间上的严格递增割线斜率。再由
\[
\lim_{x\to-1^+}q(x)=-\infty,\qquad
q\!\left(-\frac12\right)>1,
\]
可知对任意 $a\le1$，方程 $q(x)=a$ 在 $\left(-1,-\frac12\right]$ 内恰有一个解。

因此 $f(x)=0$ 在 $(-1,0)$ 内恰有一个零点；再加上 $f(0)=0$，当 $a\le1$ 时，$f(x)$ 恰有两个零点。
\end{solution}
