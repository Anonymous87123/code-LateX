\chapter{偏移题}
\section{偏移题}
\begin{example}{}{}
$f(x)=(e^(2x))+(x^2),f'(w)=0,f(u)=f(v)=a$,证明或证伪$f''(w)((u-v)^2)<8(a-f(w))$.
\end{example}
\begin{solution}
有
$$ f'(x) = 2(e^{2x} + x), \quad f''(x) = 4e^{2x} + 2. $$
因为 $f''(x)>0$ 对任意 $x\in\mathbb R$ 都成立，$f$ 在 $\mathbb R$ 上严格凸，方程 $f'(x)=0$ 只有一解 $w$，且 $w$ 是全局极小点。由 $f'(w)=0$ 得
$$ e^{2w}=-w>0,\qquad f''(w)=4e^{2w}+2=2(1-2w). $$

设 $u<v$，记
$$ m=\frac{u+v}{2},\qquad h=\frac{v-u}{2}>0, $$
则 $u=m-h,\ v=m+h,\ (v-u)^2=4h^2$。由 $f(u)=f(v)=a$ 得
$$ 0=f(m+h)-f(m-h)=e^{2m}(e^{2h}-e^{-2h})+4mh, $$
即
$$ e^{2m}\sinh(2h)=-2mh. $$
由于 $\sinh(2h)>0$，可知 $m<0$。

再算
$$ a-f(m)=f(m+h)-f(m)=e^{2m}(e^{2h}-1)+2mh+h^2. $$
代入 $2mh=-e^{2m}\sinh(2h)$，并用
$$ e^{2h}-1-\sinh(2h)=\cosh(2h)-1=2\sinh^2 h, $$
得到
$$ a-f(m)=h^2+2e^{2m}\sinh^2 h. $$

又 $f'''(x)=8e^{2x}>0$，故对任意 $h>0$，
$$ f(w+h)-f(w-h)=\frac{f'''(\xi)}{3}h^3>0. $$
而 $f(m+h)=f(m-h)$，所以 $m\ne w$，从而 $f(m)>f(w)$，即
$$ a-f(w)>a-f(m). $$
因此等价于证明
$$ f''(w)(v-u)^2<8(a-f(m)). $$

代入 $f''(w)=2(1-2w)$ 和 $(v-u)^2=4h^2$，上式等价于
$$ 8(1-2w)h^2<8(h^2+2e^{2m}\sinh^2 h), $$
也就是
$$ -wh^2<e^{2m}\sinh^2 h \iff e^{2m}\left(\frac{\sinh h}{h}\right)^2>-w. $$
再由 $e^{2m}\sinh(2h)=-2mh$，可把左边改写成
$$ e^{2m}\left(\frac{\sinh h}{h}\right)^2
=\frac{-2mh}{\sinh(2h)}\cdot\frac{\sinh^2 h}{h^2}
=\frac{-m}{h}\tanh h. $$
于是等价于证明
$$ \frac{-m}{h}\tanh h>-w, \qquad\text{即}\qquad -m>-wh\coth h. $$

固定 $h>0$，记
$$ F(x)=e^{-2x}\frac{\sinh(2h)}{2h}-x \qquad (x>0). $$
则
$$ F'(x)=-2e^{-2x}\frac{\sinh(2h)}{2h}-1<0, $$
所以 $F$ 严格递减。由 $e^{2m}\sinh(2h)=-2mh$ 可知 $F(-m)=0$，因此验证
$$ F(-wh\coth h)>0. $$
这又等价于
$$ e^{2wh\coth h}\left(\frac{\sinh h}{h}\right)^2>-w. $$
利用 $e^{2w}=-w$，上式化为
$$ \ln\left(\frac{\sinh h}{h}\right)+w(h\coth h-1)>0. $$

记
$$ \phi(h)=\ln\left(\frac{\sinh h}{h}\right)+w(h\coth h-1). $$
又 $\lim\limits_{h\to0^+}\phi(h)=0$。再证 $\phi'(h)>0$，求导得：
$$ \phi'(h)=\left(\coth h-\frac1h\right)+w\left(\coth h-\frac{h}{\sinh^2 h}\right)
=\frac1h\left[(1+w)h\coth h-w\frac{h^2}{\sinh^2 h}-1\right]. $$

先说明 $w\in\left(-\frac12,0\right)$。令 $s(x)=e^{2x}+x$，则 $s'(x)=2e^{2x}+1>0$，故 $s$ 严格递增，而
$$ s\left(-\frac12\right)=e^{-1}-\frac12<0<s(0)=1, $$
所以唯一零点 $w$ 落在 $\left(-\frac12,0\right)$，从而 $1+w>-w>0$。

再证
$$ h\coth h+\frac{h^2}{\sinh^2 h}>2 \qquad (h>0). $$
令
$$ H(h)=1+h^2+\frac{h}{2}\sinh(2h)-\cosh(2h), $$
则 $H(0)=H'(0)=0$，且
$$ H''(h)=2-2\cosh(2h)+2h\sinh(2h)=4\sinh h\,(h\cosh h-\sinh h). $$
再记 $G(h)=h\cosh h-\sinh h$，则 $G(0)=0,\ G'(h)=h\sinh h>0$，故 $G(h)>0$。于是 $H''(h)>0$，从而 $H(h)>0$。展开后便是
$$ h^2+h\sinh h\cosh h-2\sinh^2 h>0, $$
两边除以 $\sinh^2 h$ 得
$$ h\coth h+\frac{h^2}{\sinh^2 h}>2, $$
即
$$ h\coth h-1>1-\frac{h^2}{\sinh^2 h}. $$

又因为 $\sinh h>h$，所以 $1-\dfrac{h^2}{\sinh^2 h}>0$。结合 $1+w>-w>0$，有
$$ (1+w)(h\coth h-1)>-w(h\coth h-1)>-w\left(1-\frac{h^2}{\sinh^2 h}\right). $$
移项后得到
$$ (1+w)h\coth h-w\frac{h^2}{\sinh^2 h}>1, $$
故 $\phi'(h)>0$。于是 $\phi(h)$ 在 $(0,+\infty)$ 上严格递增，而 $\phi(0^+)=0$，所以
$$ \phi(h)>0 \qquad (h>0). $$

因此 $F(-wh\coth h)>0$，再由 $F$ 的单调性可得 $-m>-wh\coth h$，也就是
$$ e^{2m}\left(\frac{\sinh h}{h}\right)^2>-w. $$
于是
$$ (1-2w)h^2<a-f(m), $$
两边乘 $8$ 后得
$$ 8(1-2w)h^2<8(a-f(m)). $$
而
$$ f''(w)(v-u)^2=2(1-2w)\cdot4h^2=8(1-2w)h^2, $$
所以
$$ f''(w)(v-u)^2<8(a-f(m))<8(a-f(w)). $$
结论得证。
\end{solution}

\begin{example}{交点宽度比较}{}
已知$x_1-\ln x_1=x_2-\ln x_2=x_3\ln x_3-x_3+2=x_4\ln x_4-x_4+2,\quad x_2>x_1,\quad x_4>x_3.$证明:\[x_2+x_3>x_1+x_4\]
\end{example}
\begin{solution}
令 $f(x) = x - \ln x$ 与 $g(x) = x\ln x - x + 2$。
已知 $f(1) = g(1) = 1$。设原方程相等的各部分数值为常数 $k$（$k > 1$）。由于 $x_3 \in (0,1)$ 的存在要求常数 $k$ 必须在 $g(x)$ 于该区间的极限范围内，即 $k \in (1,2)$。然后令 $k = 1 + \frac{m^2}{2}$，则 $m$ 的取值范围为 $m \in (0, \sqrt{2})$。

先证 $x_2 > 1 + m + \frac{m^2}{3}$。
构造函数 $h_1(m) = f(x_2) - f(1 + m + \frac{m^2}{3}) = 1 + \frac{m^2}{2} - (1 + m + \frac{m^2}{3}) + \ln(1 + m + \frac{m^2}{3})$。
求导：
$$h_1'(m) = \frac{m}{3} - 1 + \frac{1 + \frac{2m}{3}}{1 + m + \frac{m^2}{3}} = \frac{(m-3)(3+3m+m^2) + 9+6m}{3(3+3m+m^2)}= \frac{m^3}{3(m^2+3m+3)} > 0$$ 
因为 $h_1(0)=0$，故当 $m > 0$ 时 $h_1(m) > 0$，即 $f(1 + m + \frac{m^2}{3}) < f(x_2)$。由 $x > 1$ 时 $f(x)$ 的单调递增性，可得 $x_2 > 1 + m + \frac{m^2}{3}$。

证明 $x_1 < 1 - m + \frac{m^2}{3}$。
注意到 $1 - m + \frac{m^2}{3}$ 的判别式 $\Delta = 1 - \frac{4}{3} < 0$，故对任意 $m$，$1 - m + \frac{m^2}{3} \in (0,1)$ 恒成立。
构造 $h_2(m) = f(x_1) - f(1 - m + \frac{m^2}{3}) = 1 + \frac{m^2}{2} - (1 - m + \frac{m^2}{3}) + \ln(1 - m + \frac{m^2}{3})$。
求导得：
$$h_2'(m) = \frac{m}{3} + 1 + \frac{-1 + \frac{2m}{3}}{1 - m + \frac{m^2}{3}} = \frac{(m+3)(3-3m+m^2) - 3+2m}{3(3-3m+m^2)}= \frac{m^3}{3(m^2-3m+3)} > 0$$
同理由 $h_2(0)=0$ 可知 $h_2(m) > 0$，即 $f(1 - m + \frac{m^2}{3}) < f(x_1)$。由 $x \in (0,1)$ 时 $f(x)$ 的单调递减性，得 $x_1 < 1 - m + \frac{m^2}{3}$。
将上述两不等式作差，得到下界：$x_2 - x_1 > 2m$。

证明 $x_4 < 1 + m + \frac{m^2}{6}$。
构造函数 
\begin{align*}
h_3(m) &= g(1 + m + \frac{m^2}{6}) - g(x_4) = (1 + m + \frac{m^2}{6})\ln(1 + m + \frac{m^2}{6}) - m - \frac{m^2}{6} + 1 - (1 + \frac{m^2}{2})\\
&= (1 + m + \frac{m^2}{6})\left[ \ln(1 + m + \frac{m^2}{6}) - \frac{m + \frac{2m^2}{3}}{1 + m + \frac{m^2}{6}} \right]=(1 + m + \frac{m^2}{6})B(m)\\
B'(m) &= \frac{(1+\frac{m}{3})(1+m+\frac{m^2}{6}) - (1+\frac{4m}{3})(1+m+\frac{m^2}{6}) + (m+\frac{2m^2}{3})(1+\frac{m}{3})}{(1+m+\frac{m^2}{6})^2}\\
&=\frac{-m(1+m+\frac{m^2}{6}) + m + m^2 + \frac{2m^3}{9}}{(1+m+\frac{m^2}{6})^2} = \frac{\frac{m^3}{18}}{(1+m+\frac{m^2}{6})^2} > 0
\end{align*}。
又因 $1 + m + \frac{m^2}{6} > 0$，所以 $h_3(m) > 0$。即 $g(1 + m + \frac{m^2}{6}) > g(x_4)$，由 $x > 1$ 时 $g(x)$ 的单调递增性，得 $x_4 < 1 + m + \frac{m^2}{6}$。

再证 $x_3 > 1 - m + \frac{m^2}{6}$。
设 $u(m) = 1 - m + \frac{m^2}{6}$。
当 $u(m) \le 0$ 时（即 $m \ge 3 - \sqrt{3}$），由定义域知 $x_3 > 0$，故 $x_3 > u(m)$。
当 $u(m) > 0$ 时（即 $0 < m < 3 - \sqrt{3}$），构造
$$h_4(m) = (1 - m + \frac{m^2}{6})\left[ \ln(1 - m + \frac{m^2}{6}) - \frac{-m + \frac{2m^2}{3}}{1 - m + \frac{m^2}{6}} \right]$$
同理可证明 $h_4(m) > 0$。即 $g(u(m)) > g(x_3)$，由 $x \in (0,1)$ 时 $g(x)$ 的单调递减性，得 $x_3 > 1 - m + \frac{m^2}{6}$。
将上述两不等式作差，得到上界：$x_4 - x_3 < 2m$。原命题得证。
\end{solution}

\begin{example}{证明$x_3+x_4>2$}{}
已知 $f(x)=(x-1)\ln x,f(x_1)=f(x_2),x_1 \neq x_2$，过$(x_1,f(x_1))$与$(x_2,f(x_2))$两点分别作切线交$x$轴于$(x_3,0),(x_4,0)$，证明$x_3+x_4>2$.
\end{example}
\begin{solution}
即证$\displaystyle\frac{(x_1-1)^2}{x_1\ln x_1 + x_1 - 1} +\frac{(x_2-1)^2}{x_2\ln x_2 + x_2 - 1}> 0$.
令
\[
\Psi(x)=\left(\frac{(x-1)^2}{x\ln x+x-1}\right)^2-\frac{60(x-1)\ln x+12\bigl((x-1)\ln x\bigr)^2}{240+43(x-1)\ln x}.
\]
再作代换 $x=e^t$，得
$$ \Psi(e^t)=\left(\frac{(e^t-1)^2}{t e^t+e^t-1}\right)^2-\frac{12t(e^t-1)(5+t(e^t-1))}{240+43t(e^t-1)}=\frac{N(t)}{D(t)}. $$
其中
$$ D(t)=(t e^t+e^t-1)^2\bigl[240+43t(e^t-1)\bigr]>0. $$
于是只需判断 $N(t)$ 的符号。提出 $-(e^t-1)$ 后，
$$ N(t)=-(e^t-1)M(t), $$
其中
$$ M(t)=12t(5+t(e^t-1))(t e^t+e^t-1)^2-(e^t-1)^3\bigl(240+43t(e^t-1)\bigr). $$
由于 $-(e^t-1)$ 与 $-t$ 同号，只要证 $M(t)<0$ 对一切 $t\ne0$ 成立即可。

把 $M(t)$ 按 $e^t$ 的幂展开：
$$ M(t)=P_0(t)+P_1(t)e^t+P_2(t)e^{2t}+P_3(t)e^{3t}+P_4(t)e^{4t}, $$
其中
\begin{align*}
 P_0(t) &= -12t^2 + 17t + 240 \\
 P_1(t) &= 24t^3 - 84t^2 + 52t - 720 \\
 P_2(t) &= -12t^4 + 12t^3 + 84t^2 - 198t + 720 \\
 P_3(t) &= 12t^4 + 24t^3 + 12t^2 + 172t - 240 \\
 P_4(t) &= -43t
\end{align*}

记微分算子 $D=\frac{\mathrm{d}}{\mathrm{d}x}$。对任意多项式 $P(x)$ 和常数 $\alpha$，都有
\[
D[P(x)e^{\alpha x}] = e^{\alpha x}(D+\alpha)P(x).
\]
反复使用便有
\[
D^m[P(x)e^{\alpha x}] = e^{\alpha x}(D+\alpha)^m P(x).
\]

另用如下事实：若 $g^{(n)}(x)<0$ 且 $g(0),g'(0),\dots,g^{(n-1)}(0)\le0$，则由牛顿--莱布尼茨公式逐次积分可得 $g(x)<0\ (x>0)$。

现在逐次消去低次项。令 $F_0(x)=M(x)$，
\[
F_1(x)=F_0^{(3)}(x)e^{-x},\qquad
F_2(x)=F_1^{(4)}(x)e^{-x},\qquad
F_3(x)=F_2^{(5)}(x)e^{-x},\qquad
F_4(x)=F_3^{(5)}(x)e^{-x}.
\]
这样一共做了 $3+4+5+5=17$ 次求导。此时只剩 $P_4(x)e^{4x}$，等价于把
\[
(D+1)^5 (D+2)^5 (D+3)^4 (D+4)^3
\]
作用到 $P_4(x)=-43x$ 上。因为它只是一阶多项式，保留常数项和一次导数项即可。计算得
\[
\mathcal{L}(0)=1^5\cdot2^5\cdot3^4\cdot4^3=165888,\qquad
\mathcal{L}'(0)=\mathcal{L}(0)\left(\frac51+\frac52+\frac43+\frac34\right)=1589760,
\]
于是
$$ F_4(x) = \mathcal{L}(0)(-43x) + \mathcal{L}'(0)(-43) = -7133184x - 68359680 $$
从而 $F_4(x)<0\ (x>0)$。

再看初值。由 $x=0$ 处的泰勒展开
\[
M(x) = -\frac{2}{3}x^8 - \frac{422}{315}x^9 + \mathcal{O}(x^{10})
\]
可知 $M$ 的前 $7$ 阶导数在 $0$ 处都为 $0$，而
$$ M^{(8)}(0)=8!\left(-\frac23\right)=-26880<0. $$
对应到上面的函数序列，$17$ 个初值依次为
\begin{align*} 0, 0, 0, 0, 0, 0, 0, 0, -26880, -271104, -1435680, \\-5466528, -17108064, -30178368, -42864768, -53035776, -61102080  \end{align*}
于是由逐次积分可得 $M(x)<0\ (x>0)$。

当 $x<0$ 时，令 $y=-x>0$，再记 $K(y)=M(-y)e^{4y}$。因为 $e^{4y}>0$，只需证 $K(y)<0$。展开得
$$ K(y) = Q_0(y) + Q_1(y)e^y + Q_2(y)e^{2y} + Q_3(y)e^{3y} + Q_4(y)e^{4y} $$
其中
\begin{align*}
 Q_0(y) &= P_4(-y) = 43y \quad \text{（最高 } 1 \text{ 次，需连求 } 2 \text{ 次导数消去）} \\
 Q_1(y) &= P_3(-y) = 12y^4 - 24y^3 + 12y^2 - 172y - 240 \quad \text{（最高 } 4 \text{ 次，需连求 } 5 \text{ 次导数消去）} \\
 Q_2(y) &= P_2(-y) = -12y^4 - 12y^3 + 84y^2 + 198y + 720 \quad \text{（最高 } 4 \text{ 次，需连求 } 5 \text{ 次导数消去）} \\
 Q_3(y) &= P_1(-y) = -24y^3 - 84y^2 - 52y - 720 \quad \text{（最高 } 3 \text{ 次，需连求 } 4 \text{ 次导数消去）} \\
 Q_4(y) &= P_0(-y) = -12y^2 - 17y + 240 \quad \text{（最高 } 2 \text{ 次）}
\end{align*}
处理完全类似：依次作 $2,5,5,4$ 次求导，并在每一步乘上 $e^{-x}$。最后只剩
\[
(D+1)^4 (D+2)^5 (D+3)^5 (D+4)^2
\]
作用到 $Q_4(y)$ 上。因为 $Q_4(y)$ 是二次多项式，只需保留到二阶导数项，整理后得
$$ G_4(y) = -1492992y^2 - 27993600y - 91756800 $$
从而 $G_4(y)<0\ (y>0)$。

而 $K(y)$ 在 $0$ 附近的展开首项仍是 $-\frac23y^8$，所以对应的 $16$ 个初值依次为
\begin{align*} &0, 0, 0, 0, 0, 0, 0, 0, -26880, -239616, -1176864,\\
     &-4403904, -14176320, -27350784, -45114624, -66790656 \end{align*}
于是同样可以逐次积分推出 $K(y)<0\ (y>0)$，也就是 $M(x)<0\ (x<0)$。

综上，$M(x)<0$ 对一切 $x\ne0$ 都成立。回到
$$ \Psi(e^x)=\frac{-(e^x-1)M(x)}{D(x)}, $$
因为 $D(x)>0$ 且 $M(x)<0$，只需看 $e^x-1$ 的符号。

故 $x>0$ 时 $\Psi(e^x)>0$，$x<0$ 时 $\Psi(e^x)<0$，原命题得证。
\end{solution}

\begin{example}{交点比较}{}
    已知$x_1-\ln x_1=x_2-\ln x_2,x_1\neq x_2$，求证\[(1)~~\frac{x_{1}\ln x_{1}}{x_{1}-1}+\frac{x_{2}\ln x_{2}}{x_{2}-1}>2~~~~~(2)~~ (x_1+\sqrt{x_1}+1)(x_2+\sqrt{x_2}+1)>9\]
\end{example}
\begin{solution}
设 $0 < x_1 < 1 < x_2$，由 $x_1 - \ln x_1 = x_2 - \ln x_2$ 移项得 $x_2 - x_1 = \ln x_2 - \ln x_1 = \ln\left(\frac{x_2}{x_1}\right)$。
令 $\frac{x_2}{x_1} = e^{2t}\ (t>0)$，则 $x_2 - x_1 = 2t$。将 $x_2 = x_1 e^{2t}$ 代回 $x_2 - x_1 = 2t$ 可得 $x_1(e^{2t}-1) = 2t$，解得：
\[ x_1 = \frac{2t}{e^{2t}-1} = \frac{2t e^{-t}}{e^t - e^{-t}} = \frac{t e^{-t}}{\sinh t},\ln x_1 = \ln\left(\frac{t}{\sinh t}\right) - t, x_2 = x_1 e^{2t} = \frac{t e^t}{\sinh t} ,\ln x_2 = \ln\left(\frac{t}{\sinh t}\right) + t\]
将上述关系代入待证不等式（1）的左端，并提取对数部分的公因式得：
\[ E = \ln\left(\frac{t}{\sinh t}\right) \left( \frac{x_1}{x_1 - 1} + \frac{x_2}{x_2 - 1} \right) + t \left( \frac{x_2}{x_2 - 1} - \frac{x_1}{x_1 - 1} \right) \]
对括号内的分式进行通分，其公共分母为 $A = (x_1 - 1)(x_2 - 1) = x_1 x_2 - (x_1 + x_2) + 1$。
代入 $x_1 x_2 = \frac{t^2}{\sinh^2 t}$ 与 $x_1 + x_2 = \frac{t(e^t+e^{-t})}{\sinh t} = \frac{2t \cosh t}{\sinh t}$ 进行计算，可得：
\[ A = \frac{t^2}{\sinh^2 t} - \frac{2t \cosh t}{\sinh t} + 1 = \frac{t^2 - 2t \sinh t \cosh t + \sinh^2 t}{\sinh^2 t} = \frac{t^2 - t \sinh 2t + \sinh^2 t}{\sinh^2 t} \]
记该式分子为 $f(t) = t^2 - t \sinh 2t + \sinh^2 t$，对其求导得 $f'(t) = 2t - (\sinh 2t + 2t \cosh 2t) + 2\sinh t \cosh t = 2t(1 - \cosh 2t)$。由于 $t > 0$ 时 $\cosh 2t > 1$，故 $f'(t) < 0$ 恒成立。结合 $f(0) = 0$ 可知对于 $t > 0$ 分子恒为负，从而恒有 $A < 0$。再化简 $E$ 中的分子部分：
\[ \frac{x_1}{x_1 - 1} + \frac{x_2}{x_2 - 1} = \frac{2x_1 x_2 - (x_1 + x_2)}{A} = \frac{\frac{2t^2}{\sinh^2 t} - \frac{2t \cosh t}{\sinh t}}{A} = \frac{2(t^2 - t \sinh t \cosh t)}{A \sinh^2 t} \]
\[ \frac{x_2}{x_2 - 1} - \frac{x_1}{x_1 - 1} = \frac{x_2(x_1 - 1) - x_1(x_2 - 1)}{A} = \frac{x_1 - x_2}{A} = \frac{\frac{t e^{-t}}{\sinh t} - \frac{t e^t}{\sinh t}}{A} = \frac{-2t}{A} \]
将以上结果代回 $E$ 中并通分整理，可得：
\[ E = \frac{2 \ln \left( \frac{t}{\sinh t} \right) (t^2 - t \sinh t \cosh t) - 2t^2 \sinh^2 t}{A \sinh^2 t} \]
由 $A \sinh^2 t < 0$，要证 $E > 2$等价于证其分子小于 $2A \sinh^2 t = 2(t^2 - 2t \sinh t \cosh t + \sinh^2 t)$。
将不等式两边同除以 $2$ 并移项整理，化为：
\[ \ln \left( \frac{t}{\sinh t} \right) (t^2 - t \sinh t \cosh t) < t^2\sinh^2 t + t^2 - 2t \sinh t \cosh t + \sinh^2 t \]
提取右端前两项的公因式得到 $t^2(\sinh^2 t + 1) = t^2\cosh^2 t$，从而右端可完全平方化为 $(t \cosh t - \sinh t)^2$。
由于 $t>0$ 时 $\sinh 2t > 2t$，故左端因式 $t^2 - t \sinh t \cosh t = t(t - \sinh t \cosh t) < 0$。将不等式两边同除以该负因式并改变不等号方向，同时利用 $\ln \left( \frac{t}{\sinh t} \right) = - \ln \left( \frac{\sinh t}{t} \right)$ 抵消负号，问题转化为证明：
\[ K(t) = \ln \left( \frac{\sinh t}{t} \right) + \frac{(t \cosh t - \sinh t)^2}{t(t - \sinh t \cosh t)} < 0 \quad (t > 0) \]

为了求导化简，记 $g(t) = t \cosh t - \sinh t$，其导数 $g'(t) = t \sinh t$；
记 $H(t) = \sinh t \cosh t - t > 0$，其导数 $H'(t) = \cosh 2t - 1 = 2 \sinh^2 t$。
原函数可简记为 $K(t) = \ln(\sinh t) - \ln t - \frac{g(t)^2}{t H(t)}$。对数部分的导数为 $\frac{\cosh t}{\sinh t} - \frac{1}{t} = \frac{t\cosh t - \sinh t}{t\sinh t} = \frac{g(t)}{t \sinh t}$，可得：
\[ K'(t) = \frac{g(t)}{t \sinh t} - \frac{2g(t)g'(t)t H(t) - g(t)^2(H(t) + t H'(t))}{t^2 H(t)^2} \]
通分并提取严格为正的公因式 $\frac{g(t)}{t^2 H(t)^2 \sinh t}$，中括号内的剩余多项式为：
\begin{align*} M(t)& = t H(t)^2 - 2t \sinh t g'(t) H(t) + g(t) \sinh t (H(t) + t H'(t)) \\
  &  = t H(t)^2 - 2t^2 \sinh^2 t H(t) + g(t) \sinh t (H(t) + 2t \sinh^2 t) \end{align*}
代入 $g(t), H(t)$ 的表达式，展开后利用双曲函数的倍角公式合并同类项（如 $\cosh 2t = 1 + 2\sinh^2 t$ 等），可得
\[ M(t) = t^3 \cosh 2t - \frac{3}{2} t^2 \sinh 2t + \frac{3}{2} t (\cosh 2t - 1) - \frac{1}{4} \sinh 2t (\cosh 2t - 1) \]

令 $u = 2t > 0$，为消除分数令 $P(u) = 8M(u/2) = u^3 \cosh u - 3u^2 \sinh u + 6u \cosh u - 6u - \sinh 2u + 2 \sinh u$。
利用 $\displaystyle\cosh u = \sum_{n=0}^\infty \frac{u^{2n}}{(2n)!},\sinh u = \sum_{n=0}^\infty \frac{u^{2n+1}}{(2n+1)!}$ 将其展开为泰勒级数 $\displaystyle P(u) = \sum_{n=0}^\infty c_n u^{2n+1}$。
对于 $n \ge 1$，提取各展开式中 $u^{2n+1}$ 的系数并提取公共分母 $(2n+1)!$：
\begin{align*} c_n &= \frac{1}{(2n-2)!} - \frac{3}{(2n-1)!} + \frac{6}{(2n)!} - \frac{2^{2n+1}}{(2n+1)!} + \frac{2}{(2n+1)!} \\&= \frac{2n(2n-1)(2n+1) - 3(2n)(2n+1) + 6(2n+1) - 2^{2n+1} + 2}{(2n+1)!} \end{align*}
化简分子，前三项多项式之和为 $(2n+1)(4n^2 - 8n + 6) = 8n^3 - 12n^2 + 4n + 6$，故合并后通项系数为 $c_n = \frac{8n^3 - 12n^2 + 4n + 8 - 2^{2n+1}}{(2n+1)!}$。
由基础低次项相抵消易知 $c_0=c_1=c_2=c_3=0$。
当 $n \ge 4$ 时，利用数学归纳法证明指数项占优，即 $2^{2n+1} > 8n^3 - 12n^2 + 4n + 8$：
基础情形：当 $n=4$ 时，$2^9 = 512 > 8(64) - 12(16) + 4(4) + 8 = 344$ 成立；
递推过程：假设 $n=k \ge 4$ 时不等式成立，则当 $n=k+1$ 时，左侧变为 $4 \times 2^{2k+1} > 4(8k^3 - 12k^2 + 4k + 8)$。只需证其大于目标值 $8(k+1)^3 - 12(k+1)^2 + 4(k+1) + 8$。两式作差得 $24k^3 - 60k^2 + 12k + 24 = 12(k^2(2k - 5) + k + 2)$，因 $k \ge 4$ 故该差值显然大于 $0$，归纳法得证。
故当 $n \ge 4$ 时恒有 $c_n < 0$。因此对于任意 $u>0$ 有 $P(u) < 0$，进而 $M(t) < 0$，即 $K'(t) < 0$。结合 $\lim_{t \to 0^+} K(t) = 0$ 可知对任意 $t > 0$ 均有 $K(t) < 0$。得证。

（2）记 $a = \sqrt{x_1}, b = \sqrt{x_2}$，即证：
\[ (a^2 + a + 1)(b^2 + b + 1) = a^2 b^2 + ab(a+b) + a^2 + b^2 + ab + a + b + 1 > 9 \]
令 $\theta = t/2 > 0$，承接前面的参数方程直接开方可得：
\[ a = e^{-\theta} \sqrt{ \frac{2\theta}{\sinh 2\theta} }, \quad b = e^{\theta} \sqrt{ \frac{2\theta}{\sinh 2\theta} } \]
令两根之积 $P = ab = \frac{2\theta}{\sinh 2\theta}$，两根之和 $S = a + b = \sqrt{\frac{2\theta}{\sinh 2\theta}} (e^{-\theta} + e^\theta) = \sqrt{\frac{2\theta}{2\sinh \theta \cosh \theta}} \cdot 2\cosh \theta = 2 \sqrt{\theta \coth \theta}$。
因 $\theta > 0$ 时 $\sinh 2\theta > 2\theta$，易知 $P \in (0, 1)$；因 $\theta \coth \theta > 1$ 恒成立，故 $S > 2$。
利用 $a^2 + b^2 = S^2 - 2P$ 将展开式化简重组，原命题等价于证明：
\[ P^2 + PS + S^2 - 2P + P + S + 1 > 9 \iff F(P, S) = P^2 + PS + S^2 - P + S - 8 > 0 \]

为严谨证明该式，按 $\theta$ 的范围分为两段进行放缩：

当 $0 < \theta \le 1.3$ 时，利用泰勒展开 $\sinh u = u + \frac{u^3}{6} + \frac{u^5}{120} + \dots = \sum_{n=0}^\infty \frac{u^{2n+1}}{(2n+1)!}$ 可得：
\[ \left(1 - \frac{u^2}{6}\right)\sinh u = u + \sum_{n=1}^\infty \left( \frac{1}{(2n+1)!} - \frac{1}{6(2n-1)!} \right) u^{2n+1} = u + \sum_{n=1}^\infty \frac{1 - 2n(2n+1)/6}{(2n+1)!} u^{2n+1} \]
当 $n=1$ 时该项系数为 $0$，当 $n \ge 2$ 时分子 $1 - n(2n+1)/3 \le 1 - 10/3 < 0$，故对任意 $u>0$ 恒有 $\left(1 - \frac{u^2}{6}\right)\sinh u < u$。令 $u = 2\theta$ 即可得严密的局部下界 $P = \frac{2\theta}{\sinh 2\theta} > 1 - \frac{2}{3}\theta^2 \triangleq P_0$。
利用 Mittag-Leffler 展开式 $\theta \coth \theta = 1 + 2\sum_{n=1}^\infty \frac{\theta^2}{\theta^2 + n^2 \pi^2}$，结合基本代数不等式 $\frac{1}{A+B} > \frac{1}{A} - \frac{B}{A^2}$ 进行放缩，并代入已知级数 $\sum_{n=1}^\infty \frac{1}{n^2} = \frac{\pi^2}{6}, \sum_{n=1}^\infty \frac{1}{n^4} = \frac{\pi^4}{90}$ 可得：
\[ \theta \coth \theta > 1 + 2\sum_{n=1}^\infty \left( \frac{\theta^2}{n^2\pi^2} - \frac{\theta^4}{n^4\pi^4} \right) = 1 + \frac{2\theta^2}{\pi^2} \cdot \frac{\pi^2}{6} - \frac{2\theta^4}{\pi^4} \cdot \frac{\pi^4}{90} = 1 + \frac{\theta^2}{3} - \frac{\theta^4}{45} \]
将多项式 $\left(1 + \frac{\theta^2}{6} - \frac{\theta^4}{40}\right)^2 = 1 + \frac{\theta^2}{3} - \frac{\theta^4}{45} - \frac{\theta^6}{120} + \frac{\theta^8}{1600}$ 与前述下界作差，差值为 $\frac{\theta^6}{120} - \frac{\theta^8}{1600} = \frac{\theta^6}{1600}\left(\frac{40}{3} - \theta^2\right)$。
在 $\theta \le 1.3$ 时 $\theta^2 \le 1.69 < \frac{40}{3}$，该差值严格大于 $0$，这说明放缩所得的下界严格大于该多项式。开方即得 $S > 2\left(1 + \frac{\theta^2}{6} - \frac{\theta^4}{40}\right) \triangleq S_0$。

为了说明 $P$ 和 $S$ 放缩后目标函数依然变小，利用代数变形分解目标函数的差值，将交叉项的差改写为 $PS - P_0 S_0 = S(P - P_0) + P_0(S - S_0)$ 并提取公因式：
\begin{align*}
F(P, S) - F(P_0, S_0) &= (P^2 - P_0^2) + (PS - P_0 S_0) + (S^2 - S_0^2) - (P - P_0) + (S - S_0) \\
&= (P - P_0)(P + P_0 + S - 1) + (S - S_0)(S + S_0 + P_0 + 1)
\end{align*}
由于 $\theta \le 1.3$ 时 $P_0 \ge 1 - \frac{2}{3}(1.3)^2 > -0.13$，结合 $P>0, S>2, S_0>2$，上述两项的后置括号均严格大于 $0$（例如 $P+P_0+S-1 > 0 - 0.13 + 2 - 1 > 0$），这就逻辑严密地确保了向下放缩的合法性，即 $F(P, S) > F(P_0, S_0)$。
将 $P_0, S_0$ 代入多项式展开并按次数整理如下：
$P_0^2 = 1 - \frac{4}{3}\theta^2 + \frac{4}{9}\theta^4$
$P_0 S_0 = \left(1 - \frac{2}{3}\theta^2\right)\left(2 + \frac{1}{3}\theta^2 - \frac{1}{20}\theta^4\right) = 2 - \theta^2 - \frac{49}{180}\theta^4 + \frac{1}{30}\theta^6$
$S_0^2 = \left(2 + \frac{1}{3}\theta^2 - \frac{1}{20}\theta^4\right)^2 = 4 + \frac{4}{3}\theta^2 - \frac{4}{45}\theta^4 - \frac{1}{30}\theta^6 + \frac{1}{400}\theta^8$
$-P_0 + S_0 - 8 = -\left(1 - \frac{2}{3}\theta^2\right) + \left(2 + \frac{1}{3}\theta^2 - \frac{1}{20}\theta^4\right) - 8 = -7 + \theta^2 - \frac{1}{20}\theta^4$
将上述四式相加后，可见常数项、\(\theta^2\) 项与 \(\theta^6\) 项都相互抵消，而 \(\theta^4\) 项系数为
\[
\frac{4}{9} - \frac{49}{180} - \frac{4}{45} - \frac{1}{20} = \frac{1}{30}.
\]
因此只剩下高阶正项：
\[ F(P_0, S_0) = \frac{1}{30}\theta^4 + \frac{1}{400}\theta^8 > 0 \]
故当 $0 < \theta \le 1.3$ 时 $F(P, S) > 0$ 严格成立。

当 $\theta > 1.3$ 时，函数 $x(\theta) = \theta \coth \theta$ 的导数 $x'(\theta) = \coth \theta - \theta \operatorname{csch}^2 \theta = \frac{\sinh 2\theta - 2\theta}{2\sinh^2\theta} > 0$，故该函数在 $(0, +\infty)$ 上单调递增。
此时 $x(\theta) > 1.3 \coth 1.3 \approx 1.5086$，从而 $S = 2\sqrt{x(\theta)} > 2\sqrt{1.5086} \approx 2.4565$。
将目标函数重构为 $F(P, S) = S^2 + S(P+1) + (P^2 - P) - 8$。
因 $P \in (0, 1)$，必有 $P+1 > 1$，且二次函数 $P^2 - P = (P-0.5)^2 - 0.25$ 恒大于其极小值 $-0.25$。代入全局界限进行放缩：
\[ F(P, S) > 2.4565^2 + 2.4565 \times 1 - 0.25 - 8 \approx 6.034 + 2.456 - 8.25 = 0.24 > 0 \]
故当 $\theta > 1.3$ 时 $F(P, S) > 0$ 同样成立。综上对于所有满足条件的 $x_1, x_2$不等式均恒成立。

\end{solution}

\begin{example}{黎曼杯T18加强}{}
     $f(x) = \e^x-x\ln x - kx - 1$．若函数 $f(x)$ 有两个零点，将其分别记为 $x_1,x_2$．\\（1）试求 $k$ 的取值范围；\\（2）证明：$\displaystyle k>x_1+\ln x_2+\frac{\sqrt2}{2}$\\
     （3）证明：$(\e-2)x_1+x_2-\ln(x_1x_2)\geqslant k-1+\ln(k+1)$
\end{example}
\begin{solution}
\textbf{(1)} 令 $g(x)=\dfrac{\e^x}{x}-\ln x-\dfrac{1}{x}$，则原方程等价于 $g(x)=k$．求导得
\[
g'(x)=\frac{(x-1)(\e^x-1)}{x^2}.
\]
由于 $x>0$ 时 $\e^x-1>0$，故 $g'(x)$ 的符号由 $x-1$ 决定：当 $0<x<1$ 时 $g'(x)<0$，当 $x>1$ 时 $g'(x)>0$．因此 $g(x)$ 在 $(0,1)$ 上单调递减，在 $(1,+\infty)$ 上单调递增，在 $x=1$ 处取得最小值 $g(1)=\e-1$．又 $\lim\limits_{x\to0^+}g(x)=+\infty$，$\lim\limits_{x\to+\infty}g(x)=+\infty$，故对任意 $k>\e-1$，方程 $g(x)=k$ 恰有两个不同的实根；而当 $k\le \e-1$ 时，方程至多有一个实根．因此 $k$ 的取值范围是 $(\e-1,+\infty)$．

\textbf{(2)} 设 $f(x)$ 的两个零点为 $\alpha,\beta$，且 $0<\alpha<1<\beta$．由 $f(\alpha)=0$ 得
\[
k=\frac{\e^\alpha}{\alpha}-\ln\alpha-\frac1\alpha.
\]
要证 $k>\alpha+\ln\beta+\dfrac{\sqrt2}{2}$，即证
\[
\frac{\e^\alpha}{\alpha}-\ln\alpha-\frac1\alpha>\alpha+\ln\beta+\frac{\sqrt2}{2},
\]
整理得
\[
\ln(\alpha\beta)<\frac{\e^\alpha}{\alpha}-\alpha-\frac1\alpha-\frac{\sqrt2}{2}.
\]
构造函数 $G(x)=\dfrac{\e^x}{x}-x-\dfrac1x-\dfrac{\sqrt2}{2}\quad(x>0)$，求导得
\[
G'(x)=\frac{(x-1)(\e^x-x-1)}{x^2}.
\]
由 $\e^x\ge x+1$（当且仅当 $x=0$ 取等），知当 $x>0$ 时 $\e^x-x-1>0$，故 $G'(x)$ 的符号由 $x-1$ 决定：$0<x<1$ 时 $G'(x)<0$，$x>1$ 时 $G'(x)>0$．因此 $G(x)$ 在 $(0,1)$ 上单调递减，在 $(1,+\infty)$ 上单调递增，最小值 $G(1)=\e-2-\dfrac{\sqrt2}{2}>0$，从而 $G(x)>0$ 恒成立．这样原不等式只剩下
\[
\ln(\alpha\beta)<0,
\]
也就是 $\alpha\beta<1$．

再看 $\alpha\beta$。考虑函数 $F(x)=I(x)-I\!\left(\dfrac1x\right)$，其中 $I(x)=\e^x-x\ln x-kx-1$．对 $x\ge1$ 求导整理得
\[
F'(x)=\frac{(x-1)\bigl(\e^x-x\e^{1/x}+x-1\bigr)}{x^2}.
\]
令 $m(x)=\e^x-x\e^{1/x}+x-1$，则
\[
m'(x)=\e^x-\e^{1/x}+\frac{\e^{1/x}}{x^2}+1>0\quad(x>1),
\]
且 $m(1)=0$，故 $m(x)>0$ 对 $x>1$ 恒成立．因此 $F'(x)>0$（$x>1$），$F(x)$ 在 $[1,+\infty)$ 上单调递增，且 $F(1)=0$，从而 $F(x)\ge0$，即 $I(x)\ge I(1/x)$ 对 $x\ge1$ 成立．取 $x=\beta>1$，得 $I(\beta)\ge I(1/\beta)$．由 $I(\beta)=0$ 知 $I(1/\beta)\le0$．

另一方面，由第(1)问定义的 $g(x)=\dfrac{\e^x}{x}-\ln x-\dfrac1x$ 在 $(0,1)$ 上单调递减，且 $g(\alpha)=k$，易知 $I(x)=x\bigl(g(x)-k\bigr)$．因为 $g(x)-k$ 在 $(0,1)$ 上严格递减且恰有一个零点 $x=\alpha$，所以当 $x\in(0,\alpha)$ 时 $g(x)-k>0$，$I(x)>0$；当 $x\in(\alpha,1)$ 时 $g(x)-k<0$，$I(x)<0$．于是由 $I(1/\beta)\le0$ 及 $1/\beta\in(0,1)$ 可得 $1/\beta\ge\alpha$，即 $\alpha\beta\le1$．若 $\alpha\beta=1$，则 $1/\beta=\alpha$，代入得 $I(\alpha)=I(\beta)=0$ 且 $\beta=1/\alpha$，但此时 $g(\alpha)=g(1/\alpha)$．容易验证 $g(x)\neq g(1/x)$ 对 $x\in(0,1)\cup(1,+\infty)$ 成立（例如由 $F(x)$ 的性质可推知），矛盾．因此 $\alpha\beta<1$．

综上，$\ln(\alpha\beta)<0$，从而原不等式得证．\\
（3）设函数 $g(x) = \frac{\e^x - 1}{x} - \ln x$，且 $k > \e - 1$．设两根为 $0 < x_1 < 1 < x_2$．由 $g(x_2) = k$ 得
\[
\frac{\e^{x_2} - 1}{x_2} - \ln x_2 = k,
\]
即
\[
\e^{x_2} = x_2\left(k + \ln x_2 + \frac{1}{x_2}\right).
\]
两边取对数，
\[
x_2 - \ln x_2 = \ln\left(k + \ln x_2 + \frac{1}{x_2}\right) > \ln(k+1).
\]
再由 $g(x_1)=k$，只要推出
\[
(\e-2)x_1 - \ln x_1 > k - 1
\]
就够了；它等价于
\[
(\e-2)x_1^2 + x_1 + 1 - \e^{x_1} > 0.
\]

设
\[
h(x) = (\e-2)x^2 + x + 1 - \e^x\qquad (0 \leqslant x \leqslant 1).
\]
有
\[
h'(x) = 2(\e-2)x + 1 - \e^x,\qquad h''(x) = 2\e - 4 - \e^x.
\]
因为 \(e^x\) 在 \([0,1]\) 上递增，所以 \(h''\) 在 \([0,1]\) 上递减。又
\[
h''(0)=2\e-5>0,\qquad h''(1)=\e-4<0,
\]
故存在唯一的 \(x_0\in(0,1)\) 使 \(h''(x_0)=0\)。于是 \(h'\) 先增后减。再结合
\[
h'(0)=0,\qquad h'(1)=\e-3<0,
\]
可知 \(h'\) 在 \((0,1)\) 内恰有一个零点 \(t_0\)，从而 \(h\) 先增后减。

最后看端点：
\[
h(0)=0,\qquad h(1)=(\e-2)+1+1-\e=0.
\]
因此 \(h(x)>0\) 对一切 \(x\in(0,1)\) 都成立，特别是
\[
h(x_1)>0.
\]
也就是
\[
(\e-2)x_1^2 + x_1 + 1 - \e^{x_1} > 0.
\]

于是原题中的不等式
\[
(\e-2)x_1+x_2-\ln(x_1x_2)\geqslant k-1+\ln(k+1)
\]
成立．
\end{solution}

\begin{example}{证明$\frac{9(x-y)^4}{8(x-1)^2(y-1)^2} > (x-y)^2 + 18$}{}
已知$x-\ln x=y-\ln y$，证明不等式 \[\frac{9(x-y)^4}{8(x-1)^2(y-1)^2} > (x-y)^2 + 18\]
\end{example}
\begin{solution}
由 $g(u)=u-\ln u$ 的单调性知，方程 $g(u)=m>1$ 有两个正根，记为 $x>1>y>0$。

令 $t = \frac{x}{y}>1$。由 $\ln x - \ln y = x - y$ 得 $x - y = \ln t$，再结合 $x = yt$，可得
$$ x = \frac{t\ln t}{t-1}, \quad y = \frac{\ln t}{t-1} $$
再令 $v = \frac{\ln t}{2}>0$，则 $x - y = 2v,\ t = e^{2v}$，并且
$$ x - 1 = \frac{v e^v - \sinh v}{\sinh v}, \quad y - 1 = \frac{v e^{-v} - \sinh v}{\sinh v} $$
从而
$$ (x-1)(y-1) = \frac{v^2 - v\sinh 2v + \sinh^2 v}{\sinh^2 v} $$
记
\[
D(v) = v\sinh 2v - \sinh^2 v - v^2.
\]
由 $(x-1)(y-1) < 0$ 得 $D(v) > 0$。原不等式化为
$$ L(v) = \frac{9(x-y)^4}{8(x-1)^2(y-1)^2} = \frac{9(2v)^4}{8 \left( -\frac{D(v)}{\sinh^2 v} \right)^2} = \frac{18 v^4 \sinh^4 v}{D(v)^2} $$
$$ R(v) = (x-y)^2 + 18 = 4v^2 + 18 $$
因此只需证 $L(v) > R(v)$。

当 $v \to 0$ 时，
\[
L(v) = 18 + 4v^2 + \frac{2}{15}v^4 + \mathcal{O}(v^6).
\]
当 $v \to +\infty$ 时，$\sinh v \sim \frac{e^v}{2}$，从而
\[
D(v) \sim \frac{2v-1}{4}e^{2v}.
\]
于是
\[
L(v) \sim \frac{18v^4(e^{4v}/16)}{((2v-1)/4)^2 e^{4v}} \sim \frac{18v^4}{4v^2} = 4.5v^2.
\]
因此取
\[
P(v) = \frac{A v^4}{B v^2 + C},
\]
使它在 $v \to 0$ 与 $v \to +\infty$ 时分别匹配首项。由
\[
\frac{A}{C} = \frac{2}{15}, \qquad \frac{A}{B} = \frac{1}{2},
\]
取最简正整数解 $A=2, B=4, C=15$，得
\[
P(v) = \frac{2v^4}{4v^2 + 15}.
\]
于是
\[
M(v) = 4v^2 + 18 + \frac{2v^4}{4v^2 + 15}.
\]
换回 $x,y$ 后，对应的中间函数为
$$ f(x,y) = (x-y)^2 + 18 + \frac{(x-y)^4}{8(x-y)^2 + 120} $$
于是只需证 $L(v) > M(v) > R(v)$。由于
\[
\frac{2v^4}{4v^2 + 15}>0 \qquad (v>0),
\]
立得 $M(v) > R(v)$。

再证 $L(v) > M(v)$。将 $M(v)$ 通分，得
$$ M(v) = \frac{(4v^2+18)(4v^2+15) + 2v^4}{4v^2 + 15} = \frac{6(3v^4 + 22v^2 + 45)}{4v^2 + 15} $$
待证不等式等价于 $\frac{18 v^4 \sinh^4 v}{D(v)^2} > \frac{6(3v^4 + 22v^2 + 45)}{4v^2 + 15}$。由于对于任意 $v > 0$ 均有 $D(v)^2 > 0$ 且 $4v^2+15 > 0$，将该不等式两端交叉相乘并约去常数因子 $6$ 后，等价于证明判定函数 $\Delta(v) > 0$ 恒成立，其中：
$$ \Delta(v) = 3v^4 (4v^2+15) \sinh^4 v - (3v^4+22v^2+45) D(v)^2 $$
为分析 $\Delta(v)$ 的符号，利用倍角公式将双曲函数的幂次降阶。代入 $\sinh^4 v = \frac{1}{8}(\cosh 4v - 4\cosh 2v + 3)$，并将 $D(v)^2 = \left(\frac{1}{2}\cosh 2v - \frac{1}{2} - v\sinh 2v + v^2\right)^2$ 完全展开。代入 $\Delta(v)$ 并同乘 $8$ 消除常数分母，合并同类项后得到关于 $2v$ 与 $4v$ 的双曲函数线性组合：
$$ 8\Delta(v) = A_0(v) + A_2(v)\cosh 2v + B_2(v)\sinh 2v + A_4(v)\cosh 4v + B_4(v)\sinh 4v $$
其中各项多项式系数分别为：
\begin{align*}
A_0(v) &= -24v^8 - 104v^6 + 30v^4 + 474v^2 - 135 \\
A_2(v) &= -72v^6 - 344v^4 - 272v^2 + 180 \\
B_2(v) &= 48v^7 + 328v^5 + 544v^3 - 360v \\
A_4(v) &= -46v^4 - 202v^2 - 45 \\
B_4(v) &= 12v^5 + 88v^3 + 180v
\end{align*}

将判定函数 $8\Delta(v)$ 在 $v=0$ 处展开为麦克劳林幂级数 $\sum_{n=0}^{\infty} c_n v^{2n}$。由于多项式 $A_0(v)$ 的最高次项为 $v^8$，当 $n \ge 5$（即对应级数中 $v^{10}$ 及更高次项）时，展开式系数 $c_n$ 完全由四项双曲函数的泰勒级数决定。提取幂级数的一般通项可表示为：
$$ c_n = \frac{1}{(2n)!} \left[ P_2(n) 2^{2n-7} + P_4(n) 4^{2n-5} \right] \quad (n \ge 5) $$
其中伴随多项式 $P_2(n)$ 与 $P_4(n)$ 经代数计算为：
\begin{align*}
P_2(n) &= 6144 n^7 - 73728 n^6 + 379904 n^5 - 1014272 n^4 + 1451904 n^3 - 1019264 n^2 + 205824 n + 23040 \\
P_4(n) &= 384 n^5 - 4864 n^4 + 23456 n^3 - 79104 n^2 + 126432 n - 46080
\end{align*}
由构造知，对于 $n \le 4$，双曲展开的低阶项与 $A_0(v)$ 的对应项正好相消，因此 $c_0 = c_1 = c_2 = c_3 = c_4 = 0$。再代入 $n=5,6$ 验算：
当 $n=5$ 时，$P_2(5) 2^3 + P_4(5) 4^5 = 306708480 - 306708480 = 0$，即 $c_5 = 0$；
当 $n=6$ 时，$P_2(6) 2^5 + P_4(6) 4^7 = 6332350464 - 6332350464 = 0$，即 $c_6 = 0$。
这表明幂级数从 $n=7$（即 $v^{14}$ 阶）开始出现非零项。

再判断 $n \ge 7$ 时的系数符号。将 $P_4(n)$ 在 $n=8$ 处平移展开，可得：
$$ P_4(n) = 384(n-8)^5 + 10496(n-8)^4 + 113568(n-8)^3 + 582144(n-8)^2 + 1267168(n-8) + 572160 $$
因各项系数均严格为正，可知当 $n \ge 8$ 时 $P_4(n) > 0$ 恒成立。同理将 $P_2(n)$ 整理为关于 $(n-7)$ 的多项式展开，经计算其各项系数亦全为正数，由此推知当 $n \ge 7$ 时 $P_2(n) > 0$ 恒成立。由于两伴随多项式在 $n \ge 8$ 时均为正，通式通项的组合必满足 $c_n > 0$。单独考察级数首个可能的非零项 $n=7$，代入伴随多项式计算得：
\begin{align*} P_2(7) 2^7 + P_4(7) 4^9 &= 785125120 \times 128 + (-216320) \times 262144 \\
&= 100496015360 - 56706990080 = 43789025280 > 0 \end{align*}
表明 $c_7 > 0$ 成立。

因此，麦克劳林幂级数 $8\Delta(v) = \sum_{n=7}^{\infty} c_n v^{2n}$ 的非零系数都严格大于 $0$。由 $v > 0$ 便知 $\Delta(v) > 0$，从而 $L(v) > M(v)$。再与 $M(v) > R(v)$ 合并，即得
$$ \frac{9(x-y)^4}{8(x-1)^2(y-1)^2} > (x-y)^2 + 18 $$
在条件 $x-\ln x = y-\ln y = m > 1$ 下恒成立。
\end{solution}

\begin{example}{曲线面积最大值}{}
    曲线$\varGamma :x^2+\ln^2y=\left(\dfrac{\e}{\e+1}\right)^2$，$AB$为$\varGamma$的一条弦.\\
    （1）求$\varGamma$的最高点$P$，最低点$Q$的坐标.\\
    （2）若$AB$的斜率为0，求$S_{\triangle{PAB}}$的最大值.\\
    （3）求$S_{\triangle{PAB}}$的最大值.
\end{example}
\begin{solution}
记 $R = \frac{e}{e+1}$，则方程化为 $x^{2} + (\ln y)^{2} = R^{2}$。由此可得参数方程：
$$\begin{cases}
x = R \cos \theta, \\
y = e^{R \sin \theta},
\end{cases}
\quad \theta \in [0, 2\pi).$$
(1) 当$\sin\theta=1$时$y$ 取最大值$e^R$,此时 $x=0$;当$\sin\theta=-1$ 时 $y$ 取最小值$e^{-R}$，此时$x=0$。故最高点为$P(0,e^R)$,最低点为$Q(0,e^{-R})$,即
$P\left(0,e^{{\frac{e}{e+1}}}\right),\quad Q\left(0,e^{{-\frac{e}{e+1}}}\right).$\\
(2) 由于直线 $l$ 斜率为 $0$，即水平线，由曲线关于 $y$ 轴对称，可设 $A$ 在左半平面，$B$ 在右半平面，且 $A$ 与 $B$ 关于 $y$ 轴对称。设 $B$ 对应的参数为 $\theta$（$\cos\theta > 0$），则 $B(R\cos\theta, \mathrm{e}^{R\sin\theta})$，$A(-R\cos\theta, \mathrm{e}^{R\sin\theta})$。于是 $\triangle PAB$ 的底边长 $AB = 2R\cos\theta$，高为 $\mathrm{e}^{R} - \mathrm{e}^{R\sin\theta}$，面积
\[
S(\theta) = R\cos\theta\,(\mathrm{e}^{R} - \mathrm{e}^{R\sin\theta}),\quad \theta\in(-\tfrac{\pi}{2},\tfrac{\pi}{2}).
\]
令$t=R\sin\theta$，则
\[
S(t) = \sqrt{R^2-t^{2}}\,(\mathrm{e}^{R} - \mathrm{e}^{t}).
\]
为求最大值，考虑函数 $\varphi(t) = \ln S(t)$：
\begin{align*}
\varphi(t) &= \frac{1}{2}\ln(R^2 - t^2) + \ln(\mathrm{e}^{R} - \mathrm{e}^{t}),~~\varphi'(t) = -\frac{t}{R^2 - t^2} - \frac{\mathrm{e}^{t}}{\mathrm{e}^{R} - \mathrm{e}^{t}}\\
\varphi'(t)=0&\Leftrightarrow t\e^R + \e^t(R^2 - t^2 - t) = 0\Leftrightarrow t\e^{\frac{\e}{\e+1}} + \e^t\left(\left(\frac{\e}{\e+1}\right)^2 - t^2 - t\right) = 0\\
&\Leftrightarrow (\e^t-\e^{-\frac1{\e+1}})\left(\left(\frac{\e}{\e+1}\right)^2 - t^2 - t\right)+\e^{-\frac1{\e+1}}\left(t+\frac1{\e+1}\right)\left(\frac{\e^2}{\e+1}-t\right)=0
\end{align*}
得到驻点$t=-\frac{1}{\e+1}$，再求二阶导：
\[
\varphi''(t) = -\frac{(R^2 - t^2) - t(-2t)}{(R^2 - t^2)^2} - \frac{\mathrm{e}^{t}(\mathrm{e}^{R} - \mathrm{e}^{t}) - \mathrm{e}^{t}(-\mathrm{e}^{t})}{(\mathrm{e}^{R} - \mathrm{e}^{t})^2} = -\frac{R^2 + t^2}{(R^2 - t^2)^2} - \frac{\mathrm{e}^{t}\mathrm{e}^{R}}{(\mathrm{e}^{R} - \mathrm{e}^{t})^2} < 0
\]
因此 $\varphi'(t)$ 在 $(-R, R)$ 上严格单调递减，故方程 $\varphi'(t) = 0$ 至多有一个实根$t=-\frac{1}{\e+1}$。由于 $S(t)$ 在端点 $t = \pm R$ 处为零，在 $(-R, R)$ 内为正，故该驻点即为最大值点。此时 $c = \mathrm{e}^{t} = \mathrm{e}^{-\frac{1}{\mathrm{e}+1}}$，代入得
\[
S_{\max} = \sqrt{R^2 - \left(-\frac{1}{\mathrm{e}+1}\right)^2}\left(\mathrm{e}^{R} - \mathrm{e}^{-\frac{1}{\mathrm{e}+1}}\right) = \sqrt{\frac{\mathrm{e}-1}{\mathrm{e}+1}} \cdot \mathrm{e}^{-\frac{1}{\mathrm{e}+1}}(\mathrm{e}-1) 
\]
(3) 对于一般直线 $l$，设 $A(R\cos\theta_1, \mathrm{e}^{R\sin\theta_1})$，$B(R\cos\theta_2, \mathrm{e}^{R\sin\theta_2})$，则 $\triangle PAB$ 面积为
\[
S(\theta_1,\theta_2) = \frac{R}{2}\left| \cos\theta_1 (\mathrm{e}^{R\sin\theta_2} - \mathrm{e}^{R}) - \cos\theta_2 (\mathrm{e}^{R\sin\theta_1} - \mathrm{e}^{R}) \right|.
\]
由面积表达式及三角不等式得
\[
S \le \frac{R}{2} \left( |\cos\theta_1| (\mathrm{e}^{R} - \mathrm{e}^{R\sin\theta_2}) + |\cos\theta_2| (\mathrm{e}^{R} - \mathrm{e}^{R\sin\theta_1}) \right) \triangleq H(\theta_1,\theta_2),
\]
等号成立当且仅当 \(\cos\theta_1\) 与 \(\cos\theta_2\) 异号。令 \(u=\sin\theta_1\), \(v=\sin\theta_2\)，则
\[
H(u,v) = \frac{R}{2} \left( \sqrt{1-u^2}\,(\mathrm{e}^{R} - \mathrm{e}^{Rv}) + \sqrt{1-v^2}\,(\mathrm{e}^{R} - \mathrm{e}^{Ru}) \right).
\]
求偏导数并令其为零，得到方程组
\[
\frac{u}{\sqrt{1-u^2}}(\mathrm{e}^{R} - \mathrm{e}^{Rv}) = -R\sqrt{1-v^2}\,\mathrm{e}^{Ru}, \quad
\frac{v}{\sqrt{1-v^2}}(\mathrm{e}^{R} - \mathrm{e}^{Ru}) = -R\sqrt{1-u^2}\,\mathrm{e}^{Rv}.
\]
由对称性知 \(u,v<0\)，令 \(a=-u>0\), \(b=-v>0\)，则方程组化为
\[
\frac{a}{\sqrt{1-a^2}}(\mathrm{e}^{R} - \mathrm{e}^{-Rb}) = R\sqrt{1-b^2}\,\mathrm{e}^{-Ra}, \quad
\frac{b}{\sqrt{1-b^2}}(\mathrm{e}^{R} - \mathrm{e}^{-Ra}) = R\sqrt{1-a^2}\,\mathrm{e}^{-Rb}.
\]
两式相除并整理得
\[
\frac{a}{b} \cdot \frac{\mathrm{e}^{R} - \mathrm{e}^{-Rb}}{\mathrm{e}^{R} - \mathrm{e}^{-Ra}} = \mathrm{e}^{-R(a-b)}.
\]
取对数后移项得到
\[
\ln a + \ln(\mathrm{e}^{R} - \mathrm{e}^{-Ra}) + Ra = \ln b + \ln(\mathrm{e}^{R} - \mathrm{e}^{-Rb}) + Rb.
\]
定义函数 \(K(t)=\ln t + \ln(\mathrm{e}^{R} - \mathrm{e}^{-Rt}) + Rt=\ln t + \ln(\mathrm{e}^{\frac{\e}{\e+1}} - \mathrm{e}^{-\frac{\e}{\e+1}t}) + \frac{\e}{\e+1}t\)，则由复合函数单调性得到 \(K(t)\) 严格递增，从而 \(a=b\)，即 \(u=v\)。因此极值点必满足 \(u=v\)，此时
\[
H(u,u)=R\sqrt{1-u^2}(\mathrm{e}^{R} - \mathrm{e}^{Ru}),
\]
与 (2) 中形式相同，由 (2) 知该函数的最大值在 $u = -\frac{1}{\mathrm{e}}$ 处取得，同时，等号条件要求 $\cos\theta_1$ 与 $\cos\theta_2$ 异号，故取 $\theta_2 = \pi - \theta_1$，此时直线为水平线 $y = \mathrm{e}^{Ru} = \mathrm{e}^{-\frac{1}{\mathrm{e}+1}}$，且 $S$ 达到该最大值。由于 $H(u,v)$ 在边界 $u,v=\pm1$ 上为零，而内点值大于零，故该驻点即为全局最大值点。综上，$\triangle PAB$ 面积的最大值为$\displaystyle  \sqrt{\frac{\mathrm{e}-1}{\mathrm{e}+1}} \cdot \mathrm{e}^{-\frac{1}{\mathrm{e}+1}}(\mathrm{e}-1) $
\end{solution}

\]

\end{solution}

\begin{example}{对称曲线比较}{}
    证明：曲线\[y=(x-\ln x)\e^{1-x}\]上不存在不同的两点关于直线$y=x$对称
\end{example}
\begin{solution}
设$f(x)=x-\ln x,g(x)=(x-\ln x)\e^{1-x}$，假设存在不同的两点关于$y=x$对称，这等价于存在$x_1< x_2$使得$x_2=g(x_1),x_1=g(x_2)$，那么
\begin{align*}
    &x_2=g(x_1)=(x_1-\ln x_1)\e^{1-x_1},x_1=g(x_2)=(x_2-\ln x_2)\e^{1-x_2}\\
\Leftrightarrow& \ln x_2=\ln(x_1-\ln x_1)+1-x_1, \ln x_1=\ln(x_2-\ln x_2)+1-x_2\\
\Rightarrow &x_1-\ln x_1-\ln(x_1-\ln x_1)=x_2-\ln x_2-\ln(x_2-\ln x_2)\\
\Leftrightarrow& f(x_1-\ln x_1)=f(x_2-\ln x_2)\\
\Leftrightarrow& f(f(x_1))=f(f(x_2))
\end{align*}
由于$x_1\neq x_2$，那么$f(x_1),f(x_2)>1$，必有
\[f(x_1)=f(x_2)\Leftrightarrow x_1-\ln x_1=x_2-\ln x_2\]
再根据对称性，进一步设$(x_1,g(x_1)),(x_2,g(x_2))$位于直线$x+y=b$上，则
\[x+g(x)=b\Leftrightarrow x_1+(x_1-\ln x_1)\e^{1-x_1}=x_2+(x_2-\ln x_2)\e^{1-x_2}\]
下面说明这两个条件不能同时成立。由
\[
x_1-\ln x_1=x_2-\ln x_2
\]
便只要比较
\[
x_1+(x_1-\ln x_1)\e^{1-x_1}>x_2+(x_2-\ln x_2)\e^{1-x_2}.
\]
令 $h(x)=g(x)-2f(x)$，则
\begin{align*}
h(x)&=(x-\ln x)\e^{1-x}+x-2(x-\ln x)=(x-\ln x)(e^{1-x}-2)+x\\
h'(x)&=\left(1-\frac1x\right)(e^{1-x}-2)+(x-\ln x)(-e^{1-x})+1=-\left[e^{1-x}\left(x-1-\ln x+\frac1x\right)-\frac2x+1\right]\\
&\leqslant -\left[\frac{e^{1-x}}{x}-\frac2x+1\right]=\frac{-1}{x}\left(e^{1-x}-(1-x)-1\right)\leqslant 0
\end{align*}
故 $h(x)$单减。由 $f(x_1)=f(x_2)$ 可得 $h(x_1)>h(x_2)$，即
\[
x_1+(x_1-\ln x_1)\e^{1-x_1}>x_2+(x_2-\ln x_2)\e^{1-x_2},
\]
与 $x_1+g(x_1)=x_2+g(x_2)$ 矛盾，结论得证。
\end{solution}

\begin{example}{导数与零点}{}
    已知函数$f(x)=x-(a+1)\ln x-\frac{a}{x},(a>1)$\\
    （1）讨论$f(x)$的单调性；\\
    （2）若$f(x_1)=f(x_2)=f(x_3),(x_1<x_2<x_3)$\\
    ~~~~（i）证明$f(x_1x_2x_3)>1-a$.\\
    ~~~~（ii）证明$f(x_1+x_2+x_3)<2e-1-a$.
\end{example}
\begin{solution}
（1）定义域为 $(0,+\infty)$。求导得
$$f'(x)=1-\frac{a+1}x+\frac a{x^2}=\frac{x^2-(a+1)x+a}{x^2}=\frac{(x-1)(x-a)}{x^2}$$
因为 $a>1$，故 $f'(x)=0$ 的两个正根是 $x=1,a$。于是 $f$ 在 $(0,1)$ 和 $(a,+\infty)$ 上递增，在 $(1,a)$ 上递减，极大值为 $f(1)=1-a$，极小值为 $f(a)=a-1-(a+1)\ln a$。\\
（2）（i）取 $t=f(x_1)=f(x_2)=f(x_3)$。因三根互异，必有 $t\in(f(a),1-a)$，并且
\[
x_1\in(0,1),\qquad x_2\in(1,a),\qquad x_3\in(a,+\infty).
\]
记 $x_0>a$ 为方程 $f(x)=1-a$ 在 $(a,+\infty)$ 上的唯一实根。由于 $f$ 在 $(a,+\infty)$ 上严格递增，问题化为证明
$$x_1x_2x_3>x_0$$
记三根乘积为关于截线高度$t$的函数$P(t)=x_1(t)\cdot x_2(t)\cdot x_3(t)$。对对数形式$\ln P(t)=\ln x_1+\ln x_2+\ln x_3$两边关于$t$求导，由隐函数求导法则$f^\prime(x_k)\cdot x_k^{\prime}(t)=1$可得：
$$\frac{P'(t)}{P(t)}=\frac{x'_1}{x_1}+\frac{x'_2}{x_2}+\frac{x'_3}{x_3}=\sum_{k=1}^3\frac1{x_kf'(x_k)}$$
直接求和很繁，改用留数定理。取
$$ g(z)=\dfrac{1}{z(f(z)-t)}. $$
各实根 $x_k$ 都是一阶极点，且
$$ \operatorname{Res}(g,x_k)=\lim_{z\to x_k}(z-x_k)g(z)=\frac{1}{x_kf'(x_k)}. $$
若能说明围道内没有别的根，就有
\[
\sum_{k=1}^3\frac1{x_kf'(x_k)}=\frac{1}{2\pi i}\oint_\Gamma g(z)\,dz.
\]

现说明：当 $t\in(f(a),1-a)$ 时，方程 $f(z)=t$ 在割去负实轴的复平面上除了这三个实根外没有其他根。设 $z=r e^{i\theta}$，$\theta\in(-\pi,\pi)$ 且 $\theta\neq0$。由 $\operatorname{Im}(f(z))=0$ 得
$$r\sin\theta-(a+1)\theta+\frac{a}{r}\sin\theta=0 \quad\Longrightarrow\quad \frac{\sin\theta}{\theta}=\frac{a+1}{r+\frac{a}{r}}.$$
因为对 $\theta\neq0$ 恒有 $\left|\frac{\sin\theta}{\theta}\right|<1$，所以复根存在的必要条件是 $\frac{a+1}{r+\frac{a}{r}}<1$，即 $r+\frac{a}{r}>a+1$，这等价于 $r\in(0,1)\cup(a,+\infty)$。因此若存在复根，其模长只能落在这两个区间内。

沿着曲线 $v(r,\theta)=\operatorname{Im}(f(z))=0$，考察实部 $u(r,\theta)=\operatorname{Re}(f(z))$ 的变化。由极坐标下的柯西--黎曼方程和隐函数求导公式，
$$\frac{du}{dr}=u_r+u_\theta\frac{d\theta}{dr}=u_r+u_\theta\left(-\frac{v_r}{v_\theta}\right)=\frac{v_\theta^2+r^2v_r^2}{r v_\theta}=\frac{r^2(u_r^2+v_r^2)}{r v_\theta}=\frac{r|f'(z)|^2}{v_\theta}.$$
再计算 $v_\theta$：
$$v_\theta=\frac{\partial}{\partial\theta}\left(r\sin\theta-(a+1)\theta+\frac{a}{r}\sin\theta\right)=\cos\theta\left(r+\frac{a}{r}\right)-(a+1).$$
将必要条件 $r+\frac{a}{r}=\frac{(a+1)\theta}{\sin\theta}$ 代入，得：
$$v_\theta=(a+1)\left(\theta\cot\theta-1\right).$$
由于 $\theta\in(-\pi,\pi)$ 且 $\theta\neq0$，由三角不等式恒有 $\theta\cot\theta<1$ 严格成立，故 $v_\theta<0$。同时在曲线上极点以外区域 $|f'(z)|\neq 0$，于是 $\frac{du}{dr}<0$，即沿复根可能存在的曲线，实部 $u$ 随 $r$ 的增大而严格单调递减。

在 $r\in(0,1)$ 分支，当 $r\to1^-$ 时有 $\theta\to0$，此时 $u\to f(1)=1-a$；又由于 $u$ 随 $r$ 递减，所以当 $r<1$ 时恒有 $u>1-a$。在 $r\in(a,+\infty)$ 分支，当 $r\to a^+$ 时同样有 $\theta\to0$，此时 $u\to f(a)$；由于 $u$ 仍随 $r$ 递减，所以当 $r>a$ 时恒有 $u<f(a)$。因此，所有非实数根的实部只能落在 $(-\infty,f(a))\cup(1-a,+\infty)$ 中。对任意 $t\in(f(a),1-a)$，方程 $f(z)=t$ 在复平面上不再出现额外的非实数根，故除这三个实根外没有其他奇点。

取一个“钥匙孔”围道 $\Gamma$，由四部分组成：
\begin{itemize}
    \item 大圆 $C_R$：$|z|=R$，逆时针方向（辐角 $-\pi \to \pi$），$R\to\infty$；
    \item 上边缘 $L_+$：从 $-R$ 到 $-\varepsilon$，沿着负实轴的上侧（辐角取 $\pi$）；
    \item 小圆 $C_{\varepsilon}$：$|z|=\varepsilon$，顺时针方向（辐角从 $\pi$ 绕到 $-\pi$），$\varepsilon\to0$；
    \item 下边缘 $L_-$：从 $-\varepsilon$ 到 $-R$，沿着负实轴的下侧（辐角取 $-\pi$）。
\end{itemize}
因 $f(z)$ 含有主分支 $\ln z$，需沿负实轴取割线，使 $\ln z$ 在围道内单值解析。同时负实轴上 $\operatorname{Im}(f(z)) = \mp(a+1)\pi \neq 0$，故割线上无根。围道内部只包含三个实根 $x_1,x_2,x_3$，没有其他极点。

由留数定理，
$$\oint_\Gamma g(z)\,dz = 2\pi i \sum_{k=1}^3 \operatorname{Res}(g,x_k)=2\pi i\sum_{k=1}^3\frac{1}{x_k f'(x_k)}.$$
再计算左边的积分。

当 $R\to\infty$ 时，$f(z)-t\sim z$，故 $g(z)\sim \frac{1}{z^2}$，由于大圆弧长为 $2\pi R$，故积分模长受控于 $O(\frac{1}{R}) \to 0$，所以 $\int_{C_R}\to0$。
当 $\varepsilon\to0$ 时，$f(z)-t\sim -\frac{a}{z}$ 主导，故 $g(z)\sim -\frac{1}{a}$ 有界，而小圆弧长趋于 0，故 $\int_{C_\varepsilon}\to0$。

接下来计算 $L_+$ 和 $L_-$ 上的积分。
在上边缘 $L_+$：$z=r e^{i\pi}$，$r$ 从 $+\infty$ 降到 $0$，$dz=-dr$，$\ln z = \ln r + i\pi$。代入计算得：
$$f(z)-t = z-(a+1)\ln z-\frac{a}{z}-t = -r-(a+1)(\ln r+i\pi)+\frac{a}{r}-t = A(r) - i(a+1)\pi,$$
其中实部函数记为 $A(r)=-r+\frac{a}{r}-(a+1)\ln r - t$。所以
$$g(z)=\frac{1}{z(f(z)-t)}=\frac{1}{r e^{i\pi}(A(r)-i(a+1)\pi)}=\frac{-1}{r(A(r)-i(a+1)\pi)}.$$
于是
$$\int_{L_+}g(z)dz = \int_{+\infty}^0 \frac{-1}{r(A(r)-i(a+1)\pi)} (-dr) = -\int_0^{+\infty} \frac{dr}{r(A(r)-i(a+1)\pi)}.$$

在下边缘 $L_-$：$z=r e^{-i\pi}$，$r$ 从 $0$ 升到 $+\infty$，$dz=-dr$，$\ln z = \ln r - i\pi$，类似可得
$$f(z)-t= -r-(a+1)(\ln r-i\pi)+\frac{a}{r}-t=A(r)+i(a+1)\pi,$$
$$g(z)=\frac{1}{r e^{-i\pi}(A(r)+i(a+1)\pi)}=\frac{-1}{r(A(r)+i(a+1)\pi)}.$$
$$\int_{L_-}g(z)dz = \int_{0}^{+\infty} \frac{-1}{r(A(r)+i(a+1)\pi)} (-dr) = \int_{0}^{+\infty} \frac{dr}{r(A(r)+i(a+1)\pi)}.$$

将 $L_+$ 和 $L_-$ 的积分相加：
$$\int_{L_+}+\int_{L_-}= \int_0^{+\infty}\frac{1}{r}\left[\frac{1}{A(r)+i(a+1)\pi} - \frac{1}{A(r)-i(a+1)\pi}\right]dr = \int_0^{+\infty}\frac{-2i(a+1)\pi}{r\left[A^2(r)+(a+1)^2\pi^2\right]}dr.$$

取极限后，
$$\oint_\Gamma g(z)dz = \int_0^{+\infty}\frac{-2i(a+1)\pi}{r\left[A^2(r)+(a+1)^2\pi^2\right]}dr.$$

于是
$$\sum_{k=1}^3\frac{1}{x_k f'(x_k)} = -\int_0^{+\infty}\frac{a+1}{r\left[A^2(r)+(a+1)^2\pi^2\right]}dr.$$

因为 $a>1$，被积函数恒正，所以
$$\sum_{k=1}^3\frac{1}{x_k f'(x_k)} < 0 \quad \text{恒成立}.$$
而 $\frac{P'(t)}{P(t)}$ 正是这个和。因实根均大于 0，故 $P(t)>0$，从而导出 $P'(t)<0$，即 $P(t)$ 关于 $t$ 严格单调递减。

由于 $P(t)$ 在 $(f(a),1-a)$ 上严格递减，对任意 $t<1-a$ 都有 $P(t) > \lim_{\tau\to(1-a)^-}P(\tau)$。当 $t\to(1-a)^-$ 时，$x_1,x_2\to1$，$x_3\to x_0$，所以
$$x_1x_2x_3 = P(t) > x_0.$$
又因 $x_0>a$ 且 $f$ 在 $(a,+\infty)$ 上严格递增，
$$f(x_1x_2x_3) > f(x_0) = 1-a.$$
故 (i) 成立。\\
（2）（ii）设三根之和为关于截线高度 $t$ 的函数 $S(t)=x_1(t)+x_2(t)+x_3(t)$。
对 $S(t)$ 两边关于 $t$ 求导，同样由隐函数求导法则 $f'(x_k)\cdot x_k'(t)=1$ 可得：
$$S'(t)=x'_1+x'_2+x'_3=\sum_{k=1}^3\frac{1}{f'(x_k)}$$
同样取
$$ h(z)=\dfrac{1}{f(z)-t}. $$
各实根 $x_k$ 都是一阶极点，且
$$ \operatorname{Res}(h,x_k)=\lim_{z\to x_k}\frac{z-x_k}{f(z)-t}=\frac{1}{f'(x_k)}. $$
因此
$$\oint_\Gamma h(z)\,dz = 2\pi i \sum_{k=1}^3 \operatorname{Res}(h,x_k)=2\pi i S'(t)$$
仍在（i）的钥匙孔围道 $\Gamma$ 上计算四段积分。

当 $R\to\infty$ 时，由洛朗展开得
\[
h(z) = \frac{1}{z\left(1 - \frac{(a+1)\ln z + a/z + t}{z}\right)} = \frac{1}{z} + O\left(\frac{\ln z}{z^2}\right).
\]
高阶项在大圆上的积分趋于 0，故
\[
\int_{C_R} h(z)dz = \int_{-\pi}^{\pi} \frac{1}{R e^{i\theta}} i R e^{i\theta} d\theta \to 2\pi i.
\]
当 $\varepsilon\to0$ 时，$f(z)-t\sim -\frac{a}{z}$，故 $h(z)\sim -\frac{z}{a}$，被积函数趋于 0，且小圆弧长趋于 0，故小圆积分 $\int_{C_\varepsilon}\to0$。

在上边缘 $L_+$：$z=r e^{i\pi}$，$dz=-dr$（$r$ 从 $+\infty$ 降到 $0$）。此时 $f(z)-t=A(r)-i(a+1)\pi$。
$$\int_{L_+}h(z)dz = \int_{+\infty}^0 \frac{1}{A(r)-i(a+1)\pi} (-dr) = \int_0^{+\infty} \frac{dr}{A(r)-i(a+1)\pi}$$
在下边缘 $L_-$：$z=r e^{-i\pi}$，$dz=-dr$（$r$ 从 $0$ 升到 $+\infty$）。此时 $f(z)-t=A(r)+i(a+1)\pi$。
$$\int_{L_-}h(z)dz = \int_{0}^{+\infty} \frac{1}{A(r)+i(a+1)\pi} (-dr) = -\int_0^{+\infty} \frac{dr}{A(r)+i(a+1)\pi}$$

将 $L_+$ 和 $L_-$ 的积分相加：
$$\int_{L_+}+\int_{L_-} = \int_0^{+\infty}\left[\frac{1}{A(r)-i(a+1)\pi} - \frac{1}{A(r)+i(a+1)\pi}\right]dr = \int_0^{+\infty}\frac{2i(a+1)\pi}{A^2(r)+(a+1)^2\pi^2}dr$$

于是整个闭合围道积分为：
$$\oint_\Gamma h(z)dz = 2\pi i + \int_0^{+\infty}\frac{2i(a+1)\pi}{A^2(r)+(a+1)^2\pi^2}dr$$
两边同除以 $2\pi i$，得到 $S'(t)$ 的精确表达式：
$$S'(t) = 1 + \int_0^{+\infty}\frac{a+1}{A^2(r)+(a+1)^2\pi^2}dr$$
因为 $a>1$ 且分母严格大于 0，被积函数严格为正。所以 $S'(t) > 1 > 0$ 恒成立，即 $S(t)$ 关于 $t$ 严格单调递增。

由于 $S(t)$ 在 $(f(a),1-a)$ 上严格递增，对于任意 $t<1-a$，有 $S(t) < \lim_{\tau\to (1-a)^-}S(\tau)$。
同（i）中的极限分析，当 $t\to (1-a)^-$ 时，$x_1\to 1$，$x_2\to 1$，$x_3\to x_0$。
故 $x_1+x_2+x_3 = S(t) < 1+1+x_0 = x_0+2$。

因为 $x_1, x_2 > 0$ 且 $x_3 > a$，有 $S(t) > a$。又因 $f(x)$ 在 $(a,+\infty)$ 上严格单调递增，且 $x_1+x_2+x_3$ 与 $x_0+2$ 均处于该区间，所以直接代入映射不改变不等号方向：
$$f(x_1+x_2+x_3) < f(x_0+2)$$
再估计 $f(x_0+2)$ 的上限。根据微积分基本定理：
$$f(x_0+2) - f(x_0) = \int_{x_0}^{x_0+2} f'(x) dx$$
已知 $f'(x) = 1 - \frac{a+1}{x} + \frac{a}{x^2} = 1 - \frac{(a+1)x-a}{x^2}$。
对于所有位于 $(x_0, x_0+2)$ 内的积分变量 $x > a > 1$，恒有 $(a+1)x - a = a(x-1) + x > a > 0$，因此 $\frac{(a+1)x-a}{x^2} > 0$，即在此区间上恒有严格的 $f'(x) < 1$。
从而 $\int_{x_0}^{x_0+2} f'(x) dx < \int_{x_0}^{x_0+2} 1 \,dx = 2$。
又因为 $f(x_0) = 1-a$，移项即可得到：
$$f(x_0+2) < f(x_0) + 2 = (1-a) + 2 = 3-a$$
又 \(3<2e-1\)，从而
$$3-a < 2e-1-a$$
综上，首尾相连写下不等号即可得：
$$f(x_1+x_2+x_3) < f(x_0+2) < 3-a < 2e-1-a$$
原不等式(ii)得证。
\end{solution}

\begin{example}{}{}
已知函数$f(x)=(x+1)\ln x,f(x_1)=f(x_2)$，过$(x_1,f(x_1)),(x_2,f(x_2))$分别作切线交$x$轴于$(x_3,0),(x_4,0)$，求证$x_3+x_4>0$.
\end{example}
\begin{solution}
设$x_-,x_+$分别表示正负根，记
$$E=x_3+x_4=\frac{x_{-}^{2}}{(x_{-}+1)f^{\prime}(x_{-})}+\frac{x_{+}^{2}}{(x_{+}+1)f^{\prime}(x_{+})}.$$
当 $y\ge 4$ 时，直接估值便有 $E>0$。下面只看 $0<y<4$。

取主值对数 $L(z)=\operatorname{Log}(1+z)$，支割线放在 $(-\infty,-1]$，并定义
\[
\Phi(z)=\frac{yz}{(z+1)L(z)(zL(z)-y)}.
\]
先看方程 $zL(z)=y$ 在割平面 $\mathbb{C}\setminus(-\infty,-1]$ 里的根。若存在非实根，设
\[
1+z=re^{i\theta}\qquad (\theta\in(-\pi,\pi)\setminus\{0\},\ r>0),
\]
则
\[
z=r\cos\theta-1+ir\sin\theta,\qquad L(z)=\ln r+i\theta.
\]
由虚部为零，
\[
\Im(zL(z))=\theta(r\cos\theta-1)+r\sin\theta\ln r=0,
\]
所以
\[
\ln r=\frac{\theta(1-r\cos\theta)}{r\sin\theta}.
\]
代回实部，
\begin{align*}
\Re(zL(z))
&=(r\cos\theta-1)\ln r-r\theta\sin\theta \\
&=(r\cos\theta-1)\frac{\theta(1-r\cos\theta)}{r\sin\theta}-\frac{r^2\theta\sin^2\theta}{r\sin\theta} \\
&=-\frac{\theta}{r\sin\theta}\bigl[(1-r\cos\theta)^2+r^2\sin^2\theta\bigr] \\
&=-\frac{\theta}{\sin\theta}\frac{|z|^2}{r}<0.
\end{align*}
这与 $y>0$ 矛盾，因此割平面内只有两个实根 $x_-,x_+$。

又因为 $L(z)\sim z$ $(z\to0)$，故 $\Phi(z)\to-1$，所以 $z=0$ 是可去奇点。对简单极点 $x_\pm$，由 $(zL(z))'=f'(z)$ 得
\[
\operatorname{Res}(\Phi;x_\pm)
=\lim_{z\to x_\pm}\frac{yz(z-x_\pm)}{(z+1)L(z)(zL(z)-y)}
=\frac{yx_\pm}{(x_\pm+1)L(x_\pm)f'(x_\pm)}
=\frac{x_\pm^2}{(x_\pm+1)f'(x_\pm)}.
\]
所以两个留数之和正好是 $E$。

取绕割线 $(-\infty,-1]$ 的匙孔形围道。大圆上的积分满足
\[
\int_{C_R}\Phi(z)\,dz=O\!\left(\frac{1}{\ln^2R}\right)\to0\qquad (R\to\infty),
\]
小圆上的积分满足
\[
\int_{C_\varepsilon}\Phi(z)\,dz=O\!\left(\frac{1}{\ln^2\varepsilon}\right)\to0\qquad (\varepsilon\to0).
\]
于是由留数定理，
\[
E=\frac{1}{2\pi i}\int_0^\infty\bigl(\Phi_+(-1-t)-\Phi_-(-1-t)\bigr)\,dt.
\]
沿割线两侧有
\[
L_\pm(-1-t)=\ln t\pm i\pi,
\]
而 $\Phi_-=\overline{\Phi_+}$，于是
\[
E=\frac1\pi\int_0^\infty \Im\Phi_+(-1-t)\,dt.
\]
把
\[
\Phi_+(-1-t)=\frac{y(-1-t)}{(-t)(\ln t+i\pi)\bigl[(-1-t)(\ln t+i\pi)-y\bigr]}
\]
化简，可得
\[
\Phi_+(-1-t)=\frac{1+t}{t}\left(\frac{1}{\ln t+\frac{y}{1+t}+i\pi}-\frac{1}{\ln t+i\pi}\right).
\]
因此
\[
\Im\Phi_+(-1-t)=\frac{\pi(1+t)}{t}\left(\frac{1}{(\ln t)^2+\pi^2}-\frac{1}{\left(\ln t+\frac{y}{1+t}\right)^2+\pi^2}\right),
\]
从而
\[
E=\int_0^\infty\frac{1+t}{t}\left(\frac{1}{(\ln t)^2+\pi^2}-\frac{1}{\left(\ln t+\frac{y}{1+t}\right)^2+\pi^2}\right)\,dt.
\]

利用恒等式
\[
\frac{1}{u^2+\pi^2}-\frac{1}{(u+a)^2+\pi^2}
=\int_0^a\frac{2(u+s)}{\bigl((u+s)^2+\pi^2\bigr)^2}\,ds,
\]
代入 $u=\ln t,\ a=\dfrac{y}{1+t}$，再令 $r=(1+t)s$，便得到
\[
E=\int_0^y J(r)\,dr,
\]
其中
\[
J(r)=\int_0^\infty\frac{2\left(\ln t+\frac{r}{1+t}\right)}{t\left(\left(\ln t+\frac{r}{1+t}\right)^2+\pi^2\right)^2}\,dt.
\]
只要证 $0<r<4$ 时 $J(r)>0$ 即可。

令 $t=e^s$，则
\[
J(r)=\int_{-\infty}^\infty\frac{2\phi_r(s)}{\bigl(\phi_r(s)^2+\pi^2\bigr)^2}\,ds,
\qquad
\phi_r(s)=s+\frac{r}{1+e^s}.
\]
因为
\[
\phi_r'(s)=1-\frac{re^s}{(1+e^s)^2}\ge 1-\frac r4>0\qquad (0<r<4),
\]
所以 $\phi_r$ 严格递增，特别地，
\[
\phi_r(s)>\phi_r(0)=\frac r2\qquad (s>0).
\]
再记
\[
A(s)=\frac{1}{\phi_r(s)^2+\pi^2},\qquad
B(s)=\frac{d}{ds}\left(\frac1{\phi_r'(s)}\right)=-\frac{\phi_r''(s)}{(\phi_r'(s))^2}.
\]
分部积分后，
\[
J(r)=\int_{-\infty}^\infty A(s)B(s)\,ds
=\int_0^\infty B(s)\bigl(A(s)-A(-s)\bigr)\,ds.
\]
这里用到了 $\phi_r'(s)$ 是偶函数，所以 $B(s)$ 是奇函数。

当 $s>0$ 时，
\[
\phi_r''(s)=\frac{re^s(e^s-1)}{(1+e^s)^3}>0,
\]
故 $B(s)<0$。另一方面，
\[
\phi_r(-s)=r-\phi_r(s),
\]
于是
\[
A(s)-A(-s)
=\frac{1}{\phi_r(s)^2+\pi^2}-\frac{1}{(r-\phi_r(s))^2+\pi^2}
=\frac{r(r-2\phi_r(s))}{\bigl(\phi_r(s)^2+\pi^2\bigr)\bigl((r-\phi_r(s))^2+\pi^2\bigr)}<0,
\]
因为 $\phi_r(s)>\dfrac r2$。因此被积函数始终为正，$J(r)>0$。

最后，
\[
E=\int_0^yJ(r)\,dr>0.
\]
因此，对任意 $y>0$ 且 $x_1\ne x_2$ 满足 $x_1\ln(1+x_1)=x_2\ln(1+x_2)=y$，都有 $E>0$，也就是 $x_3+x_4>0$。
\end{solution}

\begin{example}{$f(x_1x_2x_3)>1-a$}{}
    已知函数$f(x)=x-(a+1)\ln x-\frac{a}{x},(a>1)$\\
    （1）讨论$f(x)$的单调性；\\
    （2）若$f(x_1)=f(x_2)=f(x_3),(x_1<x_2<x_3)$，证明$f(x_1x_2x_3)>1-a$.
\end{example}
\begin{solution}
（1）定义域为 $(0,+\infty)$。求导得
$$f'(x)=1-\frac{a+1}x+\frac a{x^2}=\frac{x^2-(a+1)x+a}{x^2}=\frac{(x-1)(x-a)}{x^2}$$
故 $f$ 在 $(0,1)$ 和 $(a,+\infty)$ 上递增，在 $(1,a)$ 上递减，极大值为 $f(1)=1-a$，极小值为 $f(a)=a-1-(a+1)\ln a$。\\
（2）三根同时存在时，公共值 $c$ 必落在
\[
f(a)<c<f(1)=1-a.
\]
于是
\[
x_1<1<x_2<a<x_3.
\]
又因为 $f$ 在 $(a,+\infty)$ 上严格递增，方程 $f(x)=f(1)=1-a$ 在 $x>a$ 上只有一个解，记作 $r$。因此转为证明
\[
x_1x_2x_3>r.
\]
把三根记成 $c$ 的函数：
\[x_1=x_1(c),\quad x_2=x_2(c),\quad x_3=x_3(c),\quad c\in(f(a),1-a).\]
由于每个根都落在各自的单调区间内，且 $f'(x_i)\neq 0$，可对
\[
f(x_i(c))=c
\]
作隐函数求导，得到
\[
f'(x_i)\,x_i'(c)=1,\qquad x_i'(c)=\frac{1}{f'(x_i)}.
\]
再记
\[
P(c)=x_1x_2x_3,
\]
则
\[
\frac{P'(c)}{P(c)}=\sum_{i=1}^3\frac{x_i'(c)}{x_i}=\sum_{i=1}^3\frac{1}{x_i f'(x_i)}.
\]
令
\[
Q_c(x)=\frac{x^2-cx-a}{a+1},\qquad h_c(x)=x\ln x-Q_c(x).
\]
由于
\[
h_c(x)=-\frac{x}{a+1}(f(x)-c),
\]
所以 $h_c(x_i)=0$，且
\[
h_c'(x_i)=-\frac{x_i}{a+1}f'(x_i).
\]
从而
\[\frac{P'(c)}{P(c)}=-\dfrac{1}{a+1}\left(\frac{1}{h_c'(x_1)}+\frac{1}{h_c'(x_2)}+\frac{1}{h_c'(x_3)}\right)\]
又因为 $h_c(x_1)=h_c(x_2)=h_c(x_3)=0$，可写成
\[h_c(x)=(x-x_1)(x-x_2)(x-x_3)\,q(x),\qquad q(x)=\frac{h_c(x)}{(x-x_1)(x-x_2)(x-x_3)}\ (x\neq x_i)\]
令$p(x)=(x-x_1)(x-x_2)(x-x_3)$，得$h_c(x)=p(x)q(x),h_c(x_i)=0,h_c'(x_i)=p'(x_i)q(x_i)=-\frac{x_i}{a+1}f'(x_i)$，于是：
\begin{align*}
\frac{P'(c)}{P(c)}&=-\frac{1}{a+1}\sum_{i=1}^3\frac{1}{p'(x_i)q(x_i)}\\
\sum_{i=1}^3\frac{1}{p'(x_i)q(x_i)}&= \frac{1/q(x_1)}{(x_1-x_2)(x_1-x_3)} + \frac{1/q(x_2)}{(x_2-x_1)(x_2-x_3)} + \frac{1/q(x_3)}{(x_3-x_1)(x_3-x_2)}\\
&= \frac{1}{x_3 - x_1} \left[ \frac{1/q(x_3) - 1/q(x_2)}{x_3 - x_2} - \frac{1/q(x_2) - 1/q(x_1)}{x_2 - x_1} \right]\\
&=  \frac{1}{(x_3 - x_1)^2}\left[\frac{1/q(x_3) - 1/q(x_2)}{\frac{x_3 - x_2}{x_3-x_1}} - \frac{1/q(x_2) - 1/q(x_1)}{\frac{x_2 - x_1}{x_3-x_1}}\right]\\
&= \frac{1}{(x_3 - x_1)^2} \left[ \frac{1/q(x_3) - 1/q(x_2)}{\theta} - \frac{1/q(x_2) - 1/q(x_1)}{1-\theta} \right],\theta = \frac{x_3 - x_2}{x_3 - x_1}=1 -  \frac{x_2 - x_1}{x_3 - x_1}\\
&=\frac{1}{(x_3-x_1)^2} \frac{(1-\theta)[1/q(x_3) - 1/q(x_2)] - \theta[1/q(x_2) - 1/q(x_1)]}{\theta(1-\theta)}\\
&=\frac{1}{(x_3-x_1)^2} \frac{ \theta 1/q(x_1) + (1-\theta)1/q(x_3) - 1/q(x_2)}{\theta(1-\theta)}
\end{align*}
接着考察
\[
\frac{1}{q(x)}=\frac{(x-x_1)(x-x_2)(x-x_3)}{x\ln x-\frac{x^2-cx-a}{a+1}}
\]
的凹性。把 $g(x)=x\ln x$ 写成积分形式：
\begin{align*}
\ln x=&\int_1^x\frac1u\dd u=\int_1^x\left(\int_0^{\infty}\frac{1}{(u+t)^2}\dd t\right)\dd u=\int_0^{\infty}\int_0^x \frac{1}{(u+t)^2}\dd u\dd t\\
=& \int_0^\infty\left[ -\frac{1}{u+t} \right]_{u=1}^{u=x} \dd t= \int_0^\infty \left( \frac{1}{1+t} - \frac{1}{x+t} \right) \mathrm{d}t\\
g(x)=x\ln x=&\int_0^\infty\left(\frac{x}{1+t}-1+\frac{t}{x+t}\right)\dd t
\end{align*}
对固定 $t>0$，$\frac{x}{1+t}-1$ 只是一次式，所以三次余项全来自 $\frac{t}{x+t}$。因此只需计算 $\frac{1}{x+t}$ 的三阶差商。记 $f(x)=\frac{1}{x+t}$，则一阶差商为
\begin{align*}
f[x_1, x_2] =& \frac{f(x_2) - f(x_1)}{x_2 - x_1}=\frac{\frac{1}{x_2+t} - \frac{1}{x_1+t}}{x_2-x_1}=\frac1{x_2-x_1}\frac{(x_1+t) - (x_2+t)}{(x_1+t)(x_2+t)}\\
=&\frac1{x_2-x_1}\frac{-(x_2 - x_1)}{(x_1+t)(x_2+t)}=\frac{-1}{(x_1+t)(x_2+t)}\\
f[x_1, x_2, x_3] =& \frac{f[x_2, x_3] - f[x_1, x_2]}{x_3 - x_1}=\frac{ \frac{-1}{(x_2+t)(x_3+t)} - \frac{-1}{(x_1+t)(x_2+t)}}{x_3 - x_1}=\frac{\frac{1}{x_2+t} \left[ \frac{1}{x_1+t} - \frac{1}{x_3+t} \right]}{x_3-x_1}\\
=&\frac{\frac{1}{x_2+t} \left[ \frac{(x_3+t) - (x_1+t)}{(x_1+t)(x_3+t)} \right]}{x_3-x_1}=\frac{\frac{x_3 - x_1}{(x_1+t)(x_2+t)(x_3+t)}}{x_3-x_1}=\frac{1}{(x_1+t)(x_2+t)(x_3+t)}
\end{align*}
同理，三阶差商 $f[x_1, x_2, x_3, x]$ 为
\[
\frac{-1}{(t+x_1)(t+x_2)(t+x_3)(t+x)}.
\]
乘上常数 $t$ 再积分，便有
\[q(x)= -\int_{0}^{\infty}\frac{t\dd t}{(t+x_1)(t+x_2)(t+x_3)(t+x)}\]
因此 $q(x)<0$。设
\[
s(x)=-q(x)>0,\phi(x)=\frac{1}{s(x)}=-\frac{1}{q(x)},w(t):=\frac{t}{(t+x_1)(t+x_2)(t+x_3)}>0, (t>0),
\]
由于 $w(t)=O(t^{-2})$，可在积分号下求导：
\[s(x)=\int_0^\infty \frac{w(t)}{x+t}\dd t,~~s'(x)=-\int_0^\infty \frac{w(t)}{(x+t)^2}\dd t,~~s''(x)=2\int_0^\infty \frac{w(t)}{(x+t)^3}\dd t\]
令
\[A(t)=\frac{\sqrt{w(t)}}{(x+t)^{1/2}},B(t)=\frac{\sqrt{w(t)}}{(x+t)^{3/2}},\int_0^\infty A(t)B(t)\dd t=\int_0^\infty\frac{w(t)}{(x+t)^2}\dd t.\]
由柯西--施瓦兹不等式，
\[
\left(\int_0^\infty\frac{w(t)}{(x+t)^2}\dd t\right)^2
\leqslant
\left(\int_0^\infty\frac{w(t)}{x+t}\dd t\right)
\left(\int_0^\infty\frac{w(t)}{(x+t)^3}\dd t\right)
=\frac12\,s(x)s''(x).
\]
等价于
\[2(s'(x))^2\le s(x)s''(x)\]
而
\[
\phi(x)=\frac{1}{s(x)}\implies \phi''(x)=\frac{2(s')^2-ss''}{s^3}\le 0,
\]
所以 $\phi$ 是凹函数，而且还是严格凹的：若柯西--施瓦兹取等，则 $A(t)$ 与 $B(t)$ 必须成比例，也就是 $(x+t)$ 必须恒定，这显然不可能。
故 $\phi''(x)<0$，从而
\[
\phi(x)=-\frac{1}{q(x)}\ \text{在 }(0,\infty)\text{ 上严格凹。}
\]
而$x_2=\theta x_1+(1-\theta)x_3$，故
\[
\phi(x_2)>\theta\phi(x_1)+(1-\theta)\phi(x_3).
\]
代回 $\phi=-1/q,m>0$ 并整理得
\[\theta\frac1{q(x_1)}-\frac1{q(x_2)}+(1-\theta)\frac1{q(x_3)}>0\implies \sum_{i=1}^3\frac{1}{p'(x_i)q(x_i)}>0
\implies\frac{P'(c)}{P(c)}<0\]
即$P(c)=x_1(c)x_2(c)x_3(c)$在$(f(a),1-a)$上严格递减，从而有$f(x_1x_2x_3)=f(P)>f(r)=1-a$。
\end{solution}

\end{solution}

    \[
    f(x)=x-\ln(\ln x+1)-1.

    左侧不等式可写成
    \[
    g(x)=x^2-x(m+1)+1=x\ln(\ln x+1)-x+1,

    \[
    g'(x)=\ln(\ln x+1)+x\cdot\frac1{\ln x+1}\cdot\frac1x-1=\ln(\ln x+1)+\frac1{\ln x+1}-1.
    \]

    \frac{x^2+4x-5}{4x+2}.

\end{solution}

\begin{example}{零点比较}{}

\end{example}
\begin{solution}
分两步证明
\[
\begin{cases}
x_1+x_2>1+m+\dfrac{m^2}{2},\\
x_3>1-m-\dfrac{m^2}{2}.
\end{cases}
\]
前一个再稍微加强一点，证明
\[
x_1+x_2>\e(3-\e)m^2+m+1.
\]
这可以改写成偏移模型
\[
x_1-\ln x_1=x_2-\ln x_2=a>1,\qquad \frac1{x_1}+\frac1{x_2}>\e^a-1+\dfrac{\e(3-\e)}{\e^a}.
\]
交换 $x_1,x_2$ 后不等式不变，取 $\dfrac1{x_1}>\dfrac1{x_2}$，则 $x_1<1<x_2$。令
\[
g(x)=\frac{1}{x^2}-\left(\e^a-1+\dfrac{\e(3-\e)}{\e^a}\right)\frac{1}{x}
=\frac{1}{x^2}-\left(\frac{\e^x}{x}-1+\e(3-\e)\dfrac{x}{\e^x}\right)\frac{1}{x}.
\]
目标是证 $g(x_1)>g(x_2)$。因为 $x_1-\ln x_1=x_2-\ln x_2=a$，取指数后有
\[
\frac{\e^{x_1}}{x_1}=\frac{\e^{x_2}}{x_2}=\e^a.
\]
改写为考察
\begin{align*}
h(x)&=\frac{1}{x^2}-\frac{\frac{\mathrm{e}^x}{x}-1+\mathrm{e}(3-\mathrm{e})\frac{x}{\mathrm{e}^x}}{x}+1+\mathrm{e}(5-2\mathrm{e})\frac{x}{\mathrm{e}^x}\\

\end{align*}
若证 $h(x)$ 单调递减即可。令
\[
H(x)=x^3\e^xh'(x)=\mathrm{e}^{2x}(2-x)-(x+2)\mathrm{e}^x+\mathrm{e}x^3\left[8-3\mathrm{e}-(5-2\mathrm{e})x\right],
\]

\begin{align*}
    H'(x)&=e^{2x}(3-2x)-e^x(x+3)+3e(8-3e)x^2-4e(5-2e)x^3\\
H''(x)&=4\e^{2x}(1-x)-\e^x(x+4)+6\e(8-3\e)x-12\e(5-2\e)x^2\\
H'''(x)&=4\e^{2x}(1-2x)-\e^x(x+5)+6\e(8-3e)-24\e(5-2\e)x\\
H''''(x)&=-16x\e^{2x}-(x+6)\e^x-24\e(5-2\e)
\end{align*}

\[
H''''(0)=-6+24\e(2\e-5)>0,\qquad \lim_{x\to+\infty}H''''(x)=-\infty.

\[H'''(0)=-1+6\e(8-3\e)<0,H'''(1)=26\e^2-78\e<0,H'''\left(\frac12\right)=6\e^2-12\e-\frac{11}{2}\sqrt{\e}>0\]

可知 $H''$ 也有两个零点，记为 $\xi_4,\xi_5$，对应 $H'$ 先减后增再减。又因为
\[
H'(0)=H'(1)=0,
\]
所以 $H'$ 还有一个零点 $\xi_6$。因此 $H$ 在 $(0,\xi_6)$ 上递减，在 $(\xi_6,1)$ 上递增，在 $(1,+\infty)$ 上递减。再用
\[
H(0)=H(1)=0
\]
便得 $H(x)\le0$，从而 $h(x)$ 在 $(0,+\infty)$ 上递减。这样就证明了
\[
x_1+x_2>\e(3-\e)m^2+m+1.
\]

\[
x_3>1-m-\frac{m^2}2>1.
\]
当 $m\in\left(-\frac1{\e},0\right)$ 时，后一个不等式显然成立。剩下比较
\[
-m=x_3\ln x_3
\quad\text{与}\quad
\left(1-m-\frac{m^2}{2}\right)\ln\left(1-m-\frac{m^2}{2}\right).
\]
也就是证明
\[p(m)=\frac{m}{\frac{m^2}2+m-1}-\ln (1-m-\frac{m^2}2)>0,\forall m\in(-\frac1{\e},0)\]
令 $t=-m\in\left(0,\frac1\e\right)$，则上式化为
$$f(t)=\frac{t}{1+t-\frac{t^{2}}{2}}-\ln \left(1+t-\frac{t^{2}}{2}\right)>0,~~f'(t)=\frac{t^{2}\left(2-\frac{t}{2}\right)}{\left(1+t-\frac{t^{2}}{2}\right)^{2}}>0, \quad \forall t\in\left(0,\frac{1}{e}\right)$$
又 $f(0)=0$，故 $f(t)>0$ 在整个区间内成立，于是
\[
x_3>1-m-\frac{m^2}2>1.
\]
两部分合起来便有 $x_1+x_2+x_3>2$。
\end{solution}

\begin{example}{}{}

\end{example}
\begin{solution}

    \]

    函数是齐次的，即 $f(ka,kb,kc,kd)=f(a,b,c,d)$（$k>0$），故可令 $b+c=1$，于是 $a+d=1$。再设

    \]

    \[
    f(S,y)=\frac{y}{2-S}+\frac{1-y}{S}
    =\frac{2S-2}{S(2-S)}\,y+\frac1S.
    \]
    对固定的 $S$，这是 $y$ 的一次函数，极值落在允许区间端点。

    当 $0<S<1$ 时，系数 $\frac{2S-2}{S(2-S)}<0$，故最小值在 $y=S$ 处取得，即 $x=0$。这时
    \[
    g_1(S)=f(S,S)=\frac{S}{2-S}+\frac{1-S}{S}=\frac{2}{2-S}+\frac1S-2.
    \]

\begin{example}{指数放缩题}{}
    正实数$x,y$满足$e^x-\ln x=e^y-\ln y$，证明：$\sqrt{xy}\left(e^x+e^y\right)\leqslant2$
\end{example}
\begin{solution}
记
$$ F(t)=e^t-\ln t \qquad (t>0). $$
则
$$ F'(t)=e^t-\frac1t. $$
设 $t_0>0$ 为方程 $t e^t=1$ 的唯一解，则 $F$ 在 $(0,t_0]$ 上递减，在 $[t_0,+\infty)$ 上递增。

若 $x=y$，则只能有 $x=y=t_0$，于是
$$ \sqrt{xy}(e^x+e^y)=2t_0e^{t_0}=2. $$

下设 $0<x<t_0<y$。记
$$ m=\frac{x+y}{2},\qquad v=\frac{y-x}{2}>0,\qquad w=\frac{\ln y-\ln x}{2}>0. $$
由 $\ln\frac{y}{x}=2w$ 和 $y-x=2v$ 可得
$$ x=\frac{v e^{-w}}{\sinh w},\qquad y=\frac{v e^w}{\sinh w},\qquad \sqrt{xy}=\frac{v}{\sinh w}. $$

又因为 $e^y-\ln y=e^x-\ln x$，所以
$$ e^y-e^x=\ln y-\ln x. $$
代入 $m,v,w$ 后即为
$$ 2e^m\sinh v=2w,\qquad e^m=\frac{w}{\sinh v}. $$
因此
$$ \frac{e^x+e^y}{2}=e^m\cosh v=\frac{w}{\sinh v}\cosh v=w\coth v. $$
于是目标不等式化为
\[
\sqrt{xy}\cdot\frac{e^x+e^y}{2}<1
\iff
\frac{v}{\sinh w}\cdot w\coth v<1
\iff
v\coth v<\frac{\sinh w}{w}.
\]

记 $R=\dfrac{w}{v}$。由 $e^m=\dfrac{w}{\sinh v}$ 得
$$ R=e^m\frac{\sinh v}{v}>e^m. $$
另一方面，
$$ R=\frac{1}{2v}\ln\frac{m+v}{m-v}=\frac{1}{2v}\ln\frac{1+v/m}{1-v/m}. $$
令 $s=\dfrac{v}{m}\in(0,1)$，由
$$ \ln\frac{1+s}{1-s}>2s $$
可得
$$ R>\frac1m. $$
所以
$$ R>\max\left(e^m,\frac1m\right). $$
设 $m_0>0$ 满足 $m_0e^{m_0}=1$，则 $\max(e^m,1/m)$ 的最小值是 $e^{m_0}$，故
$$ R>e^{m_0}. $$

再看函数 $g(m)=me^m$。它严格递增，而
$$ g(\ln\sqrt2)=\frac{\ln2}{\sqrt2}<1=g(m_0), $$
所以 $\ln\sqrt2<m_0$，从而
$$ e^{m_0}>\sqrt2. $$
于是
$$ R>\sqrt2,\qquad w>\sqrt2\,v. $$

再记
$$ S(t)=\frac{\sinh t}{t}\qquad (t>0). $$
因为
$$ S'(t)=\frac{t\cosh t-\sinh t}{t^2}>0, $$
所以 $S$ 严格递增。由 $w>\sqrt2\,v$ 得
$$ \frac{\sinh w}{w}>\frac{\sinh(\sqrt2\,v)}{\sqrt2\,v}. $$
因此只需证明
$$ v\coth v\le \frac{\sinh(\sqrt2\,v)}{\sqrt2\,v}, $$
也就是
$$ \sqrt2\,v^2\cosh v\le \sinh v\,\sinh(\sqrt2\,v). $$

右边可写成
$$ \sinh v\,\sinh(\sqrt2\,v)=\frac12\bigl[\cosh((\sqrt2+1)v)-\cosh((\sqrt2-1)v)\bigr]. $$
设它的泰勒展开为
$$ \sum_{n=1}^{\infty}\frac{C_n}{(2n)!}v^{2n},\qquad
C_n=\frac12\bigl[(\sqrt2+1)^{2n}-(\sqrt2-1)^{2n}\bigr]
=\sum_{k=0}^{n-1}\binom{2n}{2k+1}(\sqrt2)^{2k+1}. $$
左边的展开为
$$ \sqrt2\,v^2\cosh v
=\sum_{n=1}^{\infty}\frac{D_n}{(2n)!}v^{2n},\qquad
D_n=\sqrt2\,(2n)(2n-1). $$

当 $n=1,2$ 时，$C_n=D_n$。当 $n\ge3$ 时，只取 $C_n$ 中 $k=0,1$ 两项就有
$$ C_n>\binom{2n}{1}\sqrt2+\binom{2n}{3}(\sqrt2)^3
=2n\sqrt2+\frac{2n(2n-1)(2n-2)}{3}\sqrt2. $$
与 $D_n$ 比较，差为
$$ \sqrt2\,(2n)\left(1+\frac{(2n-1)(2n-2)}{3}-(2n-1)\right). $$
令 $z=2n-1\ge5$，括号内就是
$$ 1+\frac{z(z-1)}{3}-z=\frac{(z-1)(z-3)}{3}>0. $$
所以 $C_n>D_n$ 对一切 $n\ge3$ 都成立。于是逐项比较可得
$$ \sinh v\,\sinh(\sqrt2\,v)>\sqrt2\,v^2\cosh v, $$
从而
$$ v\coth v<\frac{\sinh(\sqrt2\,v)}{\sqrt2\,v}<\frac{\sinh w}{w}. $$
这就推出
$$ \sqrt{xy}(e^x+e^y)<2. $$

综上，恒有
$$ \sqrt{xy}(e^x+e^y)\le2. $$
\end{solution}

\]

\[
f(1)=3a,\qquad f(3)=3a+4-4\ln3
\]

\[
\lim_{x\to0^+}f(x)=-\infty,\qquad \lim_{x\to+\infty}f(x)=+\infty.

3a>0,\qquad 3a+4-4\ln3<0.

x_1\in(0,1),\qquad x_2\in(1,3),\qquad x_3\in(3,+\infty).

\[
f\!\left(\frac1x\right)-f(x)
=\left(\frac1x-4\ln\frac1x-3x+3a+2\right)-\left(x-4\ln x-\frac3x+3a+2\right)
=4\left(\frac1x-x+2\ln x\right).
\]

\[
\phi(x)=\frac1x-x+2\ln x,
\]
则
\[
\phi'(x)=-\frac1{x^2}-1+\frac2x=-\frac{(x-1)^2}{x^2}\le0.
\]

\[

\]

\]

\[

\[
f(x_1x_2)+f(x_1x_2x_3)-3a
=\int_1^{x_1x_2}\bigl[f'(u)+x_3f'(x_3u)\bigr]\,du.
\]

&=(x_3+1)-\frac8u+\frac{3\left(1+\frac1{x_3}\right)}{u^2} \\
&=\frac{(x_3+1)u^2-8u+3\left(1+\frac1{x_3}\right)}{u^2}.
\end{align*}
记分子为
\[
N(u)=(x_3+1)u^2-8u+3\left(1+\frac1{x_3}\right).
\]
因为 $x_3>3$，$N(u)$ 的首项系数为正，而它的判别式
\[
\Delta
=64-12(x_3+1)\left(1+\frac1{x_3}\right)
=40-12\left(x_3+\frac1{x_3}\right)
\]
满足
\[
x_3+\frac1{x_3}>3+\frac13=\frac{10}{3},
\]
故 $\Delta<0$。这说明 $N(u)>0$ 对一切实数 $u$ 都成立，于是
\[
f'(u)+x_3f'(x_3u)>0\qquad (u>0).
\]
再加上 $x_1x_2>1$，积分区间确实有正长度，所以
\[
\int_1^{x_1x_2}\bigl[f'(u)+x_3f'(x_3u)\bigr]\,du>0.
\]
于是
\[
f(x_1x_2)+f(x_1x_2x_3)>3a.
\]
\end{solution}

x_i^2e^{a-x_i}=\frac{a^2}{2-x_i}\qquad(i=4,5).
\]
因此
\[
f(x_4)+f(x_5)
=a^2\left(\frac{1}{2-x_4}+\frac{1}{2-x_5}-\ln(x_4x_5)\right)-2a.
\]
记
\[
\Phi=\frac{1}{2-x_4}+\frac{1}{2-x_5}-\ln(x_4x_5)-\frac{2}{a}.
\]
原不等式等价于 \(\Phi>0\)。

当 \(a=0.1\) 时，
\[
g(1.96)>0.1^2e^{-0.1},
\]
而 \(g\) 在 \((1,2)\) 上递减，所以 \(x_5>1.96\)。又 \(0<x_4<1\)，于是
\[
\Phi>\frac12+\frac{1}{2-1.96}-\ln2-\frac{2}{0.1}>0.
\]
若结论不成立，由连续性可取某个 \(a\in(0,1)\) 使 \(\Phi=0\)。这时
\[
x_4x_5\ln(x_4x_5)+\frac{2}{a}x_4x_5-2
=(x_4+x_5-2)\left(\frac{4}{a}+2\ln(x_4x_5)-1\right).
\]
于是
\[
\begin{cases}
2\ln x_4+\ln(2-x_4)-x_4=2\ln a-a, \\
2\ln x_5+\ln(2-x_5)-x_5=2\ln a-a, \\
x_4x_5\ln(x_4x_5)+\dfrac{2}{a}x_4x_5-2=(x_4+x_5-2)\left(\dfrac{4}{a}+2\ln(x_4x_5)-1\right)
\end{cases}
\]
并且
\[
0<a<1,\qquad 0<x_4<1<x_5<2.
\]

先证 \(x_4+x_5>2\)。考察
\[
G(x)=g(1-x)-g(1+x)\qquad(0<x<1),
\]
则
\[
G(x)=e^{x-1}(1-x)(1+x)\bigl(1-x-(1+x)e^{-2x}\bigr).
\]
由 \(e^{-2x}<1-2x+2x^2\) 得
\[
1-x-(1+x)e^{-2x}<1-x-(1+x)(1-2x+2x^2)=0,
\]
所以 \(G(x)<0\)。取 \(x=1-x_4\) 可得
\[
g(x_4)<g(2-x_4).
\]
又 \(g(x_4)=g(x_5)\)，且 \(2-x_4,x_5\in(1,2)\) 上 \(g\) 递减，故
\[
2-x_4<x_5,
\]
也就是 \(x_4+x_5>2\)。

再证 \(x_4x_5<1\)。若 \(x_4\le \frac12\)，结论显然成立；以下设 \(x_4\in(\frac12,1)\)。考察
\[
H(x)=g(x)-g(1/x)\qquad\left(\frac12<x<1\right),
\]
则
\[
H(x)=x^2(2-x)e^{-x}\left(1-e^{x-1}\cdot\frac{2-\frac1x}{x^4(2-x)}\right).
\]
记
\[
h(x)=1-e^{x-1}\cdot\frac{2-\frac1x}{x^4(2-x)}\qquad\left(\frac12<x\le1\right).
\]
有
\[
h'(x)=e^{x-1}\cdot\frac{x(1-x)^2(2x^2-11x+2)}{x^7(2-x)^2}\le0,
\]
因为二次式 \(2x^2-11x+2\) 在 \((\frac12,1]\) 上为负，所以 \(h\) 在该区间递减。再结合 \(h(1)=0\)，可得 \(h(x)>0\)，故 \(H(x)>0\)。取 \(x=x_4\)，得
\[
g(x_4)>g(1/x_4).
\]
再由 \(g(x_4)=g(x_5)\) 及 \(1/x_4,x_5\in(1,2)\) 上 \(g\) 递减，便有
\[
1/x_4>x_5,
\]
即 \(x_4x_5<1\)。

记
\[
u=x_4+x_5>2,\qquad v=x_4x_5<1.
\]
把两条临界方程相加，得
\[
2\ln v+\ln(v-2u+4)-u=4\ln a-2a.
\]
而 \(\Phi=0\) 化为
\[
v\ln v+\frac{2}{a}v-2=(u-2)\left(\frac{4}{a}+2\ln v-1\right).
\]
由此解出
\[
a=\frac{2(v-2u+4)}{4-u+(2u-4-v)\ln v}.
\]
令
\[
1-v=s\in(0,1),\qquad 2u+s-4=t\in(s,1),
\]
则
\[
a=\frac{4(1-t)}{4-t+s+2(1-t)\ln\frac{1}{1-s}}
<\frac{4(1-t)}{4-t+s+2(1-t)s}
=\frac{4(1-t)}{4-t+3s-2ts}
<\frac{4(1-t)}{4-t+3s-6\sqrt{ts}}
<1.
\]
代回上面的对数式，
\[
4\ln a-2a=2\ln(1-s)+\ln(1-t)-t-s+2.
\]

记
\[
p(x)=4\ln x-2x\qquad(x\in(0,1)).
\]
因为 \(p'(x)=4/x-2>0\)，\(p\) 在 \((0,1)\) 上递增，所以
\[
2\ln(1-s)+\ln(1-t)-t-s+2
=p(a)
<p\left(\frac{4(1-t)}{4-t+3s-2ts}\right).
\]

再记
\[
q(t)=2\ln(1-s)+\ln(1-t)-t-s+2-p\left(\frac{4(1-t)}{4-t+3s-2ts}\right).
\]
利用
\[
\frac{1}{4+3s}<\frac{1}{4-t+3s-2ts}<\frac{1}{4(1-t)}
\]
可得
\[
q'(t)>\frac{s+8}{4+3s}-2(4s+12)\left(\frac{1}{4+3s}\right)^2-\frac12
=\frac{s(16-3s)}{2(4+3s)^2}>0.
\]
故 \(q\) 在 \((s,1)\) 上递增，因此
\[
q(t)>q(s)=3\ln(1-s)-2-p\left(\frac{2(1-s)}{(s+1)(2-s)}\right).
\]

最后记
\[
r(s)=3\ln(1-s)-2-p\left(\frac{2(1-s)}{(s+1)(2-s)}\right).
\]
则
\[
r'(s)
=s\frac{-7s^3+22s^2-15s+4}{(s+1)^2(2-s)^2(1-s)}
>s\frac{15s^2-15s+4}{(s+1)^2(2-s)^2(1-s)}
>0.
\]
所以 \(r\) 在 \((0,1)\) 上递增，且 \(r(s)>r(0)=0\)。因此 \(q(t)>0\)，即
\[
2\ln(1-s)+\ln(1-t)-t-s+2
<p\left(\frac{4(1-t)}{4-t+3s-2ts}\right),
\]
这与前面的等式 \(=p(a)\) 矛盾。故 \(\Phi>0\)，原不等式成立。
\end{solution}



\begin{example}{零点与极值题}{}
$f(x) = x^2 e^{a - x} - a^2 \ln x - a \left(a > \frac{1}{2}\right)$\\
$f(x_1) = f(x_2) = f(x_3) = 0 (x_1 < x_2 < x_3), \, f'(x_4) = f'(x_5) = 0 (x_4 < x_5)$\\
(1) 证明：$x_4 f(x_5) + x_5 f(x_4) < 0$；\\
(2) 证明：$(x_2 + x_3 + x_4) f(x_1 x_2 x_3) + (2x_1 + x_5) f(x_2 x_3) > 0$。
\end{example}
