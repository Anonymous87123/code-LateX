\chapter{插入中间函数}
\section{插值+配导数型}
\begin{example}{证明$\e^x-\e x\geqslant (x\ln x)^2$}{}
    求证:$\e^x-\e x\geqslant (x\ln x)^2$
\end{example}
\begin{solution}
记 $L(x) = e^x - ex$，$R(x) = (x\ln x)^2$。由于 $L(1) = R(1) = 0$ 且 $L'(1) = R'(1) = 0$，取中间函数
\[
P(x) = (x-1)^2(Ax^2 + Bx + C),
\]
再令
\[
P(2)=\frac{31}{16},\qquad P'(2)=\frac{75}{16}.
\]
解得
\[
A=\frac{1}{16},\qquad B=\frac{9}{16},\qquad C=\frac{9}{16},
\]
从而
\[
P(x)=\frac{1}{16}(x-1)^2(x^2+9x+9).
\]
于是把问题写成
\[
P(x) \ge (x\ln x)^2,\qquad e^x - ex \ge P(x).
\]

令差值函数 $F(x) = P(x) - x^2\ln^2 x$，易知 $F(1) = 0, F'(1) = 0$。
当 $0<x\le1$ 时，由
\[
-\ln x\le \frac{\left(\frac1x\right)^2-1}{2\cdot \frac1x}
\]
可得
\[
\frac{1}{16}\left(\frac{1}{x}-1\right)^2\left(\frac{1}{x^2}+\frac{9}{x}+9\right)\ge \left(\frac{-\ln x}{x}\right)^2.
\]
约去公共正因子后，剩下
\[
\sqrt{\frac{9}{x^2}+\frac{9}{x}+1}\ge 2\left(\frac1x+1\right).
\]
平方整理即得
\[
\frac{5}{x^2}+\frac{1}{x}-3\ge0,
\]
这在 $0<x\le1$ 上显然成立，故 $F(x)\ge0$。

当 $x \in [1, +\infty)$ 时，
\[
F^{(4)}(x) = \frac{3}{2} + \frac{4\ln x + 2}{x^2} > 0,
\]
故 $F'''(x)$ 递增。由 $F'''(1) < 0 < F'''(2)$，$F'''(x)$ 在 $(1,2)$ 内有唯一零点，从而 $F''(x)$ 先减后增。又由
\[
F''(1) = 0.375 > 0,\qquad F''(1.5)<0,\qquad F''(2)>0
\]
知 $F''(x)$ 恰有两个零点，于是 $F'(x)$ 先增后减再增。再用 $F'(1)=0,\ F'(2)<0$ 可知 $F'(x)$ 在 $x>2$ 处恰有一个零点，因此 $F(x)$ 在该点取得全局极小值。

对 $x \ge 2$，由于 $F'''(x) > 0$，故恒有 $F''(x) \ge F''(2) > 0.122$。应用带有拉格朗日余项的泰勒公式展开：$F(x) = F(2) + F'(2)(x-2) + \frac{F''(c)}{2}(x-2)^2 \ge F(2) + F'(2)(x-2) + \frac{0.122}{2}(x-2)^2$。该二次函数的最小值为 $F(2) - \frac{(F'(2))^2}{2 \times 0.122}$。代入已知边界估算 $F(2) > 0.0110$，$|F'(2)| < 0.015$，极小值不小于 $0.0110 - \frac{0.000225}{0.244} \approx 0.0101 > 0$。因此，当 $x \in [1, +\infty)$ 时，$F(x) \ge 0$。

令差值函数 $H(x) = e^x - ex - P(x)$，已知 $H(1) = 0, H'(1) = 0$。
当 $x \in (0, 1]$ 时，$H^{(4)}(x) = e^x - 1.5$，其唯一零点为 $\ln 1.5 \approx 0.405$。验证端点 $H'''(0) = -1.625 < 0$ 与 $H'''(1) \approx -1.407 < 0$，可知 $H'''(x)$ 先减后增且在 $(0, 1]$ 内恒负。因此 $H''(x)$ 在 $(0, 1]$ 单调递减，由 $H''(1) \approx 0.343 > 0$，知 $H''(x) > 0$ 恒成立。进而 $H'(x)$ 单调递增，由 $H'(1) = 0$ 推知 $H'(x) \le 0$ 恒成立。故 $H(x)$ 在该区间单调递减，结合 $H(1) = 0$，得 $H(x) \ge 0$。

当 $x \in [1, +\infty)$ 时，$H^{(4)}(x) = e^x - 1.5 > 0$，故 $H'''(x)$ 递增。由 $H'''(1) < 0 < H'''(2)$，$H'''(x)$ 在 $(1,2)$ 内有唯一零点，于是 $H''(x)$ 先减后增。再由
\[
H''(1) > 0,\qquad H''(1.5) < 0,\qquad H''(2) > 0
\]
知 $H''(x)$ 有两个零点，因此 $H'(x)$ 先增后减再增。又因 $H'(2) < 0$，$H'(x)$ 在 $x > 2$ 处恰有一个零点，故 $H(x)$ 在该点取得极小值。
对 $x \ge 2$，因 $H''(x) \ge H''(2)$，泰勒公式得：$H(x) \ge H(2) + H'(2)(x-2) + \frac{0.137}{2}(x-2)^2$。该二次函数的最小值为 $H(2) - \frac{(H'(2))^2}{2 \times 0.137}$。结合边界 $H(2) > 0.0115$，$|H'(2)| < 0.0195$，极小值不小于 $0.0115 - \frac{0.00038}{0.274}  > 0$。因此，当 $x \in [1, +\infty)$ 时，$H(x) \ge 0$。

于是
\[
e^x-ex\ge P(x)\ge (x\ln x)^2\qquad (x>0).
\]
\end{solution}

\begin{example}{证明$x^{1-x}(1-x)^x\le \frac{\sin\pi x}{2}$}{}
对任意 $x \in (0, 1)$，不等式 $x^{1-x}(1-x)^x\le \frac{\sin\pi x}{2}$ 均成立
\end{example}
\begin{solution}
\[
M(x)=f(t),\qquad t=x(1-x),\qquad f(t)=at^2+bt+c.
\]
当 $x\to0^+$ 时，$t\to0$，且两边都趋于 $0$，所以 $c=0$。又在 $x=\frac12$ 处，$t=\frac14$，并且
\[
L\left(\frac12\right)=R\left(\frac12\right)=\frac12,
\]
从而
\[
a + 4b = 8.
\]

由 $t\sim x$，$L(x)\sim x$，$R(x)\sim \frac{\pi}{2}x$，可知
\[
1 \le b \le \frac{\pi}{2}.
\]

令 $u=x-\frac12$，则
\[
R\left(u+\frac{1}{2}\right) = \frac{\cos \pi u}{2} = \frac{1}{2} - \frac{\pi^2}{4}u^2 + \mathcal{O}(u^4).
\]
\[
L\left(u+\frac{1}{2}\right) = \frac{1}{2} - 3u^2 + \frac{13}{3}u^4 + \mathcal{O}(u^6).
\]
\[
f(t) = \frac{1}{2} - (4-b)u^2 + au^4.
\]
故二次项必须满足
\[
-3 \le -(4-b) \le -\frac{\pi^2}{4},
\]
\[
1 \le b \le 4 - \frac{\pi^2}{4}.
\]
\(b=1\) 时有 \(a=4\)，这时左侧四次项 \(\frac{13}{3}\) 大于中间函数的 \(4\)，离开中心后会有 \(L(x)>M(x)\)。取最简单的值
\[
b = \frac{3}{2},
\]
便得 $a = 2$。于是
\[
M(x) = 2x^2(1-x)^2 + \frac{3}{2}x(1-x).
\]
三者都关于 $x=\frac12$ 对称，所以只看
\[
L(x)\le M(x)\le R(x)\qquad \left(0<x\le \frac12\right).
\]

先看左边 $L(x) \le M(x)$。
当 $x \in (0, \frac{1}{2}]$ 时，两端函数值均大于 $0$。对不等式两边同时取自然对数，即证 $(1-x)\ln x + x\ln(1-x) \le \ln\left(2x^2(1-x)^2 + \frac{3}{2}x(1-x)\right)$ 成立。
令 $t = x(1-x)$，当 $x \in (0, \frac{1}{2}]$ 时，$t \in (0, \frac{1}{4}]$ 且单调递增。不等式左边可恒等变形为 $\ln t - x\ln x - (1-x)\ln(1-x)$，右侧可化为 $\ln t + \ln\left(2t + \frac{3}{2}\right)$。消去两侧相同的 $\ln t$，原不等式等价于证明 $-x\ln x - (1-x)\ln(1-x) \le \ln\left(2t + \frac{3}{2}\right)$。
记 $H(x) = \ln\left(2t + \frac{3}{2}\right) + x\ln x + (1-x)\ln(1-x)$。证明 $L(x)\le M(x)$ 就是要说明 $H(x)\ge0$。
对 $H(x)$ 求一阶导数，注意到 $\frac{\mathrm{d}t}{\mathrm{d}x} = 1-2x$，有
$$ H'(x) = \frac{2(1-2x)}{2t + \frac{3}{2}} + \ln x + 1 - \ln(1-x) - 1 = \frac{4(1-2x)}{4t + 3} + \ln x - \ln(1-x). $$
继续对 $H'(x)$ 求二阶导数，利用除法求导法则得
$$ H''(x) = \frac{-8(4t+3) - 4(1-2x) \cdot 4(1-2x)}{(4t+3)^2} + \frac{1}{x} + \frac{1}{1-x}. $$
由于 $(1-2x)^2 = 1 - 4x + 4x^2 = 1 - 4t$，且 $\frac{1}{x} + \frac{1}{1-x} = \frac{1}{x(1-x)} = \frac{1}{t}$，可将 $H''(x)$ 完全转化为关于 $t$ 的表达式：
$$ H''(x) = \frac{-32t - 24 - 16(1-4t)}{(4t+3)^2} + \frac{1}{t} = \frac{32t - 40}{(4t+3)^2} + \frac{1}{t}. $$
通分并合并同类项，得
$$ H''(x) = \frac{t(32t - 40) + (4t+3)^2}{t(4t+3)^2} = \frac{32t^2 - 40t + 16t^2 + 24t + 9}{t(4t+3)^2} = \frac{48t^2 - 16t + 9}{t(4t+3)^2}. $$
对于分子二次多项式 $48t^2 - 16t + 9$，其判别式 $\Delta = (-16)^2 - 4 \times 48 \times 9 = 256 - 1728 < 0$，且二次项系数 $48 > 0$，故对任意实数 $t$，恒有 $48t^2 - 16t + 9 > 0$。由于在区间 $x \in (0, \frac{1}{2})$ 内 $t > 0$，因此 $H''(x) > 0$ 恒成立。
这表明一阶导数 $H'(x)$ 在区间 $(0, \frac{1}{2}]$ 上严格单调递增。结合 $H'\left(\frac{1}{2}\right) = 0 + \ln\frac{1}{2} - \ln\frac{1}{2} = 0$，可知当 $x \in (0, \frac{1}{2})$ 时，恒有 $H'(x) < 0$。
从而原函数 $H(x)$ 在 $(0, \frac{1}{2}]$ 上严格单调递减。代入端点值 $H\left(\frac{1}{2}\right) = \ln\left(2 \times \frac{1}{4} + \frac{3}{2}\right) + \frac{1}{2}\ln\frac{1}{2} + \frac{1}{2}\ln\frac{1}{2} = \ln 2 - \ln 2 = 0$，可知对于所有的 $x \in (0, \frac{1}{2}]$，均有 $H(x) \ge 0$ 成立。左侧不等式得证。

再看 $M(x)\le R(x)$。令 $u=\frac12-x$，则 $u\in[0,\frac12)$，并且
\[
t=x(1-x)=\frac14-u^2.
\]
于是
$$ M(x) = 2\left(\frac{1}{4} - u^2\right)^2 + \frac{3}{2}\left(\frac{1}{4} - u^2\right) = 2\left(\frac{1}{16} - \frac{1}{2}u^2 + u^4\right) + \frac{3}{8} - \frac{3}{2}u^2 = \frac{1}{2} - \frac{5}{2}u^2 + 2u^4. $$
又
\[
R(x)=\frac{\cos \pi u}{2}.
\]
于是变成
\[
g(u)=\cos \pi u - 1 + 5u^2 - 4u^4\ge0\qquad \left(0\le u<\frac12\right).
\]
对 $g(u)$ 依次求各阶导数：
$$ g'(u) = -\pi\sin \pi u + 10u - 16u^3, $$
$$ g''(u) = -\pi^2\cos \pi u + 10 - 48u^2, $$
$$ g'''(u) = \pi^3\sin \pi u - 96u, $$
$$ g^{(4)}(u) = \pi^4\cos \pi u - 96, $$
$$ g^{(5)}(u) = -\pi^5\sin \pi u. $$
当 $u \in (0, \frac{1}{2})$ 时，$g^{(5)}(u) < 0$ 恒成立，因此 $g^{(4)}(u)$ 在该区间内严格单调递减。考虑到 $\pi > 3.14$，有 $\pi^4 > 3.14^4 \approx 97.2 > 96$。因此 $g^{(4)}(0) = \pi^4 - 96 > 0$，又 $g^{(4)}\left(\frac{1}{2}\right) = -96 < 0$。由零点存在定理可知，$g^{(4)}(u)$ 在 $\left(0, \frac{1}{2}\right)$ 内存在唯一零点。这表明 $g'''(u)$ 在该区间内先严格单调递增，后严格单调递减。
考虑到 $\pi < 3.15$，有 $\pi^3 < 3.15^3 \approx 31.3 < 48$。因此 $g'''(0) = 0$，且 $g'''\left(\frac{1}{2}\right) = \pi^3 - 48 < 0$。因 $g'''(u)$ 从 $0$ 上升至正值后再下降至负值，故 $g'''(u)$ 在 $\left(0, \frac{1}{2}\right)$ 内存在唯一零点。同理推知，$g''(u)$ 在该区间内先严格单调递增，后严格单调递减。
考虑到 $\pi^2 < 10$，有 $g''(0) = 10 - \pi^2 > 0$，且 $g''\left(\frac{1}{2}\right) = 10 - 12 = -2 < 0$。因 $g''(u)$ 从正值上升后再下降至负值，故 $g''(u)$ 在 $\left(0, \frac{1}{2}\right)$ 内存在唯一零点。推知 $g'(u)$ 在该区间内亦先严格单调递增，后严格单调递减。
计算端点值得 $g'(0) = 0$，且 $g'\left(\frac{1}{2}\right) = -\pi + 5 - 2 = 3 - \pi < 0$。因 $g'(u)$ 从 $0$ 上升至正值后再下降至负值，故 $g'(u)$ 在 $\left(0, \frac{1}{2}\right)$ 内存在唯一零点。
所以 $g'(u)$ 在区间内先正后负，$g(u)$ 也就先增后减。
验证区间端点值：
$$ g(0) = \cos 0 - 1 + 0 = 0, $$
$$ g\left(\frac{1}{2}\right) = \cos\frac{\pi}{2} - 1 + 5 \times \frac{1}{4} - 4 \times \frac{1}{16} = 0. $$
由 $g(0)=g\left(\frac12\right)=0$，且 $g$ 先增后减，故
\[
g(u)\ge0\qquad \left(0\le u\le\frac12\right).
\]

因此
\[
x^{1-x}(1-x)^x \le 2x^2(1-x)^2 + \frac{3}{2}x(1-x) \le \frac{\sin\pi x}{2}\qquad (0<x<1).
\]
\end{solution}


\section{估阶+配导数型}
\begin{example}{证明$2\ln 2 \cdot \ln(\sqrt{x}+\sqrt{1-x}) \geq \ln x \cdot \ln(1-x)$}{}
    对于任意$x\in(0,1)$，恒成立$2\ln 2 \cdot \ln(\sqrt{x}+\sqrt{1-x}) \geq \ln x \cdot \ln(1-x)$
\end{example}
\begin{solution}
设
\[
L(x)=2\ln 2 \cdot \ln(\sqrt{x}+\sqrt{1-x}),\qquad R(x)=\ln x\ln(1-x).
\]
由对称性，取中间函数
\[
f(x) = C \sqrt{x(1-x)}.
\]
由
\[
L\left(\frac12\right)=R\left(\frac12\right)=(\ln2)^2
\]
得
\[
f(x) = 2(\ln 2)^2 \sqrt{x(1-x)}.
\]
把原不等式拆成
\[
L(x)\ge f(x),\qquad f(x)\ge R(x).
\]

令 \(w = 2\sqrt{x(1-x)}\in(0,1]\)。由
\[
(\sqrt{x}+\sqrt{1-x})^2 = 1 + w
\]
左边不等式化为
\[
\ln(1+w)\ge w\ln 2.
\]
记 \(g(w)=\ln(1+w)-w\ln2\)。因为
\[
g''(w)=-\frac{1}{(1+w)^2}<0,
\]
且 \(g(0)=g(1)=0\)，所以 \(g(w)\ge0\)，从而 \(L(x)\ge f(x)\)。

再令 \(x=\frac{1-y}{2}\)，其中 \(y\in[0,1)\)。右边不等式化为
$$ (\ln 2)^2 \sqrt{1-y^2} \geq (\ln(1-y)-\ln 2)(\ln(1+y)-\ln 2). $$
写成
$$ F(y) = (\ln 2)^2(1 - \sqrt{1-y^2}) - \ln 2 \ln(1-y^2) + \ln(1-y)\ln(1+y), $$
要证的是 \(F(y)\le0\)。这里 \(F(0)=0\)，并且由 \(\lim\limits_{z \to 0^+} z\ln z = 0\) 可得
\[
\lim_{y \to 1^-} F(y)=0.
\]

再记
\[
K(y)=(1-y^2)F'(y)
\]
为
$$ K(y) = (\ln 2)^2 y \sqrt{1-y^2} + 2y \ln 2 - (1+y)\ln(1+y) + (1-y)\ln(1-y). $$
于是 \(K(0)=0\)，\(\lim_{y\to1^-}K(y)=0\)，并且
$$ K'(y) = (\ln 2)^2 \frac{1-2y^2}{\sqrt{1-y^2}} + 2\ln 2 - 2 - \ln(1-y^2). $$
令 \(t=\sqrt{1-y^2}\in(0,1)\)，则 \(K'(y)=H(t)\)，其中
$$ H(t) = (\ln 2)^2 \frac{2t^2-1}{t} + 2\ln 2 - 2 - 2\ln t. $$
再求导，
$$ H'(t) = \frac{2(\ln 2)^2 t^2 - 2t + (\ln 2)^2}{t^2}. $$
分子是开口向上的二次式，且
\[
\Delta = 4 - 8(\ln 2)^4 > 0,
\]
所以 \(H'(t)\) 在 \((0,1)\) 内恰有一个零点，\(H(t)\) 在这一区间先增后减。

把 \(K(1)\) 补成 \(0\) 后，\(K\) 在 \([0,1]\) 上连续，在 \((0,1)\) 上可导。由罗尔定理，\(K'(y)\) 在 \((0,1)\) 内至少有一个零点，也就是 \(H(t)\) 在 \((0,1)\) 内至少有一个零点。再结合
\[
\lim_{t \to 0^+} H(t) = -\infty,\qquad H(1) = (\ln 2)^2 + 2\ln 2 - 2 < 0,
\]
可知 \(H(t)\) 的极大值必须大于 \(0\)，因此 \(H(t)\) 在 \((0,1)\) 内恰有两个零点。

于是 \(K'(y)\) 的符号依次为负、正、负，所以 \(K(y)\) 先减后增再减。
\[
K(0)=0,\qquad \lim_{y\to1^-}K(y)=0.
\]
因此 \(K\) 在 \((0,1)\) 内恰有一个零点，记为 \(y_0\)。
\[
F'(y)=\frac{K(y)}{1-y^2}
\]
在 \((0,y_0)\) 上为负，在 \((y_0,1)\) 上为正，所以 \(F(y)\) 先减后增。再用 \(F(0)=0\) 和 \(\lim_{y\to1^-}F(y)=0\)，便得 \(F(y)\le0\)。等号只在 \(y=0\) 取到，也就是 \(x=\frac12\)。

因此原不等式两端的等号都只在 \(x=\frac12\) 成立。
\end{solution}


\section{分离+泰勒型}
\begin{example}{证明$x-\ln x-\frac{\ln(x-\ln x)}{x(x-\ln x-1)}+\ln\frac{\ln(x-\ln x)}{x(x-\ln x-1)}>0$}{}
证明当 $x>1$ 时，下式成立：
$$x-\ln x-\frac{\ln(x-\ln x)}{x(x-\ln x-1)}+\ln\frac{\ln(x-\ln x)}{x(x-\ln x-1)}>0$$
\end{example}
\begin{solution}
令 $t=x-\ln x$，并记
\[
L(t)=t+\ln\frac{\ln t}{t-1}.
\]
当 $t\to1^+$ 时，$L(t)\to1$，$L'(t)\to\frac12$，因此拿
\[
\frac{t+1}{2}
\]
来比较最方便。

利用对数运算法则，原式可写成
$$x-\ln x+\ln\frac{\ln(x-\ln x)}{x-\ln x-1} > \ln x+\frac{\ln(x-\ln x)}{x(x-\ln x-1)}.$$
又因为
\[
t'=1-\frac1x>0\qquad(x>1),
\]
所以 $t>1$。

先看
\[
L(t)-\frac{t+1}{2}=\frac{t-1}{2}+\ln(\ln t)-\ln(t-1).
\]
记
$$g(t)=\frac{t-1}{2}+\ln(\ln t)-\ln(t-1).$$
则
$$g'(t) = \frac{1}{2} + \frac{1}{t\ln t} - \frac{1}{t-1} = \frac{t(t-3)\ln t + 2(t-1)}{2t(t-1)\ln t}.$$
分母恒正，只看分子
$$k(t) = t(t-3)\ln t + 2(t-1).$$
继续求导，
$$k'(t) = (2t-3)\ln t + t - 1,$$
$$k''(t) = 2\ln t + 3 - \frac{3}{t} = 2\ln t + 3\left(1 - \frac{1}{t}\right).$$
当 $t>1$ 时，$k''(t)>0$，所以 $k'(t)$ 递增。又 $k'(1)=0$，故 $k'(t)>0$；再由 $k(1)=0$，便有 $k(t)>0$，从而 $g'(t)>0$。结合
\[
\lim_{t\to 1^+}g(t)=0,
\]
可知 $g(t)>0$，也就是
\[
L(t)>\frac{t+1}{2}.
\]

另一边，
\[
\frac{t+1}{2} > \ln x+\frac{\ln t}{x(t-1)}
\]
等价于
$$\frac{x-3\ln x+1}{2} > \frac{\ln(x-\ln x)}{x(x-\ln x-1)}.$$
对任意 $u>1$，都有
\[
\frac{\ln u}{u-1} < \frac{1}{\sqrt{u}},
\]
这里可取
\[
\varphi(u)=u-1-\sqrt{u}\ln u
\]
来看单调性。代入 $u=t$，便得
\[
\frac{\ln t}{x(t-1)} < \frac{1}{x\sqrt{t}} = \frac{1}{x\sqrt{x-\ln x}}.
\]
于是比较成
$$\frac{x-3\ln x+1}{2} \ge \frac{1}{x\sqrt{x-\ln x}}.$$

记
$$P(x)=x-3\ln x+1.$$
由 $P'(x)=1-\frac3x$ 可知，$P(x)$ 在 $x=3$ 处取极小值 $4-3\ln3$。再用 $\ln27<4$，得到 $P(x)>0$。两边平方并交叉相乘后，问题变成
$$G(x)=x^2(x-\ln x)P^2(x)-4\ge0\qquad(x>1).$$
这里 $G(1)=0$，并且
$$G'(x) = xP(x)\left[(3x-2\ln x-1)P(x) + 2(x-\ln x)(x-3)\right].$$
把方括号内记为 $M(x)$，展开得
$$M(x) = 5x^2 - 13x\ln x - 4x + 6\ln^2 x + 7\ln x - 1.$$
再求导，
$$M'(x) = 10x - 13\ln x - 17 + \frac{12\ln x + 7}{x},$$
$$M''(x) = \frac{10x^2 - 13x + 5 - 12\ln x}{x^2}.$$
由 $\ln x\le x-1$，
$$10x^2 - 13x + 5 - 12\ln x \ge 10x^2 - 25x + 17.$$
而二次式 $10x^2-25x+17$ 的判别式是
\[
\Delta=-55<0,
\]
又开口向上，所以它恒正，进而 $M''(x)>0$。于是 $M'(x)$ 递增，$M'(1)=0$，故 $M'(x)>0$；再由 $M(1)=0$，便有 $M(x)>0$。因为 $xP(x)>0$，所以 $G'(x)>0$，从而 $G(x)\ge G(1)=0$。

原不等式对 $x>1$ 成立。
\end{solution}




\begin{example}{证明$\ln\left(\sqrt{1+\frac{x^2}{2}}-x\right) + e^x \le \sqrt{e^x-x}$}{}
证明$$\ln\left(\sqrt{1+\frac{x^2}{2}}-x\right) + e^x \le \sqrt{e^x-x}$$
\end{example}

\begin{solution}
记
\[
s=e^x-x-1\ge0.
\]
取
\[
C(x)=\frac{s}{2}+\ln\left(\cosh\frac{s}{2}\right).
\]
由
\[
\ln(\cosh v)=\frac{v^2}{2}+\mathcal O(v^4)
\]
可得
\[
C(x)=\frac12(e^x-x-1)+\frac18(e^x-x-1)^2+\mathcal O\!\left((e^x-x-1)^4\right),
\]
它正好和
\[
(e^x-x)-\sqrt{e^x-x}
\]
在 $x=0$ 附近同阶。再用
\[
\cosh y=\frac{e^y+e^{-y}}{2}
\]
化简，得到
\[
C(x)=\ln\left(\frac{e^{e^x-x-1}+1}{2}\right).
\]
于是设
\[
f(x)=e^x-x-C(x)=e^x-\ln\left(\frac{e^x+e^{e^x-1}}{2}\right).
\]
接着把 \(f(x)\) 卡在两边之间。

先看
\[
f(x)\le \sqrt{e^x-x}.
\]
令 $t=e^x-x$。当 $x\in(0,\sqrt2)$ 时，
\[
t'=e^x-1>0,
\]
所以 $t>1$。此时
\[
f(x)=t-\ln\left(\frac{1+e^{t-1}}{2}\right),
\]
于是记
$$G(t)=\sqrt{t}-t+\ln\left(\frac{1+e^{t-1}}{2}\right).$$
有 $G(1)=0$，并且
$$ G'(t) = \frac{1}{2\sqrt{t}} - 1 + \frac{e^{t-1}}{1+e^{t-1}} = \frac{1}{2\sqrt{t}} - \frac{1}{1+e^{t-1}}. $$
对 $t>1$，由 $e^{t-1}>1+(t-1)=t$ 以及 $1+t>2\sqrt t$，可得
\[
1+e^{t-1}>2\sqrt t.
\]
因此
\[
\frac{1}{1+e^{t-1}}<\frac{1}{2\sqrt t},
\]
从而 $G'(t)>0$。所以 $G(t)>G(1)=0$，右边不等式成立。

再看
\[
\ln\left(\sqrt{1+\frac{x^2}{2}}-x\right) + e^x \le f(x).
\]
消去两边的 $e^x$ 后，问题变成
\[
u_1(x)+u_2(x)\le0,
\]
其中
$$ u_1(x) = \ln\left(\sqrt{1+\frac{x^2}{2}}-x\right), \quad u_2(x) = \ln\left(\frac{e^x+e^{e^x-1}}{2}\right). $$
这里 $u_1(0)+u_2(0)=0$，因此转成证明 $u_1'(x)+u_2'(x)<0$。

分别求导。对 $u_1(x)$ 有
$$ -u_1'(x) = -\frac{\frac{x}{2\sqrt{1+x^2/2}} - 1}{\sqrt{1+\frac{x^2}{2}}-x} = \frac{x}{2-x^2} + \frac{2}{(2-x^2)\sqrt{1+\frac{x^2}{2}}} \triangleq A(x). $$
对 $u_2(x)$ 有
$$ u_2'(x) = \frac{e^x + e^{e^x-1} \cdot e^x}{e^x + e^{e^x-1}} = \frac{1+e^{e^x-1}}{1+e^{e^x-x-1}} = 1 + \frac{e^x-1}{1+e^{-(e^x-x-1)}} \triangleq B(x). $$
问题就落在比较 $A(x)$ 和 $B(x)$。

当 $x>0$ 时，
\[
\left(1+\frac{x^2}{4}\right)^2=1+\frac{x^2}{2}+\frac{x^4}{16}>1+\frac{x^2}{2},
\]
所以
\[
\sqrt{1+\frac{x^2}{2}}<1+\frac{x^2}{4}.
\]
代回去便有
$$ A(x) > \frac{x}{2-x^2} + \frac{2}{(2-x^2)\left(1+\frac{x^2}{4}\right)} = \frac{x^3+4x+8}{8-2x^2-x^4} \triangleq A_{\text{low}}(x). $$

再看 $B(x)$。令 $v=e^x-x-1>0$，由切线不等式 $e^{-v}\ge1-v$，
\[
e^{-(e^x-x-1)}\ge2+x-e^x,
\]
于是
\[
1+e^{-(e^x-x-1)}\ge3+x-e^x.
\]
记
\[
\psi(x)=3+x-e^x.
\]
因为 $\psi''(x)=-e^x<0$，所以 $\psi$ 在 $[0,\sqrt2]$ 上下凸；又
\[
\psi(0)=2>0,\qquad \psi(\sqrt2)=3+\sqrt2-e^{\sqrt2}>0,
\]
便知 $\psi(x)>0$。因此
$$ B(x) \le 1 + \frac{e^x-1}{3+x-e^x} = \frac{x+2}{3+x-e^x}. $$

若
\[
\frac{x+2}{3+x-e^x} \le A_{\text{low}}(x),
\]
就有 $B(x)<A(x)$。整理后等价于
$$ e^x \le 3+x - \frac{x+2}{A_{\text{low}}(x)} = \frac{(3+x)(x^3+4x+8) - (x+2)(8-2x^2-x^4)}{x^3+4x+8}. $$
右边化简为
$$ e^x \le \frac{x^5+3x^4+5x^3+8x^2+12x+8}{x^3+4x+8} \triangleq U(x). $$

这里用三阶 Padé 逼近
\[
P_3(x)=\frac{120+60x+12x^2+x^3}{120-60x+12x^2-x^3}.
\]
设
\[
S(x)=120+60x+12x^2+x^3-e^x(120-60x+12x^2-x^3).
\]
则
\[
S^{(4)}(x)=x^3e^x>0,
\]
并且
\[
S(0)=S'(0)=S''(0)=S'''(0)=0.
\]
顺次推出 $S(x)>0$，所以
\[
e^x\le P_3(x)\qquad(x>0).
\]
在 $x\in(0,\sqrt2)$ 上，分母
\[
120-60x+12x^2-x^3=60(2-x)+x^2(12-x)
\]
恒为正。

再比 $P_3(x)$ 与 $U(x)$。两边分母都为正，只看
$$ \Delta(x) = (x^5+3x^4+5x^3+8x^2+12x+8)(120-60x+12x^2-x^3) - (x^3+12x^2+60x+120)(x^3+4x+8). $$
展开后
$$ \Delta(x) = x^3(-x^5+9x^4-30x^3-20x^2+80x+80). $$
把括号里的多项式记作 $Q(x)$，并写成
\[
Q(x)=x^4(9-x)+20(-1.5x^3-x^2+4x+4).
\]
在 $x\in(0,\sqrt2)$ 上，$x^4(9-x)>0$。再记
\[
g(x)=-1.5x^3-x^2+4x+4.
\]
它的导数
\[
g'(x)=-4.5x^2-2x+4
\]
在正半轴只有一个零点
\[
x_0=\frac{2\sqrt{19}-2}{9}.
\]
所以 $g(x)$ 在 $[0,\sqrt2]$ 上先增后减。再算端点，
\[
g(0)=4>0,\qquad g(\sqrt2)=2+\sqrt2>0,
\]
便得 $g(x)>0$，从而 $Q(x)>0$，进而 $\Delta(x)>0$。这说明
\[
P_3(x)<U(x).
\]

于是
\[
e^x\le P_3(x)<U(x),
\]
从而
\[
\frac{x+2}{3+x-e^x}<A_{\text{low}}(x)<A(x),
\]
也就是 $B(x)<A(x)$。因此 $u_1'(x)+u_2'(x)<0$，再用 $u_1(0)+u_2(0)=0$，便有
\[
u_1(x)+u_2(x)<0\qquad(x\in(0,\sqrt2)).
\]
左边不等式得证。

因此原不等式在 $x\in(0,\sqrt2)$ 上成立。给定区间 $\left[\frac12,\sqrt2\right)$ 包含在其中，原命题得证。
\end{solution}


\section{pade逼近}
\begin{example}{$\frac{1}{\sqrt{h(x)}} + \frac{4}{\sqrt{h(1/x)}+1} \ge 3$}{}
已知函数 $h(x) = x e^{x-1}$，求证：对于任意 $x>0$，恒有
\[
\frac{1}{\sqrt{h(x)}} + \frac{4}{\sqrt{h(1/x)}+1} \ge 3.
\]
\end{example}
\begin{solution}
令
\[
t(x)=\frac{1}{\sqrt{h(x)}}.
\]
因为 \(\sqrt{h(1/x)}=\frac1{t(1/x)}\)，原不等式可化为
\[
t(x)+\frac{4t(1/x)}{t(1/x)+1}\ge3.
\]
再整理一下，就是
\[
(t(x)+1)(t(1/x)+1)\ge4.
\]
这一步已经完全对称，所以只看 \(0<x\le1\) 即可。

取
\[
m(x)=\frac{-x^2+22x+51}{8(7x+2)}.
\]
接下来分别看
\[
t(x)+1\ge2m(x),\qquad t(1/x)+1\ge\frac{2}{m(x)}.
\]

先看第一条。代入 \(t(x)=\frac{1}{\sqrt{x}}e^{\frac{1-x}{2}}\) 后，
\[
\frac{1}{\sqrt{x}}e^{\frac{1-x}{2}} \ge \frac{-x^2+22x+51}{4(7x+2)} - 1 = \frac{-x^2-6x+43}{4(7x+2)}.
\]
当 \(0<x\le1\) 时，右边是正的，于是可以取对数。记
\[
F(x)=\frac{1-x}{2}-\frac12\ln x-\ln(-x^2-6x+43)+\ln(28x+8).
\]
那么
\[
F'(x)= -\frac{1}{2} - \frac{1}{2x} + \frac{2x+6}{-x^2-6x+43} + \frac{28}{28x+8}
= \frac{-(x-1)^3(7x+86)}{2x(7x+2)(x^2+6x-43)}.
\]
在 \(0<x<1\) 内，分子为正，分母为负，所以 \(F'(x)<0\)。再用 \(F(1)=0\)，便得 \(F(x)\ge0\)，第一条成立。

再看第二条。把 \(t(1/x)=\sqrt{x}e^{\frac{x-1}{2x}}\) 代回去，
\[
\sqrt{x}e^{\frac{x-1}{2x}} \ge \frac{16(7x+2)}{-x^2+22x+51} - 1 = \frac{x^2+90x-19}{-x^2+22x+51}.
\]
这里分母始终为正。若 \(x^2+90x-19\le0\)，右边不大于 \(0\)，不等式立刻成立。剩下
\[
x\in(\sqrt{2044}-45,1].
\]
这时右边为正，可以再取对数。记
\[
2G(x)=\frac{x-1}{x}+\ln x-2\ln(x^2+90x-19)+2\ln(-x^2+22x+51).
\]
求导后得到
\[
2G'(x)=\frac{1}{x^2} + \frac{1}{x} - \frac{2(2x+90)}{x^2+90x-19} + \frac{2(-2x+22)}{-x^2+22x+51}
= \frac{(x-1)^3(x^2+296x-969)}{x^2(x^2-22x-51)(x^2+90x-19)}.
\]
在区间 \((\sqrt{2044}-45,1)\) 内，分子为正，分母为负，所以 \(G'(x)<0\)。又 \(G(1)=0\)，因此 \(G(x)\ge0\)，第二条也成立。

两式相乘便有
\[
(t(x)+1)(t(1/x)+1)\ge 2m(x)\cdot\frac{2}{m(x)}=4.
\]
再用对称性，原不等式对一切 \(x>0\) 都成立。
\end{solution}
